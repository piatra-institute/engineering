% Figure: Monod specific growth rate mu vs substrate concentration S,
% saturating shape, with K_S marked at half-maximum.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={->, thick},
  curve/.style={engblue, very thick},
  dashed-line/.style={densely dashed, engblack!60},
  lab/.style={font=\small},
]

\draw[axis] (0, 0) -- (8.5, 0) node[right, lab] {$S$ (substrate concentration)};
\draw[axis] (0, 0) -- (0, 4.5) node[above, lab] {$\mu$ (specific growth rate)};

% mu(S) = mu_max * S / (K_S + S); mu_max ~ 3.6 in plot units, K_S ~ 0.8
\draw[curve, smooth, samples=60, domain=0:8.0]
  plot (\x, {3.6 * \x / (0.8 + \x)});

% mu_max horizontal asymptote
\draw[dashed-line] (0, 3.6) -- (8.0, 3.6);
\node[lab, anchor=south east, engblue] at (8.0, 3.6) {$\mu_{\max}$};

% K_S point at mu = mu_max / 2 = 1.8
\draw[dashed-line] (0, 1.8) -- (0.8, 1.8);
\draw[dashed-line] (0.8, 0) -- (0.8, 1.8);
\node[lab, anchor=east] at (0, 1.8) {$\mu_{\max}/2$};
\node[lab, anchor=north] at (0.8, 0) {$K_S$};

% Sample data points (synthetic)
\fill[engred] (0.2, 0.72) circle (1.5pt);
\fill[engred] (0.4, 1.20) circle (1.5pt);
\fill[engred] (0.8, 1.80) circle (1.5pt);
\fill[engred] (1.6, 2.40) circle (1.5pt);
\fill[engred] (3.2, 2.88) circle (1.5pt);
\fill[engred] (5.0, 3.10) circle (1.5pt);
\fill[engred] (7.0, 3.32) circle (1.5pt);

\node[lab, engred, anchor=west] at (5.5, 1.2) {batch data};

\end{tikzpicture}
\caption{The Monod expression $\mu(S) = \mu_{\max} S / (K_S + S)$ as a
saturating function of substrate concentration. The maximum specific
growth rate $\mu_{\max}$ is approached as $S \to \infty$. The
half-saturation constant $K_S$ is the substrate concentration at which
$\mu = \mu_{\max}/2$; for many heterotrophs on glucose, $K_S$ falls in
the milligram-per-litre range, far below typical batch starting
substrate concentrations (so batch culture operates near $\mu_{\max}$
for most of its course). Red points are synthetic batch-culture
specific-growth-rate observations; the smooth curve is the fitted
Monod expression. The form is the same saturating shape as the
Michaelis-Menten enzyme expression, but $K_S$ is a whole-cell empirical
parameter and not a single enzyme constant.}
\label{fig:vol06ch10:monod}
\end{figure}
