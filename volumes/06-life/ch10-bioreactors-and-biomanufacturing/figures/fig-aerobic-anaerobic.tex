% Figure: aerobic vs anaerobic fermentation: contrasting energy yields
% and heat release per mole substrate.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  bar/.style={draw=engblack, thick},
  aerobic/.style={fill=engblue!40},
  anaerobic/.style={fill=englime!40},
  lab/.style={font=\small, align=center},
  axis/.style={->, thick},
]

\draw[axis] (0, 0) -- (0, 5.5) node[above, lab] {value};
\draw[axis] (0, 0) -- (10.5, 0);

% Group 1: ATP per glucose
\node[lab] at (1.5, -0.4) {ATP / glucose};
\filldraw[bar, aerobic] (0.6, 0) rectangle (1.4, 4.6);
\node[lab, anchor=south, font=\scriptsize] at (1.0, 4.6) {$\sim 30$};
\filldraw[bar, anaerobic] (1.6, 0) rectangle (2.4, 0.3);
\node[lab, anchor=south, font=\scriptsize] at (2.0, 0.3) {$2$};

% Group 2: heat MJ/mol glucose
\node[lab] at (4.5, -0.4) {heat (MJ / mol C$_6$)};
\filldraw[bar, aerobic] (3.6, 0) rectangle (4.4, 2.87);
\node[lab, anchor=south, font=\scriptsize] at (4.0, 2.87) {$2.87$};
\filldraw[bar, anaerobic] (4.6, 0) rectangle (5.4, 0.22);
\node[lab, anchor=south, font=\scriptsize] at (5.0, 0.22) {$\sim 0.22$};

% Group 3: biomass yield Y_X/S g/g
\node[lab] at (7.5, -0.4) {$Y_{X/S}$ (g / g)};
\filldraw[bar, aerobic] (6.6, 0) rectangle (7.4, 2.25);
\node[lab, anchor=south, font=\scriptsize] at (7.0, 2.25) {$0.45$};
\filldraw[bar, anaerobic] (7.6, 0) rectangle (8.4, 0.50);
\node[lab, anchor=south, font=\scriptsize] at (8.0, 0.50) {$0.10$};

% Legend
\filldraw[bar, aerobic] (8.6, 4.2) rectangle (8.9, 4.5);
\node[lab, anchor=west, font=\scriptsize] at (8.95, 4.35) {aerobic};
\filldraw[bar, anaerobic] (8.6, 3.6) rectangle (8.9, 3.9);
\node[lab, anchor=west, font=\scriptsize] at (8.95, 3.75) {anaerobic};

\end{tikzpicture}
\caption{Energy and heat accounting per mole glucose for aerobic
versus anaerobic catabolism. Complete aerobic oxidation produces
approximately 30 ATP per glucose and releases about 2.87 MJ per mole
glucose as heat; anaerobic ethanolic or lactic fermentation produces
2 ATP per glucose and releases only about 0.22 MJ per mole because
most of the chemical free energy of glucose remains in the partially
oxidised product (ethanol, lactate). Biomass yield on substrate
$Y_{X/S}$ is correspondingly higher for aerobic growth (around 0.45 g
biomass per g glucose for many heterotrophs) and lower for anaerobic
fermentation (around 0.10 g biomass per g glucose). The engineering
consequence: an aerobic reactor is heat-limited and oxygen-limited; an
anaerobic reactor is not heat-limited but has lower biomass and so
needs cheap medium or a valuable product to be economical. Numbers
order-of-magnitude per Stanier-style metabolic accounting.}
\label{fig:vol06ch10:aerobic-anaerobic}
\end{figure}
