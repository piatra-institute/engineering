% cylindrical tissue cylinder of radius R_t, oxygen partial pressure

\begin{tikzpicture}[

\caption{Krogh cylinder model (Krogh 1919): a single capillary of
radius $R_c$ supplies a coaxial cylinder of tissue of radius $R_t$
with oxygen by radial diffusion. The oxygen partial pressure falls
quadratically with radius for a tissue with uniform oxygen
consumption rate $\dot{q}$, and the maximum $R_t$ before the
oxygen partial pressure reaches zero at the cylinder boundary sets
the working diffusion-limited supply distance of $\sim
100$\,--\,$200\,\si{\micro\meter}$ in metabolically active tissue.
This number is the design constraint that all tissue-engineering
vascularisation strategies are built to satisfy.}
\label{fig:vol06ch05:krogh}
\end{figure}
