% Figure: scaling hierarchy from cell to system, showing length scale
% and the dominant transport mechanism at each level.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  level/.style={draw=engblack, rounded corners=3pt, minimum width=3.6cm,
    minimum height=1.0cm, align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblack!70},
  note/.style={font=\scriptsize, align=left, anchor=west, engblue},
]

\node[font=\bfseries] at (0, 4.2) {Hierarchy of biological scale};

\node[level, fill=englime!20]  (cell)   at (0,  3.0) {Cell\\ $\sim 10\,\si{\micro\meter}$};
\node[level, fill=englime!30]  (tissue) at (0,  1.5) {Tissue\\ $\sim 1\,\text{mm}$};
\node[level, fill=englime!40]  (organ)  at (0,  0.0) {Organ\\ $\sim 10\,\text{cm}$};
\node[level, fill=englime!50]  (system) at (0, -1.5) {Organ system\\ $\sim 1\,\text{m}$};

\draw[arrow] (cell)   -- (tissue);
\draw[arrow] (tissue) -- (organ);
\draw[arrow] (organ)  -- (system);

\node[note] at (2.4,  3.0) {membrane diffusion};
\node[note] at (2.4,  1.5) {ECM diffusion (limit $\sim 200\,\si{\micro\meter}$)};
\node[note] at (2.4,  0.0) {vascular flow + diffusion};
\node[note] at (2.4, -1.5) {inter-organ feedback};

\node[note, anchor=east] at (-2.4,  3.0) {$10^{-12}\,\text{g}$};
\node[note, anchor=east] at (-2.4,  1.5) {$10^{-6}\,\text{g}$};
\node[note, anchor=east] at (-2.4,  0.0) {$0.3\,\text{kg}$};
\node[note, anchor=east] at (-2.4, -1.5) {$70\,\text{kg}$};

\end{tikzpicture}
\caption{Hierarchy from cell to organ system: length scale, mass, and
the dominant transport mechanism at each level. Each upward
transition solves the scaling-up problem of the previous level by
inventing a new transport mechanism. The tissue level is the last at
which passive diffusion suffices; beyond a $\sim 200\,\si{\micro\meter}$
half-thickness, an internal transport network (the vasculature) is
required.}
\label{fig:vol06ch05:hierarchy}
\end{figure}
