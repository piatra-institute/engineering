% Figure: pore architecture comparison: closed pores, open pores,
% interconnected pores; with pore-size design window for various
% tissue applications.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  scaffold/.style={draw=engblack, line width=0.6pt, fill=engblack!15},
  pore/.style={draw=engblack!40, fill=white, line width=0.4pt},
  arrow/.style={-{Stealth}, thick, draw=engblue},
  label/.style={font=\scriptsize, anchor=center, align=center},
]

% Three schematic scaffold sections side by side
% (a) Closed-cell foam: pores not connected
\begin{scope}[shift={(-5.0, 0)}]
  \fill[engblack!15] (-1.4, -1.4) rectangle (1.4, 1.4);
  \foreach \x/\y in {-0.9/0.9, 0.2/0.7, -0.6/-0.3, 0.7/-0.1, 0.0/-0.9, -1.0/0.0, 0.9/0.9, -0.3/0.3} {
    \draw[pore] (\x, \y) circle (0.30);
  }
  \node[label, font=\small] at (0, -2.0) {Closed-cell foam};
  \node[label] at (0, -2.5) {pores isolated};
\end{scope}

% (b) Open-cell foam: pores partially connected
\begin{scope}[shift={(0, 0)}]
  \fill[engblack!15] (-1.4, -1.4) rectangle (1.4, 1.4);
  \foreach \x/\y in {-0.9/0.9, 0.2/0.7, -0.6/-0.3, 0.7/-0.1, 0.0/-0.9, -1.0/0.0, 0.9/0.9, -0.3/0.3} {
    \draw[pore] (\x, \y) circle (0.34);
  }
  % a few connecting channels
  \draw[white, line width=1.5pt] (-0.9, 0.9) -- (0.2, 0.7);
  \draw[white, line width=1.5pt] (0.2, 0.7) -- (0.7, -0.1);
  \draw[white, line width=1.5pt] (-0.6, -0.3) -- (-1.0, 0.0);
  \draw[white, line width=1.5pt] (0.0, -0.9) -- (0.7, -0.1);
  \node[label, font=\small] at (0, -2.0) {Open-cell foam};
  \node[label] at (0, -2.5) {partly connected};
\end{scope}

% (c) Interconnected porous scaffold
\begin{scope}[shift={(5.0, 0)}]
  \fill[engblack!15] (-1.4, -1.4) rectangle (1.4, 1.4);
  \foreach \x/\y in {-0.9/0.9, 0.2/0.7, -0.6/-0.3, 0.7/-0.1, 0.0/-0.9, -1.0/0.0, 0.9/0.9, -0.3/0.3} {
    \draw[pore] (\x, \y) circle (0.38);
  }
  % full network of channels
  \draw[white, line width=1.5pt] (-0.9, 0.9) -- (0.2, 0.7) -- (0.9, 0.9);
  \draw[white, line width=1.5pt] (-1.0, 0.0) -- (-0.3, 0.3) -- (0.2, 0.7);
  \draw[white, line width=1.5pt] (-0.6, -0.3) -- (-1.0, 0.0);
  \draw[white, line width=1.5pt] (-0.3, 0.3) -- (0.7, -0.1);
  \draw[white, line width=1.5pt] (0.0, -0.9) -- (0.7, -0.1) -- (0.2, 0.7);
  \draw[white, line width=1.5pt] (-0.6, -0.3) -- (0.0, -0.9);
  \node[label, font=\small] at (0, -2.0) {Interconnected};
  \node[label] at (0, -2.5) {network};
\end{scope}

% Pore-size window table at bottom
\node[font=\bfseries] at (0, -3.4) {Pore-size design windows};
\node[font=\scriptsize, align=left] at (0, -4.5)
  {dermal regeneration: 60\,--\,200\,$\si{\micro\meter}$\quad
   bone ingrowth: 100\,--\,400\,$\si{\micro\meter}$ \\
   vascular ingrowth: 50\,--\,300\,$\si{\micro\meter}$\quad
   nerve guidance: 5\,--\,50\,$\si{\micro\meter}$};

\node[font=\bfseries] at (0, 2.0) {Pore architecture and the connectivity requirement};
\end{tikzpicture}
\caption{Pore architecture across three scaffold classes: closed-cell
foam (pores isolated, useless for cell infiltration), open-cell foam
(pores partially connected), and an interconnected porous network
(cells, nutrients, and growing vessels can traverse the entire
volume). The pore-size design window depends on the tissue: dermal
regeneration requires 60--200\,$\si{\micro\meter}$ pores (Yannas and
Burke 1980), bone-ingrowth scaffolds require 100--400\,$\si{\micro\meter}$,
vascular-ingrowth applications 50--300\,$\si{\micro\meter}$, and
nerve-guidance channels 5--50\,$\si{\micro\meter}$.}
\label{fig:vol06ch05:scaffold}
\end{figure}
