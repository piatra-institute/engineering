% Figure: Yamanaka 2006 four-factor reprogramming: fibroblast to iPSC
% to differentiated lineages.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  cell/.style={draw=engblack, rounded corners=4pt, minimum width=2.6cm,
    minimum height=1.0cm, align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblue},
  factor/.style={font=\scriptsize, align=center, engred},
]

\node[font=\bfseries] at (0, 4.4) {Yamanaka reprogramming (2006) and downstream differentiation};

% Source cell
\node[cell, fill=engblack!10] (fib) at (-5.0, 2.0) {Adult fibroblast\\ (skin, blood)};

% Factor box
\node[factor, draw=engred, rounded corners=3pt, fill=engred!10,
      minimum width=2.6cm, minimum height=1.5cm, align=center]
  (factors) at (-1.0, 2.0)
  {Oct3/4\\ Sox2\\ Klf4\\ c-Myc};

% iPSC
\node[cell, fill=englime!25] (ipsc) at (3.0, 2.0) {iPSC\\ pluripotent};

\draw[arrow] (fib) -- (factors)
  node[midway, above, font=\scriptsize, engred] {retroviral};
\draw[arrow] (factors) -- (ipsc)
  node[midway, above, font=\scriptsize, engred] {2--4 weeks};

% Differentiation arrows down from iPSC to three lineages
\node[cell, fill=engblue!15]  (cm) at (-2.5, -0.5) {Cardiomyocytes};
\node[cell, fill=engblue!15]  (n)  at ( 0.5, -0.5) {Neurons (DA)};
\node[cell, fill=engblue!15]  (rpe) at ( 4.5, -0.5) {RPE};

\draw[arrow] (ipsc) -- (cm)
  node[midway, above left, font=\scriptsize, engblue] {Wnt, BMP4};
\draw[arrow] (ipsc) -- (n)
  node[midway, right, font=\scriptsize, engblue] {SHH, FGF8};
\draw[arrow] (ipsc) -- (rpe)
  node[midway, above right, font=\scriptsize, engblue] {Activin, Noggin};

% Application labels
\node[font=\scriptsize, align=center] at (-2.5, -2.2) {heart-failure\\ patches};
\node[font=\scriptsize, align=center] at ( 0.5, -2.2) {Parkinson's\\ trials};
\node[font=\scriptsize, align=center] at ( 4.5, -2.2) {macular\\ degeneration};

\end{tikzpicture}
\caption{Yamanaka's 2006 four-factor reprogramming: adult fibroblasts,
on retroviral delivery of Oct3/4, Sox2, Klf4, and c-Myc, dedifferentiate
to a pluripotent state over two to four weeks at a yield of order
$10^{-3}$ to $10^{-5}$ of starting cells (mouse; lower in human).
Downstream protocols direct the iPSCs to clinically relevant
lineages, including cardiomyocytes for heart-failure patches,
dopaminergic neurons for Parkinson's-disease trials, and retinal
pigment epithelium for macular degeneration; first clinical trials
in each application began in the mid-2010s.}
\label{fig:vol06ch05:yamanaka}
\end{figure}
