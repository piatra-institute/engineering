% Figure: schematic stress-strain curves for skin, cartilage, tendon,
% bone, and arterial wall on log-log axes, showing the J-shape
% strain-stiffening behaviour of collagen-containing soft tissues.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={-{Stealth}, thick, draw=engblack},
  curve/.style={thick},
  label/.style={font=\scriptsize, anchor=west},
]

% Axes
\draw[axis] (0, 0) -- (8.2, 0) node[right, font=\small] {strain $\varepsilon$};
\draw[axis] (0, 0) -- (0, 5.2) node[above, font=\small] {stress $\sigma$ (log)};

% x-axis ticks
\foreach \x/\xt in {1/0.1, 3/0.3, 5/0.5, 7/0.7} {
  \draw (\x, 0) -- (\x, -0.08) node[below, font=\scriptsize] {\xt};
}

% y-axis ticks
\foreach \y/\yt in {1/$10^{4}$, 2/$10^{5}$, 3/$10^{6}$, 4/$10^{7}$, 5/$10^{8}$} {
  \draw (0, \y) -- (-0.08, \y) node[left, font=\scriptsize] {\yt\,Pa};
}

% Skin: very compliant at small strain, strong stiffening at ~0.3
\draw[curve, engblue, smooth, samples=50, domain=0.05:6.5]
  plot (\x, {0.5 + 0.8*exp(0.45*\x)});
\node[label, engblue] at (6.6, 4.7) {skin};

% Cartilage: moderate stiffness, modest stiffening
\draw[curve, engred, smooth, samples=50, domain=0.05:6.5]
  plot (\x, {1.6 + 0.4*\x + 0.05*\x*\x});
\node[label, engred] at (6.6, 6.45-2.0) {cartilage};

% Tendon: stiff, narrow toe region, near-linear above 2% strain
\draw[curve, englime!70!black, smooth, samples=50, domain=0.05:3.5]
  plot (\x, {2.0 + 0.6*\x});
\node[label, englime!70!black] at (3.6, 4.1) {tendon};

% Artery: J-shape similar to skin but stiffer
\draw[curve, engblack!70, smooth, samples=50, domain=0.05:5.5]
  plot (\x, {1.0 + 0.6*exp(0.30*\x)});
\node[label, engblack!70] at (5.6, 3.5) {artery};

% Bone: very stiff, near-vertical
\draw[curve, engred!70!black, smooth, samples=50, domain=0.05:2.0]
  plot (\x, {3.0 + 1.0*\x});
\node[label, engred!70!black] at (2.1, 5.05) {bone};

\node[font=\bfseries] at (4.0, 5.5) {Tissue stress-strain at small to moderate strain};

\end{tikzpicture}
\caption{Schematic stress-strain behaviour of representative tissues
on semilog axes. Collagen-containing soft tissues (skin, artery) show
the characteristic J-shape: a compliant toe region at small strain
where wavy fibres straighten, then strain-stiffening as the fibres
load. Tendon's toe region is narrow because its fibres are nearly
straight at rest. Bone and tendon span four to five decades of
stiffness above skin. Curve shapes are schematic; quantitative fits
require the Fung or Holzapfel-Gasser-Ogden family of constitutive
models.}
\label{fig:vol06ch05:stressstrain}
\end{figure}
