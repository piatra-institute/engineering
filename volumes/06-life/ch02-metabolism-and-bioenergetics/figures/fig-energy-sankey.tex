% Figure: A Sankey-style energy-flow diagram from food in to heat and
% work out, summarising the cell's bulk free-energy balance.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  font=\footnotesize,
  flow/.style={draw=engblack!60, fill=engblue!20},
  flow2/.style={draw=engblack!60, fill=engamber!30},
  flow3/.style={draw=engblack!60, fill=engred!25},
  flow4/.style={draw=engblack!60, fill=englime!30},
  lab/.style={align=center, font=\footnotesize},
]
% Three columns: input | ATP | sinks
% Column 1: food (100 units)
\fill[flow] (0,0) rectangle (1.2,5);
\node[lab] at (0.6,5.4) {Food\\(100 units)};

% Column 2: split into ATP captured (40) and metabolic heat (60)
\fill[flow] (3,2.5) -- (1.2,5) -- (1.2,3) -- (3,3) -- cycle;
\fill[flow3] (3,2.5) -- (1.2,3) -- (1.2,0) -- (3,0) -- cycle;

\fill[flow] (3,3) rectangle (4.2,5);
\node[lab] at (3.6,5.4) {ATP\\($\sim 40$ units)};

\fill[flow3] (3,0) rectangle (4.2,2.5);
\node[lab] at (3.6,-0.3) {Catabolic\\heat ($\sim 60$)};

% Column 3: ATP splits into mechanical work (~10), biosynthesis (~15),
% ion pumping (~15), with the rest lost as heat.
% Mechanical (10)
\fill[flow2] (7,4.0) -- (4.2,5) -- (4.2,4.5) -- (7,4.5) -- cycle;
\fill[flow2] (7,4.0) rectangle (8.5,4.5);
\node[lab, anchor=west] at (8.6, 4.25) {Mechanical work ($\sim 10$)};

% Biosynthesis (15)
\fill[flow4] (7,3.0) -- (4.2,4.5) -- (4.2,3.5) -- (7,3.5) -- cycle;
\fill[flow4] (7,3.0) rectangle (8.5,3.5);
\node[lab, anchor=west] at (8.6,3.25) {Biosynthesis ($\sim 15$)};

% Ion / membrane work (15)
\fill[flow4] (7,2.0) -- (4.2,3.5) -- (4.2,3.0) -- (7,2.5) -- cycle;
\fill[flow4] (7,2.0) rectangle (8.5,2.5);
\node[lab, anchor=west] at (8.6,2.25) {Membrane / ion work ($\sim 15$)};

% Slip + leak heat (the residual)
\fill[flow3] (7,1.0) -- (4.2,3.0) -- (4.2,1.5) -- (7,1.5) -- cycle;
\fill[flow3] (7,1.0) rectangle (8.5,1.5);
\node[lab, anchor=west] at (8.6,1.25) {ATP-derived heat ($\sim 0$)};

% Note box
\node[draw=engblack!60, fill=engblue!8, rounded corners=2pt,
  align=left, font=\scriptsize, text width=11.5cm, inner sep=4pt]
  at (4.2,-1.3)
  {\textbf{At steady state} every joule of ATP-derived useful work is
   eventually dissipated as heat (second law); the resting body's
   total heat output is therefore $\sim 100\,\text{W}$, matching
   metabolic input. Numbers are rough fractions of a $2000$~kcal/day
   intake under sedentary conditions; the muscle fraction grows
   sharply during exercise.};
\end{tikzpicture}
\caption[Cellular energy-balance Sankey]{A schematic energy-flow
diagram from food in (left) through ATP (middle) to mechanical work,
biosynthesis, and ion-pumping work (right), normalised to a sedentary
adult's 100 arbitrary input units. About $40\,\%$ of the input
chemical energy is captured as ATP; the remainder leaves as catabolic
heat. Of the captured ATP energy, all useful work is, at steady
state, eventually dissipated as heat (the body's $\sim 100$~W resting
heat output). The diagram is intentionally lumped; per-organ
breakdowns are tabulated in the data directory.}
\label{fig:vol06:ch02:sankey}
\end{figure}
