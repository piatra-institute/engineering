% Simplified food-web schematic with trophic-level layering.
% Vol VI Ch 8 §8.4.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
    fnode/.style={
      draw=engblack, thick, circle, minimum size=1.1cm, align=center,
      font=\footnotesize, inner sep=1pt
    },
    farrow/.style={->, thick, >=stealth, draw=gray}
  ]
  % Trophic level 1: basal
  \node[fnode, fill=green!20] (Pl) at (0, 0) {Plants};
  \node[fnode, fill=green!20] (Al) at (3, 0) {Algae};
  \node[fnode, fill=brown!20] (De) at (6, 0) {Detritus};

  % Trophic level 2: primary consumers
  \node[fnode, fill=yellow!25] (Hb) at (0, 2.2) {Herbi-\\vores};
  \node[fnode, fill=yellow!25] (Zp) at (3, 2.2) {Zoo-\\plankton};
  \node[fnode, fill=yellow!25] (Dt) at (6, 2.2) {Detri-\\tivores};

  % Trophic level 3: secondary consumers
  \node[fnode, fill=engamber!40] (Ca) at (1.5, 4.4)
    {Small\\carnivores};
  \node[fnode, fill=engamber!40] (PlS) at (4.5, 4.4)
    {Plank-\\tivores};

  % Trophic level 4: apex predator
  \node[fnode, fill=red!30] (Ap) at (3, 6.6) {Apex\\predator};

  % Predation arrows
  \draw[farrow] (Pl) -- (Hb);
  \draw[farrow] (Al) -- (Zp);
  \draw[farrow] (Al) -- (Hb);
  \draw[farrow] (De) -- (Dt);
  \draw[farrow] (De) -- (Zp);
  \draw[farrow] (Hb) -- (Ca);
  \draw[farrow] (Zp) -- (PlS);
  \draw[farrow] (Dt) -- (Ca);
  \draw[farrow] (Ca) -- (Ap);
  \draw[farrow] (PlS) -- (Ap);
  \draw[farrow] (Hb) -- (Ap);
  \draw[farrow] (Zp) -- (Ap);

  % Trophic level annotations
  \node[font=\footnotesize, anchor=west] at (7.5, 0) {TL 1 (basal)};
  \node[font=\footnotesize, anchor=west] at (7.5, 2.2)
    {TL 2 (primary)};
  \node[font=\footnotesize, anchor=west] at (7.5, 4.4)
    {TL 3 (secondary)};
  \node[font=\footnotesize, anchor=west] at (7.5, 6.6) {TL 4 (apex)};
\end{tikzpicture}
\caption[Food-web schematic]{Schematic of an aquatic-terrestrial
food web. Nodes are species or guilds, edges are predation
relationships. Trophic levels are not integer in real food webs; a
species's trophic position is the weighted average of its prey's
positions plus one. Connectance (fraction of possible directed
pairs that interact) ranges 5\%--30\% across empirical webs, and
the degree distribution is heavy-tailed: a few generalist species
have many connections, most have few.}
\label{fig:vol06:ch08:food-web}
\end{figure}
