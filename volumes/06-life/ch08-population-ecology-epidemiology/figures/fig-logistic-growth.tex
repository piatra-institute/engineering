% Logistic growth curve with carrying capacity K marked.
% Vol VI Ch 8 §8.1.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm,
  height=7cm,
  xlabel={Time $t$ (arbitrary units)},
  ylabel={Population $N(t)/K$},
  xmin=0, xmax=20,
  ymin=0, ymax=1.15,
  ytick={0, 0.25, 0.5, 0.75, 1.0},
  yticklabels={$0$, $K/4$, $K/2$, $3K/4$, $K$},
  xtick={0, 5, 10, 15, 20},
  legend pos=south east,
  legend style={font=\small, fill=white, draw=engblack},
  grid=major,
  grid style={draw=gray, very thin, opacity=0.3},
  every axis plot/.append style={very thick},
  samples=200,
]
  \addplot[engblue, very thick, mark=none, domain=0:20]
    {1/(1 + 19*exp(-0.5*x))};
  \addlegendentry{Logistic: $r = 0.5$, $N_0 = 0.05K$}

  \addplot[engamber, very thick, dashed, mark=none, domain=0:9, samples=80]
    {0.05*exp(0.5*x)};
  \addlegendentry{Exponential early-phase}

  \addplot[gray, dashed, no marks, domain=0:20] {1.0};
  \node[anchor=west, font=\small] at (axis cs:14, 1.05) {$K$};

  \addplot[gray, densely dotted, no marks, domain=0:20] {0.5};
  \node[anchor=west, font=\small] at (axis cs:0.3, 0.55) {$K/2$ (inflection)};
\end{axis}
\end{tikzpicture}
\caption[Logistic growth curve with carrying capacity]{Logistic
growth curve $N(t) = K/(1 + ((K - N_0)/N_0) e^{-rt})$ for $r =
0.5$, $N_0 = 0.05 K$. The trajectory passes through the inflection
at $N = K/2$ and approaches the carrying capacity $K$
asymptotically. Exponential early-phase tracking is plotted for
reference until logistic deceleration begins to bite at $N \sim
0.3 K$.}
\label{fig:vol06:ch08:logistic}
\end{figure}
