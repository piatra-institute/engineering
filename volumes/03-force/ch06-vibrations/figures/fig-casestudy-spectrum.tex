% Figure: the measured FFT velocity spectrum of the case-study pump,
% plotted in absolute frequency (Hz) at a running speed of 24.0 Hz.
% Peaks at 1x (24 Hz, misalignment-dominated 2x larger), 2x (48 Hz),
% 3x (72 Hz), plus a raised broadband bearing zone around 120-160 Hz.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={frequency (Hz), running speed $f_{\mathrm{run}}=24.0$~Hz},
  ylabel={velocity (mm/s RMS)},
  xmin=0, xmax=200, ymin=0, ymax=7.5,
  axis lines=left,
  domain=2:200, samples=800,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% noise floor 0.35, narrow peaks at 24, 48, 72; broad bearing hump 120-160.
\addplot[engblue, thick] {
  0.35
  + 2.6*exp(-((x-24)^2)/(2*1.0))
  + 5.9*exp(-((x-48)^2)/(2*1.2))
  + 1.5*exp(-((x-72)^2)/(2*1.0))
  + 1.9*exp(-((x-140)^2)/(2*120))
  + 0.9*exp(-((x-146)^2)/(2*1.0))
};
% ISO 10816 yellow alert reference line at 4.5 mm/s
\addplot[engamber, thin, dashed, forget plot] {4.5};
\node[engamber, font=\scriptsize, anchor=south east] at (axis cs:198,4.6) {alert level};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:24,3.0) {$1\times$};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:48,6.0) {$2\times$};
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:72,1.9) {$3\times$};
\node[engorange, font=\scriptsize, anchor=south] at (axis cs:140,2.4) {bearing band};
\end{axis}
\end{tikzpicture}
\caption[Case-study pump velocity spectrum]{The measured velocity
spectrum of the case-study feed pump, RMS velocity against absolute
frequency. The running speed is $24.0$~Hz ($1440$~rpm). The dominant
feature is a $2\times$ peak at $48$~Hz that exceeds the $1\times$ peak
at $24$~Hz, the textbook signature of shaft misalignment rather than
pure imbalance. A modest $3\times$ peak at $72$~Hz and a raised
broadband zone near $120$ to $160$~Hz with a discrete line at $146$~Hz
together flag early outer-race bearing degradation. The $2\times$
component sits above the alert level (dashed), the trigger for the
corrective work the case study carries through.}
\label{fig:vol03ch06:casestudy-spectrum}
\end{figure}
