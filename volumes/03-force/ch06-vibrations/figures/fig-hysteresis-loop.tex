% Figure: force-displacement loops for the three damping mechanisms over
% one cycle of steady oscillation x = X sin(theta). The enclosed area is
% the energy dissipated per cycle. Viscous: an ellipse whose area grows
% with drive frequency. Hysteretic: an ellipse of frequency-independent
% area. Coulomb: a parallelogram (constant-magnitude force flipping sign
% with velocity). All drawn for the same displacement amplitude.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7.5cm,
  xlabel={displacement $x/X$}, ylabel={damping force $F_d$ (arb.)},
  xmin=-1.35, xmax=1.35, ymin=-1.6, ymax=1.6,
  axis lines=center,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=north east},
  every axis y label/.style={at={(current axis.above origin)}, anchor=north west},
  grid=major, grid style={enggray!15},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend cell align=left,
  legend style={font=\scriptsize},
]
% Parametrise by theta in radians; x = sin(theta), velocity ~ cos(theta).
% Viscous loop: F_d = c*omega*X*cos(theta); with x = X sin(theta) this is
% an ellipse F_d proportional to sqrt(1 - (x/X)^2) with sign of velocity.
\addplot[engblue, very thick, domain=0:360, samples=200]
  ({sin(x)}, {1.0*cos(x)});
\addlegendentry{viscous (ellipse)}
% Hysteretic loop: same elliptical shape but area independent of frequency;
% draw a narrower ellipse to distinguish, area fixed by loss factor.
\addplot[engred, very thick, dashed, domain=0:360, samples=200]
  ({sin(x)}, {0.55*cos(x)});
\addlegendentry{hysteretic (freq.-flat)}
% Coulomb loop: constant magnitude F_f flipping sign with velocity =>
% a rectangle/parallelogram. Build explicitly with a coordinate path.
\addplot[engorange, very thick] coordinates {
  (-1.0, 1.3) (1.0, 1.3) (1.0, -1.3) (-1.0, -1.3) (-1.0, 1.3)};
\addlegendentry{Coulomb (rectangle)}
\node[engorange, font=\scriptsize, anchor=south] at (axis cs:0,1.32) {$+F_f$};
\node[engorange, font=\scriptsize, anchor=north] at (axis cs:0,-1.32) {$-F_f$};
\end{axis}
\end{tikzpicture}
\caption[Damping-force loops and energy per cycle]{Force-displacement
loops traced over one cycle of steady oscillation at fixed displacement
amplitude $X$, one loop per damping mechanism. The area a loop encloses
is the energy dissipated in that cycle. The viscous force $c\dot x$
leads the displacement by a quarter cycle and traces an ellipse (blue)
whose area $\pi c\omega X^2$ climbs with drive frequency. The hysteretic
model (red, dashed) traces an ellipse of the same shape but of an area
$\pi\eta k X^2$ that does not depend on frequency, the behaviour
measured material damping shows. Coulomb friction (orange) holds a
constant magnitude $F_f$ that flips sign with the velocity, so its loop
is a rectangle of area $4F_f X$, linear in amplitude rather than
quadratic. Reading which shape a measured loop takes, and how its area
scales with amplitude and frequency, identifies the mechanism at work.}
\label{fig:vol03ch06:hysteresis-loop}
\end{figure}
