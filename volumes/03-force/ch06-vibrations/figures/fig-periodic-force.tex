% Figure: a periodic (non-sinusoidal) drive resolved into its Fourier
% harmonics, and the resonant selectivity of a lightly damped oscillator
% that amplifies whichever harmonic lands nearest its natural frequency.
% Top idea shown as a stem-style bar spectrum of the drive harmonics with
% the amplification curve overlaid; the harmonic nearest resonance is
% singled out.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={frequency ratio $\omega/\omega_n$},
  ylabel={amplitude (arb.)},
  xmin=0, xmax=5.2, ymin=0, ymax=13,
  axis lines=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend cell align=left,
  legend style={font=\scriptsize},
]
% Amplification curve for zeta = 0.05, capped for display.
\addplot[engblue, very thick, domain=0.02:5.2, samples=400]
  {min(12, 1/sqrt((1-x^2)^2 + (2*0.05*x)^2))};
\addlegendentry{amplification $\zeta=0.05$}
% Drive harmonics: a square-wave-like drive has odd harmonics with
% amplitudes 1, 1/3, 1/5, 1/7 ... placed here at ratios 0.5, 1.5, 2.5, 3.5
% so the third harmonic (ratio 1.5) sits nearest resonance. Draw as bars.
\addplot[engred, ycomb, very thick, mark=*, mark size=1.5pt] coordinates {
  (0.5, 4.0) (1.5, 1.33) (2.5, 0.80) (3.5, 0.57) (4.5, 0.44)};
\addlegendentry{drive harmonics}
\node[engred, font=\scriptsize, anchor=west] at (axis cs:1.55,3.0)
  {near resonance};
\end{axis}
\end{tikzpicture}
\caption[A periodic drive against the resonance curve]{A periodic but
non-sinusoidal drive resolved into its harmonics (red stems) laid over
the amplification curve of a lightly damped oscillator (blue). The drive
here carries odd harmonics of falling amplitude, the Fourier content of
a near-square force; their positions on the frequency axis are fixed by
the fundamental drive frequency. The oscillator amplifies each harmonic
by the value the blue curve takes at that harmonic's frequency ratio, so
the steady response is the sum of the amplified harmonics. A harmonic of
modest amplitude that happens to land near $\omega/\omega_n=1$ dominates
the response even when it is far from the largest term in the drive, the
reason a machine can ring at a harmonic of its running speed rather than
at the running speed itself.}
\label{fig:vol03ch06:periodic-force}
\end{figure}
