% Figure: response of a rotating-unbalance excited SDOF system.
% X/(m_e e / m) = r^2 / sqrt((1-r^2)^2 + (2 zeta r)^2), r = omega/omega_n.
% The non-dimensional response rises from zero, peaks near r=1, and
% approaches unity at high speed (the mass cannot follow but the
% unbalance force grows as omega^2, so the two effects cancel).

\addlegendentry{$\zeta=0.10$}

\caption[Response to rotating unbalance]{Steady-state displacement
amplitude of a machine driven by its own rotating unbalance, against
the speed ratio $r=\omega/\omega_n$. The forcing amplitude is itself
$m_e e\,\omega^2$, proportional to the square of the running speed, so
the non-dimensional response $X/(m_e e/m)$ starts at zero at low speed
(the unbalance force vanishes as $\omega\to 0$), peaks near resonance
at $1/(2\zeta\sqrt{1-\zeta^2})$, and approaches unity at high speed.
The high-speed limit is the defining property of supercritical
operation: above resonance the mass orbits about its own centre of
mass, and the residual whirl amplitude settles to the eccentricity
scaled by the mass ratio, independent of speed. Light damping (red)
gives a tall peak that a machine accelerating through resonance must
pass quickly; heavier damping (blue) caps the peak at the cost of more
transmitted force above resonance.}
\label{fig:vol03ch06:rotating-unbalance}
\end{figure}
