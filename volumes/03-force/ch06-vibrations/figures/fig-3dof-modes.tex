% Figure: the three mode shapes of a three-storey shear frame, drawn as
% the lateral deflection profile of the three floors for each mode. The
% first mode has all floors moving the same way (no internal sign
% change), the second has one sign change, the third has two, the
% standard nodal-count ordering of a chain of masses.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% three columns, one per mode. Floor heights at y = 1, 2, 3; ground at 0.
% mode shapes (normalised) for a uniform three-storey shear frame:
% phi1 ~ (0.445, 0.802, 1.000)  in-phase
% phi2 ~ (1.000, 0.555, -0.802) one node
% phi3 ~ (1.000, -1.247, 0.555) two nodes
% We draw each with a horizontal scale of 1.0 unit.
% Mode 1
\begin{scope}[xshift=0cm]
  \draw[enggray, thick] (-1.2,0) -- (1.2,0);
  \draw[enggray, dashed] (0,0) -- (0,3.3);
  \draw[engblue, very thick]
    (0,0) -- (0.445,1) -- (0.802,2) -- (1.000,3);
  \foreach \y/\x in {1/0.445, 2/0.802, 3/1.000}{
    \filldraw[engblue] (\x,\y) circle (2.2pt);
    \draw[engblue, thin] (0,\y) -- (\x,\y);
  }
  \node[font=\scriptsize, anchor=north] at (0,-0.15) {mode 1};
  \node[font=\scriptsize, anchor=north] at (0,-0.55) {$f_1$, no node};
\end{scope}
% Mode 2
\begin{scope}[xshift=4.0cm]
  \draw[enggray, thick] (-1.2,0) -- (1.2,0);
  \draw[enggray, dashed] (0,0) -- (0,3.3);
  \draw[engred, very thick]
    (0,0) -- (1.000,1) -- (0.555,2) -- (-0.802,3);
  \foreach \y/\x in {1/1.000, 2/0.555, 3/-0.802}{
    \filldraw[engred] (\x,\y) circle (2.2pt);
    \draw[engred, thin] (0,\y) -- (\x,\y);
  }
  \node[font=\scriptsize, anchor=north] at (0,-0.15) {mode 2};
  \node[font=\scriptsize, anchor=north] at (0,-0.55) {$f_2$, one node};
\end{scope}
% Mode 3
\begin{scope}[xshift=8.0cm]
  \draw[enggray, thick] (-1.4,0) -- (1.4,0);
  \draw[enggray, dashed] (0,0) -- (0,3.3);
  \draw[englime!70!engblack, very thick]
    (0,0) -- (1.000,1) -- (-1.247,2) -- (0.555,3);
  \foreach \y/\x in {1/1.000, 2/-1.247, 3/0.555}{
    \filldraw[englime!70!engblack] (\x,\y) circle (2.2pt);
    \draw[englime!70!engblack, thin] (0,\y) -- (\x,\y);
  }
  \node[font=\scriptsize, anchor=north] at (0,-0.15) {mode 3};
  \node[font=\scriptsize, anchor=north] at (0,-0.55) {$f_3$, two nodes};
\end{scope}
\end{tikzpicture}
\caption[Mode shapes of a three-storey shear frame]{The three mode
shapes of a uniform three-storey shear frame, each drawn as the lateral
deflection of the three floor levels (filled circles) joined by straight
segments, with the undeflected line dashed. The lowest mode moves all
three floors the same way and crosses zero nowhere between the floors;
the second mode reverses once, with the top floor opposed to the lower
two; the third mode reverses twice. The number of internal sign changes
rises by one with each mode, the discrete echo of the half-wave count
that orders the modes of a string. The lowest mode carries most of the
participating mass and dominates the base shear under a uniform ground
motion, which is why earthquake analysis reads it first and retains the
higher modes only until ninety percent of the mass is accounted for.}
\label{fig:vol03ch06:3dof-modes}
\end{figure}
