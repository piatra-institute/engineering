% Figure: transmissibility of a base-excited SDOF mount versus
% frequency ratio, for several damping ratios. The crossover at
% r = sqrt(2) is the isolation threshold.
% TR = sqrt( (1 + (2 zeta r)^2) / ((1-r^2)^2 + (2 zeta r)^2) ).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={$r=\omega/\omega_n$}, ylabel={transmissibility TR},
  xmin=0, xmax=4, ymin=0, ymax=6.2,
  axis lines=left,
  domain=0.001:4, samples=450,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
\addplot[engred, very thick] {sqrt((1 + (2*0.05*x)^2)/((1-x^2)^2 + (2*0.05*x)^2))};
\addlegendentry{$\zeta=0.05$}
\addplot[engorange, very thick] {sqrt((1 + (2*0.20*x)^2)/((1-x^2)^2 + (2*0.20*x)^2))};
\addlegendentry{$\zeta=0.20$}
\addplot[engblue, very thick] {sqrt((1 + (2*0.50*x)^2)/((1-x^2)^2 + (2*0.50*x)^2))};
\addlegendentry{$\zeta=0.50$}
% unity line and the sqrt(2) crossover
\addplot[enggray, thin, forget plot] {1};
\addplot[enggray, dotted, thick, forget plot] coordinates {(1.41421,0) (1.41421,6.2)};
\node[enggray, font=\scriptsize, anchor=south west, rotate=90] at (axis cs:1.41421,0.3) {$r=\sqrt2$};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:2.6,1.05) {TR $=1$};
\end{axis}
\end{tikzpicture}
\caption[Transmissibility of a vibration mount]{Transmissibility TR,
the ratio of mounted-mass amplitude to support amplitude, for a
base-excited SDOF mount, against the frequency ratio
$r=\omega/\omega_n$. All curves cross unity at $r=\sqrt2$ regardless
of damping (dotted line): below it the mount amplifies support
motion, above it the mount isolates. Isolation therefore requires the
mount natural frequency well below the lowest disturbance frequency,
so that $r>\sqrt2$. Here damping cuts both ways: more damping
($\zeta=0.50$, blue) tames the resonant peak near $r=1$ but worsens
high-frequency isolation, because the curve rolls off more slowly.
Light damping ($\zeta=0.05$, red) gives the best high-frequency
isolation at the cost of a violent resonance during startup or
transient passage through $r=1$. Computed from
$\mathrm{TR}=\sqrt{(1+(2\zeta r)^2)/[(1-r^2)^2+(2\zeta r)^2]}$.}
\label{fig:vol03ch06:transmissibility}
\end{figure}
