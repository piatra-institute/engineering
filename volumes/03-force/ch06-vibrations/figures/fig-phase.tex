% Figure: phase of the forced SDOF response versus frequency ratio,
% for several damping ratios. phi = atan( 2 zeta r / (1 - r^2) ),
% taken in [0, 180] degrees (continuous through resonance).
% We plot phi in degrees using a branch-correct piecewise expression.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={$r=\omega/\omega_n$}, ylabel={phase lag $\phi$ (deg)},
  xmin=0, xmax=2.5, ymin=0, ymax=185,
  ytick={0,45,90,135,180},
  axis lines=left,
  domain=0.001:2.5, samples=400,
  legend pos=south east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% phi = atan2(2 zeta r, 1 - r^2) in degrees, continuous 0..180.
% pgfplots atan2(y,x) returns degrees in (-180,180]; for the lower
% branch (1-r^2>0) it gives 0..90, for upper (1-r^2<0) it gives 90..180.
\addplot[engred, very thick] {atan2(2*0.05*x, 1 - x^2)};
\addlegendentry{$\zeta=0.05$}
\addplot[engorange, very thick] {atan2(2*0.10*x, 1 - x^2)};
\addlegendentry{$\zeta=0.10$}
\addplot[engblue, very thick] {atan2(2*0.25*x, 1 - x^2)};
\addlegendentry{$\zeta=0.25$}
\addplot[engamber, very thick, dashed] {atan2(2*0.50*x, 1 - x^2)};
\addlegendentry{$\zeta=0.50$}
% 90-degree reference at resonance
\addplot[enggray, thin, forget plot, domain=0:2.5] {90};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:2.45,90) {$90^{\circ}$ at $r=1$};
\end{axis}
\end{tikzpicture}
\caption[Phase of the forced oscillator]{Phase lag $\phi$ of the
steady-state displacement behind the driving force
$F_0\cos(\omega t)$, as a function of the frequency ratio
$r=\omega/\omega_n$, for four damping ratios. Below resonance the
response is nearly in phase ($\phi\to 0$); far above resonance it is
nearly out of phase ($\phi\to 180^{\circ}$). Every curve passes
through exactly $90^{\circ}$ at $r=1$, regardless of damping: the
$90^{\circ}$ crossing is the unambiguous experimental marker of a
natural frequency, since the resonant peak in amplitude shifts
slightly with damping while the phase crossing does not. Light
damping produces an abrupt phase transition; heavy damping spreads it
over a wide frequency band. Computed from
$\phi=\operatorname{atan2}(2\zeta r,\,1-r^2)$.}
\label{fig:vol03ch06:phase}
\end{figure}
