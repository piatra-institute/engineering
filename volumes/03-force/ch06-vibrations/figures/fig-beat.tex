% Figure: beating in an undamped oscillator driven just off resonance.
% Response is a slow-amplitude-modulated oscillation; envelope at the
% difference frequency. We plot a representative beat pattern.
% x(t) ~ sin(w1 t) - sin(w2 t) = 2 cos((w1+w2)/2 t) sin((w1-w2)/2 t).
% Use w1 = 1.05, w2 = 0.95 (drive near wn = 1), carrier 1.0, beat 0.05.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={$\omega_n t$}, ylabel={$x$ (arb.)},
  xmin=0, xmax=130, ymin=-2.3, ymax=2.5,
  axis lines=left,
  domain=0:130, samples=900,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% beat: 2 sin(0.05 t) carrier sin(1.0 t). Plot the product and envelope.
\addplot[engblue, thick] {2*sin(deg(0.05*x))*sin(deg(1.0*x))};
\addplot[engred, thick, dashed, forget plot] {2*sin(deg(0.05*x))};
\addplot[engred, thick, dashed, forget plot] {-2*sin(deg(0.05*x))};
\addplot[enggray, thin, forget plot] {0};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:66,2.05) {envelope};
\end{axis}
\end{tikzpicture}
\caption[Beating near resonance]{Response of a lightly damped
oscillator driven close to but not at its natural frequency. When the
drive frequency $\omega$ differs from $\omega_n$ by a small amount,
the transient free response at $\omega_n$ and the steady-state
response at $\omega$ add as two nearby sinusoids, producing a fast
oscillation at the average frequency inside a slow amplitude envelope
(red dashed) at the difference frequency $|\omega-\omega_n|$. The
slow swelling and fading is a beat. The beat period lengthens without
bound as the drive approaches exact resonance; at exact resonance of
an undamped system the envelope grows linearly without limit rather
than oscillating, the signature of pure resonant buildup. Beats are
the audible warning that a machine is running near a structural
natural frequency. Computed from $2\sin(\Delta\omega\,t/2)\sin(\bar\omega t)$.}
\label{fig:vol03ch06:beat}
\end{figure}
