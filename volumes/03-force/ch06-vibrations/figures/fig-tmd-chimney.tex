% Figure: the across-wind response of the stack with and without a
% tuned mass damper, plotted as the normalised tip displacement
% magnitude against forcing-frequency ratio. The bare lightly-damped
% stack has a single tall peak; the Den-Hartog-tuned damper splits it
% into two equal modest peaks, the classic invariant-points tuning.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={forcing-frequency ratio $\omega/\omega_n$},
  ylabel={tip-displacement magnification},
  xmin=0.75, xmax=1.30, ymin=0, ymax=42,
  axis lines=left,
  domain=0.75:1.30, samples=700,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=north east,
  restrict y to domain=0:200,
]
% Bare stack, zeta1 = 0.012: 1/sqrt((1-r^2)^2 + (2 z1 r)^2)
\addplot[enggray, thick, dashed] {1/sqrt((1-x^2)^2 + (2*0.012*x)^2)};
\addlegendentry{bare stack ($\zeta_s=0.012$)}
% TMD tuned (Den Hartog), mu=0.02, beta = 1/(1+mu)=0.9804, za=0.084.
% Standard 2-DOF absorber primary magnitude:
% num = sqrt( (beta^2 - r^2)^2 + (2 za beta r)^2 )
% den = sqrt( [ (1-r^2)(beta^2 - r^2) - mu beta^2 r^2 ]^2
%             + (2 za beta r)^2 (1 - r^2 (1+mu))^2 )
\addplot[engblue, very thick] {
  sqrt( (0.9804^2 - x^2)^2 + (2*0.084*0.9804*x)^2 )
  /
  sqrt(
    ( (1 - x^2)*(0.9804^2 - x^2) - 0.02*0.9804^2*x^2 )^2
    + (2*0.084*0.9804*x)^2 * ( 1 - x^2*(1+0.02) )^2
  )
};
\addlegendentry{with TMD ($\mu=0.02$, Den Hartog)}
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:1.13,9.5) {two equal peaks};
\draw[engblue, thin, ->] (axis cs:1.10,9.0) -- (axis cs:0.965,7.2);
\draw[engblue, thin, ->] (axis cs:1.16,9.0) -- (axis cs:1.035,7.2);
\end{axis}
\end{tikzpicture}
\caption[A tuned mass damper on a slender stack]{Across-wind
tip-displacement magnification of the stack against forcing-frequency
ratio. The bare stack carries the very light structural damping of a
welded steel cylinder ($\zeta_s=0.012$) and shows a single resonance of
magnification near $1/(2\zeta_s)\approx 42$ (grey dashed). A tuned mass
damper of two-percent mass ratio, tuned and damped by the Den Hartog
optimum, replaces that single peak with two equal, much lower peaks
straddling the natural frequency (blue). The optimum tuning is the one
that makes the two peaks equal in height, the lowest worst-case response
a passive damper of that mass ratio can give. The reduction here is from
a magnification of forty-two to about six, the factor of seven that a
two-percent damper is expected to deliver and the reason slender stacks,
chimneys, and footbridges carry these dampers against vortex-driven
resonance.}
\label{fig:vol03ch06:tmd-chimney}
\end{figure}
