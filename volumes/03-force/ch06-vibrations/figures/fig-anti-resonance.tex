% Figure: driving-point frequency response of a two-degree-of-freedom
% system, showing that a driving-point FRF alternates resonances (poles,
% where the magnitude spikes) with antiresonances (zeros, where it dips
% deeply). The antiresonance between the two resonances is the frequency
% at which the tuned absorber holds the driven mass still.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={$r=\omega/\omega_1$}, ylabel={$|H_{11}|$ (normalised, log)},
  xmin=0.2, xmax=2.4, ymin=0.02, ymax=60,
  ymode=log,
  axis lines=left,
  domain=0.2:2.4, samples=500,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% Two-DOF driving-point receptance magnitude with light damping.
% Poles near r = 0.81 and r = 1.24 (the split resonances of the equal-
% tuned absorber, mu = 0.05); antiresonance zero at r = 1 (absorber
% tuning). Build a smooth surrogate: |H| = |(1 - r^2)| /
%   sqrt( ((r^2-0.66)(r^2-1.54))^2 + (2*d*r)^2 ), a clamped receptance
% with a numerator zero at r=1 and denominator zeros at the two poles.
\addplot[engblue, very thick]
  {abs(1 - x^2) / sqrt( ((x^2-0.66)*(x^2-1.54))^2 + (0.02*x)^2 )};
% mark the antiresonance
\addplot[engred, only marks, mark=*, mark size=1.6pt] coordinates {(1.0,0.03)};
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:1.02,0.035)
  {antiresonance ($r=1$)};
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:0.81,42)
  {mode 1};
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:1.24,42)
  {mode 2};
\end{axis}
\end{tikzpicture}
\caption[Antiresonance in a driving-point FRF]{Driving-point receptance
magnitude of a two-degree-of-freedom system on a logarithmic scale,
against the frequency ratio referred to the first mode. Two resonances
(the tall peaks, the system's poles) straddle a deep dip, the
antiresonance (the system's zero, marked red). At the antiresonance the
driven coordinate holds still while the second coordinate takes up the
motion, the same cancellation the tuned absorber uses to quiet a machine
at its drive frequency. A driving-point measurement always alternates a
resonance, an antiresonance, a resonance, so the count and spacing of
the dips is itself modal information. A transfer FRF between two
different points need not carry a zero between every pair of poles, which
is one way a modal test distinguishes a driving-point measurement from a
cross measurement.}
\label{fig:vol03ch06:anti-resonance}
\end{figure}
