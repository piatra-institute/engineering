% Figure: free decay of an underdamped SDOF oscillator, showing the
% exponential envelope and the logarithmic-decrement reading.
% x(t) = X0 exp(-zeta wn t) cos(wd t), wn = 1, zeta = 0.05, wd ~ 0.99875.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={$\omega_n t$}, ylabel={$x/X_0$},
  xmin=0, xmax=40, ymin=-1.1, ymax=1.15,
  axis lines=left,
  domain=0:40, samples=500,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Underdamped response, zeta = 0.05, wd = sqrt(1-0.0025) = 0.99875.
\addplot[engblue, very thick] {exp(-0.05*x)*cos(0.99875*x)};
\addlegendentry{$x(t)/X_0$, $\zeta=0.05$}
% Upper and lower exponential envelopes.
\addplot[engred, thick, dashed] {exp(-0.05*x)};
\addlegendentry{envelope $\pm e^{-\zeta\omega_n t}$}
\addplot[engred, thick, dashed, forget plot] {-exp(-0.05*x)};
% zero line
\addplot[enggray, thin, forget plot] {0};
\end{axis}
\end{tikzpicture}
\caption[Free decay of an underdamped oscillator]{Free response of
$m\ddot{x}+c\dot{x}+kx=0$ released from $x(0)=X_0$, $\dot{x}(0)=0$,
for a damping ratio $\zeta=0.05$, on the normalised time axis
$\omega_n t$. The displacement (blue) rings at the damped frequency
$\omega_d=\omega_n\sqrt{1-\zeta^2}$, which for this light damping is
indistinguishable from $\omega_n$. The amplitude is bounded by the
exponential envelope $\pm e^{-\zeta\omega_n t}$ (red dashed). The
ratio of any peak to the peak one cycle later is
$e^{2\pi\zeta/\sqrt{1-\zeta^2}}$; its logarithm is the logarithmic
decrement $\delta$, the standard experimental reading for damping.
Over the forty normalised time units shown, the amplitude falls by
$e^{-2}\approx 0.135$, so roughly $13.5\,\%$ of the release amplitude
survives. Computed from $x(t)/X_0=e^{-\zeta\omega_n t}\cos(\omega_d t)$.}
\label{fig:vol03ch06:free-decay}
\end{figure}
