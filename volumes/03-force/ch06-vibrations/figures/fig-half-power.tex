% Figure: half-power bandwidth read off a single resonance peak.
% One lightly damped amplitude curve, with the peak, the peak/sqrt2
% level, and the two half-power frequencies f1, f2 marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={frequency ratio $r=\omega/\omega_n$},
  ylabel={amplitude (normalised)},
  xmin=0.7, xmax=1.3, ymin=0, ymax=1.15,
  axis lines=left,
  domain=0.7:1.3, samples=500,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% amplitude normalised so peak = 1. zeta = 0.04, peak ~ 1/(2 zeta)=12.5
% Normalise by multiplying by 2 zeta.
\addplot[engblue, very thick]
  {0.08/sqrt((1-x^2)^2 + (2*0.04*x)^2)};
% peak/sqrt2 horizontal line
\addplot[engred, thin, dashed, forget plot, domain=0.7:1.3] {0.7071};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.705,0.745)
  {peak$/\sqrt2$ (half power)};
% half-power frequencies at r = 1 +/- zeta = 0.96, 1.04
\addplot[enggray, thin, dotted, forget plot] coordinates {(0.96,0)(0.96,0.7071)};
\addplot[enggray, thin, dotted, forget plot] coordinates {(1.04,0)(1.04,0.7071)};
\node[enggray, font=\scriptsize, anchor=north] at (axis cs:0.96,-0.02) {$r_1$};
\node[enggray, font=\scriptsize, anchor=north] at (axis cs:1.04,-0.02) {$r_2$};
% bandwidth arrow
\draw[<->, engamber, thick] (axis cs:0.96,0.5) -- (axis cs:1.04,0.5);
\node[engamber, font=\scriptsize, anchor=south] at (axis cs:1.0,0.5)
  {$\Delta r = 2\zeta$};
\end{axis}
\end{tikzpicture}
\caption[Half-power bandwidth of a resonance]{The half-power
(three-decibel) bandwidth method for reading a damping ratio off a
measured resonance. The amplitude curve, blue, is normalised to unit
peak. The two frequencies $r_1$ and $r_2$ at which the amplitude has
fallen to the peak divided by $\sqrt2$, red dashed line, bracket the
resonance; there the response power has halved. For a lightly damped
resonance the fractional spacing between them equals twice the damping
ratio, $\Delta r=(r_2-r_1)=2\zeta$, so the damping reads directly off
three frequencies on the plot without a curve fit. The method assumes a
single isolated mode; two overlapping modes flatten the peak and make
the half-power points bracket the pair, which the internal check
$H_{\mathrm{peak}}=Q\,H(0)$ catches.}
\label{fig:vol03ch06:half-power}
\end{figure}
