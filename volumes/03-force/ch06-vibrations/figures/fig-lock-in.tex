% Figure: vortex-shedding lock-in. The Strouhal relation predicts a
% shedding frequency that rises linearly with wind speed. Near the
% structure's natural frequency the shedding frequency abandons the
% Strouhal line and locks onto the natural frequency over a band of
% wind speeds, the lock-in or synchronisation range, where the
% across-wind response is largest.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={wind speed $U$ (m/s)},
  ylabel={shedding frequency $f_s$ (Hz)},
  xmin=0, xmax=20, ymin=0, ymax=3.2,
  axis lines=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=north west,
]
% Strouhal line f = St U / D, St=0.2, D=1.6 m => slope 0.125
\addplot[engblue, very thick, domain=0:8.8, samples=2] {0.125*x};
\addlegendentry{Strouhal line $f_s=\mathit{St}\,U/D$}
% Lock-in plateau at the natural frequency 1.1 Hz from U about 8.8 to 12.8
\addplot[engred, very thick, domain=8.8:12.8, samples=2, forget plot] {1.1};
% Strouhal line resumes after lock-in
\addplot[engblue, very thick, domain=12.8:20, samples=2, forget plot] {0.125*x};
% horizontal reference at the natural frequency
\addplot[enggray, dashed, domain=0:20, samples=2, forget plot] {1.1};
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:0.3,1.25) {structure $f_n=1.1$ Hz};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:10.8,1.18) {lock-in band};
\draw[engred, thin, <->] (axis cs:8.8,0.55) -- (axis cs:12.8,0.55);
\node[engred, font=\scriptsize, anchor=north] at (axis cs:10.8,0.52) {$U_{\mathrm{cr}}\pm$};
\end{axis}
\end{tikzpicture}
\caption[Vortex-shedding lock-in]{Shedding frequency against wind
speed for a slender stack. Away from resonance the alternating
vortices shed at the Strouhal frequency $f_s=\mathit{St}\,U/D$, which
rises in proportion to wind speed (blue). As the wind speed approaches
the critical value $U_{\mathrm{cr}}$ at which the Strouhal frequency
equals the structure's across-wind natural frequency, the shedding
abandons the Strouhal line and synchronises with the structure over a
band of wind speeds (red plateau), the lock-in or capture range. Inside
that band the vortex forcing and the structural motion reinforce one
another and the across-wind amplitude reaches its peak; the Strouhal
relation resumes once the wind speed leaves the band. The design speed
that matters is therefore not the highest wind in the record but the
one that places shedding at the natural frequency, often a moderate
everyday wind.}
\label{fig:vol03ch06:lock-in}
\end{figure}
