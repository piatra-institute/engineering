% Figure: the quarter-car model, a two-mass schematic. Sprung mass
% (body) on suspension spring and damper; unsprung mass (wheel and
% axle) on tyre spring; tyre rests on the road profile. Springs drawn
% as zigzag polylines so the figure needs no decoration libraries.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  mass/.style={draw, fill=engblue!12, minimum width=2.6cm, minimum height=0.95cm, font=\small},
  arr/.style={-{Stealth}, thick},
]
% vertical spring as a zigzag from (x,ya) up to (x,yb); fixed segments.
\newcommand{\vspring}[3]{%
  \draw[enggray, thick] (#1,#2) -- (#1,#2+0.12)
    -- (#1-0.13,#2+0.24) -- (#1+0.13,#2+0.40)
    -- (#1-0.13,#2+0.56) -- (#1+0.13,#2+0.72)
    -- (#1-0.13,#2+0.88) -- (#1,#2+1.00) -- (#1,#3);}
% vertical damper as a dashpot box from (x,ya) up to (x,yb).
\newcommand{\vdamper}[3]{%
  \draw[enggray, thick] (#1,#2) -- (#1,#2+0.40);
  \draw[enggray, thick] (#1-0.16,#2+0.40) rectangle (#1+0.16,#2+0.78);
  \draw[enggray, thick] (#1,#2+0.95) -- (#1,#3);
  \draw[enggray, thick] (#1-0.20,#2+0.62) -- (#1+0.20,#2+0.62);}
% sprung mass (body)
\node[mass] (body) at (0,4.2) {$m_s$ (body)};
% suspension spring (left) and damper (right) between body and wheel
\vspring{-0.7}{2.75}{3.72}
\vdamper{0.7}{2.75}{3.72}
\node[enggray, font=\scriptsize, anchor=east] at (-0.85,3.25) {$k_s$};
\node[enggray, font=\scriptsize, anchor=west] at (0.95,3.25) {$c_s$};
% unsprung mass (wheel)
\node[mass, minimum width=1.8cm] (wheel) at (0,2.3) {$m_u$ (wheel)};
% tyre spring between wheel and road
\vspring{0}{1.05}{1.82}
\node[enggray, font=\scriptsize, anchor=west] at (0.18,1.40) {$k_t$};
% road profile (a gentle bump line)
\draw[engblack, thick] (-2.0,0.95) sin (-1.0,1.10) cos (0.0,0.95) sin (1.0,0.80) cos (2.0,0.95);
\node[enggray, font=\scriptsize, anchor=west] at (2.1,0.95) {road $u(t)$};
% coordinate arrows
\draw[arr, engred] (1.9,4.0) -- (1.9,4.6);
\node[engred, font=\scriptsize, anchor=west] at (1.95,4.3) {$x_s$};
\draw[arr, engred] (1.3,2.1) -- (1.3,2.7);
\node[engred, font=\scriptsize, anchor=west] at (1.35,2.4) {$x_u$};
\end{tikzpicture}
\caption[The quarter-car suspension model]{The quarter-car model, the
working two-mass idealisation of one corner of a vehicle suspension.
The sprung mass $m_s$ is one quarter of the body riding on the
suspension spring $k_s$ and damper $c_s$; the unsprung mass $m_u$ is
the wheel, tyre, and outboard half of the suspension links riding on
the tyre stiffness $k_t$, which couples the wheel to the road profile
$u(t)$. The model carries two natural frequencies far apart: a low
body-bounce mode near $1\,\si{\hertz}$ that sets ride comfort, and a
high wheel-hop mode near $10\,\si{\hertz}$ that sets road holding. The
suspension damper $c_s$ must control both, which is the central
compromise of suspension tuning.}
\label{fig:vol03ch06:quarter-car}
\end{figure}
