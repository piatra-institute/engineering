% Figure: the case-study isolator transmissibility curve, in absolute
% frequency, with the design natural frequency (6 Hz), the running speed
% (24 Hz), and the achieved transmissibility marked. Shows why the mount
% must clear sqrt(2)*f_n and how the running orders land in the isolated band.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={frequency (Hz)}, ylabel={transmissibility $\mathrm{TR}$},
  xmin=1, xmax=120, ymin=0.003, ymax=12,
  xmode=log, ymode=log,
  axis lines=left,
  domain=1:120, samples=500,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=south west,
]
% TR(f) with f_n = 6 Hz, zeta = 0.07. r = f/6.
\addplot[engblue, very thick] {
  sqrt( (1 + (2*0.07*(x/6))^2) / ( (1-(x/6)^2)^2 + (2*0.07*(x/6))^2 ) )
};
\addlegendentry{$f_n=6$~Hz, $\zeta=0.07$}
% TR = 1 reference
\addplot[enggray, thin, dashed, forget plot] {1};
% mark running speed and its orders
\draw[engred, thin] (axis cs:24,0.003) -- (axis cs:24,12);
\node[engred, font=\scriptsize, anchor=south, rotate=90] at (axis cs:24,0.02) {$1\times=24$~Hz};
\draw[engorange, thin, dotted] (axis cs:48,0.003) -- (axis cs:48,12);
\node[engorange, font=\scriptsize, anchor=south, rotate=90] at (axis cs:48,0.02) {$2\times=48$~Hz};
% resonance marker near 6 Hz
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:6.3,6) {startup peak $\approx 7$};
\end{axis}
\end{tikzpicture}
\caption[Case-study isolator transmissibility]{The transmissibility of
the designed isolation mount, plotted log-log against absolute
frequency. The mount natural frequency is $6$~Hz with $\zeta=0.07$. The
curve peaks near $6$~Hz at the startup amplification $1/(2\zeta)\approx
7$, crosses unity at $6\sqrt2\approx 8.5$~Hz, and falls steeply
thereafter. At the running speed $1\times=24$~Hz ($r=4$) the
transmissibility is about $0.08$, and at the dominant misalignment
order $2\times=48$~Hz it is about $0.03$, so the mount passes well
under a tenth of the disturbance force into the foundation across the
machine's whole running-order content. The machine accelerates through
the $6$~Hz resonance on startup, the unavoidable cost of placing the
mount frequency far below the operating speed.}
\label{fig:vol03ch06:casestudy-isolation}
\end{figure}
