% Figure: Rayleigh proportional damping. The modal damping ratio
% zeta(omega) = (1/2)(alpha/omega + beta*omega) is fitted to two target
% modes and rises away from the U-shaped minimum elsewhere.
% Chosen so that zeta = 0.02 at omega = 10 and omega = 60 rad/s.
% Solve: alpha/(2w)+beta w/2 = 0.02 at w1=10, w2=60.
% alpha = 2 z w1 w2 (... ) ; standard result:
% alpha = 2*z*w1*w2/(w1+w2), beta = 2*z/(w1+w2) with equal z at both.
% z=0.02, w1=10, w2=60: alpha = 2*0.02*600/70 = 0.34286,
% beta = 2*0.02/70 = 5.7143e-4.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={$\omega$ (rad/s)}, ylabel={modal damping ratio $\zeta(\omega)$},
  xmin=0, xmax=90, ymin=0, ymax=0.06,
  axis lines=left,
  domain=2:90, samples=300,
  legend pos=north west,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Total Rayleigh curve.
\addplot[engblue, very thick] {0.5*(0.34286/x + 5.7143e-4*x)};
\addlegendentry{$\zeta=\tfrac12(\alpha/\omega+\beta\omega)$}
% Mass-proportional part alpha/(2 omega).
\addplot[engamber, thick, dashed] {0.5*0.34286/x};
\addlegendentry{mass term $\alpha/2\omega$}
% Stiffness-proportional part beta omega / 2.
\addplot[englime, thick, dashed] {0.5*5.7143e-4*x};
\addlegendentry{stiffness term $\beta\omega/2$}
% target damping line at zeta = 0.02
\addplot[enggray, thin, forget plot] {0.02};
% mark the two fitted frequencies
\addplot[engred, only marks, mark=*, mark size=2pt, forget plot] coordinates {(10,0.02) (60,0.02)};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:10,0.022) {fit};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:60,0.022) {fit};
\end{axis}
\end{tikzpicture}
\caption[Rayleigh proportional damping versus frequency]{Modal damping
ratio under the Rayleigh model $C=\alpha M+\beta K$, which gives
$\zeta(\omega)=\tfrac12(\alpha/\omega+\beta\omega)$. The mass-proportional
term (amber dashed) dominates at low frequency and damps the lowest
modes; the stiffness-proportional term (green dashed) dominates at high
frequency and damps the highest modes. The sum (blue) is U-shaped, with
a minimum between the two. The two constants $\alpha$ and $\beta$ are
fitted so the curve passes through a measured damping ratio at two
frequencies (red dots, here $\zeta=0.02$ at $10$ and $60$ rad/s); every
other mode is then assigned the damping the curve gives at its own
frequency. Modes between the fit points are slightly underdamped by the
model and modes outside are overdamped, the known limitation of fitting
two constants to a whole spectrum.}
\label{fig:vol03ch06:rayleigh-damping}
\end{figure}
