% Figure: body-acceleration transmissibility of the quarter-car body
% mode for three suspension damping ratios, against road-input
% frequency. Models the sprung-mass response with the unsprung mass
% lumped: a SDOF body-on-suspension transmissibility evaluated over
% the ride band. Shows the ride-vs-handling damping compromise.
% Curve: TR = sqrt( (1 + (2 zeta r)^2) / ((1-r^2)^2 + (2 zeta r)^2) )
% with omega_n the body-bounce frequency (here normalised so r = f/f_b).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.2cm,
  xlabel={road-input frequency ratio $r=f/f_b$},
  ylabel={body transmissibility $X_s/U$},
  xmin=0, xmax=8, ymin=0, ymax=4.2,
  axis lines=left,
  domain=0.02:8, samples=500,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
\addplot[engred, very thick] {sqrt((1 + (2*0.15*x)^2)/((1-x^2)^2 + (2*0.15*x)^2))};
\addlegendentry{$\zeta=0.15$ (soft, comfort)}
\addplot[engorange, very thick] {sqrt((1 + (2*0.30*x)^2)/((1-x^2)^2 + (2*0.30*x)^2))};
\addlegendentry{$\zeta=0.30$ (compromise)}
\addplot[engblue, very thick] {sqrt((1 + (2*0.70*x)^2)/((1-x^2)^2 + (2*0.70*x)^2))};
\addlegendentry{$\zeta=0.70$ (firm, control)}
\addplot[enggray, thin, forget plot] {1};
\addplot[enggray, dotted, thick, forget plot] coordinates {(1.41421,0) (1.41421,4.2)};
\node[enggray, font=\scriptsize, anchor=south west, rotate=90] at (axis cs:1.41421,0.2) {$r=\sqrt2$};
\end{axis}
\end{tikzpicture}
\caption[Quarter-car body transmissibility and the ride-handling
trade]{Body-bounce transmissibility of the quarter-car against the
road-input frequency, normalised to the body-bounce frequency $f_b$,
for three suspension damping ratios. The soft setting ($\zeta=0.15$,
red) gives the lowest transmitted motion above resonance and so the
softest ride, at the cost of a tall resonant peak near $r=1$ that the
body crosses over every long-wavelength road undulation. The firm
setting ($\zeta=0.70$, blue) caps the resonant peak and keeps the
tyre planted for handling, at the cost of transmitting more
high-frequency road texture into the cabin. The compromise
($\zeta\approx0.3$, orange) is where most road cars sit; the curves
cross near $r=\sqrt2$, the same isolation threshold as any base-excited
mount, beyond which more damping worsens the ride. A passive damper
cannot win both ends of the band at once, which is why active and
semi-active suspensions vary $c_s$ with the road.}
\label{fig:vol03ch06:quarter-car-transmissibility}
\end{figure}
