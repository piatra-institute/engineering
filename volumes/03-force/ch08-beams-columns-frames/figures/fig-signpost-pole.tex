% Figure: a cantilever sign-post pole under combined loading. A vertical
% tubular post fixed at its base carries a sign panel near the top; wind
% pressure on the panel and on the post becomes a horizontal force and
% distributed load, while the panel and post weight act axially. The base
% section sees the largest combined stress: bending from wind plus axial
% from weight.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% ground and base fixity
\draw[engblack, line width=2pt] (-0.8,0) -- (0.8,0);
\foreach \x in {-0.8,-0.5,-0.2,0.1,0.4,0.7}{
  \draw[enggray] (\x,0) -- (\x-0.25,-0.25);
}
% the post (tube) as a tall thin rectangle
\draw[engblack, very thick, fill=engblue!8] (-0.1,0) rectangle (0.1,4.0);
\node[engblack, anchor=west, font=\scriptsize] at (0.15,2.0) {post, height $H$};
% sign panel near the top
\draw[engblack, very thick, fill=englime!20] (0.1,3.0) rectangle (1.9,3.9);
\node[engblack, font=\scriptsize] at (1.0,3.45) {sign panel};
% wind pressure on the panel
\foreach \y in {3.1,3.35,3.6,3.85}{
  \draw[engred, ->] (-0.9,\y) -- (0.05,\y);
}
\node[engred, anchor=east, font=\scriptsize] at (-0.95,3.5) {wind};
% wind line load on the post
\foreach \y in {0.4,1.0,1.6,2.2,2.8}{
  \draw[engorange, ->] (-0.7,\y) -- (-0.12,\y);
}
\node[engorange, anchor=east, font=\scriptsize] at (-0.75,1.6) {$w$};
% axial weight arrow at base
\draw[engblue, ->, very thick] (0.55,3.9) -- (0.55,0.15);
\node[engblue, anchor=west, font=\scriptsize] at (0.6,1.0) {weight $N$};
% base critical-section marker
\draw[engred, very thick] (-0.25,0.05) -- (0.25,0.05);
\node[engred, anchor=north west, font=\scriptsize] at (0.25,0.0)
  {critical section};
\end{tikzpicture}
\caption[Sign-post pole under wind and axial load]{A tubular sign-post
fixed at its base carries a sign panel near the top. Wind drives a
concentrated horizontal force on the panel and a smaller distributed
load along the post; the panel and post weight act as an axial
compression. The base section carries the full cantilever bending moment
from the wind together with the axial force, and the design checks the
combined stress there, the buckling of the post under the axial load,
and the deflection of the panel that controls serviceability.}
\label{fig:vol03ch08:signpost-pole}
\end{figure}
