% Figure: plastic collapse mechanism of a fixed-fixed beam under a
% central point load. Three plastic hinges (two at the fixed ends, one
% under the load) turn the beam into a mechanism; the upper-bound
% (kinematic) theorem equates external work to the work dissipated at
% the hinges.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% undeformed beam (faint)
\draw[enggray!50, thick] (0,0) -- (6,0);
% fixed-end wall hatching
\draw[engblack, line width=2pt] (0,-0.7) -- (0,0.7);
\draw[engblack, line width=2pt] (6,-0.7) -- (6,0.7);
\foreach \y in {-0.6,-0.3,0,0.3,0.6}{
  \draw[enggray] (0,\y) -- (-0.28,\y+0.18);
  \draw[enggray] (6,\y) -- (6.28,\y+0.18);
}
% collapse mechanism: two straight rigid segments meeting at midspan
\draw[engblue, very thick] (0,0) -- (3,-1.1);
\draw[engblue, very thick] (3,-1.1) -- (6,0);
% central load
\draw[engred, very thick, ->] (3,1.3) -- (3,-0.95);
\node[engred, anchor=west, font=\small] at (3.1,1.0) {$P$};
% plastic hinges
\fill[engamber] (0,0) circle (0.11);
\fill[engamber] (6,0) circle (0.11);
\fill[engamber] (3,-1.1) circle (0.11);
\node[engamber, font=\scriptsize, anchor=south east] at (-0.05,0.1)
  {hinge $M_p$};
\node[engamber, font=\scriptsize, anchor=south west] at (6.05,0.1)
  {hinge $M_p$};
\node[engamber, font=\scriptsize, anchor=north] at (3,-1.2)
  {hinge $M_p$};
% rotation angles
\node[engblue, font=\scriptsize, anchor=north west] at (0.5,-0.05)
  {$\theta$};
\node[engblue, font=\scriptsize, anchor=north east] at (5.5,-0.05)
  {$\theta$};
% span dimension
\draw[<->, enggray] (0,-1.7) -- (6,-1.7)
  node[midway, below, font=\scriptsize]{span $L$};
\end{tikzpicture}
\caption[Plastic collapse mechanism of a fixed-fixed beam]{The collapse
mechanism of a fixed-ended beam under a central point load. Yielding
forms three plastic hinges, one at each fixed end and one beneath the
load, each carrying the plastic moment $M_p$. The beam becomes a
mechanism of two rigid links that rotate about the end hinges through an
angle $\theta$, the central hinge rotating through $2\theta$. The
upper-bound theorem equates the external work $P\,\delta$ with the
central deflection $\delta=(L/2)\theta$ to the internal work dissipated
at the hinges, $M_p\theta+M_p(2\theta)+M_p\theta=4M_p\theta$, giving the
collapse load $P_c=8M_p/L$. This exceeds the load $P_y$ at first yield,
the reserve a ductile indeterminate beam carries between first hinge and
mechanism.}
\label{fig:vol03ch08:collapse-mechanism}
\end{figure}
