% Figure: equilibrium of a differential beam slice of width dx. Shows the
% shear V and moment M on the left face, V+dV and M+dM on the right face,
% and the distributed load w(x) acting downward over the slice. This is the
% free body from which dV/dx = -w and dM/dx = V are read.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% distributed load arrows over the slice
\foreach \x in {0.4,1.2,2.0,2.8,3.6}{
  \draw[engred, ->] (\x,2.55) -- (\x,1.85);
}
\draw[engred, thick] (0.2,2.55) -- (3.8,2.55);
\node[engred, anchor=south, font=\scriptsize] at (2.0,2.6) {$w(x)$};
% the slice body
\draw[engblack, very thick, fill=engblue!8]
  (0.2,0.0) rectangle (3.8,1.6);
% left face: shear up, moment
\draw[engblue, ->, very thick] (0.2,1.6) -- (0.2,0.55)
  node[midway, anchor=east, font=\scriptsize]{$V$};
\draw[engamber, ->, thick] (-0.35,0.8) arc (200:520:0.32);
\node[engamber, anchor=east, font=\scriptsize] at (-0.45,0.8) {$M$};
% right face: shear down, moment+dM
\draw[engblue, ->, very thick] (3.8,0.55) -- (3.8,1.6)
  node[midway, anchor=west, font=\scriptsize]{$V+dV$};
\draw[engamber, ->, thick] (4.45,0.8) arc (-20:300:0.32);
\node[engamber, anchor=west, font=\scriptsize] at (4.55,0.8) {$M+dM$};
% slice width dimension
\draw[<->, enggray] (0.2,-0.4) -- (3.8,-0.4)
  node[midway, below, font=\scriptsize]{$dx$};
% neutral axis dashed
\draw[enggray, dashed] (-0.7,0.8) -- (4.9,0.8);
\end{tikzpicture}
\caption[Equilibrium of a differential beam slice]{The free body of a
beam slice of width $dx$, carrying the distributed load $w(x)$ over its
top. The left face carries a shear $V$ and a moment $M$; the right face
carries $V+dV$ and $M+dM$. Vertical equilibrium of the slice gives
$dV/dx=-w$, and moment equilibrium about the right face, dropping the
second-order term $w\,dx^2/2$, gives $dM/dx=V$. These two relations,
combined with the moment-curvature law, generate the Euler-Bernoulli
beam equation.}
\label{fig:vol03ch08:beam-slice}
\end{figure}
