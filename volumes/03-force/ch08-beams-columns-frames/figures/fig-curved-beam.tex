% Figure: a curved beam (crane hook / C-clamp) with the shifted neutral
% axis of Winkler theory. The neutral axis moves toward the center of
% curvature, and the bending stress is hyperbolic across the depth
% rather than linear, peaking sharply at the inner fibre.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% center of curvature O
\fill[enggray] (0,0) circle (0.06);
\node[enggray, font=\scriptsize, anchor=east] at (-0.05,0) {$O$};
% inner and outer arcs of the curved bar (a C-shape opening to the right)
\draw[engblack, very thick] (150:2.0) arc (150:-150:2.0);   % inner fibre
\draw[engblack, very thick] (150:3.2) arc (150:-150:3.2);   % outer fibre
% close the two ends (top and bottom of the C)
\draw[engblack, very thick] (150:2.0) -- (150:3.2);
\draw[engblack, very thick] (-150:2.0) -- (-150:3.2);
% radial cross-section line at theta = 0 (the critical section)
\draw[enggray, dashed] (0,0) -- (3.6,0);
\node[engblack, font=\scriptsize, anchor=south] at (2.0,0.05) {$r_i$};
\node[engblack, font=\scriptsize, anchor=south] at (3.2,0.05) {$r_o$};
% centroidal radius R and neutral radius r_n (r_n < R, shifted inward)
\node[engblue, font=\scriptsize, anchor=north] at (2.6,-0.05) {$R$};
\fill[engblue] (2.6,0) circle (0.05);
\fill[engred] (2.45,0) circle (0.05);
\node[engred, font=\scriptsize, anchor=south west] at (2.42,0.08) {$r_n$};
% hyperbolic stress distribution across the section (to the right)
\begin{scope}[shift={(4.6,0)}]
  \draw[enggray] (0,-1.2) -- (0,1.2);            % section reference line
  \node[enggray, font=\scriptsize, anchor=south] at (0,1.2) {inner};
  \node[enggray, font=\scriptsize, anchor=north] at (0,-1.2) {outer};
  % hyperbolic stress: large tension at inner (top), smaller compression outer
  \draw[engred, very thick]
    plot[domain=-1.2:1.15, samples=40]
    ({ 0.9/(1.7 + 1.1*\x) - 0.42 }, {\x});
  \draw[engred, dashed] (0,0.18) -- (0.72,0.18);  % neutral axis crossing
  \node[engred, font=\scriptsize, anchor=west] at (0.75,0.9) {$\sigma$};
\end{scope}
\end{tikzpicture}
\caption[Curved-beam stress and the shifted neutral axis]{A sharply
curved bar, a crane hook or a C-clamp, bent by a moment in its own
plane. Plane sections still rotate as rigid lines, but because the inner
fibres are shorter than the outer ones the strain is no longer linear in
the distance from the centroid, and the neutral axis $r_n$ shifts inward
toward the centre of curvature $O$, away from the centroidal radius $R$.
The Winkler theory gives a hyperbolic stress distribution across the
depth, drawn at right, with the peak tension concentrated at the inner
fibre $r_i$. The straight-beam formula $\sigma=My/I$ underestimates that
inner-fibre stress, by a factor that grows as the radius of curvature
falls toward the section depth, which is why a hook is analysed as a
curved beam and not as a straight one.}
\label{fig:vol03ch08:curved-beam}
\end{figure}
