% Figure: a diving springboard modelled as a propped cantilever. The
% board is fixed (or held down) at the back, rides on an adjustable
% fulcrum, and overhangs the water as a cantilever carrying the diver's
% weight at the tip. Moving the fulcrum changes the overhang and the tip
% stiffness.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% water line
\draw[engblue!50, thick] (5.4,-1.6) -- (7.6,-1.6);
\foreach \x in {5.6,6.1,6.6,7.1} \draw[engblue!50] (\x,-1.6) .. controls (\x+0.12,-1.45) and (\x+0.25,-1.75) .. (\x+0.35,-1.6);
% support post at the back (hold-down)
\draw[enggray, very thick] (0,0) -- (0,-1.2);
\draw[engblack, thick] (-0.4,-1.2) -- (0.4,-1.2);
\foreach \x in {-0.35,-0.15,0.05,0.25} \draw[engblack] (\x,-1.2) -- (\x-0.15,-1.4);
\node[engblack, font=\scriptsize, anchor=east] at (-0.1,0) {hold-down};
% the fulcrum (roller) partway along
\fill[engamber] (2.2,-0.28) circle (0.16);
\node[engamber, font=\scriptsize, anchor=north] at (2.2,-0.5) {fulcrum};
% the board itself (a long horizontal beam), slightly deflected at tip
\draw[engblack, very thick] (0,0) -- (5.0,0);
% deflected shape under diver (dashed)
\draw[engred, very thick, dashed]
  (2.2,-0.05) .. controls (3.6,-0.15) and (4.4,-0.7) .. (5.0,-1.1);
% diver weight at the tip
\draw[engred, very thick, ->] (5.0,0.9) -- (5.0,0.05);
\node[engred, anchor=south, font=\small] at (5.0,0.95) {$W$};
% back span a and overhang L
\draw[<->, engblue] (0,0.55) -- (2.2,0.55) node[midway, above, font=\scriptsize]{$a$};
\draw[<->, engblue] (2.2,0.55) -- (5.0,0.55) node[midway, above, font=\scriptsize]{$L$};
% tip deflection
\draw[engred, ->] (5.35,0) -- (5.35,-1.1);
\node[engred, font=\scriptsize, anchor=west] at (5.4,-0.55) {$\delta$};
\end{tikzpicture}
\caption[A diving springboard as a fulcrum-supported cantilever]{A
diving springboard modelled as a beam held down at the back, riding on
an adjustable fulcrum, and overhanging the water. The diver's weight $W$
at the tip bends the overhang $L$ as a cantilever measured from the
fulcrum, while the back span $a$ is held down against the uplift the
overhang produces. The tip deflection $\delta$ and, through the same
flexural rigidity $EI$, the board's natural frequency both scale with
$L^3/EI$, so sliding the fulcrum forward shortens $L$, stiffens the
board, and raises its bounce frequency. The board is deliberately
flexible, unlike a structural beam, so that it stores and returns the
diver's energy; the design target is a tip stiffness and frequency the
diver can time to, not a deflection limit.}
\label{fig:vol03ch08:springboard}
\end{figure}
