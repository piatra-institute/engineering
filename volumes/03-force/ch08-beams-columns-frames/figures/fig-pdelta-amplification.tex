% Figure: the P-delta moment-amplification factor 1/(1 - P/Pcr) plotted
% against the axial load ratio P/Pcr. The factor is near unity for light
% axial load and diverges as the load approaches the Euler critical load,
% which is why slender columns carrying an appreciable fraction of their
% buckling load must be checked for second-order effects.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=6.6cm,
  xlabel={axial load ratio $P/P_{\text{cr}}$},
  ylabel={amplification $1/(1-P/P_{\text{cr}})$},
  xmin=0, xmax=0.9, ymin=1, ymax=8,
  xtick={0,0.2,0.4,0.6,0.8},
  ytick={1,2,3,4,5,6,7,8},
  grid=both,
  grid style={enggray!25},
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.03,0.97)}, anchor=north west, draw=enggray!50},
]
% amplification curve, sampled short of the pole at P/Pcr = 1
\addplot[engblue, very thick, domain=0:0.875, samples=120] {1/(1-x)};
\addlegendentry{$1/(1-P/P_{\text{cr}})$}
% first-order (no amplification) reference
\addplot[engamber, thick, dashed, domain=0:0.9, samples=2] {1};
\addlegendentry{first-order (no P-delta)}
% mark the 20 percent point, a common code threshold for mandatory check
\addplot[engred, only marks, mark=*, mark size=1.6pt] coordinates {(0.2,1.25)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.22,1.55)
  {$P/P_{\text{cr}}=0.2 \to 1.25$};
% mark the 60 percent point
\addplot[engred, only marks, mark=*, mark size=1.6pt] coordinates {(0.6,2.5)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.52,3.4)
  {$0.6 \to 2.5$};
\end{axis}
\end{tikzpicture}
\caption[The P-delta moment-amplification factor]{The second-order
amplification factor $1/(1-P/P_{\text{cr}})$ that multiplies the
first-order deflection and moment of a compression member carrying axial
load $P$. At light axial load the factor sits just above unity and the
member behaves as first-order theory predicts. As $P$ climbs toward the
Euler critical load the factor grows without bound, doubling the demand
near $P/P_{\text{cr}}=0.5$ and reaching two and a half at
$P/P_{\text{cr}}=0.6$. Design codes require an explicit second-order
analysis once the axial demand passes a threshold, commonly around a fifth
of the critical load, where the amplification has grown to about a quarter
above the first-order value and can no longer be neglected. The divergence
at $P=P_{\text{cr}}$ is the same instability the buckling analysis marks,
reached here through the growth of deflection rather than through an
eigenvalue.}
\label{fig:vol03ch08:pdelta-amplification}
\end{figure}
