% Figure: a beam on a Winkler elastic foundation under a concentrated
% load. The deflection is a damped cosine that decays within a few
% characteristic lengths, so a point load on a long rail or footing is
% felt only locally. The curve plotted is the closed-form
% y ~ exp(-beta x)(cos beta x + sin beta x).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={distance from load, $\beta x$}, ylabel={deflection $y/y_0$},
  xmin=-6, xmax=6, ymin=-0.35, ymax=1.1,
  axis lines=left,
  ytick={0,0.5,1.0}, yticklabels={$0$,$0.5$,$1.0$},
  xtick={-6,-4,-2,0,2,4,6},
  tick label style={font=\scriptsize},
  label style={font=\small},
  grid=major, grid style={enggray!18},
]
% undeflected line
\addplot[enggray, dashed, thick, domain=-6:6, samples=2] {0};
% Winkler point-load deflection: exp(-|s|)(cos|s| + sin|s|)
\addplot[engblue, very thick, domain=0:6, samples=100, smooth]
  {exp(-x)*(cos(deg(x)) + sin(deg(x)))};
\addplot[engblue, very thick, domain=-6:0, samples=100, smooth]
  {exp(x)*(cos(deg(-x)) + sin(deg(-x)))};
% load arrow at the origin
\addplot[engred, only marks, mark=*, mark options={fill=engred}]
  coordinates {(0,1)};
\node[engred, anchor=south, font=\small] at (axis cs:0,1.02) {$P$};
% mark the first zero crossing near beta x = 3pi/4 ~ 2.36
\addplot[engamber, only marks, mark=*, mark options={fill=engamber}]
  coordinates {(2.356,0)};
\node[engamber, anchor=south west, font=\scriptsize]
  at (axis cs:2.4,0.02) {uplift};
\end{axis}
\end{tikzpicture}
\caption[Deflection of a beam on a Winkler elastic foundation]{A
concentrated load $P$ on a long beam resting on a Winkler foundation of
modulus $k$ deflects the beam into a damped wave
$y=y_0\,e^{-\beta x}(\cos\beta x + \sin\beta x)$, where the
characteristic length $1/\beta=(4EI/k)^{1/4}$ sets the reach of the
disturbance. The deflection dies out within about two characteristic
lengths, crosses zero near $\beta x=3\pi/4$, and reverses into a small
uplift beyond, so a point load on a rail or a strip footing is carried
locally rather than shared along the whole length. Doubling the flexural
rigidity $EI$ spreads the load over a longer characteristic length and
lowers the peak deflection under the load, the design lever for a stiff
rail or a raft foundation.}
\label{fig:vol03ch08:elastic-foundation}
\end{figure}
