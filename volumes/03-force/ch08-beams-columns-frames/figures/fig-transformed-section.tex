% Figure: the transformed-section method for a composite timber beam
% reinforced with a steel plate. The actual section (timber over steel)
% is replaced by an equivalent all-timber section in which the steel is
% widened by the modular ratio n = E_s/E_t, preserving the strain
% distribution and shifting the neutral axis toward the stiffer layer.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% left: actual composite section
\begin{scope}[shift={(0,0)}]
  % timber block
  \fill[engamber!25] (0,0.6) rectangle (2,3.6);
  \draw[engblack, thick] (0,0.6) rectangle (2,3.6);
  \node[engblack, font=\scriptsize] at (1,2.1) {timber};
  \node[engblack, font=\scriptsize] at (1,1.55) {$E_t$};
  % steel plate on the underside
  \fill[engblue!35] (0,0.2) rectangle (2,0.6);
  \draw[engblack, thick] (0,0.2) rectangle (2,0.6);
  \node[engblack, font=\scriptsize, anchor=west] at (2.1,0.4) {steel $E_s$};
  \node[engblack, font=\scriptsize, anchor=north] at (1,-0.1) {actual};
\end{scope}
% right: transformed (all-timber) section
\begin{scope}[shift={(5.5,0)}]
  % timber unchanged
  \fill[engamber!25] (0,0.6) rectangle (2,3.6);
  \draw[engblack, thick] (0,0.6) rectangle (2,3.6);
  \node[engblack, font=\scriptsize] at (1,2.1) {timber};
  % steel widened by n = E_s/E_t (drawn wider, same thickness)
  \fill[engamber!45] (-1.0,0.2) rectangle (3.0,0.6);
  \draw[engblack, thick] (-1.0,0.2) rectangle (3.0,0.6);
  \node[engblack, font=\scriptsize, anchor=north] at (1,0.05)
    {steel widened by $n=E_s/E_t$};
  \node[engblack, font=\scriptsize, anchor=north] at (1,-0.45)
    {transformed};
\end{scope}
% neutral-axis line through both
\draw[engred, dashed, thick] (-1.3,1.45) -- (8.7,1.45);

\end{tikzpicture}
\caption[Transformed-section method for a composite beam]{The
transformed-section method for a timber beam reinforced with a bonded
steel plate. The two materials share the same strain at a given depth
(plane sections remain plane), so the steel carries $n=E_s/E_t$ times
the stress of timber at equal strain. Replacing the steel by an
equivalent width of timber, widened by the modular ratio $n$ while
keeping its thickness, produces an all-timber section with the same
bending stiffness and the same strain field. The neutral axis is the
centroid of the transformed section, pulled toward the stiffer steel
layer. Bending stresses follow from $\sigma=My/I$ on the transformed
section, with the steel stresses multiplied back by $n$ to recover the
real values. The same construction sizes steel-concrete composite beams
and reinforced-concrete sections, where the reinforcement is
transformed into equivalent concrete.}
\label{fig:vol03ch08:transformed-section}
\end{figure}
