% Figure: a semi-elliptic multi-leaf spring. The graded stack of leaves
% approximates a diamond-plan beam of uniform strength: each leaf is
% shorter than the one above, so the assembled depth grows toward the
% centre where the moment is largest, keeping the extreme-fibre stress
% roughly constant along the span. The eye at each end takes the axle
% shackle; the central clamp takes the axle load.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% the graded leaf stack, drawn as a nest of shallow arcs of decreasing span
\draw[engblack, very thick] (-4.0,0.0) .. controls (-2.0,0.55) and (2.0,0.55) .. (4.0,0.0);
\draw[engblue, thick]      (-3.2,0.16) .. controls (-1.6,0.60) and (1.6,0.60) .. (3.2,0.16);
\draw[engblue, thick]      (-2.4,0.30) .. controls (-1.2,0.62) and (1.2,0.62) .. (2.4,0.30);
\draw[engblue, thick]      (-1.6,0.44) .. controls (-0.8,0.62) and (0.8,0.62) .. (1.6,0.44);
\draw[engblue, thick]      (-0.9,0.54) .. controls (-0.4,0.62) and (0.4,0.62) .. (0.9,0.54);
% central clamp / U-bolt carrying the axle load
\draw[enggray, very thick] (-0.35,0.75) rectangle (0.35,0.05);
\draw[engred, very thick, ->] (0.0,1.55) -- (0.0,0.80);

\end{tikzpicture}
\caption[A multi-leaf semi-elliptic spring as a beam of uniform
strength]{A semi-elliptic multi-leaf spring. The graded stack of leaves,
each shorter than the one above it, builds an assembly whose effective
depth grows toward the central clamp where the bending moment peaks and
falls away toward the shackled ends where the moment vanishes. The
arrangement approximates a diamond-plan beam of uniform strength: with
the section modulus tracking the moment, the extreme-fibre bending stress
stays roughly constant along the span, so no leaf is worked far below its
neighbours and the material is used evenly. The dashed triangle is the
bending-moment envelope for the central axle load $P$ reacted at the two
shackles.}
\label{fig:vol03ch08:leaf-spring}
\end{figure}
