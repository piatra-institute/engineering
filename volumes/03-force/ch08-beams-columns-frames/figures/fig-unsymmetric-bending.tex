% Figure: unsymmetric bending of an equal-leg angle. The centroidal axes
% y, z aligned with the legs are not principal axes because the product
% of inertia I_yz is non-zero; the principal axes u, v are rotated, and a
% vertical load bent about the geometric axis deflects sideways as well
% as down unless it acts in a principal plane.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.1]
% equal-leg angle (L-shape), legs along +y (up) and +z (right)
\fill[engblue!20]
  (0,0) -- (3,0) -- (3,0.5) -- (0.5,0.5) -- (0.5,3) -- (0,3) -- cycle;
\draw[engblack, thick]
  (0,0) -- (3,0) -- (3,0.5) -- (0.5,0.5) -- (0.5,3) -- (0,3) -- cycle;
% centroid (approximate, for an equal-leg angle near the corner)
\coordinate (C) at (0.84,0.84);
\fill[engblack] (C) circle (0.05);
\node[engblack, font=\scriptsize, anchor=north east] at (C) {$C$};
% geometric centroidal axes y (vertical), z (horizontal)
\draw[enggray, thick, ->] (C) -- ++(0,2.3) node[above, font=\scriptsize]{$y$};
\draw[enggray, thick, ->] (C) -- ++(2.6,0) node[right, font=\scriptsize]{$z$};
\draw[enggray, thick, ->] (C) -- ++(0,-1.0);
\draw[enggray, thick, ->] (C) -- ++(-1.0,0);
% principal axes u, v rotated by alpha = 45 deg for an equal-leg angle
\draw[engred, very thick, ->] (C) -- ++({2.3*cos(45)},{2.3*sin(45)})
  node[right, font=\scriptsize]{$v$ (principal)};
\draw[engred, very thick, ->] (C) -- ++({2.0*cos(135)},{2.0*sin(135)})
  node[left, font=\scriptsize]{$u$ (principal)};
\draw[engred, very thick] (C) -- ++({2.0*cos(225)},{2.0*sin(225)});
\draw[engred, very thick] (C) -- ++({2.0*cos(315)},{2.0*sin(315)});
% angle arc between z and v
\draw[engamber, thick] ([shift={(C)}]0:0.9) arc (0:45:0.9);
\node[engamber, font=\scriptsize] at ($(C)+(22:1.15)$) {$\alpha$};
% applied vertical load (along -y) and resulting deflection direction
\draw[engorange, very thick, ->] (2.5,3.1) -- (2.5,2.0)
  node[midway, right, font=\scriptsize]{$P$};
\end{tikzpicture}
\caption[Unsymmetric bending and principal axes of an angle]{An
equal-leg angle bent about its geometric centroidal axes $y$ and $z$.
Because the section is not symmetric about either of these axes, the
product of inertia $I_{yz}$ is non-zero and they are not principal axes.
The principal axes $u$ and $v$ (red), about which the product of inertia
vanishes and the second moments are extreme, are rotated by the angle
$\alpha$ given by $\tan 2\alpha = 2I_{yz}/(I_z - I_y)$, which is exactly
$45^{\circ}$ for an equal-leg angle. A vertical load $P$ applied through
the shear centre produces bending about both principal axes, so the
section deflects sideways as well as downward unless the load is aligned
with a principal plane. Designing such a member means resolving the
moment onto the principal axes, computing the stress as the sum of the
two principal-axis contributions, and locating the neutral axis, which
no longer coincides with either geometric axis.}
\label{fig:vol03ch08:unsymmetric-bending}
\end{figure}
