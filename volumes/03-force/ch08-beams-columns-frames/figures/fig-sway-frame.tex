% Figure: a single-bay portal frame with fixed bases under a horizontal
% load at the beam level. The frame sways sideways by Delta; the columns
% bend in double curvature and the joints rotate. The deflected shape
% (dashed) shows the sidesway that the slope-deflection sway term
% captures.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% undeformed frame
\draw[enggray, thick] (0,0) -- (0,3);      % left column
\draw[enggray, thick] (4,0) -- (4,3);      % right column
\draw[enggray, thick] (0,3) -- (4,3);      % beam
% fixed bases (hatching)
\draw[engblack, thick] (-0.4,0) -- (0.4,0);
\foreach \x in {-0.35,-0.15,0.05,0.25} \draw[engblack] (\x,0) -- (\x-0.15,-0.2);
\draw[engblack, thick] (3.6,0) -- (4.4,0);
\foreach \x in {3.65,3.85,4.05,4.25} \draw[engblack] (\x,0) -- (\x-0.15,-0.2);
% deformed (swayed) shape, dashed: top translates right by Delta
\draw[engblue, very thick, dashed]
  (0,0) .. controls (0.15,1.5) and (0.55,1.5) .. (0.7,3);   % left column double curvature
\draw[engblue, very thick, dashed]
  (4,0) .. controls (4.15,1.5) and (4.55,1.5) .. (4.7,3);   % right column
\draw[engblue, very thick, dashed] (0.7,3) -- (4.7,3);       % beam translated
% horizontal load H at beam level
\draw[engred, very thick, ->] (-1.1,3) -- (-0.05,3);
\node[engred, anchor=east, font=\small] at (-1.15,3) {$H$};
% sway Delta
\draw[engamber, ->] (0,3.35) -- (0.7,3.35);
\node[engamber, font=\scriptsize, anchor=south] at (0.35,3.4) {$\Delta$};
% joint labels
\node[engblack, font=\scriptsize, anchor=south east] at (0,3) {$B$};
\node[engblack, font=\scriptsize, anchor=south west] at (4,3) {$C$};
\node[engblack, font=\scriptsize, anchor=north east] at (0,0) {$A$};
\node[engblack, font=\scriptsize, anchor=north west] at (4,0) {$D$};
% chord rotation psi = Delta / h annotation on left column
\node[engblue, font=\scriptsize, anchor=west] at (0.55,1.5) {$\psi=\Delta/h$};
\end{tikzpicture}
\caption[Sidesway of a portal frame under lateral load]{A single-bay
fixed-base portal frame carrying a horizontal load $H$ at the beam
level. The frame sways sideways through a displacement $\Delta$ (dashed),
the columns bending in double curvature between their fixed bases and the
rotating joints $B$ and $C$. The relative end displacement of each column
produces a chord rotation $\psi=\Delta/h$, the sidesway term that enters
each column's slope-deflection equation as $-3\psi$. The unknowns are the
two joint rotations and the single sway $\Delta$; the two joint-moment
equilibrium equations plus one storey-shear equation, the horizontal
equilibrium of the beam level, close the system. A frame free to sway is
always more flexible and carries larger moments than the same frame
braced against sway, which is why a lateral bracing system is added when
the sway moments or drift grow too large.}
\label{fig:vol03ch08:sway-frame}
\end{figure}
