% Figure: unit-load method applied to a portal frame. Left, the real
% structure under a horizontal load H with its bending-moment diagram on
% the members. Right, the same frame under a virtual unit horizontal load
% at the point and direction of the desired sway; the product of the two
% moment diagrams, integrated over the members, gives the sway deflection.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=0.8]
% ==== real system (left) ====
\begin{scope}[shift={(0,0)}]
\draw[engblack, very thick] (0,0) -- (0,3);   % left column
\draw[engblack, very thick] (4,0) -- (4,3);   % right column
\draw[engblack, very thick] (0,3) -- (4,3);   % beam
% pinned bases
\fill[engblack] (0,0) circle (0.08);
\fill[engblack] (4,0) circle (0.08);
\draw[enggray] (-0.25,-0.25) -- (0.25,-0.25);
\draw[enggray] (3.75,-0.25) -- (4.25,-0.25);
% real load
\draw[engred, very thick, ->] (-1.0,3) -- (-0.1,3);
\node[engred, font=\small, anchor=east] at (-1.0,3) {$H$};
% schematic moment diagram on columns (linear, peak at eaves)
\draw[engblue, thick] (0,0) -- (0.7,3);
\draw[engblue, thick] (4,0) -- (3.3,3);
\node[engblue, font=\scriptsize] at (2.0,-0.7) {real $M$};
\end{scope}
% ==== virtual system (right) ====
\begin{scope}[shift={(7,0)}]
\draw[engblack, very thick] (0,0) -- (0,3);
\draw[engblack, very thick] (4,0) -- (4,3);
\draw[engblack, very thick] (0,3) -- (4,3);
\fill[engblack] (0,0) circle (0.08);

\draw[enggray] (3.75,-0.25) -- (4.25,-0.25);
% virtual unit load at the sway point
\draw[engamber, very thick, ->] (-1.0,3) -- (-0.1,3);
\node[engamber, font=\small, anchor=east] at (-1.0,3) {$1$};
% schematic virtual moment diagram
\draw[engamber, thick] (0,0) -- (0.7,3);

\node[engamber, font=\scriptsize] at (2.0,-0.7) {virtual $m$};
\end{scope}
\end{tikzpicture}
\caption[Unit-load method for the sway of a portal frame]{The unit-load
method applied to the sidesway of a portal frame. The real system (left)
carries the actual horizontal load $H$ and develops the bending-moment
distribution $M(x)$ shown on the members. The virtual system (right) is
the same frame carrying a single unit horizontal load placed at the
joint and in the direction of the sway we want, developing its own
moment distribution $m(x)$. The sway deflection is the integral
$\Delta = \int M m / (EI)\,\mathrm{d}x$ taken over every member, which
for this frame reduces to summing the products of the two triangular
column diagrams. The method delivers a single deflection without solving
for the full deflected shape, and it generalises to any frame for which
both moment diagrams can be drawn.}
\label{fig:vol03ch08:frame-unit-load}
\end{figure}
