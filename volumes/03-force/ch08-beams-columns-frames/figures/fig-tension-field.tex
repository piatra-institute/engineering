% Figure: the tension-field action of a stiffened plate-girder web panel.
% After the web buckles in shear, the diagonal compression can grow no
% further, and the panel carries additional shear by a diagonal tension
% band anchored against the flanges and the transverse stiffeners, which
% then act like the chords and verticals of a Pratt truss.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% two web panels bounded by flanges (top, bottom) and stiffeners
\draw[engblack, line width=2pt] (0,0) -- (8,0);   % bottom flange
\draw[engblack, line width=2pt] (0,3) -- (8,3);   % top flange
\foreach \x in {0,4,8}{
  \draw[engamber, very thick] (\x,0) -- (\x,3);   % stiffeners
}
\node[engblack, font=\scriptsize, anchor=south] at (4,3.05) {flange (chord)};
\node[engblack, font=\scriptsize, anchor=north] at (4,-0.05) {flange (chord)};
\node[engamber, font=\scriptsize, anchor=east, rotate=90] at (-0.1,1.5)
  {stiffener (vertical)};
% diagonal tension band in the left panel
\draw[engred, thick, ->] (0.4,0.1) -- (3.2,2.9);
\draw[engred, thick, ->] (1.1,0.1) -- (3.9,2.9);
\draw[engred, thick, ->] (0.1,0.6) -- (2.6,3.0);
\draw[engred, thick, ->] (1.7,0.1) -- (3.95,2.4);
\node[engred, font=\scriptsize, anchor=west] at (2.1,1.0) {diagonal tension};
% buckled-out compression diagonal (faint, ineffective)
\draw[engblue, dashed, thick] (3.6,0.2) .. controls (2.0,1.5) .. (0.4,2.8);
\node[engblue, font=\scriptsize, anchor=west] at (4.15,1.5)
  {buckled diagonal (compression saturated)};
\end{tikzpicture}
\caption[Tension-field action in a stiffened web panel]{Post-buckling
strength of a stiffened plate-girder web. Once the panel buckles in
shear, the diagonal that is in compression can carry no more load, yet
the panel is not finished: it picks up further shear through a band of
diagonal tension (red) that stretches across the panel and anchors
against the flanges and the transverse stiffeners. The stiffened web
then behaves like a Pratt truss, the flanges acting as chords, the
stiffeners as compression verticals, and the diagonal tension field as
the web members. This tension-field action is the reserve that lets a
slender web be designed well past its elastic shear-buckling load, and
it is why the intermediate stiffeners must be proportioned to anchor the
tension band rather than merely to delay the initial buckle.}
\label{fig:vol03ch08:tension-field}
\end{figure}
