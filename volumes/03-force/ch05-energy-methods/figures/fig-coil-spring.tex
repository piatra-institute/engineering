% Figure: a helical coil spring under an axial load F. The wire carries a
% torque F R at every section, and the spring rate follows from the torsional
% strain energy stored along the whole wire length. An inset shows the wire
% cross-section twisted by the load arm R.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  coil/.style={draw=engblue, very thick},
  force/.style={draw=engred, -{Stealth}, very thick},
  dimline/.style={draw=enggray, {Stealth}-{Stealth}, thin},
  label/.style={font=\small},
]
% the helix, drawn as a stack of ellipses connected, side elevation
\foreach \k in {0,...,5} {
  \draw[coil] (0,{0.5*\k}) ellipse (0.9 and 0.22);
}
% connect front arcs to suggest the helix rise
\foreach \k in {0,...,4} {
  \draw[coil] (0.9,{0.5*\k}) .. controls (0.6,{0.5*\k+0.28}) .. (-0.9,{0.5*\k+0.5});
}
% top and bottom load arrows along the axis
\draw[force] (0,3.2) -- (0,3.0);
\node[engred, label, anchor=south] at (0,3.25) {$F$};
\draw[force] (0,-0.7) -- (0,-0.5);
\node[engred, label, anchor=north] at (0,-0.75) {$F$};

% mean coil radius R
\draw[dimline] (0,1.5) -- (0.9,1.5);
\node[enggray, font=\scriptsize, anchor=south] at (0.45,1.52) {$R$};

% inset: wire cross-section carrying torque T = F R
\begin{scope}[xshift=3.6cm, yshift=1.5cm]
\draw[draw=engblack, thick, fill=engamber!16] (0,0) circle (0.55);
\draw[draw=engred, -{Stealth}, very thick] (0.78,0) arc[start angle=-75, end angle=200, radius=0.78];
\node[engred, font=\scriptsize, anchor=west] at (1.0,0.0) {$T=FR$};
\node[engblack, font=\scriptsize, anchor=north] at (0,-0.65) {wire, diameter $d$};
\end{scope}

\end{tikzpicture}
\caption[Coil spring and the torque in its wire]{A helical compression spring
of mean coil radius $R$ carrying an axial load $F$. At every cross-section of
the wire the load acts at the arm $R$, so the wire is loaded in torsion by a
constant torque $T=FR$ (inset), with the direct shear $F$ a small correction
for a slender wire. The spring's whole compliance comes from the torsional
twist of the wire summed over its length, and the spring rate
$k=Gd^4/(8D^3 n)$ follows from equating the torsional strain energy stored in
the wire to the work $\tfrac12 F\delta$ done by the load, with $d$ the wire
diameter, $D=2R$ the mean coil diameter, and $n$ the number of active coils.
The energy method converts a torsion result for a straight wire into the rate
of a coiled spring in one balance.}
\label{fig:vol03ch05:coil-spring}
\end{figure}
