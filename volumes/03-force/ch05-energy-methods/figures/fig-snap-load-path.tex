% Figure: the load-displacement equilibrium path of a shallow von Mises
% truss. The path rises to a limit (snap) load, then falls through a region
% of negative slope (unstable equilibria) before stiffening again. Under
% load control the apex jumps horizontally across the unstable branch at the
% limit load: snap-through.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  stable/.style={draw=engblue, very thick},
  unstable/.style={draw=engred, very thick, dashed},
  jump/.style={draw=engamber, -{Stealth}, thick},
  guide/.style={draw=enggray, dashed, thin},
  label/.style={font=\small},
]
% x = normalised apex drop u = w/h0, range 0..2, plotted 0..8 (scale 4 per unit)
% y = normalised load P/Pmax, range -1.1..1.1, plotted -2.75..2.75 (scale 2.5)
\draw[axis] (0,0) -- (8.6,0) node[anchor=west, label] {apex drop $w/h_0$};
\draw[axis] (0,-2.9) -- (0,3.1) node[anchor=south, label] {load $P/P_\mathrm{max}$};

% x ticks at u = 0,0.5,1,1.5,2 -> plotted 0,2,4,6,8
\foreach \xp/\xv in {0/0, 2/0.5, 4/1, 6/1.5, 8/2} {
  \draw (\xp,-0.08) -- (\xp,0.08);
  \node[font=\scriptsize, below] at (\xp,-0.12) {$\xv$};
}
% y ticks at P/Pmax = -1,0,1 -> plotted -2.5,0,2.5
\foreach \yp/\yv in {-2.5/$-1$, 2.5/$1$} {
  \draw (-0.08,\yp) -- (0.08,\yp);
  \node[font=\scriptsize, left] at (-0.12,\yp) {\yv};
}

% normalised load: Pn(u) = (u^3 - 3u^2 + 2u) / (2*sqrt3/9) = (u^3-3u^2+2u)*2.598076
% plotted: X = 4*u, Y = 2.5*Pn. Plot the stable rising branch u in 0..u1.
\def\C{2.598076}   % = 9/(2 sqrt3)
% u1 = 1 - 1/sqrt3 = 0.42265 ; u2 = 1 + 1/sqrt3 = 1.57735
% Rising stable branch u: 0 .. 0.42265
\draw[stable] plot[domain=0:0.42265, samples=40]
  ({4*\x}, {2.5*\C*(\x*\x*\x - 3*\x*\x + 2*\x)});
% Unstable falling branch u: 0.42265 .. 1.57735
\draw[unstable] plot[domain=0.42265:1.57735, samples=60]
  ({4*\x}, {2.5*\C*(\x*\x*\x - 3*\x*\x + 2*\x)});
% Re-stiffening stable branch u: 1.57735 .. 2.0
\draw[stable] plot[domain=1.57735:2.0, samples=30]
  ({4*\x}, {2.5*\C*(\x*\x*\x - 3*\x*\x + 2*\x)});

% limit point (snap load) at u1 = 0.42265, Pn = +1 -> (1.6906, 2.5)
\fill[engred] ({4*0.42265},2.5) circle (2.2pt);
\node[engred, font=\scriptsize, anchor=south] at ({4*0.42265},2.6)
  {limit load $P_\mathrm{max}$};

% lower limit point at u2 = 1.57735, Pn = -1
\fill[engred] ({4*1.57735},-2.5) circle (2.2pt);

% snap jump: under load control at P = Pmax the apex leaps from u1 to the
% far stable branch at the same load. Solve u^3-3u^2+2u = 1/C-units...
% at Pn=+1 the far intersection is u ~ 2.2588 (just past u=2). For a clean
% in-frame arrow we draw the jump to the re-stiffened branch near u=2.
\draw[jump] ({4*0.42265+0.12},2.5) -- ({8.0},2.5);
\node[engamber, font=\scriptsize, anchor=south east] at (7.9,2.55) {snap-through};

% horizontal guide at P = Pmax
\draw[guide] (0,2.5) -- ({4*0.42265},2.5);

\end{tikzpicture}
\caption[Load-displacement path of a shallow snap-through truss]{The
equilibrium load-displacement path of a shallow von Mises truss, normalised
by the limit load $P_\mathrm{max}$ and the initial rise $h_0$. The apex
stiffens along the solid blue branch until the load reaches $P_\mathrm{max}$
at the upper limit point. Beyond it the equilibrium path turns down: the
dashed red branch is a continuum of unstable equilibria, equilibria that the
energy landscape marks as saddles rather than wells. A structure under load
control cannot follow the descending branch; at the limit load it jumps
horizontally (amber) to the far stable branch, the apex snapping through to
the inverted shape. The limit load and the snap displacement both fall out
of the stationarity of the total potential energy, with no time history and
no dynamic analysis.}
\label{fig:vol03ch05:snap-load-path}
\end{figure}
