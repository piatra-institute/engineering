% Figure: the idealised crush force-deflection characteristic of a vehicle
% front structure. The area under the curve is the kinetic energy absorbed;
% the dashed line is the constant mean crush force that absorbs the same
% energy over the same stroke, the design target for a near-constant
% deceleration.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  curve/.style={draw=engblue, very thick},
  meanline/.style={draw=engred, dashed, very thick},
  guide/.style={draw=enggray, dotted, thin},
  label/.style={font=\small},
]
% x = crush stroke s, plotted 0..10 (effective stroke at s = 9)
% y = crush force F, plotted 0..6
\draw[axis] (0,0) -- (10.8,0) node[anchor=west, label] {crush stroke $s$};
\draw[axis] (0,0) -- (0,6.4) node[anchor=south, label] {crush force $F$};

% effective stroke marker at s = 9 (the usable fraction of the crush space)
\draw[guide] (9,0) -- (9,5.6);
\node[font=\scriptsize, below] at (9,-0.1) {$s_{\mathrm{eff}}$};

% real crush response: an initial peak (the first buckle of the rails),
% a trough after the first fold, then a folding plateau whose ripple is the
% formation of each successive fold. All sqrt-free; arguments bounded.
\draw[curve] plot[domain=0:9, samples=240]
  ({\x},
   {2.6
    + 2.2*exp(-((\x-0.5)*(\x-0.5))/0.10)
    - 0.9*exp(-((\x-1.6)*(\x-1.6))/0.6)
    + 0.45*sin(deg(2.4*\x))
   });

% mean crush force line (constant): F_mean encloses the same area as the
% real curve over the effective stroke.
\draw[meanline] (0,2.9) -- (9,2.9);
\node[engred, label, anchor=south west] at (3.4,2.95) {$F_{\mathrm{mean}}$ (design target)};

% annotation of the equal-area idea
\node[engblue, font=\scriptsize, anchor=south west] at (4.9,3.9)
  {area $=\displaystyle\int F\,ds = E_{\mathrm{abs}}$};

% peak-force callout
\node[engblue, font=\scriptsize, anchor=west] at (0.95,5.1) {first-buckle peak};
\draw[guide] (0.55,4.85) -- (0.95,5.05);

\end{tikzpicture}
\caption[Idealised vehicle crush force-deflection characteristic]{The
force the front structure of a vehicle exerts on the decelerating mass as
it crushes, plotted against the crush stroke $s$. The real response
(blue) spikes as the first section of rail buckles, drops into a trough,
then settles into a folding plateau whose ripple is the formation of each
successive fold. The area under the curve is the energy the structure
absorbs, $E_{\mathrm{abs}} = \int F\,ds$. A constant mean crush force
$F_{\mathrm{mean}}$ (red) that encloses the same area over the same
effective stroke $s_{\mathrm{eff}}$ delivers the same energy absorption at
a constant deceleration, which is the design target: a flat force plateau
spreads the deceleration evenly over the available stroke rather than
spiking it. The early peak is suppressed in practice by crush initiators
(trigger dents or holes stamped into the rails) so that the structure
folds at a controlled load from the first instant of contact.}
\label{fig:vol03ch05:crumple-force-deflection}
\end{figure}
