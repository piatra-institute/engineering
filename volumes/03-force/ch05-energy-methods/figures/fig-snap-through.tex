% Figure: the potential-energy landscape of a shallow bistable (snap-through)
% mechanism. As the applied load rises, the single well deepens on one side,
% the barrier shrinks, and at the critical load the local minimum and the
% saddle merge: the system snaps to the far well. Three curves at three load
% levels show the well structure collapsing.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  low/.style={draw=enggray, very thick},
  mid/.style={draw=engblue, very thick},
  crit/.style={draw=engred, very thick},
  label/.style={font=\small},
]
% x = generalised displacement q from -2.4 to 2.4 (plotted directly)
% y = total potential energy U (shifted per curve for legibility)
\draw[axis] (-2.7,0) -- (2.9,0) node[anchor=west, label] {displacement $q$};
\draw[axis] (0,-0.3) -- (0,4.4) node[anchor=south, label] {potential energy $U$};

% Base double well: U0(q) = 0.25 q^4 - q^2, minima at q = +-sqrt(2) ~ +-1.414.
% Add a load-tilt term -F q. Plot three F levels, each shifted up for clarity.

% F = 0 (symmetric double well), shifted up by 3.0
\draw[low] plot[domain=-2.3:2.3, samples=120]
  ({\x}, {0.25*\x*\x*\x*\x - \x*\x + 3.0});
\node[enggray, font=\scriptsize, anchor=west] at (2.32,2.05) {$F=0$};

% F = 0.55 (tilted, two unequal wells, low barrier), shifted up by 1.5
\draw[mid] plot[domain=-2.3:2.3, samples=120]
  ({\x}, {0.25*\x*\x*\x*\x - \x*\x - 0.55*\x + 1.5});
\node[engblue, font=\scriptsize, anchor=west] at (2.32,1.0) {$F<F_\mathrm{cr}$};

% F = 1.089 (critical: shallow well and saddle merge near q ~ -0.82), no shift
% At F_cr the left local minimum becomes an inflection (snap-through).
\draw[crit] plot[domain=-2.3:2.3, samples=120]
  ({\x}, {0.25*\x*\x*\x*\x - \x*\x - 1.089*\x});
\node[engred, font=\scriptsize, anchor=west] at (2.32,-0.05) {$F=F_\mathrm{cr}$};

% mark the disappearing shallow well on the critical curve
\fill[engred] (-0.82,{0.25*0.82*0.82*0.82*0.82 - 0.82*0.82 - 1.089*(-0.82)}) circle (2pt);
\node[engred, font=\scriptsize, anchor=south east] at (-0.85,0.55) {inflection};

\end{tikzpicture}
\caption[Snap-through potential landscape of a bistable mechanism]{The total
potential energy of a shallow bistable mechanism as the applied load $F$
rises. With no load (grey) the energy has two symmetric wells separated by a
barrier, the two stable equilibria of the mechanism. A modest load (blue)
tilts the landscape: one well deepens, the other shallows, and the barrier
between them shrinks. At the critical load (red) the shallow well and the
barrier saddle merge into a single inflection, the local minimum vanishes,
and the mechanism snaps through to the remaining well. The snap is read
directly off the energy curve as the load at which a minimum and a maximum
annihilate, a fold in the equilibrium set rather than a material failure.}
\label{fig:vol03ch05:snap-through}
\end{figure}
