% Figure: path-independence of the work done by a conservative force. Two
% distinct paths from A to B carry the same work, so the line integral of a
% conservative force depends only on the endpoints. A potential difference is
% drawn as the height drop between two level curves.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  level/.style={draw=enggray, thin},
  patha/.style={draw=engblue, very thick, -{Stealth}},
  pathb/.style={draw=engred, very thick, -{Stealth}},
  pt/.style={font=\small},
]
% two level curves of the potential U (drawn as gentle arcs)
\draw[level] plot[domain=-0.4:6.4, samples=60] ({\x}, {0.6 + 0.10*sin(40*\x)});
\node[enggray, font=\scriptsize, anchor=west] at (6.5,0.6) {$U=U_A$};
\draw[level] plot[domain=-0.4:6.4, samples=60] ({\x}, {3.4 + 0.10*sin(40*\x)});
\node[enggray, font=\scriptsize, anchor=west] at (6.5,3.4) {$U=U_B$};

% endpoints A (low) and B (high)
\fill[engblack] (0.6,0.6) circle (2.2pt);
\node[pt, anchor=north east] at (0.6,0.55) {$A$};
\fill[engblack] (5.4,3.4) circle (2.2pt);
\node[pt, anchor=south west] at (5.4,3.45) {$B$};

% path 1 (blue): a direct-ish curve
\draw[patha] (0.6,0.6) .. controls (1.6,2.6) and (3.4,1.4) .. (5.4,3.4);
\node[engblue, pt, anchor=south east] at (2.2,2.3) {path 1};

% path 2 (red): a longer, wandering curve
\draw[pathb] (0.6,0.6) .. controls (2.8,0.2) and (5.6,1.2) .. (5.4,3.4);
\node[engred, pt, anchor=north west] at (4.1,1.0) {path 2};

% annotation: equal work
\node[engblack, font=\scriptsize, anchor=north] at (3.0,-0.2)
  {$\displaystyle W_{A\to B}=\int_A^B \vec F\cdot d\vec r = U_A - U_B$ on both paths};

\end{tikzpicture}
\caption[Path-independence of a conservative force]{The work a conservative
force does between two points depends only on the endpoints, not on the route.
Two paths from $A$ to $B$ (blue and red) carry the same work, equal to the
drop in potential energy $U_A-U_B$ between the level curves $U=U_A$ and
$U=U_B$. A conservative force is the negative gradient of a single-valued
potential, so its line integral around any closed loop vanishes and its work
between two points is a difference of potential values. This is the property
that lets the energy method read a speed or a deflection off the endpoints
alone, with the path's geometry dropping out of the calculation.}
\label{fig:vol03ch05:path-independence}
\end{figure}
