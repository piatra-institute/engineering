% Figure: the effective potential of a spherical pendulum. Eliminating the
% cyclic azimuthal coordinate leaves a one-dimensional problem in the polar
% angle theta governed by an effective potential that adds a centrifugal
% barrier to the gravitational well. The barrier keeps the bob away from the
% vertical axis and fixes a stable steady-precession angle at the minimum.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  grav/.style={draw=enggray, dashed, thick},
  cent/.style={draw=engamber, dash pattern=on 2pt off 2pt, thick},
  eff/.style={draw=engblue, very thick},
  guide/.style={draw=engred, dashed, thin},
  label/.style={font=\small},
]
% x in [1.4,8.6] maps to theta = 0.35*x in [0.49, 3.01] rad, so sin(theta)
% stays well away from 0 and the 1/sin^2 barrier stays bounded on the plot.
\draw[axis] (0,-2.2) -- (0,3.4) node[anchor=south, label] {energy};
\draw[axis] (0,0) -- (9.4,0) node[anchor=west, label] {polar angle $\theta$};
% ticks on theta axis
\foreach \xp/\xv in {1.5/{$\pi/6$}, 4.5/{$\pi/2$}, 7.5/{$5\pi/6$}} {
  \draw (\xp,-0.08) -- (\xp,0.08);
  \node[font=\scriptsize, below] at (\xp,-0.12) {\xv};
}
% gravitational potential -cos(theta):  Ug = -1.6*cos(theta), theta = 0.35*x
\draw[grav] plot[domain=1.4:8.6, samples=120]
  ({\x}, {-1.6*cos( deg(0.35*\x) )});
\node[enggray, font=\scriptsize, anchor=west] at (7.7,-1.55)
  {$U_g=-mgL\cos\theta$};
% centrifugal barrier  +A / sin^2(theta), A small; theta>=0.49 keeps it bounded
% barrier = 0.30 / (sin(theta))^2 with theta = 0.35*x
\draw[cent] plot[domain=1.4:8.6, samples=160]
  ({\x}, {0.30/ ( (sin( deg(0.35*\x) ))*(sin( deg(0.35*\x) )) ) });
\node[engamber, font=\scriptsize, anchor=west] at (0.35,2.9)
  {$\dfrac{p_\phi^2}{2mL^2\sin^2\theta}$};
% effective potential = sum of the two
\draw[eff] plot[domain=1.4:8.6, samples=200]
  ({\x}, { -1.6*cos( deg(0.35*\x) )
           + 0.30/ ( (sin( deg(0.35*\x) ))*(sin( deg(0.35*\x) )) ) });
\node[engblue, label, anchor=south west] at (5.0,1.6)
  {$U_\mathrm{eff}(\theta)$};
% mark the minimum (steady precession angle). Numerically near theta ~ 0.9 rad
% => x ~ 2.6, and U_eff there ~ -0.6.  Drop a guide.
\draw[guide] (2.65,-2.2) -- (2.65,-0.62);
\fill[engblue] (2.65,-0.62) circle (2pt);
\node[engred, font=\scriptsize, anchor=north] at (2.65,-2.25)
  {$\theta_0$ (steady precession)};
\end{tikzpicture}
\caption[Effective potential of a spherical pendulum]{Removing the cyclic
azimuthal coordinate $\phi$ of a spherical pendulum leaves a one-dimensional
motion in the polar angle $\theta$ governed by the effective potential
$U_\mathrm{eff}(\theta) = -mgL\cos\theta + p_\phi^2/(2mL^2\sin^2\theta)$. The
gravitational well pulls the bob down toward $\theta=0$; the conserved angular
momentum $p_\phi$ supplies a centrifugal barrier that diverges at the axis and
forbids the bob from reaching it. The two terms balance at a minimum $\theta_0$,
the angle of steady conical precession, and small departures from it oscillate
in the well. The reduction turns a two-coordinate problem into the familiar
one-coordinate well of the chapter opener, with the conserved momentum built
into the potential.}
\label{fig:vol03ch05:effective-potential}
\end{figure}
