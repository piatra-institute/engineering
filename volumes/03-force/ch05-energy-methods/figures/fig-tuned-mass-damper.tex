% Figure: a tuned mass damper. Left: the physical arrangement, a small
% absorber mass on its own spring and dashpot mounted on a large primary
% structure driven by a harmonic force. Right: the steady-state response
% amplitude of the primary mass against forcing frequency, showing the
% undamped resonance peak split by the absorber into two smaller peaks with a
% deep anti-resonance notch at the tuning frequency.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  wall/.style={draw=engblack, thick},
  spring/.style={draw=enggray, decorate,
    decoration={coil, aspect=0.5, segment length=3pt, amplitude=3pt}},
  massbig/.style={draw=engblack, fill=engblue!18, thick,
    minimum width=20mm, minimum height=14mm},
  masssm/.style={draw=engblack, fill=engred!18, thick,
    minimum width=12mm, minimum height=8mm},
  axis/.style={draw=engblack, -{Stealth}, thick},
  bare/.style={draw=enggray, dashed, thick},
  tmd/.style={draw=engblue, very thick},
  label/.style={font=\small},
]
% ===== physical system (left) =====
\draw[wall] (-0.4,-1.2) -- (1.8,-1.2);
\foreach \x in {-0.3,0.0,...,1.6} {\draw[enggray, thin] (\x,-1.2) -- (\x-0.18,-1.45);}
\draw[spring] (0.2,-1.2) -- (0.2,0.0);
\draw[spring] (1.2,-1.2) -- (1.2,0.0);
\node[massbig] (M) at (0.7,0.65) {$M$};
\node[font=\scriptsize, enggray, anchor=east] at (0.05,-0.6) {$K$};
\node[font=\scriptsize, engred, anchor=west] at (2.3,1.9) {$m,\,k,\,c$};
\draw[spring] (0.35,1.35) -- (0.35,2.15);
\node[masssm] (m) at (0.7,2.55) {$m$};
\draw[-{Stealth}, engorange, very thick] (2.0,0.65) -- (1.35,0.65);
\node[engorange, font=\scriptsize, anchor=west] at (2.0,0.65)
  {$F_0\cos\Omega t$};

% ===== response curves (right) =====
\begin{scope}[xshift=5.6cm]
\draw[axis] (0,0) -- (6.2,0) node[anchor=west, label] {forcing frequency $\Omega$};
\draw[axis] (0,0) -- (0,4.4) node[anchor=south, label] {primary amplitude};
% bare structure: single tall resonance near Omega/omega0 = 1 (x=3)
% draw a Lorentzian-like bump peaking at x=3, height 4.1
\draw[bare] plot[domain=0.4:5.8, samples=160]
  ({\x}, { 0.35 + 3.7/ (1 + 9*(\x-3)*(\x-3)) });
\node[enggray, font=\scriptsize, anchor=west] at (3.15,3.9)
  {without absorber};
% with TMD: two smaller peaks near x=2.4 and x=3.6, deep notch at x=3
\draw[tmd] plot[domain=0.4:5.8, samples=220]
  ({\x}, { 0.30
           + 1.6/ (1 + 26*(\x-2.45)*(\x-2.45))
           + 1.5/ (1 + 26*(\x-3.55)*(\x-3.55))
           + 0.15/ (1 + 60*(\x-3.0)*(\x-3.0)) });
\node[engblue, font=\scriptsize, anchor=west] at (3.7,1.9)
  {with tuned absorber};
% notch marker at tuning frequency
\draw[engred, dashed, thin] (3,0) -- (3,0.42);
\node[engred, font=\scriptsize, anchor=north] at (3,-0.05)
  {tuning $\Omega=\sqrt{k/m}$};
\end{scope}
\end{tikzpicture}
\caption[Tuned mass damper and its split response]{A tuned mass damper is a
small secondary mass $m$ on a spring $k$ and dashpot $c$ mounted on a primary
structure of mass $M$ and stiffness $K$ driven near its resonance. Tuned so
its natural frequency $\sqrt{k/m}$ matches the primary's, the absorber takes
the driving energy onto itself: at the tuning frequency the primary mass sits
almost still while the absorber swings and its dashpot dissipates the input.
The single tall resonance of the bare structure splits into two lower peaks
separated by a deep anti-resonance notch. The energy method sets the tuning
from the equal-energy condition and sizes the absorber mass from the width of
the split, the design that steadies footbridges against pedestrian pacing and
tall buildings against wind.}
\label{fig:vol03ch05:tuned-mass-damper}
\end{figure}
