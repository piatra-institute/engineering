% Figure: the effective radial potential of the Kepler problem. The
% attractive term -a/r and the centrifugal barrier b/r^2 sum to a well
% with a minimum; the energy line sets the turning points of the orbit.
% Display coefficients: b = 4.5, a = 6, so U_eff = 4.5/r^2 - 6/r,
% minimum at r0 = 2b/a = 1.5 with U(r0) = -2. Energy line E = -0.9 cuts
% the well at r = 0.86 (perihelion) and r = 5.8 (aphelion).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  curve/.style={very thick},
  guide/.style={draw=enggray, dashed, thin},
  label/.style={font=\small},
]
% axes: r from 0 to 7, potential from -3 to 3
\draw[axis] (0,0) -- (7.2,0) node[anchor=west, label] {$r$};
\draw[axis] (0,-3.2) -- (0,3.4) node[anchor=south, label] {$U_{\mathrm{eff}}$};

% centrifugal barrier 4.5/r^2 (orange), clamped to view
\draw[curve, draw=engorange] plot[domain=1.2:7, samples=160]
  ({\x}, {min(3.2, 4.5/(\x*\x))});
\node[engorange, label, anchor=south west] at (1.35, 2.5)
  {$+\dfrac{p_\phi^2}{2mr^2}$};

% attractive term -6/r (blue), clamped to view
\draw[curve, draw=engblue] plot[domain=1.95:7, samples=160]
  ({\x}, {max(-3.1, -6.0/\x)});
\node[engblue, label, anchor=north east] at (6.9, -0.75)
  {$-\dfrac{k}{r}$};

% sum: effective potential 4.5/r^2 - 6/r (red)
\draw[curve, draw=engred] plot[domain=0.78:7, samples=220]
  ({\x}, {max(-3.1, min(3.2, 4.5/(\x*\x) - 6.0/\x))});
\node[engred, label, anchor=south east] at (5.4, 0.35)
  {$U_{\mathrm{eff}}(r)$};

% minimum at r0 = 1.5, U = -2 (circular orbit)
\fill[englime] (1.5, -2.0) circle (2.4pt);
\node[englime, font=\scriptsize, anchor=west] at (1.7, -2.15) {circular orbit};

% bound-orbit energy line E = -0.9
\draw[guide] (0,-0.9) -- (7,-0.9);
\node[label, anchor=west] at (6.3,-0.72) {$E<0$};
% turning points r = 0.86 and r = 5.8
\draw[guide] (0.86,-0.9) -- (0.86,0);
\node[font=\scriptsize, anchor=south, engblack] at (0.86,0.05) {$r_{\min}$};
\draw[guide] (5.8,-0.9) -- (5.8,0);
\node[font=\scriptsize, anchor=south, engblack] at (5.8,0.05) {$r_{\max}$};

\end{tikzpicture}
\caption[Effective radial potential of the Kepler problem]{The effective
radial potential of a particle in an inverse-square attraction: the true
attractive term $-k/r$ (blue) plus the centrifugal barrier
$p_\phi^2/(2mr^2)$ (orange) that the conserved angular momentum
contributes sum to a well $U_{\mathrm{eff}}(r)$ (red) with a minimum
(green) where a circular orbit sits. A bound orbit of energy $E<0$
oscillates in radius between the two turning points $r_{\min}$ and
$r_{\max}$ where the energy line cuts the well, the perihelion and
aphelion of the ellipse; raising $E$ to zero opens the outer turning
point to infinity and the orbit becomes a parabolic escape. The
two-coordinate central-force problem has collapsed to motion in a
one-dimensional well, the cyclic azimuth folded into the barrier.}
\label{fig:vol03ch05:kepler-effective-potential}
\end{figure}
