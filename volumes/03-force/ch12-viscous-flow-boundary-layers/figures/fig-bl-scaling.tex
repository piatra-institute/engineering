% Figure: the order-of-magnitude scaling that reduces Navier=Stokes to
% the boundary-layer equations. A thin layer of thickness delta(x) growing
% over a flat plate of streamwise extent x, with the two disparate length
% scales (x along the plate, delta across it) and the associated velocity
% scales (U_inf streamwise, a much smaller v across) annotated. The point
% is the aspect ratio delta/x << 1 that licenses dropping the streamwise
% diffusion term.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=stealth, scale=1.0]
  % plate
  \draw[engblack, very thick] (0,0) -- (10,0);
  \foreach \x in {0.3,0.8,...,9.8}{
    \draw[enggray, thin] (\x,0) -- ++(-0.18,-0.18);
  }
  % boundary-layer edge: delta(x) ~ sqrt(x), exaggerated
  \draw[engblue, very thick]
    plot[domain=0:10, samples=120, smooth] ({\x}, {0.95*sqrt(\x)});
  \node[engblue, right, font=\small] at (9.0,3.0) {$\delta(x)$};
  % freestream arrows above the layer
  \foreach \y in {3.4,3.9,4.4}{
    \draw[engred, thick, ->] (0.2,\y) -- ++(1.6,0)
      node[right, font=\scriptsize, engred] {};
  }
  \node[engred, right, font=\small] at (2.0,4.4) {$U_\infty$ (freestream)};
  % streamwise length scale x
  \draw[engblack, <->] (0,-0.7) -- (7,-0.7)
    node[midway, below, font=\small] {streamwise scale $x$};
  % wall-normal length scale delta at station x=7 (0.95*sqrt(7)=2.513)
  \draw[englime, <->] (7.15,0) -- (7.15,2.513)
    node[midway, right, font=\small] {$\delta\ll x$};
  % velocity scales inside the layer
  \draw[engred, thick, ->] (3.0,0.3) -- ++(0.9,0)
    node[right, font=\scriptsize] {$u\sim U_\infty$};
  \draw[englime, thick, ->] (3.0,0.3) -- ++(0,0.5)
    node[above, font=\scriptsize] {$v\sim U_\infty\,\delta/x$};
\end{tikzpicture}
\caption[Boundary-layer length and velocity scales]{The two disparate
scales that reduce the Navier-Stokes equations to the boundary-layer
equations. Along the plate the flow varies over the streamwise length
$x$; across the layer it varies over the much smaller thickness
$\delta(x)$, and the aspect ratio $\delta/x$ is small at high Reynolds
number. Continuity then fixes the wall-normal velocity scale at
$v\sim U_\infty\,\delta/x$, far below the streamwise $U_\infty$. Carrying
these scales through each term shows the streamwise diffusion
$\nu\,\partial^2u/\partial x^2$ is smaller than the wall-normal diffusion
$\nu\,\partial^2u/\partial y^2$ by the factor $(\delta/x)^2$, which is why
the boundary-layer equation keeps only the wall-normal viscous term.}
\label{fig:vol03ch12:bl-scaling}
\end{figure}
