% Figure: the standard drag curve of a sphere across the three settling
% regimes that a terminal-velocity calculation must distinguish. The Stokes
% line CD = 24/Re, the Schiller-Naumann intermediate correlation, and the
% Newton constant-CD plateau (~0.44) are drawn on log-log axes; the regime
% boundaries near Re = 1 and Re = 1000 are marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=12cm, height=7.5cm,
  xlabel={Reynolds number $\mathrm{Re} = U d/\nu$},
  ylabel={Drag coefficient $C_D$},
  xmin=1e-2, xmax=1e5, ymin=0.1, ymax=1e3,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Stokes line CD = 24/Re.
\addplot[engblue, very thick, domain=0.01:2, samples=40] {24/x};
\addlegendentry{Stokes $C_D=24/\mathrm{Re}$}
% Schiller-Naumann CD = (24/Re)(1 + 0.15 Re^0.687), valid Re < ~1000.
\addplot[engred, very thick, domain=0.01:1000, samples=120]
  {(24/x)*(1 + 0.15*x^0.687)};
\addlegendentry{Schiller--Naumann}
% Newton plateau CD ~ 0.44 for ~1e3 < Re < ~2e5.
\addplot[englime!60!engblack, very thick] coordinates {
  (1e3,0.44) (1e4,0.44) (1e5,0.44)
};
\addlegendentry{Newton plateau $C_D\approx0.44$}
% Regime boundary markers.
\addplot[enggray, dashed, thick] coordinates {(1,0.1) (1,1e3)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:0.7,30) {Stokes limit};
\addplot[enggray, dashed, thick] coordinates {(1000,0.1) (1000,1e3)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:700,30) {Newton onset};
\end{loglogaxis}
\end{tikzpicture}
\caption[Sphere drag across settling regimes]{The drag coefficient of a
settling sphere across the regimes a terminal-velocity calculation must keep
straight. Below $\mathrm{Re}\approx1$ the Stokes law $C_D=24/\mathrm{Re}$
holds; through the intermediate range the Schiller-Naumann correlation tracks
the data; above $\mathrm{Re}\approx10^3$ the coefficient settles near the
Newton plateau $C_D\approx0.44$ until the drag crisis. The regime check is
part of the calculation: a Stokes answer used past $\mathrm{Re}\approx1$
overstates the settling velocity, sometimes by a large factor.}
\label{fig:vol03ch12:terminal-velocity-regimes}
\end{figure}
