% Figure: the stress components on a fluid element. A cube with the three
% stress components on its visible positive-x face labelled: the normal
% stress sigma_xx along x and the two shear stresses tau_xy and tau_xz
% along y and z. The sketch fixes the index convention used in the
% stress-tensor derivation of Navier=Stokes: the first index is the face
% normal, the second the direction of the traction component.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=stealth]
  % cube vertices (simple oblique projection)
  \def\s{3.2}      % cube edge
  \def\dx{1.3}     % depth offset x
  \def\dy{0.9}     % depth offset y
  % front face corners
  \coordinate (A) at (0,0);
  \coordinate (B) at (\s,0);
  \coordinate (C) at (\s,\s);
  \coordinate (D) at (0,\s);
  % back face corners
  \coordinate (A2) at (\dx,\dy);
  \coordinate (B2) at (\s+\dx,\dy);
  \coordinate (C2) at (\s+\dx,\s+\dy);
  \coordinate (D2) at (\dx,\s+\dy);
  % draw cube
  \draw[engblack, thick] (A)--(B)--(C)--(D)--cycle;
  \draw[engblack, thick] (B)--(B2)--(C2)--(C);
  \draw[engblack, thick] (D)--(D2)--(C2);
  \draw[enggray, thick, dashed] (A)--(A2)--(B2);
  \draw[enggray, thick, dashed] (A2)--(D2);
  % the positive-x face is the right face B B2 C2 C; its centre:
  \coordinate (Fc) at ($(B)!0.5!(C2)$);
  % normal stress sigma_xx (outward +x, to the right and slightly up)
  \draw[engred, very thick, ->] (Fc) -- ++(1.6,0.45)
    node[right, font=\small] {$\sigma_{xx}$};
  % shear tau_xy (along +y, up)
  \draw[engblue, very thick, ->] (Fc) -- ++(0,1.5)
    node[above, font=\small] {$\tau_{xy}$};
  % shear tau_xz (along +z, into depth: up-right along the depth axis)
  \draw[englime, very thick, ->] (Fc) -- ++(0.85,0.6)
    node[right, font=\small] {$\tau_{xz}$};
  % coordinate triad at lower left
  \coordinate (O) at (-1.6,-0.3);
  \draw[engblack, ->] (O) -- ++(1.0,0) node[right, font=\scriptsize] {$x$};
  \draw[engblack, ->] (O) -- ++(0,1.0) node[above, font=\scriptsize] {$y$};
  \draw[engblack, ->] (O) -- ++(0.55,0.40) node[right, font=\scriptsize] {$z$};
\end{tikzpicture}
\caption[Stress components on a fluid element]{The traction on the
positive-$x$ face of a fluid element resolves into one normal stress
$\sigma_{xx}$ and two shear stresses $\tau_{xy}$ and $\tau_{xz}$. The
first index names the face normal and the second the direction of the
component, so the full state of stress at a point is the nine-component
tensor $\sigma_{ij}$. Conservation of angular momentum forces the tensor
to be symmetric, $\sigma_{ij}=\sigma_{ji}$, leaving six independent
components; the Newtonian constitutive law writes those six as a linear,
isotropic function of the strain-rate tensor.}
\label{fig:vol03ch12:stress-cube}
\end{figure}
