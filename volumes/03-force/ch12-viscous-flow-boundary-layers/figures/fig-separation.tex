% Figure: boundary-layer separation under an adverse pressure gradient,
% schematic. The wall is the horizontal axis. Upstream the profile is full
% and attached; at the separation point the wall shear vanishes; downstream
% the near-wall flow reverses. Profiles are drawn as small horizontal
% velocity arrows scaled by their value.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% wall
\draw[engblack, very thick] (0,0) -- (10,0);
\node[font=\scriptsize, anchor=north] at (5,-0.1) {wall (no-slip)};
% freestream arrows above
\foreach \x in {0.5,2,3.5,5,6.5,8,9.5} {
  \draw[enggray, ->] (\x,3.1) -- (\x+0.7,3.1);
}
\node[font=\scriptsize, enggray, anchor=south] at (1,3.2) {$U_\infty$};
% adverse pressure gradient arrow
\draw[engamber, ->, thick] (1,3.7) -- (9,3.7);
\node[font=\scriptsize, engamber, anchor=south] at (5,3.7)
  {adverse gradient $\partial_x p>0$};
% attached profile at x=2 (full, positive)
\draw[engblue, thick] (2,0) .. controls (2.9,0.8) and (2.9,2.0) .. (2,2.6);
\foreach \y/\u in {0.4/0.5,0.9/0.75,1.5/0.85,2.1/0.6} {
  \draw[engblue, ->] (2,\y) -- (2+\u,\y);
}
% near-separation profile at x=5 (inflected, zero wall slope)
\draw[engred, thick] (5,0) .. controls (5.05,1.0) and (5.7,1.8) .. (5,2.6);
\foreach \y/\u in {0.4/0.05,0.9/0.4,1.5/0.7,2.1/0.5} {
  \draw[engred, ->] (5,\y) -- (5+\u,\y);
}
\node[font=\scriptsize, engred, anchor=south] at (5,2.7) {$\tau_w=0$};
% separated profile at x=8 (reversed near wall)
\draw[englime, thick] (8,0) .. controls (7.4,0.8) and (8.6,1.8) .. (8,2.6);
\draw[englime, ->] (8,0.4) -- (7.6,0.4);
\draw[englime, ->] (8,0.9) -- (7.75,0.9);
\foreach \y/\u in {1.6/0.4,2.2/0.55} {
  \draw[englime, ->] (8,\y) -- (8+\u,\y);
}
\node[font=\scriptsize, englime, anchor=south] at (8.3,2.7) {reversed};
% separation point marker
\fill[engred] (5,0) circle (1.6pt);
\node[font=\scriptsize, engred, anchor=north] at (5,-0.45)
  {separation point};
\end{tikzpicture}
\caption[Boundary-layer separation]{Boundary-layer separation under an
adverse pressure gradient. Upstream the velocity profile is full and
attached. As the pressure rises in the flow direction the near-wall fluid
decelerates; at the separation point the wall shear stress vanishes
($\tau_w=0$); downstream the near-wall flow reverses and the boundary
layer detaches, leaving a low-pressure separated wake. This is the
mechanism behind airfoil stall, diffuser stall, and bluff-body pressure
drag.}
\label{fig:vol03ch12:separation}
\end{figure}
