% Figure: the drag crisis. Drag coefficient of a smooth sphere and a smooth
% circular cylinder versus Reynolds number, zoomed on the transition near
% Re = 3e5 where CD drops abruptly as the boundary layer turns turbulent ahead
% of separation and the wake narrows. Values are transcribed standard points.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogxaxis}[
  width=12cm, height=7cm,
  xlabel={Reynolds number $\mathrm{Re} = U D/\nu$},
  ylabel={Drag coefficient $C_D$},
  xmin=1e3, xmax=1e7, ymin=0, ymax=1.4,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Cylinder CD: subcritical plateau ~1.2, crisis near 2e5-5e5, supercritical ~0.4.
\addplot[engblue, very thick] coordinates {
  (1e3,1.00) (3e3,1.05) (1e4,1.15) (3e4,1.20) (1e5,1.20)
  (2e5,1.15) (3e5,0.70) (4e5,0.35) (6e5,0.30) (1e6,0.35)
  (3e6,0.55) (1e7,0.70)
};
\addlegendentry{circular cylinder}
% Sphere CD: subcritical ~0.45-0.5, crisis near 3e5, supercritical ~0.1-0.2.
\addplot[engred, very thick] coordinates {
  (1e3,0.46) (3e3,0.40) (1e4,0.41) (3e4,0.45) (1e5,0.48)
  (2e5,0.49) (3e5,0.30) (4e5,0.12) (6e5,0.10) (1e6,0.13)
  (3e6,0.18) (1e7,0.20)
};
\addlegendentry{smooth sphere}
% Marker for the canonical crisis Reynolds number.
\addplot[enggray, dashed, thick] coordinates {(3e5,0) (3e5,1.4)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:2.4e5,0.85) {drag crisis};
\end{semilogxaxis}
\end{tikzpicture}
\caption[The drag crisis]{The drag coefficient of a sphere and a circular
cylinder drops abruptly near $\mathrm{Re}=3\times10^5$. The boundary layer
turns turbulent just upstream of separation, the turbulent layer resists the
adverse pressure gradient longer, the separation point moves rearward, the
wake narrows, and the pressure drag falls. A body whose operating Reynolds
number sits on the steep part of either curve can change its drag by a factor
of two from a small change in roughness, freestream turbulence, or speed.}
\label{fig:vol03ch12:drag-crisis}
\end{figure}
