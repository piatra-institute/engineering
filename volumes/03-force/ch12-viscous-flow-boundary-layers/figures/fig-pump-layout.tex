% Figure: schematic of the pumping system in the case study. A lower supply
% reservoir, a suction pipe up to the pump, the pump, then a discharge pipe
% rising to an upper delivery tank. The static lift Hs and the suction lift zs
% are marked; the suction lift is what sets the NPSH available.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, line width=0.8pt]
% Lower supply reservoir.
\draw[engblue, fill=engblue!12]
  (0,0) rectangle (3,1.2);
\node[font=\scriptsize, engblue] at (1.5,0.6) {supply};
% Datum line.
\draw[enggray, dashed] (-0.3,0) -- (11,0);
\node[font=\scriptsize, enggray, anchor=west] at (11,0) {datum};
% Suction pipe up to pump.
\draw[engblack] (3,0.6) -- (4.5,0.6) -- (4.5,2.2);
% Pump symbol (circle).
\draw[engred, fill=engred!12] (4.5,2.5) circle (0.45);
\node[font=\scriptsize, engred] at (4.5,2.5) {pump};
% Discharge pipe rising to upper tank.
\draw[engblack] (4.5,2.95) -- (4.5,5.5) -- (8,5.5);
% Upper delivery tank.
\draw[englime!60!engblack, fill=englime!18]
  (8,5.0) rectangle (11,6.2);
\node[font=\scriptsize, engblack] at (9.5,5.6) {delivery};
% Static lift bracket (datum to delivery free surface).
\draw[engamber, <->] (10.6,0) -- (10.6,6.0);
\node[font=\scriptsize, engamber, anchor=west, rotate=90] at (10.85,2.6)
  {static lift $H_s$};
% Suction lift bracket (supply surface to pump centreline).
\draw[engorange, <->] (6.0,1.2) -- (6.0,2.5);
\node[font=\scriptsize, engorange, anchor=west] at (6.1,1.9)
  {suction lift $z_s$};
% Supply free-surface tick.
\draw[engblue, dashed] (3,1.2) -- (6.0,1.2);
% Flow arrows.
\draw[engblack, ->] (3.7,0.6) -- (4.1,0.6);
\draw[engblack, ->] (4.5,4.2) -- (4.5,4.6);
\end{tikzpicture}
\caption[Pumping-system layout]{The case-study layout. Fluid is drawn from a
lower supply, lifted through the pump, and delivered to an upper tank. The
static lift $H_s$ is the elevation gain from datum to the delivery surface; the
suction lift $z_s$ from the supply surface to the pump centreline sets how much
the suction-side pressure falls below atmospheric, which fixes the net positive
suction head available and the margin against cavitation.}
\label{fig:vol03ch12:pump-layout}
\end{figure}
