% Figure: drag coefficient of a smooth sphere versus Reynolds number, the
% standard log-log curve. Stokes branch C_D = 24/Re at low Re, a broad
% plateau near C_D approx 0.4 to 0.5, and the drag crisis: a sharp drop
% near Re approx 3e5 as the boundary layer trips to turbulence and the
% wake narrows. Tabulated standard-curve values.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=13cm, height=8cm,
  xlabel={Reynolds number $\mathrm{Re}=\rho U d/\mu$},
  ylabel={drag coefficient $C_D$},
  xmin=0.05, xmax=1e6, ymin=0.06, ymax=500,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Stokes branch C_D = 24/Re, Re 0.05..2.
\addplot[engblue, very thick, domain=0.05:2, samples=40] {24/x};
\addlegendentry{Stokes $C_D=24/\mathrm{Re}$}
% Standard sphere drag curve, tabulated values (Schlichting / White).
\addplot[engred, very thick] coordinates {
  (1,26.5) (2,14.4) (5,6.9) (10,4.1) (20,2.55) (50,1.50)
  (100,1.07) (200,0.77) (500,0.55) (1000,0.46) (5000,0.39)
  (1e4,0.40) (5e4,0.49) (1e5,0.50) (2e5,0.49) (3e5,0.20)
  (5e5,0.10) (1e6,0.12)
};
\addlegendentry{standard sphere curve}
% drag-crisis marker
\addplot[engamber, thick, dashed] coordinates {(3e5,0.06) (3e5,500)};
\node[font=\scriptsize, engamber, anchor=west] at (axis cs:3.3e5,3)
  {drag crisis};
\end{loglogaxis}
\end{tikzpicture}
\caption[Sphere drag coefficient versus Reynolds number]{The drag
coefficient of a smooth sphere across nine decades of Reynolds number. At
$\mathrm{Re}\ll1$ the Stokes law $C_D=24/\mathrm{Re}$ holds; the curve
flattens to a plateau near $C_D\approx0.4$ to $0.5$ through the
intermediate range; near $\mathrm{Re}\approx3\times10^{5}$ the drag crisis
drops $C_D$ abruptly as the boundary layer trips to turbulence and the
separated wake narrows.}
\label{fig:vol03ch12:sphere-drag}
\end{figure}
