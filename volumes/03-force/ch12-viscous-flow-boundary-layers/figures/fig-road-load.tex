% Figure: road-load power of a passenger car versus speed, split into the
% rolling-resistance part (linear in speed, P = Crr m g U) and the aerodynamic
% part (cubic in speed, P = 0.5 rho Cd A U^3), with their sum. The crossover
% speed where aerodynamic power overtakes rolling power is marked. Numbers
% match the vehicle case study: m = 1500 kg, Crr = 0.012, Cd*A = 0.66 m^2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={Road speed $U$ (\si{\meter\per\second})},
  ylabel={Power at the wheels $P$ (\si{\kilo\watt})},
  xmin=0, xmax=45, ymin=0, ymax=45,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Rolling power P_roll = Crr m g U = 0.012*1500*9.81*U = 176.6 U  (watts) -> kW.
\addplot[engamber, very thick, domain=0:45, samples=60] {0.1766*x};
\addlegendentry{rolling $C_{rr}mgU$}
% Aerodynamic power P_aero = 0.5 rho Cd A U^3 = 0.5*1.2*0.66*U^3 = 0.396 U^3 W -> kW.
\addplot[engblue, very thick, domain=0:45, samples=80] {0.000396*x^3};
\addlegendentry{aerodynamic $\tfrac12\rho C_D A\,U^3$}
% Total road-load power.
\addplot[engred, very thick, domain=0:45, samples=80] {0.1766*x + 0.000396*x^3};
\addlegendentry{total road load}
% Crossover speed where aero overtakes rolling: 0.396 U^3 = 176.6 U -> U^2 = 446 -> U = 21.1 m/s.
\addplot[enggray, dashed, thick] coordinates {(21.1,0) (21.1,45)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:20.0,22) {aero overtakes rolling};
\end{axis}
\end{tikzpicture}
\caption[Road-load power versus speed]{Road-load power splits into a
rolling-resistance term that grows linearly with speed and an aerodynamic term
that grows as the cube of speed. The two are equal near $21\,\si{\meter\per
\second}$ ($76\,\text{km/h}$) for an ordinary passenger car; above that speed
the aerodynamic term dominates and sets the slope of the fuel-versus-speed
curve, which is why a drag reduction pays off most at highway speed and why the
rolling term governs city driving.}
\label{fig:vol03ch12:road-load}
\end{figure}
