% Figure: geometry of a hydrodynamic journal bearing. A shaft (journal) of
% radius r turns inside a bush of radius r + c, where c is the radial
% clearance. Under load the journal rides eccentric to the bush by an
% eccentricity e, so the oil film is thick on the unloaded side and thin on the
% loaded side; the minimum film thickness h_min = c - e = c(1 - epsilon) is
% where seizure first threatens. The converging wedge on the loaded side is
% what generates the hydrodynamic pressure that carries the load.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% Bush (outer circle), centre at O_b = (0,0).
\draw[engblack, very thick] (0,0) circle (2.6);
% Journal (shaft), centre at O_j shifted down by the eccentricity.
\draw[engblue, very thick, fill=engblue!8] (0,-0.9) circle (2.05);
% Centres.
\fill[engblack] (0,0) circle (1.6pt);
\node[font=\scriptsize, anchor=south west] at (0.05,0.05) {$O_b$};
\fill[engblue] (0,-0.9) circle (1.6pt);
\node[font=\scriptsize, engblue, anchor=north west] at (0.05,-0.95) {$O_j$};
% Eccentricity vector.
\draw[engred, thick, -{Latex}] (0,0) -- (0,-0.9);
\node[font=\scriptsize, engred, anchor=west] at (0.08,-0.45) {$e$};
% Load arrow downward (line of centres aligns roughly with load for this sketch).
\draw[engorange, ultra thick, -{Latex}] (0,3.5) -- (0,2.7);
\node[font=\scriptsize, engorange, anchor=west] at (0.1,3.1) {load $W$};
% Minimum film thickness at the bottom (loaded side), between the two circles.
\draw[enggray, thick, {Latex}-{Latex}] (1.0,-2.9) -- (1.0,-2.39);
\node[font=\scriptsize, anchor=west] at (1.05,-2.65) {$h_{\min}=c-e$};
% Maximum film thickness at the top (unloaded side).
\draw[enggray, thick, {Latex}-{Latex}] (-1.0,2.6) -- (-1.0,1.55);
\node[font=\scriptsize, anchor=east] at (-1.05,2.1) {$h_{\max}=c+e$};
% Rotation arrow on the journal.
\draw[engblue, thick, -{Latex}] (1.3,-0.9) arc (0:75:1.3);
\node[font=\scriptsize, engblue] at (1.55,-0.1) {$\omega$};
% Converging wedge label on the loaded (lower) side.
\node[font=\scriptsize, align=center] at (-2.7,-1.6)
  {converging\\ wedge};
\end{tikzpicture}
\caption[Journal-bearing geometry]{A loaded journal rides eccentric inside its
bush. The journal centre $O_j$ sits a distance $e$ from the bush centre $O_b$,
so the oil film converges from a maximum thickness $h_{\max}=c+e$ on the
unloaded side to a minimum $h_{\min}=c-e=c(1-\varepsilon)$ on the loaded side,
where $c$ is the radial clearance and $\varepsilon=e/c$ the eccentricity ratio.
The journal's rotation drags oil into the converging wedge, and the pressure
the wedge builds is what carries the load $W$ without metal-to-metal contact.
The minimum film thickness is the design quantity: it must stay above the
combined surface roughness of journal and bush or the bearing wipes.}
\label{fig:vol03ch12:journal-bearing}
\end{figure}
