% Figure: the Hagen=Poiseuille parabolic velocity profile in a circular
% pipe, drawn as a profile across the diameter with the pipe walls marked.
% r runs from -R to +R; u = u_max (1 - (r/R)^2). The mean velocity (half
% the peak) is marked as a vertical reference.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={$u_z / u_{\max}$}, ylabel={$r/R$},
  xmin=0, xmax=1.15, ymin=-1.1, ymax=1.1,
  axis lines=left,
  legend pos=outer north east,
  legend cell align=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Parabola: u = 1 - (r/R)^2, parametric in r/R from -1 to 1.
\addplot[engblue, very thick, domain=-1:1, samples=160] ({1 - x*x}, {x});
\addlegendentry{$u_z(r)=u_{\max}(1-(r/R)^2)$}
% mean velocity = 0.5 u_max, vertical dashed line.
\addplot[engred, thick, dashed] coordinates {(0.5,-1) (0.5,1)};
\addlegendentry{mean $\bar u = \tfrac12 u_{\max}$}
% pipe walls
\addplot[engblack, very thick] coordinates {(0,1) (1.15,1)};
\addplot[engblack, very thick] coordinates {(0,-1) (1.15,-1)};
\end{axis}
\end{tikzpicture}
\caption[The Hagen=Poiseuille parabolic profile]{The Hagen=Poiseuille
velocity profile in fully developed laminar pipe flow. The profile is a
parabola, zero at the wall (no-slip) and peaking on the centreline. The
cross-section average is exactly half the centreline maximum, so the
volume flow rate is $Q=\tfrac12 u_{\max}\,\pi R^2$, and the flow scales
as the fourth power of the diameter at fixed pressure gradient.}
\label{fig:vol03ch12:poiseuille-profile}
\end{figure}
