% Figure: the linear-pad (Reynolds) slider bearing. A flat slider of length B
% moves at speed U beneath a stationary pad whose film thickness converges from
% h1 at inlet to h2 at outlet. The wedge generates a positive pressure hump
% that carries the load. Top panel: the converging geometry; lower curve: the
% dimensionless film pressure p(x) rising from zero at inlet to zero at outlet
% with an interior maximum, the load-carrying hump.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
% Geometry sketch (upper).
% Stationary inclined pad: top boundary of the film, converging left to right.
\draw[engblack, very thick] (0,2.0) -- (6,1.1);
\node[engblack, font=\scriptsize, anchor=south] at (3,1.7) {stationary pad};
% Moving runner (lower boundary), horizontal.
\draw[engblack, very thick] (0,0) -- (6,0);
\draw[->, engblue, very thick] (4.6,-0.35) -- (5.8,-0.35);
\node[engblue, font=\scriptsize, anchor=north] at (5.2,-0.4) {runner, $U$};
% Film thickness labels.
\draw[<->, enggray] (-0.25,0) -- (-0.25,2.0);
\node[enggray, font=\scriptsize, anchor=east] at (-0.3,1.0) {$h_1$};
\draw[<->, enggray] (6.25,0) -- (6.25,1.1);
\node[enggray, font=\scriptsize, anchor=west] at (6.3,0.55) {$h_2$};
% Length label.
\draw[<->, enggray] (0,2.35) -- (6,2.35);
\node[enggray, font=\scriptsize, anchor=south] at (3,2.35) {pad length $B$};
% Pressure curve (lower).
\begin{scope}[yshift=-3.2cm]
\draw[->, enggray] (0,0) -- (6.4,0) node[anchor=west, font=\scriptsize] {$x$};
\draw[->, enggray] (0,0) -- (0,2.2) node[anchor=south, font=\scriptsize] {film pressure $p$};
% Pressure hump: zero at both ends, peak shifted toward the outlet.
\draw[engred, very thick]
  plot[smooth, domain=0:6, samples=60]
  (\x, {1.8*sin(deg(3.1416*\x/6))*(0.55 + 0.45*\x/6)});
\node[engred, font=\scriptsize, anchor=south] at (3.9,1.85) {load-carrying hump};
\node[enggray, font=\scriptsize, anchor=north] at (0,0) {$p=0$};
\node[enggray, font=\scriptsize, anchor=north] at (6,0) {$p=0$};
\end{scope}
\end{tikzpicture}
\caption[The linear-pad slider bearing]{A flat runner moving beneath a slightly
inclined pad drags fluid into a converging wedge. The wedge raises the film
pressure to a hump that is zero at the inlet and outlet and peaks toward the
narrow end; the integral of that pressure over the pad area is the load the
film carries without metal contact. The same converging-wedge mechanism, wrapped
around a shaft, is what lets the journal bearing of the second case study float
its rotor on a film of oil.}
\label{fig:vol03ch12:slider-bearing}
\end{figure}
