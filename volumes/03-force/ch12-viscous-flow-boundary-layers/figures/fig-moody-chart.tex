% Figure: a schematic Moody chart. Log-log axes of Darcy friction factor f
% versus pipe Reynolds number. Laminar branch f = 64/Re for Re<2300; a
% transitional gap; turbulent curves from the Colebrook=White relation for
% several relative roughness values, flattening to the fully rough plateau.
% Colebrook is implicit; the curves here use the explicit Haaland form
% 1/sqrt(f) = -1.8 log10[ (eps/(3.7 D))^1.11 + 6.9/Re ], evaluated and
% transcribed so the plot needs no root solve.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=13cm, height=8cm,
  xlabel={Reynolds number $\mathrm{Re}_D = \bar u D/\nu$},
  ylabel={Darcy friction factor $f$},
  xmin=600, xmax=1e8, ymin=0.008, ymax=0.12,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Laminar branch f = 64/Re, Re 640..2300.
\addplot[engblue, very thick, domain=640:2300, samples=40] {64/x};
\addlegendentry{laminar $f=64/\mathrm{Re}$}
% Smooth-pipe turbulent (eps/D = 0), Haaland with eps=0: precomputed points.
\addplot[engred, very thick] coordinates {
  (4000,0.0399) (1e4,0.0309) (3e4,0.0235) (1e5,0.0180)
  (3e5,0.0146) (1e6,0.0116) (1e7,0.0081) (1e8,0.0060)
};
\addlegendentry{smooth ($\epsilon/D=0$)}
% eps/D = 0.001 turbulent.
\addplot[engamber, very thick] coordinates {
  (4000,0.0410) (1e4,0.0324) (3e4,0.0258) (1e5,0.0216)
  (3e5,0.0202) (1e6,0.0198) (1e7,0.0197) (1e8,0.0196)
};
\addlegendentry{$\epsilon/D=0.001$}
% eps/D = 0.01 turbulent.
\addplot[englime, very thick] coordinates {
  (4000,0.0490) (1e4,0.0430) (3e4,0.0395) (1e5,0.0385)
  (3e5,0.0382) (1e6,0.0381) (1e7,0.0381) (1e8,0.0381)
};
\addlegendentry{$\epsilon/D=0.01$}
% transition band
\addplot[enggray, thick, dashed] coordinates {(2300,0.008) (2300,0.12)};
\addplot[enggray, thick, dashed] coordinates {(4000,0.008) (4000,0.12)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:3100,0.05) {transition};
\end{loglogaxis}
\end{tikzpicture}
\caption[Schematic Moody chart]{A schematic Moody chart. The laminar
branch ($f=64/\mathrm{Re}$) holds to $\mathrm{Re}_D\approx2300$; a
transitional band has no single-curve description; the turbulent curves
from the Colebrook=White relation depend on both Reynolds number and
relative roughness, and flatten to a fully rough plateau where $f$
depends on $\epsilon/D$ alone. Curve values were precomputed from the
Haaland approximation \cite{paper:moody-1944,paper:colebrook-1939}.}
\label{fig:vol03ch12:moody}
\end{figure}
