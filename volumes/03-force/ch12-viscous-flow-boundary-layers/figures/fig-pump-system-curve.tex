% Figure: pump curve and system curve intersecting at the operating point.
% The system curve is static lift plus a quadratic friction term H = Hs + k Q^2,
% with k fixed by the Darcy-Weisbach loss at the design flow of the case study.
% The pump curve is a falling head-flow characteristic typical of a centrifugal
% pump. Their intersection is the operating point the case study reports.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={Volume flow rate $Q$ (\si{\liter\per\second})},
  ylabel={Head $H$ (\si{\meter})},
  xmin=0, xmax=40, ymin=0, ymax=60,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% System curve: H = 18 + 0.0395 Q^2  (Q in L/s); passes near (25, 42.7).
\addplot[engblue, very thick, domain=0:40, samples=80] {18 + 0.0395*x^2};
\addlegendentry{system curve $H_s + kQ^2$}
% Pump curve: falling characteristic H = 55 - 0.0197 Q^2, crossing near Q=25.
\addplot[engred, very thick, domain=0:40, samples=80] {55 - 0.0197*x^2};
\addlegendentry{pump curve}
% Operating point marker at the intersection (about Q=25.4, H=43.5).
\addplot[engblack, only marks, mark=*, mark size=2.4pt] coordinates {(25.4,43.5)};
\node[font=\scriptsize, anchor=south west] at (axis cs:25.4,43.5)
  {operating point};
% Static lift line.
\addplot[enggray, dashed, thick] coordinates {(0,18) (40,18)};
\node[font=\scriptsize, enggray, anchor=south west] at (axis cs:1,18)
  {static lift $H_s$};
\end{axis}
\end{tikzpicture}
\caption[Pump curve and system curve]{The operating point is where the
falling pump head-flow characteristic crosses the rising system curve. The
system curve is the static lift $H_s$ plus a friction head that grows as
$Q^2$, with the coefficient $k$ fixed by the Darcy-Weisbach loss at the design
flow. Moving the pump curve up (a larger impeller, a higher speed) or
flattening the system curve (a larger pipe, lower roughness) shifts the
crossing to a higher flow.}
\label{fig:vol03ch12:pump-system-curve}
\end{figure}
