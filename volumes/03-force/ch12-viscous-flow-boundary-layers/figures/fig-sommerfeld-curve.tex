% Figure: design chart for a hydrodynamic journal bearing. The eccentricity
% ratio epsilon = e/c and the minimum-film fraction h_min/c = 1 - epsilon are
% plotted against the Sommerfeld number S = (mu N / P)(r/c)^2. A heavily loaded
% bearing (small S) runs at high eccentricity with a thin film; a lightly
% loaded or fast bearing (large S) runs nearly concentric with a thick film.
% The curves are the standard short-bearing trend, transcribed as smooth
% monotone points for the working range.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogxaxis}[
  width=12cm, height=7cm,
  xlabel={Sommerfeld number $S=\dfrac{\mu N}{P}\left(\dfrac{r}{c}\right)^2$},
  ylabel={ratio (dimensionless)},
  xmin=1e-2, xmax=1e1, ymin=0, ymax=1,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Eccentricity ratio epsilon vs S: high at low S, low at high S.
\addplot[engred, very thick] coordinates {
  (1e-2,0.95) (2e-2,0.92) (5e-2,0.85) (1e-1,0.76) (2e-1,0.64)
  (5e-1,0.45) (1e0,0.31) (2e0,0.20) (5e0,0.10) (1e1,0.06)
};
\addlegendentry{eccentricity ratio $\varepsilon=e/c$}
% Minimum-film fraction h_min/c = 1 - epsilon.
\addplot[engblue, very thick] coordinates {
  (1e-2,0.05) (2e-2,0.08) (5e-2,0.15) (1e-1,0.24) (2e-1,0.36)
  (5e-1,0.55) (1e0,0.69) (2e0,0.80) (5e0,0.90) (1e1,0.94)
};
\addlegendentry{film fraction $h_{\min}/c=1-\varepsilon$}
% Marker for a representative duty point used in the worked case study.
\addplot[engblack, only marks, mark=*, mark size=2.4pt] coordinates {(0.2,0.36)};
\node[font=\scriptsize, anchor=south west] at (axis cs:0.2,0.36)
  {case-study duty};
\end{semilogxaxis}
\end{tikzpicture}
\caption[Journal-bearing design chart]{The eccentricity ratio and the minimum
film fraction read off the Sommerfeld number $S$. A heavily loaded or
slow-turning bearing sits at small $S$, runs at high eccentricity, and leaves a
thin film close to the surface roughness; a lightly loaded or fast-turning
bearing sits at large $S$, runs nearly concentric, and floats on a thick film.
The design task is to keep the duty point at an $S$ large enough that the
minimum film clears the combined roughness with margin, yet not so large that
the bearing whirls. The marked point is the duty the case study sizes.}
\label{fig:vol03ch12:sommerfeld-curve}
\end{figure}
