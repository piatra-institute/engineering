% Figure: the Blasius similarity velocity profile f'(eta) = u/U_infty as a
% function of the similarity variable eta = y sqrt(U/(nu x)). Tabulated
% values from the numerical integration of the Blasius equation
% 2 f''' + f f'' = 0. The 99-percent thickness is at eta approx 5.0.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={$u/U_\infty = f'(\eta)$}, ylabel={$\eta = y\sqrt{U_\infty/(\nu x)}$},
  xmin=0, xmax=1.05, ymin=0, ymax=7,
  axis lines=left,
  legend pos=south east,
  legend cell align=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Blasius f'(eta) tabulated (Schlichting). x = f', y = eta.
\addplot[engblue, very thick, mark=*, mark size=1pt] coordinates {
  (0.0000,0.0) (0.1328,0.4) (0.2647,0.8) (0.3938,1.2) (0.5168,1.6)
  (0.6298,2.0) (0.7290,2.4) (0.8115,2.8) (0.8761,3.2) (0.9233,3.6)
  (0.9555,4.0) (0.9759,4.4) (0.9878,4.8) (0.9943,5.2) (0.9975,5.6)
  (0.9990,6.0) (0.9999,6.8)
};
\addlegendentry{Blasius $f'(\eta)$}
% 99% line and eta=5.0 marker
\addplot[engred, thick, dashed] coordinates {(0.99,0) (0.99,7)};
\node[font=\scriptsize, engred, anchor=west] at (axis cs:0.55,5.0)
  {$\eta_{99}\approx5.0$};
\addplot[engamber, thick, dotted] coordinates {(0,5.0) (1.05,5.0)};
\end{axis}
\end{tikzpicture}
\caption[The Blasius similarity profile]{The Blasius similarity solution
for the laminar flat-plate boundary layer. Plotting $u/U_\infty$ against
the similarity variable $\eta$ collapses the profiles at every streamwise
station onto a single curve. The velocity reaches $99\%$ of the
freestream at $\eta_{99}\approx5.0$, which gives the standard thickness
estimate $\delta\approx5.0\sqrt{\nu x/U_\infty}$.}
\label{fig:vol03ch12:blasius}
\end{figure}
