% Figure: the two terms of the Ergun equation for packed-bed pressure drop
% versus superficial velocity. The viscous (Blake-Kozeny) term is linear in U;
% the inertial (Burke-Plummer) term is quadratic; their sum is the total. The
% crossover, where the two are equal, marks the transition from the
% viscous-dominated (Darcy) regime to the inertia-dominated regime. Constants
% match worked example 28: viscous slope 507/0.30 = 1690 Pa/m per (m/s),
% inertial coefficient 591/0.30^2 = 6567 Pa/m per (m/s)^2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={superficial velocity $U_s$ (\si{\meter\per\second})},
  ylabel={pressure gradient $\Delta p/L$ (\si{\pascal\per\meter})},
  xmin=0, xmax=0.6, ymin=0, ymax=3600,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Viscous (Blake-Kozeny) term: linear, slope 1690 Pa/m per (m/s).
\addplot[engblue, very thick, domain=0:0.6, samples=40] {1690*x};
\addlegendentry{viscous term (linear in $U_s$)}
% Inertial (Burke-Plummer) term: quadratic, coefficient 6567.
\addplot[engamber, very thick, domain=0:0.6, samples=60] {6567*x^2};
\addlegendentry{inertial term (quadratic in $U_s$)}
% Total Ergun sum.
\addplot[engred, very thick, domain=0:0.6, samples=60] {1690*x + 6567*x^2};
\addlegendentry{Ergun sum}
% Mark the operating point of worked example 28 at U_s = 0.30.
\addplot[enggray, dashed, thick] coordinates {(0.30,0) (0.30,3600)};
\node[font=\scriptsize, enggray, rotate=90, anchor=south]
  at (axis cs:0.285,1600) {worked example 28};
\end{axis}
\end{tikzpicture}
\caption[Ergun packed-bed pressure drop]{The Ergun equation splits the
packed-bed pressure gradient into a viscous term linear in the superficial
velocity (the creeping-flow Blake-Kozeny limit) and an inertial term quadratic
in it (the Burke-Plummer limit). At low velocity the viscous term dominates and
the drop is Darcy-linear; at high velocity the inertial term dominates and the
drop is quadratic. At the operating point of worked example~28 the two terms
are comparable, the transitional pore regime where neither limit alone
suffices, and the void fraction $\varepsilon$ enters both terms through a strong
$\varepsilon^{-3}$ factor.}
\label{fig:vol03ch12:ergun}
\end{figure}
