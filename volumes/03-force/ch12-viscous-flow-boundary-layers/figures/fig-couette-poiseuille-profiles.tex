% Figure: the three canonical laminar velocity profiles between or in a
% channel. Linear Couette (plate drag), parabolic Poiseuille (pressure
% driven), and a combined Couette=Poiseuille profile with an adverse
% pressure gradient producing near-wall reversal. y is the gap-normal
% coordinate, normalised; u is normalised to its peak magnitude.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={$u/u_{\mathrm{ref}}$}, ylabel={$y/h$},
  xmin=-0.35, xmax=1.1, ymin=0, ymax=1,
  axis lines=left,
  domain=0:1, samples=200,
  legend pos=outer north east,
  legend cell align=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Couette: u = y (linear). Plot with y as the running variable: x=u(y), y=y.
\addplot[engblue, very thick] coordinates {(0,0) (1,1)};
\addlegendentry{Couette (linear)}
% Poiseuille: u = 4 y (1-y), peak 1 at y=0.5. Parametric in y.
\addplot[engred, very thick, domain=0:1, samples=120] ({4*x*(1-x)}, {x});
\addlegendentry{Poiseuille (parabola)}
% Combined with adverse gradient: u = y - 1.6 y (1-y). At small y the
% pressure term dominates: u(0.1)=0.1-1.6*0.1*0.9=0.1-0.144=-0.044 (reversed).
\addplot[engamber, very thick, dashed, domain=0:1, samples=120]
  ({x - 1.6*x*(1-x)}, {x});
\addlegendentry{combined, adverse $\partial_x p>0$}
% zero reference line
\addplot[enggray, thin] coordinates {(0,0) (0,1)};
\end{axis}
\end{tikzpicture}
\caption[Canonical laminar profiles: Couette, Poiseuille, combined]{The
three canonical laminar velocity profiles. The linear Couette profile is
driven by wall motion alone; the parabolic Poiseuille profile is driven
by a pressure gradient alone; the combined profile superposes the two,
and under a strong enough adverse pressure gradient develops a near-wall
reversed-flow region (the part of the dashed curve at $u<0$). That
near-wall reversal is the simplest analytical signature of incipient
separation.}
\label{fig:vol03ch12:couette-poiseuille}
\end{figure}
