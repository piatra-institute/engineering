% Figure: polynomial boundary-layer velocity profiles under the three pressure
% gradients, drawn as u/Ue against y/delta. The favourable-gradient profile is
% full (convex, high wall shear); the zero-gradient (flat-plate) profile is the
% reference cubic; the adverse-gradient profile is inflected and approaches zero
% wall slope, the shape that precedes separation. The Pohlhausen quartic family
% supplies the shapes through the gradient parameter Lambda.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=9.5cm, height=8cm,
  xlabel={$u/U_e$},
  ylabel={$y/\delta$},
  xmin=0, xmax=1.05, ymin=0, ymax=1.05,
  legend pos=south east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  minor grid style={enggray!8},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Plot u/Ue (x) versus eta = y/delta (y). Pohlhausen quartic:
% u/Ue = 2 eta - 2 eta^3 + eta^4 + (Lambda/6) eta (1-eta)^3.
% Favourable gradient, Lambda = +7 (full profile, high wall shear).
\addplot[engblue, very thick, domain=0:1, samples=80]
  ({2*x - 2*x^3 + x^4 + (7/6)*x*(1-x)^3}, {x});
\addlegendentry{favourable ($\Lambda>0$)}
% Zero gradient, Lambda = 0 (the flat-plate reference).
\addplot[engblack, very thick, domain=0:1, samples=80]
  ({2*x - 2*x^3 + x^4}, {x});
\addlegendentry{zero ($\Lambda=0$)}
% Adverse gradient at incipient separation, Lambda = -12 (zero wall slope).
\addplot[engred, very thick, domain=0:1, samples=80]
  ({2*x - 2*x^3 + x^4 + (-12/6)*x*(1-x)^3}, {x});
\addlegendentry{adverse ($\Lambda=-12$)}
\end{axis}
\end{tikzpicture}
\caption[Boundary-layer profiles under pressure gradient]{The Pohlhausen
quartic profile family, drawn as velocity against height across the layer for
three pressure gradients. A favourable (accelerating) gradient fills the
profile and steepens the wall slope, raising the wall shear; the zero-gradient
profile is the flat-plate reference; an adverse (decelerating) gradient
inflects the profile and flattens the wall slope. At $\Lambda=-12$ the wall
slope reaches zero, the wall shear vanishes, and the layer is at the point of
separation, beyond which the near-wall flow reverses.}
\label{fig:vol03ch12:momentum-profiles}
\end{figure}
