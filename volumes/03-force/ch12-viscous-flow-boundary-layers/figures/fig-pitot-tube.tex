% Figure: schematic of a Pitot-static tube. A forward-facing total-pressure
% port sits at the stagnation point; side static ports read the freestream
% static pressure; the two feed a differential-pressure sensor that reads the
% dynamic pressure 0.5*rho*U^2, from which the speed is recovered. Pure TikZ
% schematic, no axis.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>={Latex}]
% Freestream arrows.
\foreach \y in {0.55,0.9,1.25} {
  \draw[engred, ->] (-2.4,\y) -- (-1.4,\y);
  \draw[engred, ->] (-2.4,-\y) -- (-1.4,-\y);
}
\node[font=\scriptsize, engred, anchor=east] at (-2.4,0) {flow $U$};
% Probe body: an outer static tube and an inner total tube.
\draw[enggray, thick] (-1.2,0.28) -- (3.2,0.28) -- (3.2,-0.28) -- (-1.2,-0.28);
% Rounded nose (stagnation point) at the left.
\draw[enggray, thick] (-1.2,0.28) .. controls (-1.55,0.12) and (-1.55,-0.12) .. (-1.2,-0.28);
% Total-pressure port at the nose.
\fill[engblue] (-1.5,0) circle (1.6pt);
\node[font=\scriptsize, engblue, anchor=south] at (-1.5,0.35) {total $p_0$};
% Inner total tube.
\draw[engblue, thick] (-1.45,0) -- (2.0,0) -- (2.0,0.9) -- (2.6,0.9);
% Static ports on the side.
\fill[engorange] (0.9,0.28) circle (1.4pt);
\fill[engorange] (0.9,-0.28) circle (1.4pt);
\node[font=\scriptsize, engorange, anchor=south] at (0.9,0.36) {static $p_\infty$};
% Static annulus tube.
\draw[engorange, thick] (0.9,0.28) -- (0.9,0.55) -- (2.6,0.55) -- (2.6,0.55);
\draw[engorange, thick] (0.9,-0.28) -- (0.9,-0.9) -- (2.6,-0.9);
% Differential sensor box.
\draw[enggray, thick] (2.6,-1.2) rectangle (3.9,1.2);
\node[font=\scriptsize, align=center] at (3.25,0) {$\Delta p$\\ sensor};
\draw[engblue, thick] (2.6,0.9) -- (2.6,0.9);
\draw[engorange, thick] (2.6,-0.9) -- (2.6,-0.9);
% Output reading.
\node[font=\scriptsize, anchor=west] at (4.0,0)
  {$U=\sqrt{\dfrac{2(p_0-p_\infty)}{\rho}}$};
\end{tikzpicture}
\caption[The Pitot-static tube]{A Pitot-static tube reads speed from the
dynamic pressure. A forward-facing port at the stagnation point measures the
total pressure $p_0 = p_\infty + \tfrac12\rho U^2$; side ports flush with the
body measure the static pressure $p_\infty$; a differential sensor returns the
difference $\tfrac12\rho U^2$, from which the speed follows once the density is
known. The reading is clean only while the ports stay clear and the probe
Reynolds number stays high; icing of the forward port or a very low Reynolds
number corrupts the stagnation reading, the failure at the root of the Air
France 447 unreliable-airspeed sequence.}
\label{fig:vol03ch12:pitot-tube}
\end{figure}
