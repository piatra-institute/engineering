% Figure: the wall-mounted bracket crane of the worked example. Jib AC
% pinned at A, tie BC from a point B above A, load W hung at C.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, x=1.4cm, y=1.4cm]
  % wall
  \draw[thick] (0,-0.6) -- (0,1.6);
  \foreach \y in {-0.5,-0.2,...,1.5} { \draw[enggray, thin] (0,\y) -- (-0.25,\y-0.18); }
  % points
  \coordinate (A) at (0,0);     % pin
  \coordinate (B) at (0,1);     % tie anchor, 1 m above A
  \coordinate (C) at (2,0);     % jib outer end, 2 m out
  % jib
  \draw[engblue, very thick] (A) -- (C);
  % tie
  \draw[engamber, very thick] (B) -- (C);
  % joints
  \fill[engblack] (A) circle (1.8pt); \node[anchor=north east, font=\scriptsize] at (A) {$A$};
  \fill[engblack] (B) circle (1.8pt); \node[anchor=south east, font=\scriptsize] at (B) {$B$};
  \fill[engblack] (C) circle (1.8pt); \node[anchor=north west, font=\scriptsize] at (C) {$C$};
  % load W at C
  \draw[->, very thick, engred] (C) -- ++(0,-0.9);
  \node[engred, anchor=west, font=\scriptsize] at ($(C)+(0.05,-0.5)$) {$W$};
  % tie tension direction on jib at C (toward B)
  \draw[->, thick, englime!70!black] (C) -- ($(C)!0.45!(B)$);
  \node[englime!60!black, font=\scriptsize, anchor=south] at ($(C)!0.25!(B)$) {$T$};
  % angle theta at C
  \draw[enggray, thin] ($(C)+(-0.55,0)$) arc (180:153:0.55);
  \node[enggray, font=\scriptsize] at ($(C)+(-0.75,0.12)$) {$\theta$};
  % dimensions
  \draw[<->, enggray, thin] (0,-0.75) -- (2,-0.75);
  \node[enggray, font=\scriptsize, anchor=north] at (1,-0.75) {$L=2\,\si{\meter}$};
  \draw[<->, enggray, thin] (-0.45,0) -- (-0.45,1);
  \node[enggray, font=\scriptsize, anchor=east] at (-0.45,0.5) {$1\,\si{\meter}$};
\end{tikzpicture}
\caption[Wall-mounted bracket crane]{The wall-mounted bracket crane of
the worked example. The horizontal jib $AC$ is pinned to the wall at $A$
and held up by the tie $BC$ running from an anchor $B$ one metre above
$A$ to the jib's outer end $C$, where the load $W$ hangs. Taking moments
about the pin $A$ removes both pin-reaction components and leaves a single
equation in the tie tension $T$. The geometry sets the tie angle
$\theta=\arctan(1/2)\approx 26.6^\circ$, and because the tie pulls at this
shallow angle the tension comes out more than twice the hung load. The
horizontal pin reaction at $A$ then equals the horizontal pull of the
tie, so the pin must be sized for a force much larger than $W$ alone, the
recurring surprise of bracket and cable geometry: the supporting members
routinely carry several times the load they hold.}
\label{fig:vol03ch02:bracket-crane}
\end{figure}
