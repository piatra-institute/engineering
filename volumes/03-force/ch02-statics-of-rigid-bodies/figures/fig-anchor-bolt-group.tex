% Figure: plan view of the signpost base plate and its four-bolt anchor
% group, showing how the overturning moment about the across (x) axis is
% carried as a tension-compression couple between the downwind and upwind
% bolt pairs. The wind blows in +y; the base plate is square; bolts sit at
% the four corners. The neutral axis runs through the plate centre along x.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, x=1.2cm, y=1.2cm]
  % base plate (plan)
  \draw[engblack, fill=enggray!12] (-1.5,-1.5) rectangle (1.5,1.5);
  % bolt positions (corners), at +/- s in each direction
  \def\s{1.0}
  \coordinate (b1) at (-\s, \s);   % upwind-left
  \coordinate (b2) at ( \s, \s);   % upwind-right
  \coordinate (b3) at (-\s,-\s);   % downwind-left
  \coordinate (b4) at ( \s,-\s);   % downwind-right
  % neutral axis (about which the moment acts), along x through centre
  \draw[enggray, dashed, thin] (-1.7,0) -- (1.7,0);
  \node[enggray, font=\tiny, anchor=west] at (1.7,0) {neutral axis};
  % downwind bolts in compression (engblue), upwind in tension (engred)
  \fill[engred]  (b1) circle (3pt);
  \fill[engred]  (b2) circle (3pt);
  \fill[engblue] (b3) circle (3pt);
  \fill[engblue] (b4) circle (3pt);
  \node[engred,  font=\tiny, anchor=south] at ($(b1)+(0,0.12)$) {$T$};
  \node[engred,  font=\tiny, anchor=south] at ($(b2)+(0,0.12)$) {$T$};
  \node[engblue, font=\tiny, anchor=north] at ($(b3)+(0,-0.12)$) {$C$};
  \node[engblue, font=\tiny, anchor=north] at ($(b4)+(0,-0.12)$) {$C$};
  % wind direction arrow (+y, downwind, drawn up the page here = into page top)
  \draw[->, very thick, engamber] (0,1.9) -- (0,2.6);
  \node[engamber, font=\scriptsize, anchor=west] at (0,2.25) {wind ($+y$)};
  % gauge distance dimension between bolt rows
  \draw[<->, enggray, thin] (1.35,-\s) -- (1.35,\s);
  \node[enggray, font=\scriptsize, anchor=west] at (1.35,0) {$d_b$};
  % bolt-row labels
  \node[font=\tiny, anchor=east] at (-1.5,\s)  {upwind row};
  \node[font=\tiny, anchor=east] at (-1.5,-\s) {downwind row};

  \draw[->, engblack, thin] (-2.2,-1.9) -- (-1.6,-1.9); \node[font=\tiny, anchor=west] at (-1.6,-1.9) {$x$};

\caption[Anchor-bolt group under overturning moment]{Plan view of the
signpost base plate and its four-bolt anchor group. The wind blows in the
$+y$ (downwind) direction, so the base carries an overturning moment about
the across ($x$) axis. The plate tends to rotate about a neutral axis
through its centre: the upwind bolts (red) are pulled into tension $T$ as
that edge lifts, and the downwind bolts (blue) are pushed into compression
$C$ as that edge bears down. The couple of the two bolt rows, each row at a
lever arm $d_b/2$ from the neutral axis, balances the overturning moment, so
the bolt tension follows from the moment divided by the row spacing, not from
the wind force directly. A rigid base plate with bolts equidistant from the
neutral axis shares the tension equally between the two upwind bolts. The
direct downwind shear is carried by all four bolts together, and where the
plate also twists about the vertical axis the bolts pick up a further
in-plane force proportional to their distance from the plate centre.}
\label{fig:vol03ch02:anchor-bolt-group}
\end{figure}
