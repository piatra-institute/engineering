% Figure: a cantilevered roadside signpost under a horizontal wind load,
% the three-dimensional case study. A vertical pole rises from a fixed base
% and carries a sign panel offset to one side; the wind blows horizontally
% across the panel. The base must supply three reaction forces and three
% reaction moments. Drawn in an oblique (cabinet) projection: the in-plane
% x (across) and z (up) axes are orthogonal on the page, and the depth y
% (downwind) is drawn receding to the upper right.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, x=1.0cm, y=1.0cm]
  % oblique unit for the receding (downwind, +y) direction
  \def\dx{0.5}
  \def\dy{0.32}
  % ground plane (a parallelogram in oblique projection)
  \fill[enggray!12]
    (-1.6,0) -- (3.2,0) -- ($(3.2,0)+(2*\dx,2*\dy)$) -- ($(-1.6,0)+(2*\dx,2*\dy)$) -- cycle;
  \draw[enggray, thin]
    (-1.6,0) -- (3.2,0) -- ($(3.2,0)+(2*\dx,2*\dy)$) -- ($(-1.6,0)+(2*\dx,2*\dy)$) -- cycle;
  % the pole, rising from the base O at the origin
  \coordinate (O) at (0,0);
  \draw[engblue, line width=3pt] (O) -- (0,4.2);
  % sign panel, offset to the right of the pole near the top
  \coordinate (P) at (0,3.6);          % attachment on the pole
  \fill[engamber!22] (P) rectangle (2.2,4.2);
  \draw[engamber, very thick] (P) rectangle (2.2,4.2);
  \node[engamber!70!black, font=\scriptsize] at (1.1,3.9) {sign panel};
  % short connecting arm pole-to-panel
  \draw[engblue, very thick] (0,3.9) -- (P -| 0,3.9);
  % centroid of the panel (where the wind resultant acts)
  \coordinate (G) at (1.1,3.9);
  \fill[engblack] (G) circle (1.4pt);
  \node[font=\tiny, anchor=south west] at (G) {$G$};
  % wind load: horizontal resultant F acting downwind (+y), drawn receding
  \draw[->, very thick, engred] (G) -- ($(G)+(2.0*\dx,2.0*\dy)$);
  \node[engred, font=\scriptsize, anchor=south west] at ($(G)+(2.0*\dx,2.0*\dy)$) {$F$ (wind)};
  % base support block (fixed)
  \draw[engblack, fill=enggray!30] (-0.45,-0.55) rectangle (0.45,0);
  \draw[engblack] (-0.65,-0.55) -- (0.65,-0.55);
  \foreach \xx in {-0.55,-0.25,...,0.55} { \draw[enggray,thin] (\xx,-0.55) -- (\xx-0.16,-0.75); }
  % anchor-bolt circle at the base (four bolts), drawn as small dots on a rectangle
  \fill[engblack] (-0.30,-0.05) circle (1.2pt);
  \fill[engblack] ( 0.30,-0.05) circle (1.2pt);
  \fill[engblack] ($(-0.30,-0.05)+(\dx,\dy)$) circle (1.2pt);
  \fill[engblack] ($( 0.30,-0.05)+(\dx,\dy)$) circle (1.2pt);
  \node[font=\tiny, anchor=north] at (0,-0.78) {anchor bolts};
  \node[engblue, font=\scriptsize, anchor=east] at (-0.1,2.0) {pole};
  \node[font=\scriptsize, anchor=east] at (O) {$O$};
  % reaction triad at the base
  \draw[->, englime!70!black, thick] (O) -- (-1.3,0);
  \node[englime!60!black, font=\tiny, anchor=east] at (-1.3,0) {$R_x$};
  \draw[->, englime!70!black, thick] (O) -- (0,-1.0) ++(0,0);
  % height dimension
  \draw[<->, enggray, thin] (3.0,0) -- (3.0,3.9);
  \node[enggray, font=\scriptsize, anchor=west] at (3.0,1.95) {$H$};
  % offset dimension
  \draw[<->, enggray, thin] (0,4.45) -- (1.1,4.45);
  \node[enggray, font=\scriptsize, anchor=south] at (0.55,4.45) {$e$};
  % axis triad
  \draw[->, engblack, thin] (-1.6,4.4) -- (-1.0,4.4); \node[font=\tiny, anchor=west] at (-1.0,4.4) {$x$};
  \draw[->, engblack, thin] (-1.6,4.4) -- (-1.6,5.0); \node[font=\tiny, anchor=south] at (-1.6,5.0) {$z$};
  \draw[->, engblack, thin] (-1.6,4.4) -- ($(-1.6,4.4)+(\dx,\dy)$); \node[font=\tiny, anchor=south west] at ($(-1.6,4.4)+(\dx,\dy)$) {$y$};
\end{tikzpicture}
\caption[Cantilevered signpost under wind load]{A roadside signpost as a
three-dimensional cantilever. The pole rises a height $H$ from a fixed base
at $O$ and carries a sign panel whose centroid $G$ sits an offset $e$ to one
side of the pole axis. The wind blows horizontally across the panel and
delivers a resultant force $F$ in the downwind ($y$) direction at $G$. The
base, a plate bolted down by a group of anchor bolts, is a fixed support: it
supplies three reaction force components and three reaction moments. Because
the wind force is offset from the pole axis by $e$, it produces not only a
downwind shear and an overturning moment at the base but also a twisting
moment about the pole's own vertical axis, the torsional reaction that an
analysis treating the post as a planar cantilever would miss. The drawing
uses an oblique projection: the across ($x$) and up ($z$) directions are
true on the page, and the downwind ($y$) direction recedes to the upper
right.}
\label{fig:vol03ch02:signpost-3d}
\end{figure}
