% Figure: a band brake. A flexible band wraps a rotating drum through a
% wrap angle beta; the tight side carries T1, the slack side T2, related
% by the capstan equation. A lever pivoted at O sets the band tension.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, x=1.0cm, y=1.0cm]
  % drum
  \coordinate (Ctr) at (0,0);
  \def\R{1.6}
  \draw[engblue, very thick] (Ctr) circle (\R);
  \fill[engblack] (Ctr) circle (1.5pt);
  \node[font=\scriptsize, anchor=north east] at (Ctr) {drum};
  % rotation arrow
  \draw[->, thick, enggray] (135:0.7) arc (135:45:0.7);
  \node[enggray, font=\scriptsize] at (90:0.95) {$\omega$};
  % band wraps from angle -45 (slack, lower right) around the top to 225 (tight, lower left):
  % draw the wrapped arc as a thicker band, wrap beta = 270 deg
  \draw[engamber, line width=2.2pt] (-45:\R) arc (-45:225:\R);
  % tight side T1 leaves at 225 deg downward-left to anchor near pivot
  \coordinate (Pa) at (225:\R);
  \coordinate (O)  at (-3.4,-2.6);   % lever pivot
  \draw[engred, very thick] (Pa) -- ($(Pa)+(-1.0,-1.0)$);
  \node[engred, font=\scriptsize, anchor=south east] at ($(Pa)+(-0.55,-0.5)$) {$T_1$ (tight)};
  % slack side T2 leaves at -45 deg downward-right to the lever
  \coordinate (Pb) at (-45:\R);
  \draw[englime!70!black, very thick] (Pb) -- ($(Pb)+(1.0,-1.0)$);
  \node[englime!55!black, font=\scriptsize, anchor=south west] at ($(Pb)+(0.5,-0.5)$) {$T_2$ (slack)};
  % wrap-angle label
  \draw[->, enggray, thin] (10:\R+0.25) arc (10:170:\R+0.25);
  \node[enggray, font=\scriptsize] at (90:\R+0.55) {$\beta$};
  % lever: from pivot O out to a handle, with band slack end attached close to pivot
  \coordinate (H) at (2.6,-3.2);
  \draw[engblack, very thick] (O) -- (H);
  \fill[engblack] (O) circle (1.8pt);
  \node[font=\scriptsize, anchor=north east] at (O) {$O$};
  % pivot ground
  \draw[thick] ($(O)+(-0.28,-0.28)$) -- (O) -- ($(O)+(0.28,-0.28)$) -- cycle;
  % applied lever force at handle
  \draw[->, very thick, engred] (H) -- ++(0,-0.9);
  \node[engred, font=\scriptsize, anchor=west] at ($(H)+(0.05,-0.5)$) {$F$};
  % moment arms a and l marked along lever
  \node[enggray, font=\scriptsize, anchor=south] at ($(O)!0.18!(H)+(0,0.12)$) {$a$};
  \node[enggray, font=\scriptsize, anchor=south] at ($(O)!0.85!(H)+(0,0.12)$) {$\ell$};
\end{tikzpicture}
\caption[Band brake and the capstan relation]{A band brake. A flexible
band wraps the drum through a wrap angle $\beta$; the side that the
drum's rotation drags taut carries the tight tension $T_1$ and the other
the slack tension $T_2$. Belt friction relates the two by the capstan
equation $T_1 = T_2\, e^{\mu\beta}$, an exponential in the wrap angle.
The lever pivoted at $O$ attaches the slack end at the short arm $a$ and
takes the operator's force $F$ at the long arm $\ell$, so a modest hand
force sets a large band tension and a large braking torque
$M=(T_1-T_2)r$. The exponential wrap is what lets a sailor hold a loaded
line with a few turns around a capstan and what lets a small brake lever
stop a heavy drum.}
\label{fig:vol03ch02:band-brake}
\end{figure}
