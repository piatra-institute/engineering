% Figure: the three-force graphical (Culmann) construction. A jib carries a
% load at the tip, a tie, and a wall pin. The three forces are concurrent, and
% the closed force triangle to the right returns the three magnitudes without a
% component equation. Left: the space diagram with the concurrency point. Right:
% the force triangle laid head to tail.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, x=1.0cm, y=1.0cm]
  % space diagram (left)
  % wall
  \draw[thick] (0,-0.4) -- (0,3.2);
  \foreach \y in {-0.3,0.0,...,3.0} { \draw[enggray, thin] (0,\y) -- (-0.25,\y-0.18); }
  \coordinate (A) at (0,0);      % pin
  \coordinate (B) at (0,2.4);    % tie anchor
  \coordinate (C) at (3.2,0);    % jib tip
  % concurrency point: load line (vertical through C) meets tie line (B--C extended)
  % tie line B--C; load line x=C_x. They meet at C itself for this geometry,
  % so use a jib that carries the load short of the tip to show concurrency.
  \coordinate (D) at (2.2,0);    % load application point on jib
  % jib A--C
  \draw[engblue, very thick] (A) -- (C);
  % tie B--C
  \draw[engamber, very thick] (B) -- (C);
  % load at D
  \draw[->, very thick, engred] (D) -- ++(0,-1.2);
  \node[engred, font=\scriptsize, anchor=west] at ($(D)+(0.05,-0.7)$) {$W$};
  % concurrency point P: intersection of load line (x=2.2) and tie line B--C
  % tie line: from (0,2.4) to (3.2,0): y = 2.4 - 0.75 x; at x=2.2, y=0.75
  \coordinate (P) at (2.2,0.75);
  \fill[englime!60!black] (P) circle (1.6pt);
  \node[englime!60!black, font=\scriptsize, anchor=south west] at (P) {$P$};
  % pin-reaction line: must pass through A and P
  \draw[engblack, dashed] (A) -- ($(A)!1.35!(P)$);
  \node[engblack, font=\scriptsize, anchor=south east] at ($(A)!1.15!(P)$) {$R_A$ line};
  % dotted extensions of load and tie lines up to P
  \draw[engred, dotted] (D) -- (P);
  % joints
  \fill[engblack] (A) circle (1.8pt); \node[font=\scriptsize, anchor=north east] at (A) {$A$};
  \fill[engblack] (B) circle (1.8pt); \node[font=\scriptsize, anchor=south east] at (B) {$B$};
  \fill[engblack] (C) circle (1.8pt); \node[font=\scriptsize, anchor=north west] at (C) {$C$};
  \node[font=\scriptsize\itshape, anchor=north] at (1.6,-0.6) {space diagram};

  % force triangle (right)
  \begin{scope}[shift={(6.6,0.4)}]
    % W straight down (length 1.2), then tie force, then pin reaction closing.
    \coordinate (o) at (0,2.0);
    % W: down
    \draw[->, very thick, engred] (o) -- ++(0,-1.2);
    \node[engred, font=\scriptsize, anchor=west] at (0.05,1.4) {$W$};
    \coordinate (m) at (0,0.8);
    % tie force T: along direction of tie C->B = (-3.2, 2.4)/4 = (-0.8,0.6)
    \draw[->, very thick, engamber] (m) -- ++(-1.6,1.2);
    \node[engamber, font=\scriptsize, anchor=north east] at (-0.8,1.4) {$T$};
    \coordinate (n) at (-1.6,2.0);
    % pin reaction R_A closing the triangle back to o
    \draw[->, very thick, engblue] (n) -- (o);
    \node[engblue, font=\scriptsize, anchor=south] at (-0.8,2.0) {$R_A$};
    \node[font=\scriptsize\itshape, anchor=north] at (-0.8,-0.5) {force triangle};
  \end{scope}
\end{tikzpicture}
\caption[Three-force graphical construction]{The three-force construction for
a jib carrying a tip load. On the space diagram (left) the load line and the
tie line meet at the point $P$, and equilibrium forces the pin reaction $R_A$
to pass through $P$ as well, which fixes its direction by geometry before any
number is computed. The force triangle (right) then lays the three vectors head
to tail: the known load $W$ down, the tie force $T$ along the tie's own line,
and the pin reaction $R_A$ closing the figure. A triangle that closes is the
drawn statement of equilibrium, and the two unknown magnitudes follow from the
sine rule on the triangle rather than from a pair of component equations. A
triangle that fails to close signals a dropped force, the fastest visual check
on a three-force free body.}
\label{fig:vol03ch02:three-force-graphical}
\end{figure}
