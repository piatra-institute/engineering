% Figure: the free surface of a liquid in a bucket rotating about its
% vertical axis is a paraboloid, z(r) = z0 + omega^2 r^2 / (2 g). Shows the
% parabola for two spin rates. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={radius $r$ (m)}, ylabel={surface height $z$ (m)},
  xmin=-1, xmax=1, ymin=0, ymax=1.4,
  axis lines=left,
  legend pos=north west,
  legend cell align=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  samples=80, domain=-1:1,
]
% slow spin: omega = 3 rad/s -> z = 0.2 + (9/19.62) r^2 = 0.2 + 0.4587 r^2
\addplot[engblue, very thick] {0.2 + 0.4587*x^2};
\addlegendentry{$\omega=3$ rad/s}
% faster spin: omega = 4.43 rad/s -> coeff (19.62/19.62)=1.0 -> z=0.2 + 1.0 r^2
\addplot[engred, very thick] {0.2 + 1.0*x^2};
\addlegendentry{$\omega=4.43$ rad/s}
% wall lines
\draw[engblack, thick] (axis cs:-1,0) -- (axis cs:-1,1.4);
\draw[engblack, thick] (axis cs:1,0) -- (axis cs:1,1.4);
% axis of rotation
\draw[enggray, dashed] (axis cs:0,0) -- (axis cs:0,1.4);
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:0,1.4) {axis};
\end{axis}
\end{tikzpicture}
\caption[Rotating-bucket paraboloid]{The free surface of a liquid in a
bucket rotating steadily about its vertical axis. Hydrostatics in the
rotating frame, with the apparent gravity tilted outward by the
centrifugal term, makes the surface everywhere normal to the apparent
gravity, which forces the surface into a paraboloid
$z(r)=z_0+\omega^2r^2/(2g)$. Doubling the spin rate quadruples the rise
at the rim. The same shape is the figure of a self-gravitating rotating
planet and the working principle of the spinning-mercury parabolic
mirror, which casts a near-perfect parabola without grinding. Computed
from $z(r)=z_0+\omega^2r^2/(2g)$ with $z_0=0.2\,$m.}
\label{fig:vol03ch10:rotating-paraboloid}
\end{figure}
