% Figure: hydrostatic pressure rising linearly with depth for three fluids
% (water, seawater, mercury). Gauge pressure p = rho g h. Slopes differ by
% density. All comments use = or words, never em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7.5cm,
  xlabel={gauge pressure $p=\rho g h$ (kPa)},
  ylabel={depth $h$ below surface (m)},
  xmin=0, xmax=140, ymin=0, ymax=10,
  y dir=reverse,
  axis lines=left,
  domain=0:10, samples=2,
  legend pos=south east,
  legend cell align=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% water: p(kPa) = 9.81*h
\addplot[engblue, very thick] coordinates {(0,0) (98.1,10)};
\addlegendentry{fresh water, $\rho=1000$}
% seawater: p = 10.06*h
\addplot[englime, very thick] coordinates {(0,0) (100.6,10)};
\addlegendentry{seawater, $\rho=1025$}
% mercury truncated to fit: slope 133.4 kPa/m, draw to depth ~1 m
\addplot[engred, very thick] coordinates {(0,0) (133.4,1)};
\addlegendentry{mercury, $\rho=13600$}
% one-atmosphere guide at 101.325 kPa
\draw[engamber, dashed, thin] (axis cs:101.325,0) -- (axis cs:101.325,10);
\node[engamber, font=\scriptsize, anchor=south, rotate=90] at (axis cs:101.325,5) {$1\,\mathrm{atm}$};
\end{axis}
\end{tikzpicture}
\caption[Pressure rises linearly with depth]{Gauge pressure
$p=\rho g h$ as a function of depth for three fluids. The slope is the
specific weight $\rho g$: water adds about $9.81\,\mathrm{kPa}$ per
metre, so a $10\,\mathrm{m}$ column adds one atmosphere (amber guide).
Seawater is slightly denser and rises faster. Mercury, $13.6$ times
denser, reaches one atmosphere in only $0.76\,\mathrm{m}$, which is why
a mercury barometer is short and a water barometer would be ten metres
tall. The figure carries the single rule of fluid statics: pressure is
linear in depth with slope set by density. Computed from $p=\rho g h$
with $g=9.81\,\mathrm{m/s^2}$.}
\label{fig:vol03ch10:pressure-depth}
\end{figure}
