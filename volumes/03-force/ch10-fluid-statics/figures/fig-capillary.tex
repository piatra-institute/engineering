% both against the outside free-surface level. The contact angle sets the sign:
% theta_c < 90 deg gives a concave meniscus and a rise; theta_c > 90 deg gives a
% convex meniscus and a depression. No em-dashes.

\centering
\begin{tikzpicture}[scale=1.0]

% outer bath
\fill[engblue!15] (0.0,0.0) rectangle (4.0,1.4);
\draw[engblack, thick] (0.0,0.0) -- (0.0,1.4);
\draw[engblack, thick] (4.0,0.0) -- (4.0,1.4);
\draw[engblack, thick] (0.0,0.0) -- (4.0,0.0);
\draw[engblue, thick] (0.0,1.4) -- (4.0,1.4);
\node[engblue, font=\scriptsize, anchor=east] at (0.0,1.4) {outside level};
% capillary tube walls
\draw[enggray, thick] (1.7,0.6) -- (1.7,3.6);
\draw[enggray, thick] (2.1,0.6) -- (2.1,3.6);
% raised column inside the tube
\fill[engblue!30] (1.7,0.6) rectangle (2.1,2.6);
% concave meniscus at top of the raised column
\draw[engblue, thick] (1.7,2.6) .. controls (1.9,2.42) .. (2.1,2.6);
% rise height marker
\draw[engred, thick, <->] (2.35,1.4) -- (2.35,2.55);
\node[engred, font=\scriptsize, anchor=west] at (2.4,1.97) {$h>0$};
\node[engblack, font=\small, anchor=south] at (1.9,3.6) {$\theta_c<90^{\circ}$};
\node[engblack, font=\scriptsize, anchor=north] at (2.0,-0.15) {wetting: rise};
% ---- right panel: non-wetting liquid (mercury in glass), depression ----
\begin{scope}[xshift=6.0cm]
\fill[engamber!30] (0.0,0.0) rectangle (4.0,1.4);
\draw[engblack, thick] (0.0,0.0) -- (0.0,1.4);

% depression marker
\draw[engred, thick, <->] (2.35,0.8) -- (2.35,1.4);
\node[engred, font=\scriptsize, anchor=west] at (2.4,1.1) {$h<0$};
\node[engblack, font=\small, anchor=south] at (1.9,3.2) {$\theta_c>90^{\circ}$};

\end{scope}

\caption[Capillary rise and depression]{The contact angle sets both the sign and
the size of the capillary height $h=2\sigma\cos\theta_c/(\rho g r)$. A wetting
liquid (left, water in glass, $\theta_c<90^{\circ}$) forms a concave meniscus
and climbs above the outside level, pulled up by the wall tension around the
circumference. A non-wetting liquid (right, mercury in glass,
$\theta_c>90^{\circ}$) forms a convex meniscus and is pushed below the outside
level, because $\cos\theta_c$ is now negative. The rise scales as the inverse of
the tube radius, so the effect is invisible in a wide vessel and dominant in a
fine bore, the crossover set by the capillary length $\ell_c=\sqrt{\sigma/(\rho
g)}$, about $2.7$ mm for water.}
\label{fig:vol03ch10:capillary}
\end{figure}
