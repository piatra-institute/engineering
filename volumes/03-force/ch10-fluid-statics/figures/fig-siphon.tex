% Figure: a siphon carrying liquid over a crest, with the pressure profile
% along the tube. The pressure falls to its minimum at the crest and must
% stay above the vapour pressure or the siphon breaks. No em-dashes.

\begin{tikzpicture}[scale=1.0]

% lower reservoir
\fill[engblue!14] (7.0,0.4) rectangle (9.2,1.6);
\draw[engblue, thick] (7.0,1.6) -- (9.2,1.6);
\node[engblue, font=\scriptsize, anchor=south] at (8.1,1.6) {delivery level};
% siphon tube: up over a crest, then down
\draw[engblack, very thick] (1.4,4.0) -- (1.4,5.3)
  .. controls (1.4,6.0) and (4.6,6.0) .. (4.6,6.0)

\caption[Siphon geometry and its low-pressure crest]{A siphon lifts
liquid from a supply level over a crest a height $h_c$ above the supply
surface and delivers it to a lower level, driven by the net drop
$\Delta z$ between the two free surfaces. Pressure falls along the tube
to a minimum at the crest, where the liquid has been raised highest
against gravity. The siphon runs only while that minimum stays above the
liquid's vapour pressure; raise the crest too high, or warm the liquid,
and the column cavitates at the crest and the flow breaks. The static
ceiling on $h_c$ is set by $p_{\mathrm{atm}}-p_{\mathrm{vap}}=\rho g
h_{c,\max}$, about ten metres of water at ordinary temperature.}
\label{fig:vol03ch10:siphon}
\end{figure}
