% Figure: hydrostatic pressure prism on a vertical wall of depth H. The
% pressure grows linearly from 0 at the surface to rho*g*H at the base; the
% triangular load has resultant 0.5*rho*g*H^2 per unit width acting at H/3
% above the base. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% wall (vertical line)
\draw[engblack, very thick] (0,0) -- (0,4);
\node[font=\small, anchor=east, rotate=90] at (-0.25,2) {wall};
% free surface
\draw[engblue, thick] (-0.6,4) -- (0,4);
\node[engblue, font=\scriptsize, anchor=south] at (-0.3,4) {free surface};
% pressure prism: triangle pointing right, base at bottom
\fill[engblue!25] (0,0) -- (0,4) -- (2.6,0) -- cycle;
\draw[engblue, thick] (0,4) -- (2.6,0);
% pressure arrows at several depths (proportional to depth)
\foreach \y/\len in {3.5/0.32, 2.8/0.71, 2.1/1.10, 1.4/1.49, 0.7/1.88, 0.1/2.53}{
  \draw[engred, ->] (0,\y) -- (\len,\y);
}
% labels
\node[font=\small, anchor=west] at (2.7,0) {$p=\rho g H$};
\node[font=\small, anchor=west] at (0.05,4.1) {$p=0$};
% H dimension on left
\draw[enggray, <->] (-0.9,0) -- (-0.9,4);
\node[enggray, font=\small, anchor=east] at (-0.95,2) {$H$};
% resultant arrow at H/3
\draw[engamber, ultra thick, ->] (0,1.333) -- (1.7,1.333);
\node[engamber, font=\small, anchor=west] at (1.75,1.5) {$F=\tfrac12\rho gH^2$};
\node[engamber, font=\scriptsize, anchor=west] at (1.75,1.05) {at $H/3$};
% H/3 marker
\draw[enggray, dashed] (-0.9,1.333) -- (0,1.333);
\end{tikzpicture}
\caption[The hydrostatic pressure prism]{The pressure distribution on a
vertical wall of depth $H$ holding back a fluid to its top. Gauge
pressure (red arrows) grows linearly from zero at the free surface to
$\rho gH$ at the base, sweeping out a triangular pressure prism (blue).
The resultant per unit width is the area of the triangle, $F=\tfrac12
\rho gH^2$ (amber), and it acts through the centroid of the triangle, at
$H/3$ above the base. Because the load grows as $H^2$ and its lever arm
grows as $H$, the overturning moment about the base grows as $H^3$: this
single scaling governs dams, basement walls, and tank walls alike.}
\label{fig:vol03ch10:pressure-prism}
\end{figure}
