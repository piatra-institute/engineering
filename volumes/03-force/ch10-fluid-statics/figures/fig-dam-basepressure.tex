% Figure: base-pressure distribution under a gravity dam for two cases. The
% trapezoidal (no-tension) case keeps the resultant inside the middle third;
% the triangular case has the resultant at the kern edge; the heel-tension case
% (resultant outside the middle third) lifts the heel and is disallowed.
% Plotted as a small bar/axes diagram. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
\begin{axis}[
  width=11cm, height=5.4cm,
  xlabel={position across base (heel $\to$ toe), $x/B$},
  ylabel={base pressure (relative)},
  xmin=0, xmax=1, ymin=0, ymax=2.4,
  xtick={0,0.5,1}, xticklabels={$0$ (heel),$0.5$,$1$ (toe)},
  ytick={0,1,2},
  legend pos=north west,
  legend style={font=\scriptsize, draw=enggray},
  tick label style={font=\scriptsize},
  label style={font=\scriptsize},
  axis lines=left,
]
% Case 1: resultant at centroid -> uniform-ish trapezoid (safe, no tension)
\addplot[engblue, very thick] coordinates {(0,0.7) (1,1.3)};
\addlegendentry{resultant in middle third (no tension)}
% Case 2: resultant at kern edge -> triangular, zero at heel
\addplot[engamber, very thick, dashed] coordinates {(0,0.0) (1,2.0)};
\addlegendentry{resultant at kern edge (zero at heel)}
% Case 3: resultant outside middle third -> heel would go into tension (disallowed)
% concrete cannot take tension, so the real pressure block is truncated; show the
% notional negative branch lightly to mark the disallowed state.
\addplot[engred, very thick, dotted] coordinates {(0.25,0.0) (1,2.3)};
\addlegendentry{resultant outside (heel lifts, disallowed)}
\end{axis}
\end{tikzpicture}
\caption[Base-pressure distribution under a gravity dam]{Base-pressure
distribution under the heel-to-toe width of a gravity dam, for three
positions of the resultant of self-weight and uplift. When the resultant
falls within the middle third of the base (the kern), the pressure stays
compressive everywhere (blue trapezoid). When it reaches the kern edge,
the pressure is triangular and just touches zero at the heel (amber). When
it falls outside the middle third, the heel would be driven into tension
(red); concrete masonry cannot carry tension across a horizontal joint, so
the heel lifts and the contact area shrinks, raising the toe pressure
further. The no-tension rule, resultant within the middle third, is the
base-pressure check in the dam stability calculation.}
\label{fig:vol03ch10:dam-basepressure}
\end{figure}
