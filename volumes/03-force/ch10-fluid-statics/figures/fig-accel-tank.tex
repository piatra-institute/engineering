% Figure: liquid in a closed rectangular tank under steady horizontal
% acceleration a. The constant-pressure surfaces tilt by alpha = atan(a/g)
% toward the rear; the effective gravity is the vector sum of g (down) and the
% inertial reaction -a (rearward). Shows the tilted free surface and the
% effective-gravity vector. Schematic; no axis environment. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.05]
% tank walls (closed rectangular tank), motion direction to the right
\draw[engblack, very thick] (-2.6,0) rectangle (2.6,2.2);
% direction-of-acceleration arrow above the tank
\draw[engorange, ultra thick, ->] (-1.0,2.7) -- (1.0,2.7);
\node[engorange, font=\small, anchor=south] at (0,2.75) {acceleration $a$};
% tilted free surface: high at the rear (left), low at the front (right)
% tilt angle alpha; here drawn with a modest slope for legibility
\fill[engblue!22] (-2.6,0) -- (2.6,0) -- (2.6,0.7) -- (-2.6,1.5) -- cycle;
\draw[engblue, thick] (-2.6,1.5) -- (2.6,0.7);
\node[engblue, font=\scriptsize, anchor=south east] at (-1.0,1.35) {tilted surface};
% rear and front labels
\node[engblack, font=\scriptsize, anchor=north] at (-2.6,-0.05) {rear};
\node[engblack, font=\scriptsize, anchor=north] at (2.6,-0.05) {front};
% effective-gravity construction at a point inside the liquid
\coordinate (P) at (0.6,0.55);
\fill[engblack] (P) circle (1.3pt);
% gravity component (down)
\draw[engred, thick, ->] (P) -- ($(P)+(0,-0.95)$);
\node[engred, font=\scriptsize, anchor=west] at ($(P)+(0.03,-0.55)$) {$g$};
% inertial reaction component (rearward, i.e. to the left), magnitude a
\draw[engred, thick, ->] (P) -- ($(P)+(-0.85,0)$);
\node[engred, font=\scriptsize, anchor=south] at ($(P)+(-0.45,0.02)$) {$a$};
% resultant effective gravity g_eff (down and rearward)
\draw[engamber, ultra thick, ->] (P) -- ($(P)+(-0.85,-0.95)$);
\node[engamber, font=\scriptsize, anchor=north west] at ($(P)+(-0.85,-0.95)$) {$g_{\mathrm{eff}}$};
% tilt angle marker between vertical and g_eff
\node[engamber, font=\scriptsize] at ($(P)+(-0.18,-0.55)$) {$\alpha$};
\end{tikzpicture}
\caption[Liquid in an accelerating tank]{Liquid in a closed rectangular
tank under steady horizontal acceleration $a$ (orange, to the right). In
the tank frame each fluid element feels an effective gravity
$\vec g_{\mathrm{eff}}=\vec g-\vec a$, the vector sum of gravity $g$
downward and an inertial reaction $a$ pointing toward the rear (amber). The
constant-pressure surfaces, including the free surface (blue), stand normal
to $\vec g_{\mathrm{eff}}$ and so tilt from horizontal by
$\alpha=\arctan(a/g)$, rising toward the rear. The pressure rises along
$\vec g_{\mathrm{eff}}$ at the rate $\rho\,|\vec g_{\mathrm{eff}}|$, so the
rear-bottom corner carries the highest wall load. In a closed full tank the
free surface is replaced by the same family of tilted isobars, and the rear
wall still carries the larger thrust.}
\label{fig:vol03ch10:accel-tank}
\end{figure}
