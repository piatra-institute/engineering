% Figure: metacentric stability of a heeled floating body. Shows the body
% heeled by a small angle; the centre of buoyancy B shifts to B'; the line of
% action of buoyancy meets the centreline at the metacentre M. With M above G
% the couple is restoring. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.1]
% waterline (horizontal)
\draw[engblue, thick] (-2.6,0) -- (2.6,0);
\node[engblue, font=\scriptsize, anchor=west] at (2.1,-0.18) {waterline};
% heeled rectangular hull: rotate a box by 12 degrees about its centre
\begin{scope}[rotate around={12:(0,0)}]
  % hull cross-section (box), submerged part below original waterline
  \draw[engblack, very thick] (-1.4,-1.0) rectangle (1.4,1.2);
  % centreline of the body (vertical in body frame)
  \draw[enggray, dashed] (0,-1.0) -- (0,2.4);
  % centre of mass G on centreline
  \fill[engred] (0,0.35) circle (1.8pt);
  \node[engred, font=\small, anchor=west] at (0.08,0.45) {$G$};
\end{scope}
% submerged-volume centroid B' shifts to the low (down-heel) side
\fill[englime] (-0.42,-0.42) circle (1.8pt);
\node[englime, font=\small, anchor=north east] at (-0.42,-0.42) {$B'$};
% vertical line of action of buoyancy through B' (true vertical)
\draw[englime, thick] (-0.42,-0.42) -- (-0.42+0.32, 2.1);
% metacentre M where buoyancy line meets body centreline (above G)
\fill[engamber] (0.10,1.45) circle (2.0pt);
\node[engamber, font=\small, anchor=west] at (0.16,1.5) {$M$};
% GM bracket
\draw[engamber, <->] (0.55,0.55) -- (0.55,1.4);
\node[engamber, font=\scriptsize, anchor=west] at (0.6,1.0) {$\overline{GM}>0$};
% restoring couple arrows
\draw[engorange, ->] (1.5,0.0) arc (0:55:1.5);
\node[engorange, font=\scriptsize, anchor=south] at (1.2,1.4) {restoring};
\end{tikzpicture}
\caption[Metacentric stability of a heeled hull]{A floating body heeled
by a small angle. The centre of buoyancy moves to the low side, to $B'$
(green), because more volume is submerged there. The vertical line of
action of the buoyancy force crosses the body centreline at the
metacentre $M$ (amber). When $M$ lies above the centre of mass $G$
(red), the metacentric height $\overline{GM}=I_{\mathrm{wp}}/
V_{\mathrm{disp}}-\overline{BG}$ is positive and the buoyancy couple
rotates the body back toward upright (orange). When $M$ falls below $G$
the same couple drives the body further over and it capsizes. A wide
hull (large waterplane second moment $I_{\mathrm{wp}}$) is stable; a
tall hull with mass carried high is not.}
\label{fig:vol03ch10:metacentre}
\end{figure}
