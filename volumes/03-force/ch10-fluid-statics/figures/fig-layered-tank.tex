% Figure: a tank holding three immiscible layers (oil over water over a brine
% sediment) with the resulting piecewise-linear gauge-pressure profile. The slope
% of each segment is the local specific weight rho*g, so the profile kinks
% downward at each interface as the denser fluid takes over. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% left panel: the tank with three layers
% tank outline
\draw[engblack, very thick] (-0.2,0) -- (-0.2,4.2) (1.8,0) -- (1.8,4.2);
\draw[engblack, very thick] (-0.2,0) -- (1.8,0);
% oil layer: top, z from 3.0 to 4.2 (1.2 m thick), lightest
\fill[engamber!20] (-0.2,3.0) rectangle (1.8,4.2);
\node[engamber, font=\scriptsize] at (0.8,3.6) {oil};
% water layer: middle, z from 1.0 to 3.0 (2.0 m thick)
\fill[engblue!18] (-0.2,1.0) rectangle (1.8,3.0);
\node[engblue, font=\scriptsize] at (0.8,2.0) {water};
% brine layer: bottom, z from 0 to 1.0 (1.0 m thick), densest
\fill[englime!25] (-0.2,0) rectangle (1.8,1.0);
\node[englime!60!engblack, font=\scriptsize] at (0.8,0.5) {brine};
% interface lines
\draw[engblack, thin] (-0.2,4.2) -- (1.8,4.2);
\node[engblack, font=\scriptsize, anchor=south] at (0.8,4.2) {free surface};
\draw[engblack, thin, dashed] (-0.2,3.0) -- (1.8,3.0);
\draw[engblack, thin, dashed] (-0.2,1.0) -- (1.8,1.0);
% right panel: the gauge-pressure profile p(depth), depth increasing downward
% reuse the same vertical scale; pressure increases to the right
% axis
\draw[enggray, ->] (3.0,4.2) -- (3.0,-0.2);
\node[enggray, font=\scriptsize, anchor=south] at (3.0,4.25) {$z$};
\draw[enggray, ->] (3.0,4.2) -- (6.4,4.2);
\node[enggray, font=\scriptsize, anchor=west] at (6.4,4.2) {$p_{\text{gauge}}$};
% segment 1 (oil): from (p=0 at z=4.2) to (p1 at z=3.0); gentle slope
% segment 2 (water): from (p1 at z=3.0) to (p2 at z=1.0); steeper
% segment 3 (brine): from (p2 at z=1.0) to (p3 at z=0); steepest
\coordinate (A) at (3.0,4.2);     % surface, p=0
\coordinate (B) at (3.55,3.0);    % after oil
\coordinate (C) at (5.0,1.0);     % after water
\coordinate (D) at (5.95,0.0);    % bottom, after brine
\draw[engamber, ultra thick] (A) -- (B);
\draw[engblue, ultra thick] (B) -- (C);
\draw[englime!60!engblack, ultra thick] (C) -- (D);
\fill[engblack] (B) circle (1.4pt) (C) circle (1.4pt) (D) circle (1.4pt);
\node[engblack, font=\scriptsize, anchor=west] at (B) {$p_1$};
\node[engblack, font=\scriptsize, anchor=west] at (C) {$p_2$};
\node[engblack, font=\scriptsize, anchor=west] at (D) {$p_3$};
\end{tikzpicture}
\caption[Pressure profile in a layered tank]{Left: a tank holding three
immiscible fluids, lightest (oil) on top, densest (brine) at the bottom.
Right: the gauge-pressure profile against height. Pressure rises with depth
at the local specific weight $\rho g$, so the profile is piecewise linear and
steepens at each interface as a denser fluid takes over. The pressure at the
base is the sum of the three column contributions,
$p_3=\rho_{\text{oil}}g\,h_{\text{oil}}+\rho_{\text{water}}g\,h_{\text{water}}+\rho_{\text{brine}}g\,h_{\text{brine}}$,
the bookkeeping that a path trace performs one layer at a time.}
\label{fig:vol03ch10:layered-tank}
\end{figure}
