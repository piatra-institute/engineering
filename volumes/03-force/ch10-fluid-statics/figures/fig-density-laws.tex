% Figure: one hydrostatic integral, three density laws. Pressure versus vertical
% coordinate for the incompressible liquid (constant density, linear p), the
% slightly compressible ocean (rising density, faintly super-linear p), and the
% isothermal atmosphere (falling density, exponential p). Curves normalised for
% shape comparison. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
\begin{axis}[
  width=12cm, height=6.2cm,
  xlabel={depth below surface / normalised},
  ylabel={pressure / normalised},
  xmin=0, xmax=1, ymin=0, ymax=1.15,
  xtick={0,0.25,0.5,0.75,1.0},
  ytick={0,0.25,0.5,0.75,1.0},
  tick label style={font=\scriptsize},
  label style={font=\scriptsize},
  legend pos=north west,
  legend style={font=\scriptsize, draw=enggray},
  axis lines=left,
  samples=120, domain=0:1,
]
% Incompressible liquid: p linear in depth. p = x.
\addplot[engblue, very thick] ({x}, {x});
\addlegendentry{incompressible liquid ($\rho$ const)}
% Slightly compressible ocean: density rises with depth, p faintly super-linear.
% Model p = x + 0.12*x^2 (curvature exaggerated for visibility), renormalised.
\addplot[englime, very thick] ({x}, {(x + 0.12*x*x)/1.12});
\addlegendentry{compressible ocean ($\rho$ rising)}
% Isothermal atmosphere: here read x as height, p = exp(-k x). Falling density.
% Plot as decreasing curve normalised to 1 at x=0. Use k=2.3 so it drops to ~0.1.
\addplot[engamber, very thick] ({x}, {exp(-2.3*x)});
\addlegendentry{atmosphere ($\rho$ falling)}
\end{axis}
\end{tikzpicture}
\caption[One integral, three density laws]{The single hydrostatic integral
$p=p_0+g\int\rho\,dz$ produces three shapes according to how the density behaves.
An incompressible liquid holds a constant density and its pressure rises linearly
with depth (blue). The deep ocean compresses under its own load, so the density
rises with depth and the pressure curves faintly above the straight line (green,
curvature exaggerated for visibility). The atmosphere expands as the pressure
drops, so read against height its density falls and the pressure decays
exponentially (amber). Identify the element, write the balance, integrate: only
the density law inside the integral changes across the three cases.}
\label{fig:vol03ch10:density-laws}
\end{figure}
