% Figure: an inclined rectangular gate hinged at its top edge, holding back
% water. Shows the gate inclined at angle theta, the centroid depth h_C, and
% the centre of pressure below the centroid. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% free surface
\draw[engblue, thick] (-0.5,4) -- (5.5,4);
\node[engblue, font=\scriptsize, anchor=south] at (-0.3,4) {free surface};
% water region shading (left of gate)
\fill[engblue!12] (-0.5,0) rectangle (5.5,4);
% gate inclined at 60 deg, hinge at top edge (1, 3.4), going down-right
% top edge depth h_T below surface; gate length b along the slope
\coordinate (T) at (1.5,3.4);     % top edge (hinge)
\coordinate (B) at (3.3,0.28);    % bottom edge
\draw[engblack, ultra thick] (T) -- (B);
\node[engblack, font=\small, anchor=south west] at (T) {hinge};
% gate centroid
\coordinate (C) at (2.4,1.84);
\fill[engred] (C) circle (1.6pt);
\node[engred, font=\scriptsize, anchor=west] at (C) {centroid $h_C$};
% centre of pressure below centroid along the gate
\coordinate (P) at (2.65,1.41);
\fill[engamber] (P) circle (1.6pt);
\node[engamber, font=\scriptsize, anchor=north west] at (P) {$\!$c.\,of pressure $h_P$};
% resultant force perpendicular to gate (pointing into gate, to lower-left)
\draw[engamber, ultra thick, ->] (P) -- ($(P)+(-0.86,-0.5)$);
\node[engamber, font=\small, anchor=east] at ($(P)+(-0.86,-0.5)$) {$F=p_CA$};
% depth dimension h_C
\draw[enggray, dashed] (C) -- (5.0,1.84);
\draw[enggray, <->] (5.0,4) -- (5.0,1.84);
\node[enggray, font=\scriptsize, anchor=west] at (5.0,2.9) {$h_C$};
% angle theta at top edge with horizontal
\draw[engblack] ($(T)+(0.6,0)$) arc (0:-60:0.6);
\node[font=\scriptsize] at ($(T)+(0.55,-0.45)$) {$\theta$};
\draw[enggray, dashed] (T) -- ($(T)+(1.0,0)$);
\end{tikzpicture}
\caption[Inclined submerged gate]{A rectangular gate hinged at its top
edge, inclined at angle $\theta$ to the horizontal, holding back water.
The resultant hydrostatic force equals the pressure at the gate's
centroid times the gate area, $F=p_CA=\rho gh_CA$, and acts
perpendicular to the gate. Its line of action, the centre of pressure
(amber), lies below the centroid (red) because pressure increases with
depth; the offset is the second-moment correction
$I_{xx,C}\sin\theta/(h_CA)$ measured along the gate. The hinge reaction
and the latch reaction together carry $F$ and the moment it produces
about the hinge.}
\label{fig:vol03ch10:gate}
\end{figure}
