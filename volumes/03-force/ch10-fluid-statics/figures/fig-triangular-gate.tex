% Figure: two triangular gates of equal area and equal submergence, one apex
% up and one apex down, showing that the apex-down triangle carries its load
% deeper and so takes a larger force and a lower centre of pressure. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% common free surface
\draw[engblue, thick] (-0.5,5.0) -- (9.5,5.0);
\node[engblue, font=\scriptsize, anchor=south] at (-0.3,5.0) {free surface};
\fill[engblue!8] (-0.5,0.4) rectangle (9.5,5.0);
% Apex-up triangle (left): base at top, apex at bottom
\fill[engamber!25, draw=engblack, thick] (0.6,4.2) -- (3.4,4.2) -- (2.0,1.4) -- cycle;
\node[font=\scriptsize, anchor=south] at (2.0,4.2) {base up};
% centroid at one-third of height below the base (apex-up: centroid nearer the base)
\fill[engred] (2.0,3.27) circle (1.6pt);
\node[engred, font=\scriptsize, anchor=west] at (2.1,3.27) {centroid};
\node[font=\scriptsize, anchor=north] at (2.0,1.4) {apex-up gate};
% Apex-down triangle (right): base at bottom, apex at top
\fill[engamber!25, draw=engblack, thick] (6.6,1.4) -- (9.4,1.4) -- (8.0,4.2) -- cycle;
\node[font=\scriptsize, anchor=north] at (8.0,1.4) {base down};
% centroid at one-third of height above the base (apex-down: centroid nearer the base, deeper)
\fill[engred] (8.0,2.33) circle (1.6pt);
\node[engred, font=\scriptsize, anchor=west] at (8.1,2.33) {centroid};
\node[font=\scriptsize, anchor=south] at (8.0,4.2) {apex-down gate};
% depth guides
\draw[enggray, dashed] (2.0,3.27) -- (5.3,3.27);
\draw[enggray, dashed] (8.0,2.33) -- (5.3,2.33);
\draw[enggray, <->] (5.3,5.0) -- (5.3,3.27);
\node[enggray, font=\scriptsize, anchor=east] at (5.3,4.1) {$h_{C,\mathrm{up}}$};
\draw[enggray, <->] (5.7,5.0) -- (5.7,2.33);
\node[enggray, font=\scriptsize, anchor=west] at (5.7,3.6) {$h_{C,\mathrm{dn}}$};
\end{tikzpicture}
\caption[Triangular gates, apex up and apex down]{Two triangular gates of
equal area and equal top submergence carry different loads because the
distribution of area with depth differs. The apex-down gate puts its wide
base in the deeper, higher-pressure water, so its centroid sits deeper
($h_{C,\mathrm{dn}}>h_{C,\mathrm{up}}$), it takes the larger resultant
$F=\rho g h_C A$, and its centre of pressure lies lower. The centroid of a
triangle is one-third of the height from the base toward the apex; reading
which edge is the base fixes the centroidal depth, and the same $p_CA$ and
second-moment machinery gives the force and its line of action for either
orientation.}
\label{fig:vol03ch10:triangular-gate}
\end{figure}
