% Figure: how the centre-of-pressure offset below the centroid shrinks as a plate
% is lowered to greater depth. Same rectangular plate at three centroidal depths;
% the offset I/(h_C A) falls as 1/h_C, so a shallow plate loads well below its
% centroid and a deep one loads almost through it. No em-dashes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
\begin{axis}[
  width=11cm, height=6.0cm,
  xlabel={centroidal depth $h_C$ (m)},
  ylabel={offset $h_P-h_C$ (m)},
  xmin=0.5, xmax=8, ymin=0, ymax=0.65,
  xtick={1,2,3,4,5,6,7,8},
  ytick={0,0.2,0.4,0.6},
  tick label style={font=\scriptsize},
  label style={font=\scriptsize},
  legend pos=north east,
  legend style={font=\scriptsize, draw=enggray},
  axis lines=left,
  samples=120, domain=0.75:8,
]
% offset = I_C/(h_C A) = (b^2/12)/h_C for a rectangle of height b, since I_C=w b^3/12
% and A = w b, so I_C/A = b^2/12. Plot for three plate heights b.
\addplot[engblue, very thick] ({x}, {(3*3)/12/x});
\addlegendentry{$b=3$ m}
\addplot[engamber, very thick] ({x}, {(2*2)/12/x});
\addlegendentry{$b=2$ m}
\addplot[englime, very thick] ({x}, {(1*1)/12/x});
\addlegendentry{$b=1$ m}
\node[enggray, font=\scriptsize, anchor=north east] at (axis cs:7.8,0.6)
  {offset $=\dfrac{b^2}{12\,h_C}$};
\end{axis}
\end{tikzpicture}
\caption[Centre-of-pressure offset versus depth]{The centre of pressure lies
below the centroid of a submerged plate by $I_{xx,C}/(h_C A)$, which for a
rectangle of height $b$ reduces to $b^2/(12\,h_C)$, independent of width. The
offset falls as the inverse of the centroidal depth: a large plate near the
surface (upper left) throws its resultant well below its geometric centre,
while the same plate lowered to great depth (right) loads almost through its
centroid because the pressure field across it is nearly uniform. The three
curves are plates of height $1$, $2$, and $3$ metres; the offset grows as the
square of the plate size and shrinks as the depth, the scaling that decides
where a gate's hinge and framing must carry the load.}
\label{fig:vol03ch10:cp-shift}
\end{figure}
