% Figure: Tresca and von Mises yield loci in the plane-stress (sigma_1, sigma_2)
% plane, normalized by sigma_y. Von Mises is an ellipse, Tresca an inscribed
% hexagon. Shows the two agree at uniaxial points and differ most in biaxial/
% pure-shear states.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10cm, height=10cm,
  axis equal image,
  xlabel={$\sigma_1/\sigma_y$}, ylabel={$\sigma_2/\sigma_y$},
  xmin=-1.5, xmax=1.5, ymin=-1.5, ymax=1.5,
  axis lines=middle,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.02,0.02)}, anchor=south west},
]
% von Mises ellipse in plane stress: s1^2 - s1 s2 + s2^2 = 1
% parametrize: rotate coordinates. Use samples of the implicit curve via param.
% Standard parametrization: s1 = (2/sqrt3) cos(t - pi/6), s2 = (2/sqrt3) cos(t + pi/6)?
% Simpler: ellipse with axes along (1,1) and (1,-1).
% s1^2 - s1 s2 + s2^2 = 1. Let u=(s1+s2)/sqrt2, v=(s1-s2)/sqrt2.
% Then s1^2+s2^2 = u^2+v^2, s1 s2 = (u^2 - v^2)/2.
% => u^2+v^2 - (u^2-v^2)/2 = 1 => u^2/2 + 3v^2/2 = 1 => u^2/2 + v^2/(2/3)=1
% semi-axis along u: sqrt2 ; along v: sqrt(2/3).
\addplot[engblue, very thick, samples=120, domain=0:360]
  ({ (sqrt(2)*cos(x))/sqrt(2) + (sqrt(2/3)*sin(x))/sqrt(2) },
   { (sqrt(2)*cos(x))/sqrt(2) - (sqrt(2/3)*sin(x))/sqrt(2) });
\addlegendentry{von Mises}
% Tresca hexagon vertices (plane stress): (1,0),(1,1),(0,1),(-1,0),(-1,-1),(0,-1)
\addplot[engred, very thick] coordinates {
  (1,0) (1,1) (0,1) (-1,0) (-1,-1) (0,-1) (1,0)
};
\addlegendentry{Tresca}
% mark uniaxial points where they coincide
\filldraw[enggray] (axis cs:1,0) circle [radius=1.4pt];
\filldraw[enggray] (axis cs:0,1) circle [radius=1.4pt];
\end{axis}
\end{tikzpicture}
\caption[Tresca and von Mises yield loci in plane stress]{Yield loci in
the plane-stress $(\sigma_1,\sigma_2)$ plane, normalised by the uniaxial
yield stress $\sigma_y$. The von Mises criterion (blue) is the ellipse
$\sigma_1^2-\sigma_1\sigma_2+\sigma_2^2=\sigma_y^2$; the Tresca criterion
(red) is the hexagon inscribed in it. The two coincide at the uniaxial
points $(\pm1,0)$ and $(0,\pm1)$ and at equibiaxial tension; they differ
most in pure shear ($\sigma_1=-\sigma_2$), where Tresca predicts yield at
$\sigma_1=\sigma_y/2$ and von Mises at $\sigma_1=\sigma_y/\sqrt3\approx
0.577\,\sigma_y$. Tresca lies inside von Mises everywhere, which is why it
is the conservative choice.}
\label{fig:vol03ch03:yield-loci}
\end{figure}
