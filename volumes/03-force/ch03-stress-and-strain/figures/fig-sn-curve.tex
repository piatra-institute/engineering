% Figure: the S-N (Wohler) curve. Alternating stress amplitude versus
% cycles to failure on a semi-log axis (log cycles). Two curves: a
% ferrous steel that shows a distinct endurance limit (the curve flattens
% to a horizontal asymptote near 1e6 cycles), and an aluminium alloy that
% has no true endurance limit (the curve keeps falling). The endurance
% limit S_e is marked. The high-cycle regime is the province of the
% fatigue checks in the chapter's vignettes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={cycles to failure $N$ (log scale)},
  ylabel={alternating stress amplitude $\sigma_a$ (MPa)},
  xmode=log,
  xmin=1e3, xmax=1e9, ymin=0, ymax=520,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
  log basis x=10,
]
% Steel: high-cycle branch falling from ~460 at 1e3 to a knee at ~180 near
% 1e6, then flat (endurance limit). Build from explicit points.
\addplot[engblue, very thick] coordinates {
  (1e3,460) (1e4,360) (1e5,255) (5e5,200) (1e6,180) (1e7,180) (1e8,180) (1e9,180)
};
\addlegendentry{steel (endurance limit)}
% Aluminium: no true limit, keeps falling.
\addplot[engred, very thick] coordinates {
  (1e3,340) (1e4,270) (1e5,205) (1e6,155) (1e7,120) (1e8,95) (1e9,78)
};
\addlegendentry{aluminium alloy (no limit)}
% endurance-limit line for steel
\addplot[engblue, dashed, samples=2, domain=1e3:1e9] {180};
\node[engblue, font=\scriptsize, anchor=south west] at (axis cs:2e6,182) {$S_e = 180$ MPa};
% knee marker
\filldraw[engorange] (axis cs:1e6,180) circle[radius=1.8pt];
\node[engorange, font=\scriptsize, anchor=south] at (axis cs:1e6,192) {knee $\approx 10^6$};
\end{axis}
\end{tikzpicture}
\caption[The S-N (W\"ohler) fatigue curve]{Alternating stress amplitude
against cycles to failure for a steel and an aluminium alloy, on a
logarithmic cycle axis. The steel curve flattens to a horizontal
endurance limit near $10^6$ cycles: below that stress the steel runs
indefinitely, which is the design target of an infinite-life fatigue
check. The aluminium curve has no such plateau; it keeps falling, so an
aluminium part is designed to a finite life at a chosen number of cycles
rather than to an endurance limit. The endurance limit that the padeye
and bolted-joint vignettes compare their alternating stress against is the
flat part of the steel curve, already corrected for surface finish, size,
and reliability.}
\label{fig:vol03ch03:sn-curve}
\end{figure}
