% Figure: hoop and radial stress through the thickness of a spinning
% solid disk, plotted against radius. Both stresses peak at the centre
% (where they are equal); the hoop stress stays high to the rim while the
% radial stress falls to zero at the free outer edge. Curves are the
% closed-form rotating-disk solution for nu = 0.3 evaluated at the case-
% study speed. Outer radius b = 0.25 m, density 7800 kg/m^3, omega from
% 6000 rpm = 628 rad/s. Centre stresses near 144 MPa; rim hoop near
% 47 MPa. Numbers match the worked case study in the prose.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={radius $r$ (m)},
  ylabel={stress (MPa)},
  xmin=0, xmax=0.25, ymin=0, ymax=170,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
  samples=80, domain=0:0.25,
]
% Solid spinning disk, nu = 0.3, rho omega^2 = 7800 * 628^2 = 3.076e9 Pa/m^2.
% sigma_theta(r) = (rho omega^2 /8)[(3+nu) b^2 - (1+3nu) r^2]
%   prefactor = 3.076e9/8 = 3.845e8; (3+nu)=3.3; (1+3nu)=1.9; b^2=0.0625
%   in MPa: 0.3845*(3.3*0.0625 - 1.9*r^2)*1e3  => keep as direct MPa expression
% sigma_r(r) = (rho omega^2 /8)(3+nu)(b^2 - r^2)
\addplot[engblue, very thick] {384.5*(3.3*0.0625 - 1.9*x*x)};
\addlegendentry{hoop stress $\sigma_\theta$}
\addplot[engred, very thick] {384.5*3.3*(0.0625 - x*x)};
\addlegendentry{radial stress $\sigma_r$}
% mark the centre value (both equal) and the rim hoop value
\filldraw[engorange] (axis cs:0,79.3) circle[radius=2pt];
\node[engorange, font=\scriptsize, anchor=south west] at (axis cs:0.005,80) {centre $\sigma_\theta=\sigma_r$};
\filldraw[englime!60!black] (axis cs:0.25,33.6) circle[radius=2pt];
\node[englime!60!black, font=\scriptsize, anchor=south east] at (axis cs:0.245,35) {rim hoop};
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:0.245,2) {$\sigma_r=0$ at free rim};
\end{axis}
\end{tikzpicture}
\caption[Hoop and radial stress in a spinning solid disk]{Hoop and
radial stress through a spinning solid disk of outer radius $b = 0.25\
\text{m}$ turning at the case-study speed, from the closed-form
rotating-disk solution with $\nu = 0.3$. Both stresses reach their
maximum at the centre, where they are equal; the radial stress falls to
zero at the free outer edge while the hoop stress stays high to the rim.
The centre is the governing point for a solid disk, which is why a
central bore (a bolt hole or shaft hole) is the worst thing one can put
in a flywheel: it sits exactly where the stress peaks and concentrates
it further.}
\label{fig:vol03ch03:rotating-disk-stresses}
\end{figure}
