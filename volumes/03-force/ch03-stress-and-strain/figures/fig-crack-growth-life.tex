% Figure: crack depth versus number of pressurisation cycles, from
% integrating the Paris law da/dN = C (Delta K)^m for the case study.
% Closed form for m=3 with a constant geometry factor Y and constant
% stress range: a(N) = (a0^{-1/2} - (1/2) P N)^{-2}, with P = C(Y dsig
% sqrt(pi))^3. Numbers: a0=1.5 mm, B=18 mm, Y=1.12, dsig=90 MPa,
% C=6.9e-12 m/cycle. b0 = a0^{-1/2} = 25.82 m^{-1/2}; the per-thousand-
% cycle decrement of a^{-1/2} is (1/2)P*1000 = 0.019675 m^{-1/2}. So with
% x = cycles in thousands, a(x) in mm = 1000*(25.82 - 0.019675 x)^{-2}.
% Leak (a=B=18 mm) at x = 933 thousand cycles. The crack grows slowly
% at first and accelerates as Delta K rises with sqrt(a).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={pressurisation cycles $N$ (thousands)},
  ylabel={crack depth $a$ (mm)},
  xmin=0, xmax=950, ymin=0, ymax=20,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
]
% a(x) in mm = 1000*(25.82 - 0.019675 x)^{-2}, x in thousands of cycles.
\addplot[engblue, very thick, samples=160, domain=0:932]
  {1000/((25.82 - 0.019675*x)*(25.82 - 0.019675*x))};
\addlegendentry{Paris-law integration ($m=3$)}
% wall thickness limit (leak)
\addplot[englime!60!black, dashed, very thick, samples=2, domain=0:950] {18};
\addlegendentry{wall thickness $B$ (leak)}
% initial flaw level
\addplot[enggray, dotted, thick, samples=2, domain=0:950] {1.5};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:14,2.0) {initial flaw $a_0 = 1.5$ mm};
% leak point marker at x = 933
\filldraw[engred] (axis cs:933,18) circle[radius=2.4pt];
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:930,18.2)
  {leak at $\approx 9.3\times 10^{5}$ cycles};
% mark the 64% line: a=5mm reached near x=593k
\filldraw[engorange] (axis cs:593,5) circle[radius=2pt];
\node[engorange, font=\scriptsize, anchor=north west] at (axis cs:470,5.6)
  {$\sim 64\%$ of life below $a = 5$ mm};
% annotate the accelerating region
\node[engblue, font=\scriptsize, anchor=west, rotate=62] at (axis cs:855,8)
  {$\Delta K \propto \sqrt{a}$};
\end{axis}
\end{tikzpicture}
\caption[Crack depth versus cycles from the Paris law]{Crack depth
against pressurisation cycles for the case study, from integrating the
Paris law $da/dN = C(\Delta K)^m$ with $m = 3$ from the initial flaw
$a_0 = 1.5\ \text{mm}$. The growth is slow at first and accelerates as
the crack deepens, because the stress-intensity range $\Delta K$ rises
with $\sqrt{a}$ and the growth rate rises with its cube. The crack
reaches the full wall thickness $B = 18\ \text{mm}$, and so leaks, at
roughly $9.3\times 10^{5}$ cycles; the critical depth for fast fracture
lies well beyond the wall, so the vessel leaks before it breaks. About
two thirds of the life passes while the crack is below $5\ \text{mm}$
deep, which is why early detection of a small flaw buys so much remaining
life and a flaw found late, in the steep part of the curve, leaves little
life to act on.}
\label{fig:vol03ch03:crack-growth-life}
\end{figure}
