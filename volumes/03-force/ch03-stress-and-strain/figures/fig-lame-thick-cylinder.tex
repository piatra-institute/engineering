% Figure: the Lame thick-walled-cylinder stress field. Plots the radial
% and hoop stress through the wall of a thick cylinder under internal
% pressure, normalised to the internal pressure. The hoop stress is
% largest at the bore and always exceeds the pressure; the radial stress
% runs from minus the pressure at the bore to zero at the free outer
% surface. The dashed line marks the thin-wall pr/t estimate for contrast.
% Geometry: a/b = 0.5 (bore radius half the outer radius), so the
% coordinate x = r/a runs from 1 (bore) to 2 (outer surface).
% Lame constants for internal pressure p_i, zero outer pressure:
%   A = p a^2/(b^2-a^2),  B = p a^2 b^2/(b^2-a^2)
% with a=1, b=2:  A = p/3,  B = 4p/3.
%   sigma_r(x)     = A - B/x^2 = p(1/3 - (4/3)/x^2)
%   sigma_theta(x) = A + B/x^2 = p(1/3 + (4/3)/x^2)
% Bore hoop = p(1/3+4/3) = 5/3 p; outer hoop = p(1/3+1/3) = 2/3 p.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={radius through the wall, $r/a$},
  ylabel={stress, in units of internal pressure $p$},
  xmin=1, xmax=2, ymin=-1.2, ymax=2.0,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
  domain=1:2, samples=80,
]
% hoop stress: p(1/3 + (4/3)/x^2)
\addplot[engblue, very thick] {1/3 + (4/3)/(x*x)};
\addlegendentry{hoop $\sigma_\theta/p = A + B/r^2$}
% radial stress: p(1/3 - (4/3)/x^2)
\addplot[engred, very thick] {1/3 - (4/3)/(x*x)};
\addlegendentry{radial $\sigma_r/p = A - B/r^2$}
% zero line
\addplot[enggray, dashed, samples=2, domain=1:2] {0};
% thin-wall estimate: pr/t with t=b-a=a, r_mean=1.5a  ->  sigma/p = 1.5
\addplot[englime!60!black, dotted, very thick, samples=2, domain=1:2] {1.5};
\addlegendentry{thin-wall $pr/t = 1.5\,p$}
% mark bore hoop = 5/3
\filldraw[engblue] (axis cs:1,1.6667) circle[radius=1.6pt];
\node[engblue, font=\scriptsize, anchor=south west] at (axis cs:1.02,1.62) {$5p/3$ at bore};
% mark bore radial = -1
\filldraw[engred] (axis cs:1,-1) circle[radius=1.6pt];
\node[engred, font=\scriptsize, anchor=north west] at (axis cs:1.02,-0.95) {$-p$ at bore};
\end{axis}
\end{tikzpicture}
\caption[The Lam\'e thick-cylinder stress field]{Radial and hoop stress
through the wall of a thick cylinder under internal pressure, for a wall
whose outer radius is twice the bore radius. The hoop stress is largest
at the bore, $5p/3$ here, and falls toward the outer surface while never
dropping below $2p/3$; the radial stress runs from $-p$ at the pressurised
bore to zero at the free outer surface. Their sum stays constant across
the wall, the signature of the two-constant Lam\'e field. The thin-wall
estimate $pr/t$ (dotted) is a single flat value that misses both the
elevated bore stress and the through-wall gradient, which is why the
thin-wall formula is trusted only when the wall is a small fraction of
the radius.}
\label{fig:vol03ch03:lame-thick-cylinder}
\end{figure}
