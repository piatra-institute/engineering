% Figure: Hertzian contact. Left, two curved bodies pressed together over
% a small contact patch of half-width a; right, the elliptical (semi-
% circular in the line-contact case) contact-pressure distribution p(x)
% peaking at p_0 in the centre and falling to zero at the patch edges,
% with the sub-surface maximum shear marked below the surface.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
% Left: two bodies in contact
\begin{scope}[shift={(0,0)}]
  % upper body (convex down)
  \draw[thick, fill=engblue!6] (-1.6,2.6) arc (200:340:1.7) -- (1.5,3.4) -- (-1.5,3.4) -- cycle;
  \node[font=\scriptsize] at (0,3.05) {body 1, $R_1$};
  % lower body (flat-ish, convex up)
  \draw[thick, fill=engred!6] (-1.6,0.2) arc (160:20:1.7) -- (1.5,-0.6) -- (-1.5,-0.6) -- cycle;
  \node[font=\scriptsize] at (0,-0.3) {body 2, $R_2$};
  % contact patch half-width
  \draw[engorange, {Latex}-{Latex}] (-0.5,1.42) -- (0.5,1.42);
  \node[engorange, font=\scriptsize, anchor=south] at (0,1.45) {$2a$};
  % applied load arrows
  \draw[engblack, -{Latex}, thick] (0,4.0) -- (0,3.5);
  \node[font=\scriptsize, anchor=west] at (0.05,3.85) {$F$};
\end{scope}
% Right: pressure distribution
\begin{scope}[shift={(4.0,0)}]
  \begin{axis}[
    width=7.5cm, height=6cm,
    xlabel={position across patch $x/a$},
    ylabel={contact pressure $p/p_0$},
    xmin=-1.4, xmax=1.4, ymin=0, ymax=1.2,
    axis lines=left,
    tick label style={font=\scriptsize},
    label style={font=\small},
  ]
  % semi-elliptical pressure p = p0 sqrt(1 - (x/a)^2), clamped domain.
  \addplot[engblue, very thick, samples=101, domain=-1:1] {sqrt(max(0,1-x*x))};
  \node[engblue, font=\scriptsize, anchor=south] at (axis cs:0,1.0) {$p_0$};
  % edge markers
  \filldraw[engorange] (axis cs:-1,0) circle[radius=1.6pt];
  \filldraw[engorange] (axis cs:1,0) circle[radius=1.6pt];
  \node[engorange, font=\scriptsize, anchor=north] at (axis cs:1,0) {$+a$};
  \node[engorange, font=\scriptsize, anchor=north] at (axis cs:-1,0) {$-a$};
  % note the sub-surface shear peak
  \node[engred, font=\scriptsize, anchor=north west] at (axis cs:-1.3,0.55)
    {$\tau_{\max}\approx 0.3\,p_0$};
  \node[engred, font=\scriptsize, anchor=north west] at (axis cs:-1.3,0.40)
    {at depth $\approx 0.5a$};
  \end{axis}
\end{scope}
\end{tikzpicture}
\caption[Hertzian contact pressure]{Two curved bodies pressed together
touch over a small patch of half-width $a$, and the contact pressure is
distributed semi-elliptically, peaking at $p_0$ in the centre and falling
to zero at the patch edges. The peak pressure exceeds the nominal
force-over-area by a wide factor because the load crowds into a patch far
smaller than either body. The maximum shear stress sits not at the
surface but a short distance below it, near $0.5a$ deep and about
$0.3\,p_0$ in magnitude, which is why rolling-contact fatigue in bearings
and gears initiates as a sub-surface crack that later spalls to the
surface.}
\label{fig:vol03ch03:hertz-contact}
\end{figure}
