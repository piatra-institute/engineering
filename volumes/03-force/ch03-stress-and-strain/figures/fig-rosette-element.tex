% Figure: a 0-45-90 rectangular strain-gauge rosette bonded to a free
% surface. Three gauges measure normal strain along three directions
% (0, 45, 90 degrees from the x-axis). The three measured strains reduce
% to the in-plane strain tensor (eps_xx, eps_yy, gamma_xy) and from there,
% through Hooke's law, to the surface principal stresses. The sketch shows
% the gauge orientations on a small element and labels the measured
% strains eps_a, eps_b, eps_c.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% surface element
\draw[engblack, very thick] (-2,-2) rectangle (2,2);
\node[engblack, font=\scriptsize, anchor=north east] at (2,-2.05) {free surface element};
% common origin of the rosette
\coordinate (O) at (0,0);
\filldraw[enggray] (O) circle[radius=1.5pt];
% gauge a at 0 degrees (along x)
\draw[engblue, very thick, -{Latex}] (O) -- (1.7,0);
\node[engblue, font=\scriptsize, anchor=west] at (1.75,0) {$\varepsilon_a$ ($0^\circ$)};
% gauge b at 45 degrees
\draw[englime!60!black, very thick, -{Latex}] (O) -- (1.2,1.2);
\node[englime!60!black, font=\scriptsize, anchor=south west] at (1.2,1.2) {$\varepsilon_b$ ($45^\circ$)};
% gauge c at 90 degrees (along y)
\draw[engred, very thick, -{Latex}] (O) -- (0,1.7);
\node[engred, font=\scriptsize, anchor=south] at (0,1.75) {$\varepsilon_c$ ($90^\circ$)};
% angle arc for the 45-degree gauge
\draw[enggray, -{Latex}] (0.8,0) arc[start angle=0, end angle=45, radius=0.8];
\node[enggray, font=\scriptsize] at (1.05,0.3) {$45^\circ$};
% x and y axis labels
\node[engblack, font=\scriptsize, anchor=north west] at (1.9,-0.05) {$x$};
\node[engblack, font=\scriptsize, anchor=south east] at (-0.05,1.9) {$y$};
\end{tikzpicture}
\caption[A rectangular strain-gauge rosette]{A $0$--$45$--$90$
rectangular strain-gauge rosette on a free surface. Each gauge reports
the normal strain along its own direction: $\varepsilon_a$ at $0^\circ$,
$\varepsilon_b$ at $45^\circ$, $\varepsilon_c$ at $90^\circ$. A single
gauge cannot resolve the surface strain state, because the normal strain
in one direction is one number out of the three that define the in-plane
strain tensor. Three gauges at known angles supply three equations, and
the strain-transformation rule inverts them to the tensor
$(\varepsilon_{xx},\varepsilon_{yy},\gamma_{xy})$, from which Hooke's law
gives the surface principal stresses. The rosette is the experimental
counterpart of the Mohr-circle algebra: it measures the stress state at
a point that no formula could supply from the loads alone.}
\label{fig:vol03ch03:rosette-element}
\end{figure}
