% Figure: thermal stress in a fully restrained bar. Shows the bar between
% two rigid walls, the free thermal expansion that would occur if
% unrestrained, and the compressive stress that builds because the walls
% prevent it. Plots compressive stress versus temperature rise for two
% materials (steel and aluminium), with the yield thresholds marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={temperature rise $\Delta T$ (K)},
  ylabel={restrained compressive stress $|\sigma|$ (MPa)},
  xmin=0, xmax=200, ymin=0, ymax=400,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
]
% steel: sigma = E alpha DT = 200e3 MPa * 12e-6 /K * DT = 2.4 DT MPa/K
\addplot[engblue, very thick, samples=2, domain=0:200] {2.4*x};
\addlegendentry{steel ($E\alpha = 2.4$ MPa/K)}
% aluminium: 70e3 * 23e-6 = 1.61 MPa/K
\addplot[engred, very thick, samples=2, domain=0:200] {1.61*x};
\addlegendentry{aluminium ($E\alpha = 1.61$ MPa/K)}
% steel yield line
\addplot[engblue, dashed, samples=2, domain=0:200] {250};
\node[engblue, font=\scriptsize, anchor=south east] at (axis cs:200,250) {steel yield $250$};
% aluminium yield line
\addplot[engred, dashed, samples=2, domain=0:200] {275};
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:200,275) {Al 6061 yield $275$};
% mark the steel yield-onset temperature: 250/2.4 = 104 K
\filldraw[engorange] (axis cs:104,250) circle[radius=2pt];
\node[engorange, font=\scriptsize, anchor=west] at (axis cs:106,235) {$\Delta T \approx 104$ K};
\end{axis}
\end{tikzpicture}
\caption[Thermal stress in a fully restrained bar]{Compressive stress
that builds in a bar held between rigid supports as its temperature
rises, $\sigma = E\alpha\,\Delta T$, for steel and for 6061 aluminium.
The stress is independent of the bar length and cross-section; it
depends only on the modulus, the expansion coefficient, and the
temperature change. Steel reaches its yield stress at a temperature
rise near $104\ \text{K}$, so a fully restrained steel member in a fire
or a sun-heated rail yields in compression and then buckles. The lesson
is that restraint converts a temperature change into a stress with no
external load at all.}
\label{fig:vol03ch03:thermal-stress}
\end{figure}
