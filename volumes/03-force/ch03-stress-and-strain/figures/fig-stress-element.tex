% Figure: the 2D stress element. A square element with normal stresses on the
% x- and y-faces and the complementary shear stresses, showing the sign
% convention sigma_ij = force per area in direction j on the face with normal i.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
% the element
\draw[thick, fill=enggray!8] (0,0) rectangle (3,3);
% center label
\node at (1.5,1.5) {\small element at point $P$};
% axes
\draw[->, enggray] (-1.6,-1.6) -- (-0.6,-1.6) node[below] {\small $x$};
\draw[->, enggray] (-1.6,-1.6) -- (-1.6,-0.6) node[left] {\small $y$};
% normal stress on +x face (right)
\draw[->, very thick, engblue] (3,1.5) -- (4.1,1.5) node[right] {\small $\sigma_{xx}$};
% normal stress on -x face (left)
\draw[->, very thick, engblue] (0,1.5) -- (-1.1,1.5);
% normal stress on +y face (top)
\draw[->, very thick, engblue] (1.5,3) -- (1.5,4.1) node[above] {\small $\sigma_{yy}$};
% normal stress on -y face (bottom)
\draw[->, very thick, engblue] (1.5,0) -- (1.5,-1.1);
% shear on +x face (direction +y)
\draw[->, very thick, engred] (3,0.6) -- (3,2.4) node[midway, right] {\small $\sigma_{xy}$};
% shear on -x face (direction -y)
\draw[->, very thick, engred] (0,2.4) -- (0,0.6);
% shear on +y face (direction +x)
\draw[->, very thick, engred] (0.6,3) -- (2.4,3) node[midway, above left=-2pt] {\small $\sigma_{yx}$};
% shear on -y face (direction -x)
\draw[->, very thick, engred] (2.4,0) -- (0.6,0);
\end{tikzpicture}
\caption[The two-dimensional stress element]{The two-dimensional
stress element at a point $P$. Each face carries a normal stress
(blue) acting along its outward normal and a shear stress (red)
acting parallel to the face. The convention is $\sigma_{ij}$ =
force per unit area in direction $j$ on the face whose outward
normal points in direction $i$. Moment equilibrium of the element
forces the complementary shear stresses to be equal,
$\sigma_{xy}=\sigma_{yx}$, which is why the stress tensor is
symmetric. The element is drawn at finite size for clarity; the
state of stress is defined in the limit as the element shrinks to
the point.}
\label{fig:vol03ch03:stress-element}
\end{figure}
