% Figure: the bolted-joint force-deflection (joint) diagram. The bolt is
% the soft member (low slope k_b); the clamped members are stiff (steep
% slope k_m, drawn descending from the preload). An external load P_ext
% raises the bolt force by only C*P_ext, where C = k_b/(k_b+k_m) is the
% joint stiffness factor; the rest unloads the members until separation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=8cm,
  xlabel={deflection $\delta$},
  ylabel={force $F$},
  xmin=0, xmax=5.2, ymin=0, ymax=5.2,
  axis lines=left,
  xtick=\empty, ytick=\empty,
  clip=false,
]
% bolt line: gentle slope from origin, reaching preload F_i at delta_b
\addplot[engblue, very thick, domain=0:2.0, samples=2] {1.5*x};
\node[engblue, font=\scriptsize, anchor=south east] at (axis cs:1.9,2.85)
  {bolt $k_b$};
% member line: steep slope, descending from preload point toward separation
\addplot[engred, very thick, domain=2.0:3.5, samples=2] {3.0 - 2.0*(x-2.0)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:2.05,1.4)
  {members $k_m$};
% preload level
\draw[enggray, dashed] (axis cs:0,3.0) -- (axis cs:2.0,3.0);
\node[enggray, font=\scriptsize, anchor=east] at (axis cs:0,3.0)
  {$F_i$};
\filldraw[engblack] (axis cs:2.0,3.0) circle [radius=1.4pt];
% external load raises bolt force by C*P_ext to new point
\addplot[engblue, very thick, dashed, domain=2.0:2.7, samples=2] {1.5*x};
\filldraw[engorange] (axis cs:2.7,4.05) circle [radius=1.6pt];
\draw[engorange, dashed] (axis cs:0,4.05) -- (axis cs:2.7,4.05);
\node[engorange, font=\scriptsize, anchor=east] at (axis cs:0,4.05)
  {$F_b$};
% bracket showing the small bolt-force increment
\draw[engorange, <->] (axis cs:3.1,3.0) -- (axis cs:3.1,4.05);
\node[engorange, font=\scriptsize, anchor=west] at (axis cs:3.15,3.5)
  {$C\,P_{\text{ext}}$};
% bracket showing the large member relief
\draw[engred, <->] (axis cs:2.0,2.2) -- (axis cs:2.7,2.2);
\node[engred, font=\scriptsize, anchor=north] at (axis cs:2.35,2.15)
  {$(1{-}C)P_{\text{ext}}$};
\end{axis}
\end{tikzpicture}
\caption[The bolted-joint force-deflection diagram]{The joint diagram. The
preload $F_i$ stretches the bolt along its soft line (slope $k_b$) and
compresses the clamped members along their stiff line (slope $k_m$), the
two meeting at the assembled state (black dot). An external tensile load
$P_{\text{ext}}$ shares between the two paths in proportion to their
stiffnesses: the bolt force rises by only $C\,P_{\text{ext}}$ with $C =
k_b/(k_b+k_m)$, while the members lose $(1-C)P_{\text{ext}}$ of their
compression. Because the members are far stiffer than the bolt, $C$ is
small (often $0.1$ to $0.3$), so most of an external load relieves the
clamp rather than loading the bolt, and the bolt's cyclic stress, the
fatigue driver, is a small fraction of the applied range.}
\label{fig:vol03ch03:bolt-joint-diagram}
\end{figure}
