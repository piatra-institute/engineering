% Figure: the same stress state on two element orientations. Left: a
% general element carrying sigma_xx, sigma_yy and shear sigma_xy. Right:
% the same point rotated by theta_p to the principal axes, where the shear
% vanishes and only sigma_1, sigma_2 remain.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
% ==== Left: general element ====
\begin{scope}[shift={(0,0)}]
  \draw[engblack, thick] (0,0) rectangle (2,2);
  % normal stresses
  \draw[engblue, ->, thick] (2,1) -- (2.7,1);
  \draw[engblue, ->, thick] (0,1) -- (-0.7,1);
  \node[engblue, font=\scriptsize, anchor=west] at (2.7,1) {$\sigma_{xx}$};
  \draw[engblue, ->, thick] (1,2) -- (1,2.7);
  \draw[engblue, ->, thick] (1,0) -- (1,-0.7);
  \node[engblue, font=\scriptsize, anchor=south] at (1,2.7) {$\sigma_{yy}$};
  % shear arrows
  \draw[engred, ->, thick] (2,0.4) -- (2,1.6);
  \draw[engred, ->, thick] (0,1.6) -- (0,0.4);
  \draw[engred, ->, thick] (0.4,2) -- (1.6,2);
  \draw[engred, ->, thick] (1.6,0) -- (0.4,0);
  \node[engred, font=\scriptsize, anchor=west] at (2.05,1.65) {$\sigma_{xy}$};
  \node[font=\small, anchor=north] at (1,-1.0) {general axes};
\end{scope}
% ==== Arrow between ====
\draw[enggray, ->, thick] (3.3,0.7) -- (4.7,0.7);
\node[enggray, font=\scriptsize, anchor=south] at (4.0,0.75)
  {rotate $\theta_p$};
% ==== Right: principal element (rotated) ====
\begin{scope}[shift={(7.2,1.0)}, rotate=28]
  \draw[engblack, thick] (-1,-1) rectangle (1,1);
  \draw[engorange, ->, thick] (1,0) -- (1.8,0);
  \draw[engorange, ->, thick] (-1,0) -- (-1.8,0);
  \node[engorange, font=\scriptsize, anchor=west] at (1.85,0) {$\sigma_1$};
  \draw[englime!60!black, ->, thick] (0,1) -- (0,1.8);
  \draw[englime!60!black, ->, thick] (0,-1) -- (0,-1.8);
  \node[englime!60!black, font=\scriptsize, anchor=south] at (0,1.85)
    {$\sigma_2$};
\end{scope}
\node[font=\small, anchor=north] at (7.2,-1.0) {principal axes};
\end{tikzpicture}
\caption[The principal-axis orientation removes the shear]{The stress at a
point on two orientations of the same element. On general axes (left) the
faces carry normal stresses $\sigma_{xx}$, $\sigma_{yy}$ and a shear
$\sigma_{xy}$. Rotating the element by the principal angle $\theta_p$
(right) brings it to the orientation where the shear vanishes and the
normal stresses reach their extreme values, the principal stresses
$\sigma_1$ and $\sigma_2$. The principal-axis representation is the same
physical stress with the off-diagonal entries rotated away; finding it is
the two-dimensional eigenproblem of the stress tensor.}
\label{fig:vol03ch03:principal-rotation}
\end{figure}
