% Figure: stress distributions across a beam cross-section. Three panels:
% uniform axial stress N/A, linear bending stress -My/I, parabolic shear VQ/Ib.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
% ==== Panel 1: axial, uniform ====
\begin{scope}[shift={(0,0)}]
  \draw[thick] (0,-1.5) rectangle (0.5,1.5);
  \node[font=\small] at (0.25,-2.1) {axial};
  \node[font=\scriptsize] at (0.25,-2.55) {$\sigma=N/A$};
  % uniform arrows
  \foreach \y in {-1.2,-0.6,0,0.6,1.2}{
    \draw[->, engblue, thick] (0.5,\y) -- (1.5,\y);
  }
  \draw[engblue] (1.5,-1.5) -- (1.5,1.5);
\end{scope}
% ==== Panel 2: bending, linear ====
\begin{scope}[shift={(4.5,0)}]
  \draw[thick] (0,-1.5) rectangle (0.5,1.5);
  \node[font=\small] at (0.25,-2.1) {bending};
  \node[font=\scriptsize] at (0.25,-2.55) {$\sigma=-My/I$};
  % neutral axis
  \draw[dashed, enggray] (-0.3,0) -- (1.8,0) node[right, font=\scriptsize] {N.A.};
  % linear: compression top, tension bottom
  \draw[->, engred, thick] (0.5,1.2) -- (1.46,1.2);
  \draw[->, engred, thick] (0.5,0.6) -- (0.98,0.6);
  \draw[->, engblue, thick] (1.46,-1.2) -- (0.5,-1.2);
  \draw[->, engblue, thick] (0.98,-0.6) -- (0.5,-0.6);
  \draw[engred] (0.5,0) -- (1.5,1.5);
  \draw[engblue] (0.5,0) -- (1.5,-1.5);
\end{scope}
% ==== Panel 3: shear, parabolic ====
\begin{scope}[shift={(9.5,0)}]
  \draw[thick] (0,-1.5) rectangle (0.5,1.5);
  \node[font=\small] at (0.25,-2.1) {shear};
  \node[font=\scriptsize] at (0.25,-2.55) {$\tau=VQ/(Ib)$};
  % parabola: max at center (y=0), zero at +-1.5
  \draw[englime!60!black, thick, samples=40, domain=-1.5:1.5, variable=\y]
    plot ({0.5 + 1.0*(1 - (\y*\y)/2.25)}, {\y});
  \draw[enggray] (0.5,-1.5) -- (0.5,1.5);
  \node[font=\scriptsize] at (1.7,0) {$\tau_{\max}$};
\end{scope}
\end{tikzpicture}
\caption[Stress distributions across a beam cross-section]{The three
canonical stress distributions across a beam cross-section. Axial force
$N$ produces uniform normal stress $\sigma=N/A$. Bending moment $M$
produces a linear normal stress $\sigma=-My/I$, zero on the neutral axis
and extreme at the top and bottom fibres (compression on one face,
tension on the other). Transverse shear $V$ produces a parabolic shear
stress $\tau=VQ/(Ib)$ for a rectangular section, maximum on the neutral
axis and zero at the free top and bottom surfaces. Under combined
loading the contributions superpose, valid because elasticity is linear
in this regime. The bending and shear distributions are developed in
Chapter 8.}
\label{fig:vol03ch03:axial-bending-distribution}
\end{figure}
