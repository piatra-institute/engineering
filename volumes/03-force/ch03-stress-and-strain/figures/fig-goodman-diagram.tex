% Figure: modified-Goodman and Soderberg fatigue diagrams in the
% mean-alternating stress plane. The Goodman line runs from the endurance
% limit on the alternating axis to the ultimate strength on the mean axis;
% Soderberg runs to the yield strength. The padeye operating point is
% plotted; its position relative to each line gives the fatigue safety
% factor. Numbers from the case study: Se=180, Sy=350, Su=500 MPa,
% operating point (sigma_m, sigma_a) = (95, 86) MPa.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={mean stress $\sigma_m$ (MPa)},
  ylabel={alternating stress $\sigma_a$ (MPa)},
  xmin=0, xmax=520, ymin=0, ymax=200,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
]
% Goodman line: from (0, Se=180) to (Su=500, 0)
\addplot[engblue, very thick] coordinates {(0,180) (500,0)};
\addlegendentry{modified Goodman ($S_u=500$)}
% Soderberg line: from (0, Se=180) to (Sy=350, 0)
\addplot[engred, very thick] coordinates {(0,180) (350,0)};
\addlegendentry{Soderberg ($S_y=350$)}
% yield line (first-cycle): sigma_m + sigma_a = Sy
\addplot[enggray, dashed, very thick] coordinates {(0,350) (350,0)};
\addlegendentry{first-cycle yield}
% operating point (95, 86)
\addplot[only marks, mark=*, engorange, mark size=2.6pt] coordinates {(95,86)};
\node[engorange, font=\scriptsize, anchor=south west] at (axis cs:98,88) {operating point $(95,86)$};
% load line from origin through the operating point, extended to Goodman
% slope = 86/95 = 0.905; meets Goodman where sigma_a = 180(1 - sigma_m/500)
% 0.905 sigma_m = 180 - 0.36 sigma_m  => sigma_m = 142.0, sigma_a=128.5
\addplot[engorange, dotted, very thick] coordinates {(0,0) (142,128.5)};
\node[engorange, font=\scriptsize, anchor=north west] at (axis cs:145,128.5) {Goodman limit $(142,129)$};
\end{axis}
\end{tikzpicture}
\caption[Goodman and Soderberg fatigue diagrams]{Mean-alternating
stress plane for the padeye case study, with the modified-Goodman line
(endurance limit $S_e = 180$ to ultimate $S_u = 500\ \text{MPa}$), the
more conservative Soderberg line (to yield $S_y = 350\ \text{MPa}$), and
the first-cycle yield line $\sigma_m + \sigma_a = S_y$. The operating
point $(\sigma_m,\sigma_a) = (95, 86)\ \text{MPa}$ sits below both
fatigue lines, so the padeye survives. The load line from the origin
through the operating point meets the Goodman line at $(142, 129)$; the
ratio of that distance to the operating-point distance is the Goodman
fatigue factor of safety, here about $1.5$.}
\label{fig:vol03ch03:goodman-diagram}
\end{figure}
