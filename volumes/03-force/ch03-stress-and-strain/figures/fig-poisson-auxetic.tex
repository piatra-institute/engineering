% Figure: Poisson's ratio through three materials. Each panel shows a
% square element (dashed, undeformed) pulled horizontally, and the
% deformed shape (solid) it takes for a positive, zero, and negative
% Poisson ratio. A conventional material (nu>0) thins as it stretches; a
% zero-nu material (cork) keeps its width; an auxetic material (nu<0)
% widens as it stretches. The horizontal stretch is the same in all three;
% only the transverse response differs. Strains are exaggerated for
% legibility.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% common: undeformed unit square from (-0.6,-0.6) to (0.6,0.6)
% horizontal stretch factor applied to all: x -> 1.35 x
% panel 1: nu = 0.5 (thins strongly): y -> 0.68 y
\begin{scope}[shift={(0,0)}]
\draw[enggray, dashed] (-0.6,-0.6) rectangle (0.6,0.6);
\draw[engblue, very thick] (-0.81,-0.41) rectangle (0.81,0.41);
\draw[engblack, {Latex}-{Latex}] (-1.15,0) -- (1.15,0);
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.0) {$\nu \approx 0.5$ (rubber)};
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.35) {thins as it stretches};
\end{scope}
% panel 2: nu = 0 (cork): y unchanged
\begin{scope}[shift={(3.4,0)}]
\draw[enggray, dashed] (-0.6,-0.6) rectangle (0.6,0.6);
\draw[englime!60!black, very thick] (-0.81,-0.6) rectangle (0.81,0.6);
\draw[engblack, {Latex}-{Latex}] (-1.15,0) -- (1.15,0);
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.0) {$\nu \approx 0$ (cork)};
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.35) {keeps its width};
\end{scope}
% panel 3: nu < 0 (auxetic): y -> 1.2 y (widens)
\begin{scope}[shift={(6.8,0)}]
\draw[enggray, dashed] (-0.6,-0.6) rectangle (0.6,0.6);
\draw[engred, very thick] (-0.81,-0.72) rectangle (0.81,0.72);
\draw[engblack, {Latex}-{Latex}] (-1.15,0) -- (1.15,0);
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.0) {$\nu < 0$ (auxetic)};
\node[engblack, font=\scriptsize, anchor=north] at (0,-1.35) {widens as it stretches};
\end{scope}
\end{tikzpicture}
\caption[Poisson's ratio: conventional, zero, and auxetic]{Three
materials pulled by the same horizontal stretch, with the undeformed
element dashed and the deformed element solid. A conventional material
with a positive Poisson ratio thins in the transverse direction as it
stretches (rubber, near the incompressible limit $\nu \to 1/2$, thins the
most). A material with $\nu \approx 0$, such as cork, keeps its transverse
dimension, which is why a cork slides into a bottle neck without bulging.
An auxetic material with $\nu < 0$, realised in re-entrant foams and
engineered lattices, widens as it is pulled. The admissible range
$-1 < \nu < 1/2$ set by the positivity of the shear and bulk moduli spans
all three behaviours; the transverse response, not the axial stretch, is
what the sign of $\nu$ controls.}
\label{fig:vol03ch03:poisson-auxetic}
\end{figure}
