% Figure: Ramberg-Osgood stress-strain curves for two hardening exponents.
% The total strain is the elastic part sigma/E plus a plastic part that
% grows as a power of stress. A low hardening exponent gives a rounded
% knee; a high exponent gives a sharp near-elastic-perfectly-plastic
% corner. The linear-elastic line sigma = E*epsilon is drawn for contrast.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={strain $\epsilon$ (\%)},
  ylabel={stress $\sigma$ (MPa)},
  xmin=0, xmax=1.2, ymin=0, ymax=460,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=south east,
  legend style={font=\scriptsize},
]
% Linear-elastic reference: sigma = E*epsilon. With E = 200 GPa, at
% epsilon = 0.2% sigma = 400 MPa. Plot sigma vs epsilon(%): sigma = 200000*eps
% with eps as fraction; eps(%) = 100*eps so sigma = 2000*eps(%).
\addplot[enggray, dashed, thick, samples=2, domain=0:0.2] {2000*x};
\addlegendentry{linear elastic $\sigma=E\epsilon$}
% Ramberg-Osgood: epsilon = sigma/E + alpha*(sigma/sigma_0)^n * sigma_0/E.
% We plot sigma on y and epsilon(%) on x by sampling sigma and computing
% epsilon. Do it with explicit points to stay compile-safe.
% Parameters: E = 200000 MPa, sigma_0 = 350 MPa, alpha = 0.5.
% Low exponent n = 5 (rounded knee):
\addplot[engblue, very thick] coordinates {
  (0,0) (0.05,100) (0.10,200) (0.16,300) (0.205,350)
  (0.29,390) (0.45,420) (0.70,440) (1.05,455)
};
\addlegendentry{Ramberg-Osgood, $n=5$ (rounded)}
% High exponent n = 20 (sharp knee, near elastic-perfectly-plastic):
\addplot[engred, very thick] coordinates {
  (0,0) (0.075,150) (0.15,300) (0.185,345) (0.205,350)
  (0.40,356) (0.70,360) (1.05,363)
};
\addlegendentry{Ramberg-Osgood, $n=20$ (sharp)}
% mark 0.2% offset yield
\draw[engorange, dashed] (axis cs:0.2,0) -- (axis cs:0.2,350);
\filldraw[engorange] (axis cs:0.205,350) circle[radius=2pt];
\node[engorange, font=\scriptsize, anchor=west] at (axis cs:0.23,350)
  {$0.2\%$ offset yield $\sigma_0$};
\end{axis}
\end{tikzpicture}
\caption[Ramberg-Osgood stress-strain curves]{The Ramberg-Osgood law
writes the total strain as an elastic part $\sigma/E$ plus a plastic
part that grows as a power $n$ of the stress, giving a single smooth
curve with no sharp yield point. A low hardening exponent produces a
gradual, rounded knee; a high exponent produces a sharp corner
approaching elastic-perfectly-plastic behaviour. The $0.2\%$ offset
construction locates a conventional yield stress $\sigma_0$ on either
curve by shifting the elastic line to a strain of $0.002$ and reading the
crossing. The law is the standard smooth fit to a measured tension curve
and the one most finite-element codes use when a full plastic response is
needed.}
\label{fig:vol03ch03:ramberg-osgood}
\end{figure}
