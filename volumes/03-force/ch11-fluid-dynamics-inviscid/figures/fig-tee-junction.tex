% Figure: a dividing tee. One inlet stream splits into two outlets; continuity
% and the momentum theorem give the reaction the junction restraint must carry.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  pipe/.style={draw=engblack, very thick},
  stream/.style={draw=engblue, -{Stealth}, thick},
  force/.style={draw=engred, -{Stealth}, very thick},
  label/.style={font=\scriptsize},
]
% main run (horizontal) with a branch going up
\draw[pipe] (-4,0.5) -- (4,0.5);
\draw[pipe] (-4,-0.5) -- (0.5,-0.5);
\draw[pipe] (1.5,-0.5) -- (4,-0.5);
% branch pipe rising from the gap
\draw[pipe] (0.5,-0.5) -- (0.5,-2.6);
\draw[pipe] (1.5,-0.5) -- (1.5,-2.6);
% inlet stream
\draw[stream] (-3.4,0) -- (-2.4,0);
\node[label, engblue, anchor=south] at (-2.9,0.05) {$\dot m_1,\ u_1$};
% outlet 1 (straight through)
\draw[stream] (2.5,0) -- (3.5,0);
\node[label, engblue, anchor=south] at (3.0,0.05) {$\dot m_2,\ u_2$};
% outlet 2 (branch, downward)
\draw[stream] (1.0,-1.6) -- (1.0,-2.4);
\node[label, engblue, anchor=west] at (1.1,-2.0) {$\dot m_3,\ u_3$};
% reaction force on the junction restraint
\draw[force] (0.0,0) -- (-1.4,0.9);
\node[label, engred, anchor=south east] at (-1.2,0.8) {$\vec R$ (restraint)};
% mass balance annotation
\node[label, engblack, anchor=north] at (0,-2.8)
  {$\dot m_1 = \dot m_2 + \dot m_3$};
\end{tikzpicture}
\caption[Momentum balance at a dividing tee]{A dividing tee splits one
inlet stream into two outlets. Continuity fixes the mass split,
$\dot m_1 = \dot m_2 + \dot m_3$, and the pressures on the three faces
plus the restraint force $\vec R$ balance the net change in momentum
flux. The straight-through branch carries most of the streamwise
momentum, while the side branch removes a component transverse to the
main run, so the junction feels a reaction with both a streamwise and
a transverse part. The momentum theorem sizes the restraint that keeps
the fitting from walking off its supports; it makes no inviscid
assumption and holds whatever the losses inside the tee.}
\label{fig:vol03ch11:tee-junction}
\end{figure}
