% Figure: wind over a low building. The flow speeds up over the roof, dropping
% the external pressure into suction; the internal pressure stays near stagnation.
% The net uplift is the difference, and it can exceed the roof's dead weight.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  wall/.style={draw=engblack, very thick},
  stream/.style={draw=engblue, thick},
  up/.style={draw=engred, -{Stealth}, very thick},
  label/.style={font=\scriptsize},
]
% building box
\draw[wall, fill=enggray!12] (-1.8,0) rectangle (1.8,1.4);
\node[label, engblack] at (0,0.7) {building};
% ground
\draw[enggray, thick] (-4.2,0) -- (4.2,0);
% wind streamlines accelerating over the roof (crowded above the ridge)
\draw[stream] (-4.2,1.9) .. controls (-2.0,1.95) and (-0.8,2.35) .. (0,2.4)
   .. controls (0.8,2.35) and (2.0,1.95) .. (4.2,1.9);
\draw[stream] (-4.2,2.5) .. controls (-2.0,2.55) and (-0.8,2.85) .. (0,2.9)
   .. controls (0.8,2.85) and (2.0,2.55) .. (4.2,2.5);
\draw[engblue,-{Stealth},thick] (-4.0,1.65) -- (-3.2,1.65);
\node[label, engblue, anchor=south] at (-3.6,1.7) {$U$};
% external suction on the roof (arrows pulling up, outside)
\draw[up] (-1.0,1.4) -- (-1.0,2.05);
\draw[up] (0.0,1.4) -- (0.0,2.05);
\draw[up] (1.0,1.4) -- (1.0,2.05);
\node[label, engred, anchor=south] at (0,2.5) {external suction $p_e < p_\infty$};
% internal pressure near stagnation (arrow pushing up from inside)
\draw[engamber,-{Stealth},thick] (0,0.35) -- (0,1.05);
\node[label, engamber, anchor=north] at (0,0.3) {internal $p_i \approx p_\infty$};
% net uplift annotation
\node[label, engblack, anchor=west] at (2.1,1.4)
  {net uplift $=(p_i-p_e)A_\text{roof}$};
\end{tikzpicture}
\caption[Wind uplift on a low-building roof]{Wind blowing across a low
building accelerates over the roof, so by Bernoulli the external
pressure there drops into suction, $p_e < p_\infty$. If the building
is not tightly sealed, or has a windward opening, the internal
pressure sits near the windward stagnation value, close to or above
ambient. The roof then carries a net uplift equal to the pressure
difference times the roof area, directed outward. For a strong wind
the uplift can exceed the roof's dead weight, which is why roofs are
lost in storms by lifting off rather than by being pushed in, and why
building codes size roof fixings for suction, not for downforce.}
\label{fig:vol03ch11:roof-pressure}
\end{figure}
