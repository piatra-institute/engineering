% Figure: the drag polar of a finite wing, total drag = profile drag plus
% induced drag, showing the parabolic rise of induced drag with lift coefficient
% and the benefit of higher aspect ratio.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={drag coefficient $C_D$},
  ylabel={lift coefficient $C_L$},
  xmin=0, xmax=0.10, ymin=0, ymax=1.4,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=south east,
]
% profile drag floor
\addplot[enggray, dashed, very thick, samples=2, domain=0:1.4] coordinates {(0.012,0) (0.012,1.4)};
\addlegendentry{profile drag $C_{D,0}=0.012$}
% AR = 7, e = 0.9: CD = 0.012 + CL^2/(pi*7*0.9), invert for CL on x-axis by plotting CD(CL)
% plot CL on vertical axis, CD on horizontal: parametric in CL
\addplot[engblue, very thick, samples=80, domain=0:1.3, variable=\t]
  ({0.012 + (\t)^2/(3.14159*7*0.9)}, {\t});
\addlegendentry{$AR=7$ (stubby wing)}
% AR = 18, e = 0.95: slender sailplane wing
\addplot[engred, very thick, samples=80, domain=0:1.3, variable=\t]
  ({0.012 + (\t)^2/(3.14159*18*0.95)}, {\t});
\addlegendentry{$AR=18$ (sailplane wing)}
% mark a working CL
\draw[engblack, dotted] (axis cs:0,0.8) -- (axis cs:0.10,0.8);
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:0.055,0.82) {cruise $C_L\approx0.8$};
\filldraw[engblue] (axis cs:0.0443,0.8) circle (1.6pt);
\filldraw[engred] (axis cs:0.0239,0.8) circle (1.6pt);
\end{axis}
\end{tikzpicture}
\caption[Drag polar of a finite wing]{The drag polar plots lift
coefficient against drag coefficient. The total drag is the profile
drag (a roughly constant floor set by skin friction and pressure drag
on the section) plus the induced drag, which grows as the square of
the lift coefficient and inversely with the aspect ratio. At a cruise
lift coefficient near $0.8$ the stubby $AR=7$ wing carries almost
twice the drag of the slender $AR=18$ sailplane wing, the gap being
entirely induced drag. The induced part is an inviscid result: it
survives in a flow with no viscosity at all, because it is the kinetic
energy left in the trailing vortex sheet. The profile floor, by
contrast, is purely viscous and is the business of the next chapter.}
\label{fig:vol03ch11:drag-polar}
\end{figure}
