% Figure: two point vortices. An equal-and-opposite pair translates in
% a straight line at a self-induced speed; an equal-sign pair orbits a
% common centre. The figure shows the counter-rotating (translating)
% pair, the model of a wingtip vortex pair.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=5.6cm,
  xlabel={$x$}, ylabel={$y$},
  xmin=-2.6, xmax=2.6, ymin=-1.6, ymax=1.6,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% The two vortices of opposite sign, separation d = 1
\addplot[engblue, only marks, mark=*, mark size=2pt]
  coordinates {(0,0.5)};
\addplot[engred, only marks, mark=*, mark size=2pt]
  coordinates {(0,-0.5)};
% Rotation-sense arcs
\draw[engblue, -{Stealth[length=1.6mm]}]
  (axis cs:0.32,0.5) arc[start angle=0, end angle=270, radius=0.32];
\draw[engred, -{Stealth[length=1.6mm]}]
  (axis cs:0.32,-0.5) arc[start angle=0, end angle=-270, radius=0.32];
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:0.45,0.5) {$+\Gamma$};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.45,-0.5) {$-\Gamma$};
% Translation direction (self-induced) along +x
\draw[engblack, -{Stealth[length=2.2mm]}, very thick]
  (axis cs:-1.9,0) -- (axis cs:-1.0,0);
\node[engblack, font=\scriptsize, anchor=south] at (axis cs:-1.45,0.03)
  {$V=\dfrac{\Gamma}{2\pi d}$};
% Separation bracket
\draw[enggray, {Stealth[length=1.4mm]}-{Stealth[length=1.4mm]}]
  (axis cs:1.4,0.5) -- (axis cs:1.4,-0.5);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:1.45,0) {$d$};
\end{axis}
\end{tikzpicture}
\caption[A counter-rotating vortex pair translates]{Two point vortices
of equal and opposite circulation, separation $d$, each sit in the
velocity the other induces. The upper vortex feels the downward-then-
leftward field of the lower one and vice versa, and the pair translates
together at the self-induced speed $V=\Gamma/(2\pi d)$ perpendicular to
the line joining them, holding their separation fixed. An equal-sign
pair instead orbits its midpoint. The counter-rotating pair is the
model of the two trailing vortices behind a wing: they descend together
below the flight path at just this speed, and the descent is why a
following aircraft is warned to stay above the wake of the one ahead.
The construction uses only the point-vortex velocity field of the
potential-flow section, with each vortex a passive marker carried by
the other's flow.}
\label{fig:vol03ch11:vortex-pair}
\end{figure}
