% Figure: pressure forces on a small fluid element, the face-by-face
% bookkeeping behind the -grad p net surface force in Euler's equation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  elem/.style={draw=engblack, very thick, fill=engblue!6},
  pforce/.style={draw=engblue, -{Stealth}, very thick},
  dim/.style={draw=enggray, {Stealth}-{Stealth}, thin},
  label/.style={font=\small},
]
% the cubic element (drawn as a square in the x-y plane)
\draw[elem] (0,0) rectangle (3,3);
% centre marker
\filldraw[engred] (1.5,1.5) circle (1.4pt);
\node[label, engred, anchor=south west] at (1.5,1.5) {$p(x,y)$};
% pressure on the left face (pushing right, inward)
\draw[pforce] (-1.4,1.5) -- (-0.05,1.5);
\node[label, engblue, anchor=east] at (-1.45,1.5) {$\left(p-\dfrac{\partial p}{\partial x}\dfrac{dx}{2}\right)dy\,dz$};
% pressure on the right face (pushing left, inward)
\draw[pforce] (4.4,1.5) -- (3.05,1.5);
\node[label, engblue, anchor=west] at (4.45,1.5) {$\left(p+\dfrac{\partial p}{\partial x}\dfrac{dx}{2}\right)dy\,dz$};
% pressure on the bottom face (pushing up)
\draw[pforce] (1.5,-1.2) -- (1.5,-0.05);
\node[label, engblue, anchor=north] at (1.5,-1.25) {$\left(p-\dfrac{\partial p}{\partial y}\dfrac{dy}{2}\right)dx\,dz$};
% pressure on the top face (pushing down)
\draw[pforce] (1.5,4.2) -- (1.5,3.05);
\node[label, engblue, anchor=south] at (1.5,4.25) {$\left(p+\dfrac{\partial p}{\partial y}\dfrac{dy}{2}\right)dx\,dz$};
% edge dimensions
\draw[dim] (0,-0.45) -- (3,-0.45);
\node[label, anchor=north] at (1.5,-0.5) {$dx$};
\draw[dim] (3.45,0) -- (3.45,3);
\node[label, anchor=west] at (3.5,1.5) {$dy$};
\end{tikzpicture}
\caption[Pressure forces on a fluid element]{Pressure forces on the
faces of a small rectangular fluid element of volume $dx\,dy\,dz$. The
pressure on each face is the centre value $p$ plus a first-order Taylor
correction across the half-width, and the force on each face is that
pressure times the face area, directed inward. The two $x$-faces leave a
net force $-(\partial p/\partial x)\,dx\,dy\,dz$ in the $x$-direction;
the two $y$-faces leave $-(\partial p/\partial y)\,dx\,dy\,dz$. Summing
the three coordinate directions and dividing by the volume gives the net
pressure force per unit volume $-\nabla p$, the surface force that drives
the inviscid Euler equation. The opposite faces nearly cancel; only the
gradient survives.}
\label{fig:vol03ch11:differential-element}
\end{figure}
