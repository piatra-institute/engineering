% Figure: pressure (hydraulic grade line) through a Venturi metering station,
% showing the deep throat dip, near-full recovery, and the small permanent
% pressure loss that an orifice plate would not recover.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={distance along the meter $x$ (m)},
  ylabel={static pressure $p$ (kPa, gauge)},
  xmin=0, xmax=1.0, ymin=0, ymax=160,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=south east,
]
% Venturi pressure: high upstream, deep dip at the throat (x=0.3), strong
% recovery in the diffuser, ending a little below the inlet (permanent loss).
\addplot[engblue, very thick, smooth, samples=80, domain=0:1.0] {
  150 - 95*exp(-((x-0.3)^2)/0.004)*( x<0.3 ? 1 : 1 ) - 8*x
};
\addlegendentry{Venturi (gentle diffuser)}
% Orifice plate: same throat dip but poor recovery -> large permanent loss.
\addplot[engred, dashed, very thick, smooth, samples=80, domain=0:1.0] {
  150 - 95*exp(-((x-0.3)^2)/0.004) - 55*( x>0.3 ? (1-exp(-((x-0.3))/0.12)) : 0 )
};
\addlegendentry{orifice plate (abrupt)}
% Throat marker
\addplot[enggray, dotted, very thick, samples=2] coordinates {(0.3,0) (0.3,150)};
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:0.3,2) {throat};
% Permanent-loss bracket annotations
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:0.78,138) {small loss};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.62,90) {large loss};
\node[font=\scriptsize, anchor=south] at (axis cs:0.3,58) {measured $\Delta p$};
\end{axis}
\end{tikzpicture}
\caption[Pressure trace through a metering station]{Static pressure
along a flow-metering station. Both the Venturi and an orifice plate
drop the pressure sharply at the throat, and that throat dip is the
differential the transducer reads. The two devices part company
downstream: the Venturi's gentle diffuser recovers nearly all of the
dynamic pressure, leaving a small permanent loss, while the orifice
plate's abrupt expansion separates and dissipates, leaving a large
permanent loss. The measured differential is the same; the energy bill
the station charges the pump is not. Choosing the meter trades capital
cost and length against the permanent loss the plant pays for as long
as it runs.}
\label{fig:vol03ch11:metering-egl}
\end{figure}
