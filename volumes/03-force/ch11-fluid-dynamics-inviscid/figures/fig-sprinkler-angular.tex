% Figure: lawn-sprinkler angular-momentum balance: two arms with bent
% nozzles eject jets tangentially, the reaction torque spins the head.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  arm/.style={draw=engblack, very thick},
  jet/.style={draw=engblue, -{Stealth}, very thick},
  react/.style={draw=englime, -{Stealth}, very thick},
  spin/.style={draw=engred, -{Stealth}, thick},
  label/.style={font=\small},
]
% central pivot
\filldraw[enggray] (0,0) circle (3pt);
\node[label, anchor=north] at (0,-0.15) {pivot (supply)};
% two arms along x
\draw[arm] (-2.6,0) -- (2.6,0);
% bent nozzles at each tip, pointing tangentially (perpendicular to arm)
% right tip: nozzle bends downward, jet leaves downward (tangential)
\draw[arm] (2.6,0) -- (2.6,-0.55);
\draw[jet] (2.6,-0.6) -- (2.6,-1.9);
\node[label, engblue, anchor=west] at (2.65,-1.4) {$v_\text{rel}$};
% left tip: nozzle bends upward, jet leaves upward (tangential, opposite)
\draw[arm] (-2.6,0) -- (-2.6,0.55);
\draw[jet] (-2.6,0.6) -- (-2.6,1.9);
\node[label, engblue, anchor=east] at (-2.65,1.4) {$v_\text{rel}$};
% radius dimension
\draw[draw=enggray,{Stealth}-{Stealth},thin] (0,0.35) -- (2.6,0.35);
\node[label, anchor=south] at (1.3,0.4) {$R$};
% reaction force on each arm (opposite to jet) producing the couple
\draw[react] (2.6,0.0) -- (2.6,0.9);
\node[label, englime, anchor=south west] at (2.6,0.9) {reaction};
\draw[react] (-2.6,0.0) -- (-2.6,-0.9);
\node[label, englime, anchor=north east] at (-2.6,-0.9) {reaction};
% spin arc
\draw[spin] (1.5,1.3) arc (50:130:2.3);
\node[label, engred, anchor=south] at (0,2.0) {$\Omega$ (steady spin)};
\end{tikzpicture}
\caption[Lawn-sprinkler angular-momentum balance]{A two-arm lawn
sprinkler. Each nozzle of radius $R$ ejects a jet tangentially at speed
$v_\text{rel}$ relative to the arm; by Newton's third law the jet's
reaction drives the arm the opposite way, and the two reactions form a
couple that spins the head at angular speed $\Omega$. The
angular-momentum form of the control-volume theorem balances the
applied torque against the net flux of angular momentum carried out by
the jets. At the free-running steady speed the bearing torque is small,
and the head spins until the absolute tangential velocity of the
ejected water, $v_\text{rel} - \Omega R$, leaves nearly zero net
angular-momentum flux, the working principle behind the spin speed
computed in the vignette.}
\label{fig:vol03ch11:sprinkler-angular}
\end{figure}
