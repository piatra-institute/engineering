% Figure: the ideal Euler head of a centrifugal pump falls linearly
% with flow for backward-curved blades, and the real head curve droops
% below it because of slip and internal losses.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.0cm,
  xlabel={volume flow $Q$ (arb. units)},
  ylabel={head $H$ (arb. units)},
  xmin=0, xmax=1.0, ymin=0, ymax=1.15,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=north east,
]
% Ideal Euler head: linear fall with flow for backward-curved blades
\addplot[engblue, very thick, samples=2, domain=0:1.0] {1.0 - 0.85*x};
\addlegendentry{ideal Euler head}
% Head after slip correction (shifted down, still roughly linear)
\addplot[engamber, dashed, very thick, samples=2, domain=0:1.0] {0.82 - 0.80*x};
\addlegendentry{after slip}
% Real head curve: droops further, curved, from friction + shock losses
\addplot[engred, very thick, samples=120, domain=0:0.95]
  {0.80 - 0.35*x - 0.55*x*x};
\addlegendentry{real curve (with losses)}
% Best-efficiency operating point marker
\addplot[engblack, only marks, mark=*, mark size=1.6pt]
  coordinates {(0.55,0.42)};
\node[engblack, font=\scriptsize, anchor=south west]
  at (axis cs:0.56,0.43) {best efficiency};
\end{axis}
\end{tikzpicture}
\caption[Head-flow curve of a centrifugal pump]{The Euler turbomachine
equation gives the ideal head a backward-curved impeller adds to the
flow, and for those blades the head falls linearly as the flow rises
(solid), because more throughflow cuts the swirl the blade imparts.
Two corrections lower the real curve. Slip, the failure of the fluid to
follow the blade angle exactly, drops the whole line (dashed); internal
friction and off-design incidence losses, smallest near the design
flow, bend it into the drooping real curve (lower). The gap between the
ideal Euler line and the real curve is the pump's internal loss, the
same energy-not-conserved bookkeeping the sudden-expansion example
carried, and the best-efficiency point sits where that gap is
narrowest. The inviscid Euler line sets the ceiling the real pump
chases, exactly as Bernoulli sets the ceiling for a nozzle jet.}
\label{fig:vol03ch11:pump-head-flow}
\end{figure}
