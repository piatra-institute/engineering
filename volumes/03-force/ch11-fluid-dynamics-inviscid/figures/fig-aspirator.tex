% Figure: a jet aspirator (eductor). A high-speed primary jet through a throat
% drops the local pressure below ambient and entrains a secondary flow from a
% side port. Bernoulli sets the throat suction; the momentum theorem mixes it.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  pipe/.style={draw=engblack, very thick},
  jet/.style={draw=engblue, -{Stealth}, thick},
  suck/.style={draw=engred, -{Stealth}, thick},
  label/.style={font=\scriptsize},
]
% body: converging nozzle into throat, then diverging diffuser
\draw[pipe] (-4,0.8) -- (-1.4,0.8) -- (-0.5,0.32) -- (0.5,0.32) -- (2.2,0.9) -- (4,0.9);
\draw[pipe] (-4,-0.8) -- (-1.4,-0.8) -- (-0.5,-0.32) -- (0.5,-0.32) -- (2.2,-0.9) -- (4,-0.9);
% side suction port entering just upstream of the throat
\draw[pipe] (-0.9,-0.8) -- (-0.9,-2.2);
\draw[pipe] (-0.1,-0.32) -- (-0.1,-2.2);
% primary jet (fast, through the throat)
\draw[jet] (-3.4,0) -- (-2.4,0);
\node[label, engblue, anchor=south] at (-2.9,0.05) {primary jet $u_1$};
\draw[jet] (-0.35,0) -- (0.45,0);
\node[label, engblue, anchor=south] at (0.1,0.35) {throat $u_t$ (fast)};
% entrained secondary flow drawn up into the low-pressure throat
\draw[suck] (-0.5,-2.0) -- (-0.5,-0.9);
\node[label, engred, anchor=north] at (-0.5,-2.1) {entrained $\dot m_s$ (suction)};

\draw[jet] (2.6,0) -- (3.5,0);
\node[label, engblue, anchor=south] at (3.0,0.05) {mixed discharge};

\end{tikzpicture}
\caption[A jet aspirator entrains flow through a throat suction]{A jet
aspirator, or eductor, drives a high-speed primary jet through a
throat. Bernoulli on the primary stream drops the throat static
pressure below the ambient pressure at the side port, and that
suction draws a secondary flow $\dot m_s$ into the throat, where the
two streams mix and discharge through a diffuser. No moving part is
involved; the pump is the pressure field of the accelerated jet. The
same device is the laboratory water aspirator, the carburettor that
meters fuel into an engine's intake air, and the ejector that pulls a
vacuum on a condenser. The throat suction is inviscid Bernoulli; the
mixing that raises the entrained stream back to discharge pressure is
a momentum-theorem process with an unavoidable loss.}
\label{fig:vol03ch11:aspirator}
\end{figure}
