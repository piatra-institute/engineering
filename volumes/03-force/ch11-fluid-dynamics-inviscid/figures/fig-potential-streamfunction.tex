% Figure: the orthogonal grid of equipotential lines and streamlines
% for a source-plus-uniform-flow, illustrating the Cauchy-Riemann
% orthogonality of phi and psi.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  equipot/.style={draw=engred, thick},
  stream/.style={draw=engblue, very thick},
  label/.style={font=\small},
]
\begin{scope}
\clip (-4.2,-3.0) rectangle (4.2,3.0);
% streamlines (psi = const) for a uniform flow: nearly horizontal,
% deflected around a source at the origin. drawn as smooth curves.
\foreach \c in {-2.4,-1.6,-0.8,0.8,1.6,2.4} {
  \draw[stream] plot[domain=-4.2:4.2,samples=80,smooth]
    (\x, {\c + 0.55*(1 + tanh(1.4*\x)) - 0.55});
}
% the central dividing streamline through the stagnation point
\draw[stream] plot[domain=-4.2:-1.0,samples=60,smooth]
  (\x, {0.0 + 0.0*\x});
% equipotentials (phi = const): nearly vertical, curving near source
\foreach \d in {-3.2,-2.4,-1.6,1.6,2.4,3.2} {
  \draw[equipot] plot[domain=-2.6:2.6,samples=80,smooth]
    ({\d - 0.35*(1/(1+0.18*\x*\x))*\d/abs(\d)}, \x);
}
% the source
\filldraw[engblack] (0,0) circle (2pt);
\node[label, engblack, anchor=south west] at (0.1,0.08) {source};
\end{scope}
% frame
\draw[enggray, thin] (-4.2,-3.0) rectangle (4.2,3.0);
% legend
\node[label, engblue, anchor=west] at (-4.1,3.4) {streamlines $\psi=\text{const}$};
\node[label, engred, anchor=west] at (1.0,3.4) {equipotentials $\phi=\text{const}$};
\end{tikzpicture}
\caption[Orthogonal potential and stream-function nets]{Equipotential
lines $\phi=\text{const}$ (in red) and streamlines $\psi=\text{const}$
(in blue) for a point source set in a uniform stream. The two families
cross at right angles everywhere, the geometric signature of the
Cauchy-Riemann relations $\partial\phi/\partial x =
\partial\psi/\partial y$ and $\partial\phi/\partial y =
-\partial\psi/\partial x$: the gradient of $\phi$ points along the flow
and the gradient of $\psi$ points across it, so the level sets are
mutually perpendicular. A flow net drawn by hand exploits this
orthogonality, the squares of the net shrink where the flow speeds up,
giving the velocity field directly from the spacing. The picture is a
sketch of the topology, not a contour plot to scale.}
\label{fig:vol03ch11:potential-streamfunction}
\end{figure}
