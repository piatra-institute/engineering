% Figure: a free liquid jet striking a stationary vane that turns it through an
% angle theta; the control-volume momentum balance gives the force on the vane.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  wall/.style={draw=engblack, very thick},
  stream/.style={draw=engblue, -{Stealth}, thick},
  force/.style={draw=engred, -{Stealth}, very thick},
  label/.style={font=\scriptsize},
]
% Incoming jet
\draw[stream] (-4,0) -- (-1.1,0);
\draw[stream] (-4,0.22) -- (-1.2,0.22);
\draw[stream] (-4,-0.22) -- (-1.2,-0.22);
\node[label, engblue, anchor=south] at (-3.0,0.28) {$\dot m,\ u_1$};
% Vane: a curved surface turning the jet through angle theta (here ~120 deg)
\draw[wall] (-1.0,-0.6) .. controls (0.4,-0.6) and (1.2,0.2) .. (1.1,1.6);
\draw[wall] (-1.0,-0.2) .. controls (0.0,-0.2) and (0.7,0.2) .. (0.7,1.6);
% Outgoing jet (turned)
\draw[stream] (0.9,1.7) -- (1.7,3.1);
\node[label, engblue, anchor=west] at (1.4,2.6) {$u_2=u_1$};
% Turning-angle arc
\draw[engamber, thick, -{Stealth}] (1.6,0.0) arc[start angle=0, end angle=110, radius=1.6];
\node[label, engamber] at (1.2,1.2) {$\theta$};
% Reaction force on the vane (pointing back along the bisector)
\draw[force] (0.0,0.4) -- (-1.6,-0.7);
\node[label, engred, anchor=north east] at (-1.4,-0.6) {$\vec R$ on vane};
% Datum / centreline
\draw[enggray, dashed] (-4,0) -- (1.9,0);
\node[label, anchor=north] at (0,-0.9) {turning vane (fixed)};
\end{tikzpicture}
\caption[Jet impingement on a fixed turning vane]{A free liquid jet
of mass flow $\dot m$ and speed $u_1$ strikes a fixed vane that turns
it through angle $\theta$. The jet stays at atmospheric pressure
throughout, so for an inviscid vane the speed is unchanged,
$u_2=u_1$, and only the direction of the momentum flux changes. The
steady momentum theorem applied to a control volume around the vane
gives the force the vane exerts on the fluid as $\dot m(\vec u_2-\vec
u_1)$, and the reaction $\vec R$ on the vane is its negative. The
streamwise component is $\dot m\,u_1(1-\cos\theta)$, largest for a
half-pipe bucket that reverses the jet ($\theta=180^\circ$), the
geometry of a Pelton turbine.}
\label{fig:vol03ch11:jet-vane}
\end{figure}
