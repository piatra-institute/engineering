% Figure: streamlines around a rotating cylinder in a uniform stream. The bound
% circulation crowds the streamlines over the top (fast, low pressure) and opens
% them out below (slow, high pressure), giving the transverse Magnus force.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  stream/.style={draw=engblue, thick},
  force/.style={draw=engred, -{Stealth}, very thick},
  spin/.style={draw=engamber, -{Stealth}, thick},
  label/.style={font=\scriptsize},
]
% freestream arrows on the left
\foreach \y in {-1.8,-1.2,-0.6,0.6,1.2,1.8}{
  \draw[engblue,-{Stealth},thick] (-4.3,\y) -- (-3.6,\y);
}
\node[label, engblue, anchor=east] at (-4.3,0) {$U_\infty$};
% cylinder
\filldraw[fill=enggray!20, draw=engblack, very thick] (0,0) circle (0.75);
% rotation sense
\draw[spin] (0,0) ++(35:1.05) arc[start angle=35, end angle=125, radius=1.05];
\node[label, engamber] at (0,1.35) {$\Gamma$};
% streamlines: crowded above, open below (hand-drawn smooth curves)
\draw[stream] (-3.6,0.6) .. controls (-1.4,0.9) and (-0.9,1.35) .. (0,1.42)
   .. controls (0.9,1.35) and (1.4,0.9) .. (3.6,0.6);
\draw[stream] (-3.6,1.2) .. controls (-1.4,1.45) and (-0.9,1.85) .. (0,1.95)
   .. controls (0.9,1.85) and (1.4,1.45) .. (3.6,1.2);
\draw[stream] (-3.6,1.8) -- (3.6,1.8);
\draw[stream] (-3.6,-0.6) .. controls (-1.6,-0.75) and (-1.0,-0.98) .. (0,-1.02)
   .. controls (1.0,-0.98) and (1.6,-0.75) .. (3.6,-0.6);
\draw[stream] (-3.6,-1.2) .. controls (-1.6,-1.28) and (-1.0,-1.42) .. (0,-1.45)
   .. controls (1.0,-1.42) and (1.6,-1.28) .. (3.6,-1.2);
\draw[stream] (-3.6,-1.8) -- (3.6,-1.8);
% fast / slow labels
\node[label, engblack, anchor=south] at (0,2.05) {fast, low $p$ (crowded)};
\node[label, engblack, anchor=north] at (0,-2.0) {slow, high $p$ (open)};
% transverse force
\draw[force] (0,0.75) -- (0,2.55);
\node[label, engred, anchor=west] at (0.1,2.35) {$L' = \rho U_\infty \Gamma$};
\end{tikzpicture}
\caption[The Magnus force on a rotating cylinder]{A cylinder spinning
in a uniform stream carries a bound circulation $\Gamma$ that adds to
the freestream over the top and subtracts below. The streamlines
crowd together above the cylinder, so the flow there runs fast and the
pressure falls; below, the streamlines open out, the flow slows, and
the pressure rises. The pressure difference integrated over the
surface is the transverse Magnus force $L' = \rho U_\infty\Gamma$, the
Kutta-Joukowski lift with the circulation supplied by the spin rather
than by camber and angle of attack. The same mechanism curves a
sliced golf ball and a topspin tennis shot.}
\label{fig:vol03ch11:magnus-cylinder}
\end{figure}
