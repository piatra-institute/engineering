% Figure: the parabolic free surface of a liquid in solid-body rotation
% (forced vortex), z(r) = z0 + omega^2 r^2 / (2g).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.2cm,
  xlabel={radius $r$ (m)},
  ylabel={free-surface height $z$ (m)},
  xmin=-0.3, xmax=0.3, ymin=0, ymax=0.5,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  legend pos=north west,
]
% Container walls
\addplot[engblack, very thick, samples=2] coordinates {(-0.25,0) (-0.25,0.5)};
\addplot[engblack, very thick, samples=2] coordinates {(0.25,0) (0.25,0.5)};
% Paraboloid surface: z = 0.08 + omega^2 r^2 / (2g), with omega^2/(2g) chosen
% so the rise across r=0.25 is about 0.28 m (omega ~ 14.8 rad/s).
\addplot[engblue, very thick, samples=120, domain=-0.25:0.25] {0.08 + 4.5*x*x};
\addlegendentry{$z(r)=z_0+\dfrac{\omega^2 r^2}{2g}$}
% Rest level (no rotation) for reference, conserving volume
\addplot[engamber, dashed, very thick, samples=2, domain=-0.25:0.25] {0.22};
\addlegendentry{rest level (no spin)}
% Centreline depression annotation
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:-0.23,0.10) {centre dips};
\node[engblue, font=\scriptsize, anchor=east] at (axis cs:0.23,0.40) {wall climbs};
\end{axis}
\end{tikzpicture}
\caption[Free surface of a forced vortex]{A liquid rotating as a
rigid body at angular speed $\omega$ in an open cylinder takes a
paraboloidal free surface $z(r)=z_0+\omega^2 r^2/(2g)$. The pressure
field carries a radial term $\tfrac12\rho\omega^2 r^2$ that balances
the centripetal acceleration, so the surface rises toward the wall
and dips at the axis while the mean level (dashed) is fixed by volume
conservation. The shape is a true parabola in $r$; spinning a bucket
of water and reading the depth at the centre against the depth at the
wall measures $\omega$ without a tachometer. The same paraboloid is
the working surface of a spin-cast mirror.}
\label{fig:vol03ch11:forced-vortex}
\end{figure}
