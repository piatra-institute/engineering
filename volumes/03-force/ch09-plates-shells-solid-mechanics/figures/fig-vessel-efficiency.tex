% Figure: required wall thickness for sphere vs cylinder vs hemispherical
% cap as a function of pressure, at fixed radius and allowable stress.
% t = pR/(2 sigma) for sphere/cap, t = pR/sigma for cylinder hoop.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={internal pressure $p$ (MPa)},
  ylabel={required wall thickness $t$ (mm)},
  xmin=0, xmax=6, ymin=0, ymax=14,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize}, legend pos=north west,
]
% R = 0.5 m = 500 mm, sigma_allow = 120 MPa.
% cylinder hoop: t = pR/sigma = p*500/120 mm (p in MPa) = 4.167*p
\addplot[engred, very thick, domain=0:6, samples=2] {4.167*x};
\addlegendentry{cylinder (hoop)}
% sphere / hemispherical cap: t = pR/(2 sigma) = 2.083*p
\addplot[engblue, very thick, domain=0:6, samples=2] {2.083*x};
\addlegendentry{sphere / hemispherical cap}
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:3.1,13.0)
  {twice the wall};
\node[engblue, font=\scriptsize, anchor=north west] at (axis cs:4.2,8.0)
  {half the wall};
\end{axis}
\end{tikzpicture}
\caption[Wall thickness: sphere versus cylinder]{Required wall
thickness against internal pressure for a sphere and a cylinder of
the same radius ($R = 0.5\,\text{m}$) and allowable stress
($\sigma_{\text{allow}} = 120\,\text{MPa}$), from the membrane
results $t = pR/(2\sigma)$ for the sphere and $t = pR/\sigma$ for the
cylinder hoop. The sphere needs exactly half the wall thickness of
the cylinder, the quantitative statement of the sphere being the
most material efficient pressure containing geometry. The
hemispherical end cap follows the sphere line, which is why a
hemispherical cap on a cylindrical body can be made thinner than the
cylinder wall it closes.}
\label{fig:vol03ch09:vessel-efficiency}
\end{figure}
