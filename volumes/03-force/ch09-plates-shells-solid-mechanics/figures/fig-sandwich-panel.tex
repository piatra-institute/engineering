% Figure: cross-section of a sandwich panel. Two thin stiff facesheets
% separated by a thick light core. The bending stiffness scales with the
% square of the separation, so a sandwich is stiff per unit weight; the
% core carries the transverse shear. Schematic, not to scale.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % core
  \fill[enggray!15] (0,0) rectangle (8,1.6);
  \draw[enggray, thick] (0,0) rectangle (8,1.6);
  % top facesheet
  \fill[engblue!30] (0,1.6) rectangle (8,1.85);
  \draw[engblue, very thick] (0,1.6) rectangle (8,1.85);
  % bottom facesheet
  \fill[engblue!30] (0,-0.25) rectangle (8,0);
  \draw[engblue, very thick] (0,-0.25) rectangle (8,0);
  % core honeycomb hint (vertical hatching)
  \foreach \x in {0.5,1.0,...,7.5}{
    \draw[enggray!50, thin] (\x,0.05) -- (\x,1.55);
  }
  % facesheet thickness callout
  \draw[engblue, <->] (8.25,1.6) -- (8.25,1.85);
  \node[engblue, font=\scriptsize, anchor=west] at (8.35,1.72) {face $t_f$};
  % core thickness callout
  \draw[enggray, <->] (8.25,0) -- (8.25,1.6);
  \node[enggray, font=\scriptsize, anchor=west] at (8.35,0.8) {core $c$};
  % separation between face centroids
  \draw[engred, <->] (-0.3,0.125) -- (-0.3,1.725);
  \node[engred, font=\scriptsize, anchor=east, rotate=90] at (-0.45,0.92)
    {centroid spacing $d$};
  % labels
  \node[engblue, font=\scriptsize, anchor=west] at (0.2,1.72) {stiff facesheet};
  \node[enggray, font=\scriptsize, anchor=west] at (0.2,0.8) {light shear core};
  \node[engblue, font=\scriptsize, anchor=west] at (0.2,-0.13) {stiff facesheet};
\end{tikzpicture}
\caption[Sandwich-panel cross-section]{Cross-section of a sandwich
panel: two thin stiff facesheets of thickness $t_f$ bonded to a thick
light core of thickness $c$, with centroid spacing $d = c + t_f$. The
facesheets carry the in-plane membrane forces of bending, tension on one
face and compression on the other, and the core carries the transverse
shear and holds the faces apart. Moving the same face material outward
raises the bending stiffness as the square of the separation, so a
sandwich is far stiffer per unit weight than a solid plate of the same
mass. The price is a new failure mode the solid plate does not have:
face wrinkling, core shear, and debonding, each of which the solid-plate
bending check never sees.}
\label{fig:vol03ch09:sandwich-panel}
\end{figure}
