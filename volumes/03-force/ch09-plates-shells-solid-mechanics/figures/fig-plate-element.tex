% Figure: an infinitesimal plate element with bending and twisting moments
% M_xx, M_yy, M_xy and transverse shears Q_x, Q_y per unit length. No
% em-dashes anywhere; use = dividers or words in comments.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=2.2,>=latex]
  % midsurface element, drawn in oblique projection
  \coordinate (A) at (0,0);
  \coordinate (B) at (1.6,0);
  \coordinate (C) at (1.6+0.6,0.5);
  \coordinate (D) at (0.6,0.5);
  \draw[engblack, thick] (A)--(B)--(C)--(D)--cycle;
  \node[engblack, font=\scriptsize] at (0.8,-0.18) {$dx$};
  \node[engblack, font=\scriptsize] at (1.95,0.28) {$dy$};
  % bending moment M_xx on the +x face (double arrow = moment)
  \draw[engblue, ultra thick, ->] (B)++(0.05,0.05) arc (200:340:0.18);
  \node[engblue, font=\scriptsize] at (2.0,0.0) {$M_{xx}$};
  % bending moment M_yy on the +y face
  \draw[engred, ultra thick, ->] (C)++(-0.1,0.05) arc (20:160:0.18);
  \node[engred, font=\scriptsize] at (1.4,0.62) {$M_{yy}$};
  % twisting moment M_xy
  \draw[engamber, ultra thick, ->] (D)++(0.05,0.0) arc (160:300:0.14);
  \node[engamber, font=\scriptsize] at (0.35,0.30) {$M_{xy}$};
  % transverse shear Q_x as out-of-plane arrow at the +x face
  \draw[englime, ultra thick, ->] (1.7,0.25) -- (1.7,0.62);
  \node[englime, font=\scriptsize] at (1.78,0.5) {$Q_x$};
  % coordinate triad
  \draw[enggray,->] (-0.4,0) -- (0.0,0) node[below left,font=\tiny]{$x$};
  \draw[enggray,->] (-0.4,0) -- (-0.25,0.18) node[above,font=\tiny]{$y$};
  \draw[enggray,->] (-0.4,0) -- (-0.4,0.4) node[left,font=\tiny]{$z,w$};
\end{tikzpicture}
\caption[Stress resultants on a plate element]{The bending moments
$M_{xx}$, $M_{yy}$, the twisting moment $M_{xy}$, and the transverse
shear $Q_x$ acting per unit length on the faces of an infinitesimal
plate element of size $dx \times dy$. The moments are the through
thickness first moments of the in plane stresses; the plate bending
equation $D\nabla^4 w = q$ follows from transverse force equilibrium
of this element combined with the moment curvature relations.}
\label{fig:vol03ch09:plate-element}
\end{figure}
