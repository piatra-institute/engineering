% Figure: shear locking. A bending-dominated thin element modelled with a
% single linear element through the thickness cannot represent pure bending:
% straight edges stay straight, producing spurious shear strain.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0,>=latex]
  % LEFT: true bending of a fibre (curved)
  \begin{scope}[shift={(0,0)}]
    \draw[enggray, thin] (0,0) rectangle (2,1);
    \draw[engblue, very thick] (0,0.5) .. controls (0.7,0.95) and (1.3,0.95) .. (2,0.5);
    \node[font=\scriptsize, anchor=north] at (1,-0.15) {true bending};
    \node[engblue, font=\tiny] at (1,1.15) {fibre curves; faces stay normal};
  \end{scope}
  % RIGHT: linear element approximation (straight edges) => parasitic shear
  \begin{scope}[shift={(3.5,0)}]
    % deformed linear quad: top edge tilts, but edges stay straight lines
    \draw[engred, very thick] (0,0) -- (2,0.0) -- (2,1.0) -- (0,1.0) -- cycle;
    % show the originally-vertical edge now sheared
    \draw[engred, very thick, dashed] (0.0,0.0) -- (0.4,1.0);
    \draw[engred, very thick, dashed] (2.0,0.0) -- (2.4,1.0);
    \draw[engamber,->] (1.0,0.5) -- (1.5,0.5);
    \node[engamber, font=\tiny, anchor=west] at (1.5,0.5) {spurious $\gamma_{xz}$};
    \node[font=\scriptsize, anchor=north] at (1,-0.15) {linear element};
    \node[engred, font=\tiny] at (1,1.25) {straight edges cannot curve};
  \end{scope}
\end{tikzpicture}
\caption[Shear locking in a linear element]{Why a single linear solid
element through the thickness over predicts the stiffness of a thin
bending member. In true bending (left) the fibres curve and cross
sections rotate while staying normal to the midsurface. A linear
element (right) interpolates displacement with straight edges, so it
cannot represent the curvature; it instead develops a parasitic
transverse shear strain $\gamma_{xz}$ that absorbs energy and makes
the element behave far too stiffly. This is shear locking; the cure
is shell elements, reduced integration, or quadratic elements with
several through the thickness.}
\label{fig:vol03ch09:shear-locking}
\end{figure}
