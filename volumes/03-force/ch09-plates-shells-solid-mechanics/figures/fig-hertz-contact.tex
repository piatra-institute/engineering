% Figure: subsurface stress in Hertzian point/line contact.
% Plots the normalised principal-shear stress tau_max/p_max along the depth
% axis beneath the centre of a two-cylinder (line) contact, showing that the
% peak shear sits BELOW the surface, at about z = 0.78 b for line contact,
% where p_max is the peak contact pressure and b the half-width.
% The buried maximum is why rolling-contact fatigue spalls from below.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=6.5cm,
  xlabel={depth below surface $\ z/b$},
  ylabel={stress / $p_{\max}$},
  xmin=0, xmax=3.0, ymin=-1.05, ymax=0.05,
  axis lines=left,
  y dir=reverse,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.98,0.06)}, anchor=south east, draw=enggray!40},
]
% normal stress sigma_z / p_max = -1/sqrt(1+(z/b)^2)  (line contact, on axis)
\addplot[engblue, very thick, domain=0:3, samples=120]
  {-1/sqrt(1 + x*x)};
\addlegendentry{$\sigma_z/p_{\max}$}
% in-plane sigma_x / p_max = -( sqrt(1+2(z/b)^2)/sqrt(1+(z/b)^2) - 2 z/b )
\addplot[engred, very thick, domain=0:3, samples=120]
  {-( sqrt(1 + 2*x*x)/sqrt(1 + x*x) - 2*x )};
\addlegendentry{$\sigma_x/p_{\max}$}
% max shear tau = (sigma_x - sigma_z)/2, magnitude, plotted negative to sit
% on the reversed axis with the compressions.
\addplot[engamber, very thick, dashed, domain=0:3, samples=120]
  {-0.5*abs( ( sqrt(1 + 2*x*x)/sqrt(1 + x*x) - 2*x ) - 1/sqrt(1 + x*x) )};
\addlegendentry{$-\tau_{\max}/p_{\max}$}
% mark the buried shear peak at z/b ~ 0.78, tau_max ~ 0.30 p_max
\addplot[engamber, only marks, mark=*, mark size=1.6pt] coordinates {(0.78,-0.30)};
\node[engamber, font=\scriptsize, anchor=west] at (axis cs:0.86,-0.30)
  {buried $\tau_{\max}\approx 0.30\,p_{\max}$};
\end{axis}
\end{tikzpicture}
\caption[Subsurface stress in Hertzian contact]{On-axis stress beneath the
centre of a Hertzian line contact of half-width $b$ and peak pressure
$p_{\max}$, plotted against depth (the axis is reversed so the surface is at
the top). The direct pressure $\sigma_z$ (blue) and the in-plane stress
$\sigma_x$ (red) are both largest at the surface, but the maximum shear stress
$\tau_{\max}=(\sigma_x-\sigma_z)/2$ (amber, dashed) does not peak at the
surface. It reaches its largest value of about $0.30\,p_{\max}$ at a depth near
$0.78\,b$, buried below the contact. A ductile material yields first at that
buried point, and a rolling contact under repeated passes initiates its fatigue
crack there and spalls a flake from below, which is why rolling-contact fatigue
of bearings and gears is a subsurface failure rather than a surface one.}
\label{fig:vol03ch09:hertz-contact}
\end{figure}
