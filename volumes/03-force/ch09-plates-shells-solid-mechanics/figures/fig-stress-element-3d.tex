% Figure: a 3D stress element (cube) with the six independent components of
% the symmetric Cauchy stress tensor labelled on the visible faces.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=2.4,>=latex]
  % cube vertices (oblique)
  \coordinate (o)  at (0,0);
  \coordinate (x)  at (1,0);
  \coordinate (y)  at (0.5,0.4);
  \coordinate (z)  at (0,1);
  \coordinate (xy) at (1.5,0.4);
  \coordinate (xz) at (1,1);
  \coordinate (yz) at (0.5,1.4);
  \coordinate (xyz) at (1.5,1.4);
  % visible faces
  \draw[engblack, thick] (o)--(x)--(xy)--(y)--cycle;       % bottom
  \draw[engblack, thick] (x)--(xz)--(xyz)--(xy)--cycle;    % right (+x)
  \draw[engblack, thick] (y)--(xy)--(xyz)--(yz)--cycle;    % back (+y)
  \draw[engblack, thick] (z)--(xz)--(xyz)--(yz)--cycle;    % top (+z)
  \draw[engblack, thick] (o)--(x); \draw[engblack, thick] (o)--(z);
  % normal stress on +x face
  \draw[engblue, ultra thick, ->] (1.25,0.7) -- (1.65,0.7);
  \node[engblue, font=\scriptsize] at (1.78,0.7) {$\sigma_{xx}$};
  % shear on +x face (along z) and (along y)
  \draw[engred, ultra thick, ->] (1.25,0.7) -- (1.25,1.05);
  \node[engred, font=\scriptsize] at (1.25,1.15) {$\tau_{xz}$};
  \draw[engamber, ultra thick, ->] (1.25,0.7) -- (1.45,0.85);
  \node[engamber, font=\scriptsize] at (1.55,0.92) {$\tau_{xy}$};
  % normal stress on +z (top)
  \draw[engblue, ultra thick, ->] (0.75,1.4) -- (0.75,1.75);
  \node[engblue, font=\scriptsize] at (0.75,1.85) {$\sigma_{zz}$};
  \draw[engamber, ultra thick, ->] (0.75,1.4) -- (1.1,1.4);
  \node[engamber, font=\scriptsize] at (1.2,1.42) {$\tau_{zx}$};
  % normal stress on +y (back)
  \draw[engblue, ultra thick, ->] (0.95,0.95) -- (1.2,1.15);
  \node[engblue, font=\scriptsize] at (1.05,1.2) {$\sigma_{yy}$};
  % axes
  \draw[enggray,->] (0,0) -- (-0.35,0) node[left,font=\tiny]{$x$};
  \draw[enggray,->] (0,0) -- (-0.2,-0.16) node[below,font=\tiny]{$y$};
  \draw[enggray,->] (0,0) -- (0,-0.35) node[below,font=\tiny]{$z$};
\end{tikzpicture}
\caption[The three dimensional stress element]{The Cauchy stress
tensor on an infinitesimal cube. The three normal components
$\sigma_{xx}, \sigma_{yy}, \sigma_{zz}$ and the three independent
shear components $\tau_{xy}, \tau_{yz}, \tau_{zx}$ together give the
six independent entries of the symmetric stress tensor
$\sigma_{ij}=\sigma_{ji}$. Symmetry follows from moment equilibrium
of the element: an unbalanced shear couple would spin the cube. These
six numbers at every point, with the strain displacement and
constitutive relations, are what the finite element method solves
for.}
\label{fig:vol03ch09:stress-element-3d}
\end{figure}
