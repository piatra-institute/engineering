% Figure: solving the clamped circular plate by superposition of two
% simpler solved problems. The clamped case (zero edge slope) is built
% from the simply supported plate under the same load (which has a free
% edge rotation) plus an edge moment chosen to cancel that rotation.
% Three small deflection profiles side by side, schematic.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % helper: a half deflected-plate profile over [0,2] sagging by depth d
  % drawn as a smooth bowl. We draw three panels.

  % == panel A: clamped (target), flat at edge ==
  \begin{scope}[shift={(0,0)}]
    \draw[enggray, thick] (-0.1,0) -- (2.1,0);          % undeformed midline
    % clamped bowl: zero slope at x=2 (edge). use (x-2)^2 shape mirrored
    \draw[engblue, very thick, domain=0:2, samples=60, smooth]
      plot (\x, {-0.55*(1 - ((\x*\x)/4))*(1 - ((\x*\x)/4))});
    \draw[engblue!60, line width=2pt] (1.95,0.12) -- (2.05,-0.12); % wall hatch
    \node[font=\scriptsize, anchor=north] at (1.0,-0.75) {clamped: $w=0,\ w'=0$ at edge};
    \node[engblue, font=\scriptsize, anchor=south] at (1.0,0.02) {target};
  \end{scope}

  % plus sign
  \node[font=\large] at (2.55,-0.3) {$=$};

  % == panel B: simply supported under load (has edge rotation) ==
  \begin{scope}[shift={(3.0,0)}]
    \draw[enggray, thick] (-0.1,0) -- (2.1,0);
    % deeper bowl with nonzero slope at edge: 1 - (x/2)^2
    \draw[englime!80!black, very thick, domain=0:2, samples=60, smooth]
      plot (\x, {-0.75*(1 - (\x*\x)/4)});
    % edge-rotation arc indicator
    \draw[engred, ->] (2.0,-0.05) arc (-30:25:0.35);
    \node[engred, font=\scriptsize, anchor=west] at (2.15,0.0) {edge rotates};
    \node[font=\scriptsize, anchor=north] at (1.0,-0.95) {simply supported, same $q$};
  \end{scope}

  % plus sign
  \node[font=\large] at (5.55,-0.3) {$+$};

  % == panel C: edge moment that cancels the rotation ==
  \begin{scope}[shift={(6.0,0)}]
    \draw[enggray, thick] (-0.1,0) -- (2.1,0);
    % upward-curving correction (edge moment lifts edge slope back to zero)
    \draw[engamber, very thick, domain=0:2, samples=60, smooth]
      plot (\x, {0.20*(\x*\x)/4});
    % applied edge moment arrows
    \draw[engamber, ->] (2.05,0.25) arc (60:-30:0.25);
    \node[engamber, font=\scriptsize, anchor=south west] at (1.55,0.35) {$M_0$};
    \node[font=\scriptsize, anchor=north] at (1.0,-0.95) {edge moment $M_0$ alone};
  \end{scope}
\end{tikzpicture}
\caption[Clamped plate by superposition]{The clamped circular plate solved
by superposition of two cases that are each easier to write down. A clamped
edge holds both the deflection and the slope to zero. The simply supported
plate under the same distributed load (centre) satisfies zero deflection at
the rim but lets the rim rotate. Adding an edge moment $M_0$ applied alone
(right), with no transverse load, bends the plate the other way and produces
a rim rotation of its own. Choosing $M_0$ so that its rim rotation exactly
cancels the simply supported rim rotation rebuilds the clamped condition
(left). The clamped centre deflection is then the simply supported value
reduced by the contribution of the cancelling moment, which is why the
clamped plate deflects about a quarter as much as the simply supported one
under the same load. The method needs only two catalogue solutions and one
algebraic condition, and it generalises to any edge restraint between the
two extremes by scaling the edge moment.}
\label{fig:vol03ch09:clamped-superposition}
\end{figure}
