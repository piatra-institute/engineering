% Figure: imperfection sensitivity of shell buckling. Knock-down factor
% (ratio of real buckling load to classical linear prediction) versus
% imperfection amplitude / thickness. Schematic of the well-known
% lower-bound decay for axially compressed cylinders.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={imperfection amplitude / wall thickness $\ \delta/h$},
  ylabel={knock-down factor $p_{\mathrm{cr}}/p_{\mathrm{cr,linear}}$},
  xmin=0, xmax=2.0, ymin=0, ymax=1.05,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% schematic decay: rho = 1/(1 + 2.5*sqrt(delta/h)) style curve
\addplot[engblue, very thick, domain=0:2.0, samples=81]
  {1/(1 + 2.5*sqrt(x))};
% design lower-bound band
\addplot[engred, dashed, domain=0:2.0, samples=2] {0.3};
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:1.0,0.31)
  {typical design knock-down $\sim 0.2$ to $0.4$};
\node[engblue, font=\scriptsize, anchor=north east] at (axis cs:0.55,0.9)
  {classical theory at $\delta=0$};
\draw[engblue,->] (axis cs:0.25,0.88) -- (axis cs:0.03,0.97);
\end{axis}
\end{tikzpicture}
\caption[Imperfection sensitivity of shell buckling]{The buckling
load of a thin shell in compression collapses as the geometric
imperfection grows, expressed here as the knock-down factor against a
normalised imperfection amplitude $\delta/h$ (schematic). The
classical linear theory (top left) assumes a perfect shell; a dimple
or flat spot of the order of one wall thickness drops the real
buckling load to $20$ to $40\,\%$ of the linear prediction. No
engineer designs a shell against the linear value; design codes use
the empirically calibrated lower band (red, dashed). Imperfection
sensitivity is the single property that separates shell stability
from plate and column buckling.}
\label{fig:vol03ch09:shell-buckling}
\end{figure}
