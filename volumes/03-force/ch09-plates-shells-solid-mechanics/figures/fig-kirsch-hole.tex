% Figure: stress concentration around a circular hole (Kirsch). Hoop stress
% on the hole boundary, sigma_theta/sigma_inf = 1 - 2cos(2theta), peaking at
% 3 at theta=90 deg (the sides of the hole, transverse to the load).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={angle $\theta$ from load axis (deg)},
  ylabel={$\sigma_\theta / \sigma_\infty$ on hole edge},
  xmin=0, xmax=180, ymin=-1.5, ymax=3.5,
  xtick={0,30,60,90,120,150,180},
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% sigma_theta/sigma_inf = 1 - 2 cos(2theta)
\addplot[engblue, very thick, domain=0:180, samples=181]
  {1 - 2*cos(2*x)};
% zero line and the K=3 marker
\addplot[enggray, dashed, domain=0:180, samples=2] {0};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:90,3.05)
  {$K_t = 3$ at $\theta=90^\circ$};
\draw[engred,->] (axis cs:90,2.9) -- (axis cs:90,3.0);
\node[engamber, font=\scriptsize, anchor=north] at (axis cs:0,-1.0)
  {$-1$ at $\theta=0^\circ$ (compression)};
\end{axis}
\end{tikzpicture}
\caption[Stress concentration at a circular hole]{Hoop stress around
the edge of a small circular hole in a wide plate under remote
uniaxial tension $\sigma_\infty$, the Kirsch (1898) solution
$\sigma_\theta/\sigma_\infty = 1 - 2\cos 2\theta$. At the sides of the
hole ($\theta = 90^\circ$, transverse to the load) the stress is
three times the remote stress, the classic stress concentration
factor $K_t = 3$. At the poles ($\theta = 0^\circ$, in line with the
load) the edge is in compression at $-\sigma_\infty$. The hole
triples the local stress without changing the average, which is why
cutouts, holes, and sharp corners are where structures crack first.}
\label{fig:vol03ch09:kirsch-hole}
\end{figure}
