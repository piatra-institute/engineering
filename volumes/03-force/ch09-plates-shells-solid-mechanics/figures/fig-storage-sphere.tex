% Figure: a spherical gas-storage vessel (a Horton sphere) on its leg
% supports, with the membrane stress sigma = pR/2t marked, and a small
% inset comparing the required wall thickness of a sphere and a cylinder of
% the same radius and allowable stress (the sphere needs half the wall).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% === the sphere ===
\draw[engblue, very thick] (0,2.3) circle (1.6);
% equator (perspective ellipse)
\draw[enggray, thin] (0,2.3) ellipse (1.6 and 0.42);
% internal pressure arrows (radial, outward)
\foreach \a in {30,90,150,210,270,330}{
  \draw[engred, ->] ({0.7*cos(\a)},{2.3+0.7*sin(\a)}) -- ({1.25*cos(\a)},{2.3+1.25*sin(\a)});
}
\node[engred, font=\scriptsize] at (0,2.3) {$p$};
% membrane stress label
\draw[engamber, <->, thick] (1.6,2.3) -- (2.3,2.3);
\node[engamber, font=\scriptsize, anchor=west] at (2.3,2.3)
  {$\sigma=\dfrac{pR}{2t}$};
% leg supports
\draw[engblack, thick] (-1.1,0.85) -- (-1.35,-0.7);
\draw[engblack, thick] (1.1,0.85) -- (1.35,-0.7);
\draw[engblack, thick] (-0.35,0.55) -- (-0.45,-0.7);
% ground
\draw[enggray, thick] (-2.0,-0.7) -- (2.0,-0.7);
\foreach \x in {-1.7,-1.0,-0.3,0.4,1.1,1.8}{
  \draw[enggray, thin] (\x,-0.7) -- (\x-0.25,-1.0);
}
% === inset: sphere vs cylinder wall ===
\begin{scope}[xshift=5.0cm, yshift=0.2cm]
\node[font=\scriptsize, anchor=south] at (0.9,3.0) {same $R$, same allowable};
% cylinder wall (thicker)
\draw[engblue, fill=engblue!12] (0,0) rectangle (0.36,2.6);
\node[font=\scriptsize, anchor=north] at (0.18,-0.05) {cylinder};
\node[font=\scriptsize, anchor=south] at (0.18,2.62) {$t$};
% sphere wall (half)
\draw[engblue, fill=engblue!12] (1.6,0) rectangle (1.78,2.6);
\node[font=\scriptsize, anchor=north] at (1.69,-0.05) {sphere};
\node[font=\scriptsize, anchor=south] at (1.69,2.62) {$t/2$};
\end{scope}
\end{tikzpicture}
\caption[A spherical gas-storage vessel]{A spherical gas-storage vessel (a
Horton sphere) carries internal pressure $p$ by equal biaxial membrane
stress $\sigma=pR/2t$ at every point, the lowest membrane stress any
pressure-containing geometry achieves for a given radius. The inset states
the design consequence: for the same radius and the same allowable stress,
the sphere needs half the wall thickness of a cylinder, whose hoop stress
$pR/t$ is twice the sphere value. The penalty the sphere pays is the cost of
forming and welding a doubly curved surface from many petal plates, and the
need for leg or column supports that introduce local bending where they meet
the shell, the same discontinuity-stress mechanism that governs every real
support detail.}
\label{fig:vol03ch09:storage-sphere}
\end{figure}
