% Figure: von Mises and Tresca yield envelopes in the 2D principal-stress
% plane (sigma1, sigma2), normalised by yield stress.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=9.5cm, height=9.5cm,
  xlabel={$\sigma_1/\sigma_Y$}, ylabel={$\sigma_2/\sigma_Y$},
  xmin=-1.6, xmax=1.6, ymin=-1.6, ymax=1.6,
  axis lines=middle,
  axis equal image,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.02,0.02)}, anchor=south west},
]
% von Mises ellipse: s1^2 - s1 s2 + s2^2 = 1, parametrised.
% Use rotated ellipse param: along 45 deg axis radius sqrt(2), across sqrt(2/3).
\addplot[engblue, very thick, samples=200, domain=0:360]
  ({sqrt(2)*cos(x)*cos(45) - sqrt(2/3)*sin(x)*sin(45)},
   {sqrt(2)*cos(x)*sin(45) + sqrt(2/3)*sin(x)*cos(45)});
\addlegendentry{von Mises}
% Tresca hexagon vertices in (s1,s2)
\addplot[engred, dashed, very thick] coordinates {
  (1,0) (1,1) (0,1) (-1,0) (-1,-1) (0,-1) (1,0)
};
\addlegendentry{Tresca}
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:0.7,0.95)
  {safe inside};
\end{axis}
\end{tikzpicture}
\caption[Yield envelopes in principal stress space]{The von Mises
(blue ellipse) and Tresca (red hexagon) yield criteria in the plane
of the two in plane principal stresses, normalised by the uniaxial
yield stress $\sigma_Y$. A stress state inside the curve is elastic;
a state on the curve is at yield. The von Mises criterion,
$\sigma_1^2 - \sigma_1\sigma_2 + \sigma_2^2 = \sigma_Y^2$, is the one
that finite element post processors report as von Mises stress; it is
the criterion to compare against the allowable when reading an FEM
contour plot. The Tresca hexagon is inscribed in the ellipse and is
the more conservative of the two.}
\label{fig:vol03ch09:von-mises-envelope}
\end{figure}
