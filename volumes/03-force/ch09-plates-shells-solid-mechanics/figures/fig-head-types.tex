% Figure: comparison of pressure-vessel head profiles and their
% relative required thickness at equal pressure, radius, and allowable
% stress. Hemispherical thinnest, ellipsoidal next, torispherical next,
% flat thickest. Bar chart of a representative thickness multiplier.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  ybar,
  bar width=22pt,
  xlabel={head type},
  ylabel={wall thickness relative to cylinder},
  ymin=0, ymax=4.2,
  symbolic x coords={hemispherical, ellipsoidal, torispherical, flat},
  xtick=data,
  x tick label style={font=\scriptsize, rotate=20, anchor=north east},
  tick label style={font=\scriptsize},
  label style={font=\small},
  nodes near coords,
  nodes near coords style={font=\scriptsize},
  grid=major, grid style={enggray!20},
  axis lines=left,
]
% representative multipliers relative to the cylinder hoop thickness:
% hemisphere 0.5 (carries pR/2h), 2:1 ellipsoidal ~1.0, torispherical
% ~1.1 (governed by the knuckle), flat head several times thicker.
\addplot[fill=engblue!70, draw=engblue] coordinates {
  (hemispherical, 0.5)
  (ellipsoidal, 1.0)
  (torispherical, 1.1)
  (flat, 3.8)
};
\end{axis}
\end{tikzpicture}
\caption[Pressure-vessel head thickness by type]{Wall thickness of a
pressure-vessel head relative to the cylindrical shell it closes, for
equal pressure, radius, and allowable stress (representative values).
The hemispherical head carries $pR/2h$ and needs about half the shell
thickness; a $2{:}1$ ellipsoidal head needs roughly the shell
thickness; a torispherical head is a little more because the knuckle
arc concentrates stress; a flat head carries the pressure in bending
rather than membrane and is several times thicker. The progression is
the design reason hemispherical and ellipsoidal heads dominate where
geometry and cost allow, and flat heads survive only on small, low
pressure closures.}
\label{fig:vol03ch09:head-types}
\end{figure}
