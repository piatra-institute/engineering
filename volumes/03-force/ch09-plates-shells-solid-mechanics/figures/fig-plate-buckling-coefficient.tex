% Figure: the buckling coefficient k of a simply supported rectangular plate
% under uniaxial in-plane compression, plotted against the aspect ratio a/b.
% The classic scalloped curve: k = (m b/a + a/(m b))^2 minimised over the
% integer half-wave number m, with a floor at k = 4 and cusps where the
% buckle switches from m to m+1 half-waves at a/b = sqrt(m(m+1)).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={aspect ratio $\ a/b$},
  ylabel={buckling coefficient $\ k$},
  xmin=0.4, xmax=4.2, ymin=3.8, ymax=7.2,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east, draw=enggray!40},
]
% branch m=1: k = (b/a + a/b)^2, valid up to a/b = sqrt(2) ~ 1.414
\addplot[engblue, very thick, domain=0.4:1.4142, samples=120]
  {(1/x + x)^2};
% branch m=2: k = (2 b/a + a/(2 b))^2, valid sqrt(2) to sqrt(6) ~ 2.449
\addplot[engblue, very thick, domain=1.4142:2.4495, samples=120]
  {(2/x + x/2)^2};
% branch m=3: valid sqrt(6) to sqrt(12) ~ 3.464
\addplot[engblue, very thick, domain=2.4495:3.4641, samples=120]
  {(3/x + x/3)^2};
% branch m=4: valid sqrt(12) to sqrt(20) ~ 4.472
\addplot[engblue, very thick, domain=3.4641:4.2, samples=120]
  {(4/x + x/4)^2};
% the k=4 floor
\addplot[engred, dashed, domain=0.4:4.2, samples=2] {4};
\addlegendentry{$k=4$ floor}
% mark the cusps where m switches
\addplot[engamber, only marks, mark=*, mark size=1.4pt] coordinates
  {(1.4142,4.5) (2.4495,4.1667) (3.4641,4.0833)};
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:0.75,6.5) {$m=1$};
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:1.9,4.55) {$m=2$};
\node[engblue, font=\scriptsize, anchor=south] at (axis cs:2.95,4.35) {$m=3$};
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:0.45,4.02)
  {long-plate limit};
\end{axis}
\end{tikzpicture}
\caption[Plate buckling coefficient versus aspect ratio]{Buckling
coefficient $k$ of a simply supported rectangular plate under uniaxial
in-plane compression, as a function of the aspect ratio $a/b$ (load along
$a$). The critical stress is $\sigma_{\mathrm{cr}} = k\,\pi^2 D/(b^2 h)$,
so $k$ carries the whole geometric dependence. Each branch (blue) is one
integer number $m$ of buckle half-waves along the loaded direction; the
plate always chooses the $m$ that gives the lowest $k$, so the operating
curve is the lower envelope, a scalloped line that dips to $k=4$ at every
cusp (amber) where the buckle adds a half-wave, at $a/b = \sqrt{m(m+1)}$.
A long plate settles at the floor $k=4$ (red dashed) and buckles into
roughly square panels whatever its length, which is the reason a long
compressed plate is stiffened at a spacing near its width rather than left
to span its full length. The floor $k=4$ is the number a designer carries
for any plate longer than it is wide.}
\label{fig:vol03ch09:plate-buckling-coefficient}
\end{figure}
