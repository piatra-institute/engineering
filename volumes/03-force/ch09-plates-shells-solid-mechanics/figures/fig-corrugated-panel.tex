% Figure: a corrugated (trapezoidally folded) sheet and its section. The fold
% moves material away from the midplane, so the effective bending stiffness in
% the corrugation direction rises with the profile depth squared while the
% mass per area rises only a little. A flat sheet is drawn below for contrast.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0,>=latex]
  % --- corrugated profile (trapezoidal) ---
  \begin{scope}[shift={(0,0.9)}]
    \draw[engblue, very thick]
      (0,0) -- (0.5,0) -- (0.9,0.7) -- (1.5,0.7) -- (1.9,0)
      -- (2.5,0) -- (2.9,0.7) -- (3.5,0.7) -- (3.9,0)
      -- (4.5,0) -- (4.9,0.7) -- (5.5,0.7) -- (5.9,0);
    % depth callout
    \draw[engamber,<->] (6.15,0) -- (6.15,0.7) node[midway,right,font=\scriptsize]{$d$};
    % neutral-plane dashed line
    \draw[enggray, dashed] (0,0.35) -- (6.0,0.35);
    \node[engblue, font=\small, anchor=south west] at (0,0.85) {corrugated: stiff along the folds};
  \end{scope}
  % --- flat sheet for contrast ---
  \begin{scope}[shift={(0,-0.4)}]
    \fill[engred!25] (0,-0.04) rectangle (6.0,0.04);
    \draw[engred, very thick] (0,0) -- (6.0,0);
    \draw[engamber,<->] (6.15,-0.06) -- (6.15,0.06) node[midway,right,font=\scriptsize]{$h$};
    \node[engred, font=\small, anchor=north west] at (0,-0.15) {flat sheet of the same metal, same mass};
  \end{scope}
\end{tikzpicture}
\caption[Corrugated panel stiffness]{A corrugated sheet (blue) and a flat
sheet of the same metal and nearly the same mass per unit area (red). Folding
the sheet into a trapezoidal profile of depth $d$ moves the material away from
the panel's neutral plane (grey dashed), so the second moment of area per unit
width, and with it the bending stiffness along the corrugation direction,
rises roughly as the square of the profile depth $d$ while the mass grows only
by the small extra sheet length the folds consume. A corrugation a few
centimetres deep can be one or two orders of magnitude stiffer in bending than
the flat sheet it was rolled from, which is why roofing sheet, cardboard,
shipping containers, and aircraft floor panels are corrugated rather than
flat. The stiffness is strongly directional: the panel is stiff along the
folds and much softer across them, so a corrugated panel is an orthotropic
plate, and it is designed with the corrugations running along the span it has
to carry. The price is the same set of local modes a sandwich panel has, the
thin walls of the individual folds can buckle or crush before the panel as a
whole is in trouble.}
\label{fig:vol03ch09:corrugated-panel}
\end{figure}
