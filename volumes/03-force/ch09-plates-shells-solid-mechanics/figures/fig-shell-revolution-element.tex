% Figure: an axisymmetric element of a shell of revolution, showing the
% meridional resultant N_phi and the circumferential (hoop) resultant
% N_theta acting on the two pairs of edges, and the two principal radii of
% curvature r1 (meridian) and r2 (circumference). The figure makes the two
% membrane equilibrium equations readable: the load is carried by the two
% resultants weighted by the reciprocals of the two radii.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% === left: the shell of revolution with the element marked ===
% a dome profile (quarter ellipse) revolved; draw the meridian and a
% schematic element patch
\draw[engblue, very thick] (0,0) .. controls (0,2.4) and (1.6,3.2) .. (3.2,3.2);
\draw[engblue, very thick, dashed] (0,0) .. controls (0,-0.5) .. (0.6,-0.7);
% axis of revolution
\draw[enggray, dash dot] (0,-1.0) -- (0,4.0) node[above, font=\scriptsize]{axis};
% element patch on the meridian near the middle
\fill[englime!25] (1.55,2.55) -- (2.05,2.85) -- (2.25,2.55) -- (1.78,2.28) -- cycle;
\draw[engblack, thin] (1.55,2.55) -- (2.05,2.85) -- (2.25,2.55) -- (1.78,2.28) -- cycle;
% meridional resultant arrows (along the meridian)
\draw[engred, ->, thick] (2.05,2.85) -- (2.5,3.2) node[right, font=\scriptsize]{$N_\phi$};
\draw[engred, ->, thick] (1.78,2.28) -- (1.33,1.93) node[left, font=\scriptsize]{$N_\phi$};
% hoop resultant arrows (around the parallel circle)
\draw[engamber, ->, thick] (2.25,2.55) -- (2.7,2.4) node[right, font=\scriptsize]{$N_\theta$};
\draw[engamber, ->, thick] (1.55,2.55) -- (1.1,2.7) node[left, font=\scriptsize]{$N_\theta$};
% normal pressure on the patch
\draw[engblue, ->, thick] (1.9,2.55) -- (2.35,2.9);
\node[engblue, font=\scriptsize, anchor=west] at (2.35,2.95) {$p$};
\node[enggray, font=\scriptsize, anchor=north] at (1.7,-0.9) {meridian};
% === right: the principal-radii inset ===
\begin{scope}[xshift=5.4cm, yshift=0.4cm]
% normal at a point, the two radii to the two centres of curvature
\draw[engblack, very thick] (0,0) .. controls (0.4,1.6) .. (1.6,2.4);
\node[engblack, font=\scriptsize, anchor=north] at (0.8,0.0) {surface};
% point P
\fill[engblack] (0.7,1.2) circle (1.2pt);
\node[font=\scriptsize, anchor=south east] at (0.7,1.2) {$P$};
% meridian radius r1 (tangent-plane curvature in the meridian)
\draw[engred, ->] (0.7,1.2) -- (-0.9,2.0) node[left, font=\scriptsize]{$r_1$ (meridian)};
% hoop radius r2 (to the axis along the normal)
\draw[engamber, ->] (0.7,1.2) -- (2.3,0.4) node[right, font=\scriptsize]{$r_2$ (hoop)};
\node[enggray, font=\scriptsize, anchor=north west, align=left] at (-0.9,3.3)
  {$\dfrac{N_\phi}{r_1}+\dfrac{N_\theta}{r_2}=p$};
\end{scope}
\end{tikzpicture}
\caption[Membrane resultants on a shell of revolution]{An axisymmetric
element of a shell of revolution carries a normal load $p$ by two in-plane
stress resultants: the meridional resultant $N_\phi$, acting along the
generating curve, and the circumferential (hoop) resultant $N_\theta$,
acting around the parallel circle. The local shape is set by two principal
radii of curvature, $r_1$ in the meridian plane and $r_2$ measured along the
normal to the axis. Normal equilibrium of the element reduces to the single
Laplace-Young relation $N_\phi/r_1 + N_\theta/r_2 = p$, the membrane
master equation from which the sphere, the cylinder, and the cone all
follow by inserting their radii. The second membrane equation, force
balance along the meridian, fixes $N_\phi$ from the load carried below the
parallel circle, after which $N_\theta$ follows from the relation shown.}
\label{fig:vol03ch09:shell-revolution-element}
\end{figure}
