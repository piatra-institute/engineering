% Figure: schematic of the ring-stiffened vacuum vessel of the second
% case study. A horizontal cylindrical shell carrying external pressure,
% with internal ring stiffeners at spacing L_s that shorten the
% unsupported length and raise the inter-ring buckling pressure.
% Schematic only; proportions are not to scale.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % cylindrical shell as a rectangle
  \draw[engblue, very thick] (0,-1.1) rectangle (8,1.1);
  % dished ends (arcs)
  \draw[engblue, very thick] (0,1.1) .. controls (-1.0,0.7) and (-1.0,-0.7) .. (0,-1.1);
  \draw[engblue, very thick] (8,1.1) .. controls (9.0,0.7) and (9.0,-0.7) .. (8,-1.1);
  % internal ring stiffeners (vertical bars inside the shell)
  \foreach \x in {1.6,3.2,4.8,6.4}{
    \draw[engred, very thick] (\x,-1.1) -- (\x,1.1);
    \draw[engred, line width=1.6pt] (\x,-0.75) -- (\x,0.75);
  }
  % ring-spacing dimension between two rings
  \draw[enggray, <->] (1.6,1.45) -- (3.2,1.45);
  \node[enggray, font=\scriptsize, anchor=south] at (2.4,1.5) {ring spacing $L_s$};
  % shell radius dimension
  \draw[enggray, <->] (-0.3,-1.1) -- (-0.3,1.1);
  \node[enggray, font=\scriptsize, anchor=east, rotate=90] at (-0.45,0.0) {$D = 2R$};
  % external pressure arrows pushing inward
  \foreach \x in {0.8,2.4,4.0,5.6,7.2}{
    \draw[engamber, ->] (\x,1.75) -- (\x,1.2);
    \draw[engamber, ->] (\x,-1.75) -- (\x,-1.2);
  }
  \node[engamber, font=\scriptsize, anchor=west] at (8.3,1.5) {external $p$};
  % one ring callout
  \draw[engred, ->] (4.8,-1.55) -- (4.8,-1.15);
  \node[engred, font=\scriptsize, anchor=north] at (4.8,-1.6) {ring stiffener};
\end{tikzpicture}
\caption[Ring-stiffened vacuum vessel layout]{Layout of the
ring-stiffened cylindrical vessel sized against external pressure in the
second case study. Internal ring stiffeners at spacing $L_s$ shorten the
unsupported shell length, raising the inter-ring buckling pressure from
the bare long-cylinder value toward the short-cylinder regime. The
external pressure $p$ presses inward all around the shell; the governing
failure is inward buckling of the shell between rings, not material
yield. The rings themselves carry the line load shed by the shell and
must be sized against their own general-instability collapse. The figure
fixes the geometry the case study sizes; it is schematic and not to
scale.}
\label{fig:vol03ch09:ring-stiffened-vessel}
\end{figure}
