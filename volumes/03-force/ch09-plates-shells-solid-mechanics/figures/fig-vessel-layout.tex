% Figure: schematic of the worked air-receiver pressure vessel. A
% horizontal cylindrical shell closed by two torispherical heads, with a
% nozzle on the top and the governing dimensions labelled. Schematic only;
% proportions are not to scale.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % cylindrical shell as a rectangle
  \draw[engblue, very thick] (0,-1.0) rectangle (6,1.0);
  % left torispherical head (an arc bulging left)
  \draw[engblue, very thick] (0,1.0) .. controls (-1.4,0.7) and (-1.4,-0.7) .. (0,-1.0);
  % right torispherical head (an arc bulging right)
  \draw[engblue, very thick] (6,1.0) .. controls (7.4,0.7) and (7.4,-0.7) .. (6,-1.0);
  % knuckle radius callout on right head
  \draw[engred, thick] (6.0,1.0) .. controls (6.5,0.95) .. (6.7,0.55);
  \node[engred, font=\scriptsize, anchor=west] at (6.8,0.75) {knuckle $r$};
  % nozzle on top
  \draw[engamber, very thick] (3.0,1.0) rectangle (3.6,1.7);
  \node[engamber, font=\scriptsize, anchor=south] at (3.3,1.75) {nozzle};
  % reinforcing pad around nozzle
  \draw[engamber, thick, dashed] (2.6,1.0) -- (4.0,1.0);
  % diameter dimension
  \draw[enggray, <->] (-0.3,-1.0) -- (-0.3,1.0);
  \node[enggray, font=\scriptsize, anchor=east, rotate=90] at (-0.45,0.0) {$D_i = 1.2$ m};
  % length dimension
  \draw[enggray, <->] (0,-1.4) -- (6,-1.4);
  \node[enggray, font=\scriptsize, anchor=north] at (3.0,-1.45) {tangent length $L = 3.0$ m};
  % wall thickness callout
  \draw[engblue, ->] (5.9,-0.6) -- (5.4,-0.6);
  \node[engblue, font=\scriptsize, anchor=west] at (6.0,-0.6) {wall $t$};
  % internal pressure arrows
  \foreach \y in {-0.5,0,0.5}{
    \draw[englime, ->] (3.0,\y) -- (3.5,\y);

\end{tikzpicture}
\caption[Worked air-receiver layout]{Layout of the worked
air-receiver vessel: a horizontal cylindrical shell of inner diameter
$D_i = 1.2\,\text{m}$ and tangent length $L = 3.0\,\text{m}$, closed by
two torispherical heads and pierced by a single top nozzle. The
governing membrane stress is the hoop stress in the cylindrical shell;
the heads, the nozzle opening, and the head-to-shell junction each add a
local check the membrane formula alone does not supply. The figure
fixes the geometry the case study sizes; it is schematic and not to
scale.}
\label{fig:vol03ch09:vessel-layout}
\end{figure}
