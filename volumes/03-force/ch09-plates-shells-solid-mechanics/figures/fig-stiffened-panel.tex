% Figure: a stiffened panel (skin plus longitudinal stiffeners) in cross
% section, with the smeared-stiffness idea: the discrete stiffeners are
% replaced by an equivalent orthotropic plate of higher effective
% bending rigidity in the stiffener direction.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % skin
  \draw[engblue, very thick] (0,0) rectangle (8,0.3);
  \node[engblue, font=\scriptsize, anchor=east] at (-0.1,0.15) {skin $t$};
  % three blade stiffeners hanging below the skin
  \foreach \x in {1.2,4.0,6.8}{
    \draw[engred, very thick] (\x,0) -- (\x,-1.0);
    \draw[engred, very thick] (\x-0.35,-1.0) -- (\x+0.35,-1.0);
  }
  \node[engred, font=\scriptsize, anchor=north] at (4.0,-1.1) {stiffener (pitch $b$)};
  % pitch dimension
  \draw[enggray, <->] (1.2,0.55) -- (4.0,0.55);
  \node[enggray, font=\scriptsize, anchor=south] at (2.6,0.55) {$b$};
  % arrow to the smeared equivalent
  \draw[engblack, ->, thick] (8.4,-0.35) -- (9.4,-0.35);
  \node[engblack, font=\scriptsize, anchor=south] at (8.9,-0.3) {smear};
  % equivalent orthotropic plate
  \draw[englime, very thick, fill=englime!15] (9.7,-0.2) rectangle (12.0,0.3);
  \node[englime, font=\scriptsize, anchor=north, align=center] at (10.85,-0.3)
    {orthotropic plate\\$D_x \gg D_y$};
\end{tikzpicture}
\caption[Stiffened panel and its smeared equivalent]{A skin-stiffened
panel (left): a thin plate carrying transverse load is reinforced by
discrete longitudinal stiffeners at pitch $b$. For a global deflection
analysis the discrete stiffeners are smeared into an equivalent
orthotropic plate (right) whose bending rigidity in the stiffener
direction $D_x$ is much larger than in the transverse direction $D_y$,
because the stiffeners add second moment of area along their length
only. The smeared model gives the panel deflection and the global
stress; a separate local check sizes the skin panel between stiffeners
and guards against local buckling, which the smeared model cannot see.}
\label{fig:vol03ch09:stiffened-panel}
\end{figure}
