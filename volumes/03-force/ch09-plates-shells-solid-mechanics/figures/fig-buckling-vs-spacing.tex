% Figure: inter-ring buckling pressure as a function of ring spacing for
% a cylinder under external pressure. The bare long-cylinder value is the
% horizontal asymptote at large spacing; shortening the unsupported
% length by adding rings lifts the capacity steeply. A schematic of the
% Mises / short-cylinder behaviour, normalised so the long-cylinder
% value is 1.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={ring spacing / radius $\ L_s/R$},
  ylabel={buckling pressure / long-cylinder value},
  xmin=0.4, xmax=6.0, ymin=0, ymax=12,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% schematic short-cylinder lift: capacity ~ 1 + a/(L_s/R)^2 toward the
% long-cylinder asymptote of 1 at large spacing.
\addplot[engblue, very thick, domain=0.5:6.0, samples=120]
  {1 + 2.6/(x*x)};
% long-cylinder asymptote
\addplot[engred, dashed, domain=0.4:6.0, samples=2] {1};
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:5.8,1.1)
  {bare long-cylinder capacity};
% design spacing marker
\draw[enggray, dashed] (axis cs:1.0,0) -- (axis cs:1.0,3.6);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:1.05,4.4)
  {chosen $L_s/R \approx 1$};
\addplot[engamber, only marks, mark=*, mark size=1.6pt] coordinates {(1.0,3.6)};
\end{axis}
\end{tikzpicture}
\caption[Buckling pressure versus ring spacing]{Inter-ring buckling
pressure of an externally pressurised cylinder, normalised by the bare
long-cylinder value, against the ring spacing measured in radii
(schematic). At wide spacing the shell buckles as a long cylinder and
the capacity flattens to the long-cylinder asymptote (red, dashed).
Closing the rings up lifts the capacity steeply, because the
short-cylinder buckling mode is stiffer than the long-cylinder ring mode.
The design chooses a spacing (amber marker) that lifts the inter-ring
capacity, with its own knock-down applied, above the required external
differential. The curve is the reason ring stiffeners are the standard
fix for an externally loaded shell too thin to carry the pressure bare.}
\label{fig:vol03ch09:buckling-vs-spacing}
\end{figure}
