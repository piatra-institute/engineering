% Figure: deflection profile w(r) of a clamped circular plate under uniform
% load, normalised by w_max. w(r)/w_max = (1 - (r/a)^2)^2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={radial position $r/a$}, ylabel={deflection $w/w_{\max}$},
  xmin=-1.02, xmax=1.02, ymin=-0.05, ymax=1.08,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize}, legend pos=south east,
]
% clamped circular plate: w/wmax = (1 - x^2)^2
\addplot[engblue, very thick, domain=-1:1, samples=121] {(1 - x^2)^2};
\addlegendentry{clamped: $(1-(r/a)^2)^2$}
% simply supported circular plate (for comparison, normalised shape)
% w_ss(r) = (a^2-r^2)[(5+nu)/(1+nu) a^2 - r^2]; here drawn schematically,
% normalised, with nu=0.3 => factor (5.3/1.3)=4.077
\addplot[engred, dashed, very thick, domain=-1:1, samples=121]
  {((1-x^2)*(4.077 - x^2))/3.077};
\addlegendentry{simply supported (shape)}
% mark clamped edge slope = 0
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:0.42,0.55)
  {edge slope $=0$};
\draw[engblue,->] (axis cs:0.55,0.5) -- (axis cs:0.92,0.06);
\end{axis}
\end{tikzpicture}
\caption[Clamped circular plate deflection profile]{Normalised
deflection of a uniformly loaded circular plate. The clamped plate
(blue) has the closed form $w(r) = w_{\max}(1-(r/a)^2)^2$ with
$w_{\max} = q_0 a^4/(64D)$; its slope vanishes at the edge, the
defining feature of the clamped boundary condition. The simply
supported plate (red, dashed, shape only) deflects about four times
as far at the centre for the same load because its edge is free to
rotate. The clamped edge is where the bending moment peaks,
$M_{rr} = -q_0 a^2/8$, and is therefore the first place to check
for yield.}
\label{fig:vol03ch09:circular-plate-deflection}
\end{figure}
