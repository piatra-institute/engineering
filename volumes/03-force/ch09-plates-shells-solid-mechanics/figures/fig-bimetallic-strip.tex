% expansion. Heated, the high-expansion layer (top) grows more, so the
% strip curves toward the low-expansion layer (bottom). Schematic: a flat
% cold strip and a curved hot strip, with the neutral behaviour annotated.

\centering
\begin{tikzpicture}[scale=1.0,>=latex]
  % --- cold, flat strip (left) ---
  \begin{scope}[shift={(0,0)}]
    \fill[engred!30] (0,0) rectangle (3,0.18);
    \fill[engblue!30] (0,-0.18) rectangle (3,0);
    \draw[engblack, thick] (0,-0.18) rectangle (3,0.18);
    \draw[engblack] (0,0) -- (3,0);
    \node[font=\scriptsize, anchor=south] at (1.5,0.22) {high $\alpha$ (top)};
    \node[font=\scriptsize, anchor=north] at (1.5,-0.22) {low $\alpha$ (bottom)};
    \node[font=\small, anchor=north] at (1.5,-0.6) {cold: flat};
  \end{scope}
  % --- hot, curved strip (right), curving down toward the low-alpha side ---
  \begin{scope}[shift={(5.2,0.1)}]

    % centre at top, bends down; draw two offset arcs for the two layers
    \draw[engred!30, line width=3.6pt] (-1.7,-0.5) .. controls (-0.6,0.35) and (0.6,0.35) .. (1.7,-0.5);

    \draw[engblack, thick] (-1.7,-0.5) .. controls (-0.6,0.35) and (0.6,0.35) .. (1.7,-0.5);

\caption[Bimetallic strip curvature]{A bimetallic strip: two bonded layers
whose thermal expansion coefficients differ. Cold, the strip is flat (left).
Heated by $\Delta T$, the high-expansion layer (red, top) wants to grow more
than the low-expansion layer (blue, bottom), and because the two are bonded
they cannot slide, so the strip curves toward the low-expansion side (right).
The curvature is proportional to the expansion mismatch $\Delta\alpha$ and the
temperature rise $\Delta T$ and inversely proportional to the strip thickness,
so a thin strip of two well-separated metals gives a large, repeatable
deflection per degree. That deflection, read against a contact or a scale, is
the working principle of the thermostat, the thermal cut-out, and the older
dial thermometer. The same bonded-layer mismatch, driven by cure shrinkage
rather than temperature, is what warps a composite laminate cured with an
unbalanced stacking sequence.}
\label{fig:vol03ch09:bimetallic-strip}
\end{figure}
