% Figure: shell-course thickness of a flat-bottom liquid storage tank.
% The hydrostatic hoop stress falls linearly with height up the wall, so
% each welded plate course is sized for the head at its own base and the
% wall steps down in thickness from bottom to top. A top wind girder
% stiffens the open rim against ovalling and wind buckling. Schematic;
% the stepped profile and the linear pressure triangle are the point, not
% the proportions.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % == stepped tank wall, drawn as a stack of courses on the right ==
  % each course is a vertical band; thickness shown by the band width.
  % bottom course thickest, top course thinnest.
  \def\xc{6.0}            % wall inner face x
  \draw[enggray, dashed] (\xc,0) -- (\xc,6.2);   % inner face reference
  % course 1 (bottom): widest
  \filldraw[engblue!20, draw=engblue, very thick]
    (\xc,0) rectangle (\xc+0.85,1.5);
  % course 2
  \filldraw[engblue!20, draw=engblue, very thick]
    (\xc,1.5) rectangle (\xc+0.62,3.0);
  % course 3
  \filldraw[engblue!20, draw=engblue, very thick]
    (\xc,3.0) rectangle (\xc+0.44,4.5);
  % course 4 (top): thinnest
  \filldraw[engblue!20, draw=engblue, very thick]
    (\xc,4.5) rectangle (\xc+0.30,6.0);
  % thickness labels
  \node[engblue, font=\scriptsize, anchor=west] at (\xc+0.9,0.75) {$t_1$ (thickest)};
  \node[engblue, font=\scriptsize, anchor=west] at (\xc+0.66,2.25) {$t_2$};
  \node[engblue, font=\scriptsize, anchor=west] at (\xc+0.48,3.75) {$t_3$};
  \node[engblue, font=\scriptsize, anchor=west] at (\xc+0.34,5.25) {$t_4$ (thinnest)};
  % top wind girder
  \filldraw[engred, draw=engred] (\xc-0.05,5.95) rectangle (\xc+0.55,6.2);
  \node[engred, font=\scriptsize, anchor=south west] at (\xc+0.05,6.25) {wind girder};
  % base annular plate
  \filldraw[enggray!40, draw=enggray] (\xc-0.6,-0.18) rectangle (\xc+0.9,0);
  \node[enggray, font=\scriptsize, anchor=north] at (\xc+0.2,-0.2) {bottom plate / foundation};

  % == hydrostatic pressure triangle on the left, max at base ==
  \def\xp{0.0}
  \draw[engblack, thick, ->] (\xp,0) -- (\xp,6.4) node[anchor=south east, font=\scriptsize] {height $z$};
  \draw[engblack, thick, ->] (\xp,0) -- (4.6,0) node[anchor=north, font=\scriptsize] {hoop stress $\sigma_\theta \propto (H-z)$};
  % linear pressure / hoop-stress triangle (zero at top, max at base)
  \filldraw[engamber!25, draw=engamber, thick]
    (\xp,6.0) -- (\xp,0) -- (4.2,0) -- cycle;
  \node[engamber, font=\scriptsize, anchor=west] at (3.0,0.35) {$\sigma_{\theta,\max}=\rho g H R/t_1$};

  % liquid level marker
  \draw[engblue, dashed] (\xp,6.0) -- (\xc,6.0);
  \node[engblue, font=\scriptsize, anchor=south] at (2.6,6.0) {liquid level $H$};

  % connect each course to its design head (one-foot design point)
  \foreach \zb/\zt in {0/1.5, 1.5/3.0, 3.0/4.5, 4.5/6.0}{
    \draw[enggray, dotted] (\xp,\zb) -- (\xc,\zb);
  }
\end{tikzpicture}
\caption[Shell-course thickness of a flat-bottom storage tank]{Wall-course
sizing of a flat-bottom liquid storage tank. The hydrostatic pressure rises
linearly with depth, so the hoop stress in the wall is largest at the base
and falls to nearly zero at the liquid surface (amber triangle, left). The
wall is built from welded plate courses, and each course is sized for the
head acting a short distance above its own bottom seam, the design point
that the variable-design-point and one-foot methods make precise. The wall
therefore steps down in thickness from the thick bottom course $t_1$ to the
thin top course $t_4$ (right). The open top rim carries a wind girder (red)
that stiffens it against ovalling and wind-induced buckling, and the base
course transfers the load to the annular bottom plate and the foundation.
The stepped profile is the storage-tank counterpart of the constant-wall
pressure vessel: the load varies along the wall, so the wall does too.}
\label{fig:vol03ch09:tank-courses}
\end{figure}
