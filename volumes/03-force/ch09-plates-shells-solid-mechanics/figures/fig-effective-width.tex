% Figure: post-buckling stress redistribution in a compressed plate and the
% von Karman effective width. Left: the actual longitudinal membrane stress
% across the plate width after buckling is non-uniform, sagging in the middle
% (which has shed load by buckling) and peaking at the braced edges. Right: the
% idealisation replaces it with two strips of width b_e/2 each carrying the
% uniform edge stress sigma_edge. We plot the actual stress profile and the
% equivalent block.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xlabel={position across the plate width $\ y/b$},
  ylabel={membrane stress $\ \sigma_x/\sigma_{\mathrm{cr}}$},
  xmin=0, xmax=1.0, ymin=0, ymax=3.2,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.5,0.98)}, anchor=north, draw=enggray!40},
]
% Actual post-buckling stress: high at the edges (y=0 and y=1), low in the
% middle. Model as sigma = a + c*cos(2 pi y)  with edge value ~3, mid value ~1.
% a = 2, c = 1  -> edges 3, middle 1.
\addplot[engblue, very thick, domain=0:1, samples=120]
  {2 + 1*cos(deg(2*pi*x))};
\addlegendentry{actual stress profile}
% edge stress line at 3
\addplot[engred, thick, dashed, domain=0:1, samples=2] {3};
\addlegendentry{$\sigma_{\mathrm{edge}}$}
% effective-width blocks: two strips of width 0.25 each at each edge carrying
% sigma_edge=3, drawn as filled steps. Left strip 0 to 0.25, right 0.75 to 1.
\addplot[englime, fill=englime!25, draw=englime, thick]
  coordinates {(0,0) (0,3) (0.25,3) (0.25,0)} \closedcycle;
\addplot[englime, fill=englime!25, draw=englime, thick]
  coordinates {(0.75,0) (0.75,3) (1.0,3) (1.0,0)} \closedcycle;
\node[englime!60!black, font=\scriptsize, anchor=south] at (axis cs:0.125,0.15)
  {$b_e/2$};
\node[englime!60!black, font=\scriptsize, anchor=south] at (axis cs:0.875,0.15)
  {$b_e/2$};
\end{axis}
\end{tikzpicture}
\caption[Effective width of a post-buckled plate]{The longitudinal membrane
stress across a simply supported plate compressed past its buckling load, in
units of the critical stress $\sigma_{\mathrm{cr}}$. The buckled middle of the
plate sheds load and carries little stress (blue), while the two edges, braced
by the unloaded side supports, carry a stress that climbs toward the edge
value $\sigma_{\mathrm{edge}}$ (red, dashed). The von Karman idealisation
discards the sagging middle and replaces the true profile with two strips of
total width $b_e$ (green) each carrying the uniform edge stress, chosen so the
block carries the same total force as the real profile. The plate therefore
keeps carrying load after it buckles, but only over an effective width that
shrinks as the edge stress rises, the mechanism that lets a thin stiffened
skin survive well past the buckling of its individual bays.}
\label{fig:vol03ch09:effective-width}
\end{figure}
