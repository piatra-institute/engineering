% Figure: the A- and C-weighting relative-response curves versus frequency,
% the frequency-dependent gains that convert a physical SPL spectrum into a
% level weighted toward human sensitivity. Smooth analytic forms of the
% standard IEC 61672 weightings, plotted in dB versus log frequency.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={frequency (Hz, log scale)},
  ylabel={relative weighting (dB)},
  xmode=log, log basis x=10,
  xmin=16, xmax=20000, ymin=-50, ymax=6,
  axis lines=left,
  domain=16:20000, samples=300,
  legend pos=south east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% A-weighting (IEC 61672), R_A(f) in dB, normalised to 0 dB at 1 kHz:
% R_A = 20 log10( 12194^2 f^4 / [ (f^2+20.6^2) sqrt((f^2+107.7^2)(f^2+737.9^2)) (f^2+12194^2) ] ) + 2.00
\addplot[engblue, very thick]
  {20*log10( (12194^2 * x^4) /
     ( (x^2 + 20.6^2) * sqrt( (x^2 + 107.7^2)*(x^2 + 737.9^2) ) * (x^2 + 12194^2) ) )
   + 2.00};
\addlegendentry{A-weighting}
% C-weighting (IEC 61672), R_C(f) in dB, normalised to 0 dB at 1 kHz:
% R_C = 20 log10( 12194^2 f^2 / [ (f^2+20.6^2)(f^2+12194^2) ] ) + 0.06
\addplot[engred, very thick]
  {20*log10( (12194^2 * x^2) / ( (x^2 + 20.6^2)*(x^2 + 12194^2) ) ) + 0.06};
\addlegendentry{C-weighting}
% the 0 dB reference at 1 kHz
\addplot[enggray, thin, dashed] {0};
\addplot[enggray, only marks, mark=*, mark size=1.4pt, forget plot]
  coordinates {(1000,0)};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:1000,0.3)
  {$0\,$dB at $1\,$kHz};
\end{axis}
\end{tikzpicture}
\caption[A- and C-weighting curves]{The standard A- and C-weighting
relative-response curves, the frequency-dependent gains a sound-level meter
applies before summing a spectrum into a single number. Both are normalised
to zero decibels at $1\,\si{\kilo\hertz}$. The A-weighting (blue) discounts
low-frequency content steeply, by about $16\,\si{\decibel}$ at
$125\,\si{\hertz}$ and more than $30\,\si{\decibel}$ at $31.5\,\si{\hertz}$,
matching the ear's reduced sensitivity to bass at moderate levels, and it
rolls the top octave off as well. The C-weighting (red) is nearly flat
across the mid band and falls only at the extremes, the right choice for
peak and impulsive levels where the low-frequency energy is real load. A
weighting is a receiver model: the A curve models the ear, and a structural
problem demands its own transfer function in place of either curve.}
\label{fig:vol03ch07:weighting-curves}
\end{figure}
