% Figure: schematic Lamb-wave phase-velocity dispersion curves for the
% lowest symmetric and antisymmetric modes in a plate, versus the
% frequency-thickness product, approaching the shear and Rayleigh speeds
% at high frequency-thickness.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={frequency$\times$thickness product $fh$ (MHz$\cdot$mm)},
  ylabel={phase velocity (km/s)},
  xmin=0, xmax=6, ymin=0, ymax=6.5,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Antisymmetric A0: starts near zero, rises toward the Rayleigh speed ~2.9 km/s.
% Schematic saturating curve: cR - (cR)*exp(-k*fh), cR = 2.9.
\addplot[engblue, very thick, domain=0.05:6, samples=120]
  {2.9*(1 - exp(-1.4*x))};
\addlegendentry{$A_0$ (antisymmetric, flexural)}
% Symmetric S0: starts near the plate extensional speed ~5.4 km/s, dips
% toward the shear speed ~3.2 km/s. Schematic descending curve.
\addplot[engred, very thick, domain=0.05:6, samples=120]
  {3.2 + 2.2*exp(-1.1*x)};
\addlegendentry{$S_0$ (symmetric, extensional)}
% shear speed asymptote reference
\addplot[enggray, thin, dashed] coordinates {(0,3.2) (6,3.2)};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:5.9,3.3) {shear speed $c_T$};
\end{axis}
\end{tikzpicture}
\caption[Lamb-wave dispersion in a plate]{Phase velocity of the two
lowest guided (Lamb) modes in a plate, drawn schematically against the
product of frequency and plate thickness. The antisymmetric $A_0$ mode is
the flexural wave: at a small frequency-thickness product it travels far
slower than the bulk shear speed and carries the bending vibration of thin
panels, then rises toward the surface (Rayleigh) speed as the wavelength
shrinks below the plate thickness. The symmetric $S_0$ mode is the
extensional wave: it starts near the plate longitudinal speed and falls
toward the shear speed. Because each mode's speed depends on frequency, a
broadband pulse spreads as it travels, in contrast to the non-dispersive
bulk P and S waves. An ultrasonic inspection of a pipe or an aircraft skin
picks a mode and a frequency-thickness point on these curves to trade
penetration, sensitivity to a given defect, and the distance the mode runs
before dispersion smears it beyond use.}
\label{fig:vol03ch07:lamb-dispersion}
\end{figure}
