% Figure: source-path-receiver level budget for the worked enclosure case study.
% A waterfall of levels from machine sound-power-derived SPL down through
% the enclosure path to the receiver, with the exposure limit drawn across.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12.5cm, height=7cm,
  ybar,
  bar width=22pt,
  xlabel={accounting stage},
  ylabel={level at receiver (dB(A))},
  symbolic x coords={untreated,mass law,leakage,absorption,received},
  xtick=data,
  x tick label style={rotate=20, anchor=east, font=\scriptsize},
  ymin=60, ymax=105,
  axis lines=left,
  enlarge x limits=0.14,
  nodes near coords,
  nodes near coords style={font=\scriptsize},
  tick label style={font=\scriptsize},
  label style={font=\small},
  grid=major, grid style={enggray!20},
]
\addplot[draw=engblue, fill=engblue!35] coordinates {
  (untreated,98) (mass law,68) (leakage,76) (absorption,73) (received,73)};
% OSHA 85 dB(A) action/limit line
\addplot[engred, sharp plot, thick, dashed] coordinates {
  (untreated,85) (received,85)};
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:received,85.4) {$85\,$dB(A) limit};
\end{axis}
\end{tikzpicture}
\caption[Source-path-receiver level budget for the worked enclosure]{Level
budget for the noise-control enclosure of the case study, read left to
right as an accounting waterfall. The untreated receiver level is
$98\,\si{\decibel}$(A). The bare mass-law transmission loss of the panel
would drop it to $68\,\si{\decibel}$(A), but a realistic leakage and
flanking path limits the achievable insertion loss and lifts the level
back to $76\,\si{\decibel}$(A). Lining the enclosure interior with
absorber removes the reverberant buildup inside the box and recovers
about $3\,\si{\decibel}$, giving a received level of
$73\,\si{\decibel}$(A). The final bar sits below the $85\,\si{\decibel}$(A)
exposure limit (dashed), so the design passes, with margin to spare
against the dose calculation in the text. The bar that controls the
outcome is leakage, not the panel mass, which is the recurring lesson of
enclosure design.}
\label{fig:vol03ch07:enclosure-budget}
\end{figure}
