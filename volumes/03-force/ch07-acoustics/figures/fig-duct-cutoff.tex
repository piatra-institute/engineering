% Figure: phase and group velocity of the first higher-order mode in a
% rigid rectangular duct, normalised to c, versus frequency above cutoff.
% Above the cutoff frequency the mode propagates dispersively.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={frequency ratio $f/f_c$}, ylabel={velocity ratio $v/c$},
  xmin=1, xmax=4, ymin=0, ymax=4,
  axis lines=left,
  domain=1.001:4, samples=300,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\small},
]
% phase velocity c_ph / c = 1 / sqrt(1 - (fc/f)^2); clamp the sqrt argument.
\addplot[engred, very thick] {1/sqrt( max(0.0001, 1 - (1/x)^2) )};
\addlegendentry{phase velocity $c_{\mathrm{ph}}/c$}
% group velocity c_gr / c = sqrt(1 - (fc/f)^2).
\addplot[engblue, very thick] {sqrt( max(0.0001, 1 - (1/x)^2) )};
\addlegendentry{group velocity $c_{\mathrm{gr}}/c$}
% the free-medium speed c as reference: product c_ph c_gr = c^2.
\addplot[enggray, thin, dashed] {1};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:3.9,1.08)
  {free-medium speed $c$};
\end{axis}
\end{tikzpicture}
\caption[Duct-mode dispersion above cutoff]{Phase velocity (red) and
group velocity (blue) of a higher-order acoustic mode in a rigid-walled
duct, each normalised to the free-medium speed $c$, plotted against
frequency in units of the mode cutoff frequency $f_c$. Below $f_c$ the
mode does not propagate; it decays evanescently along the duct. Above
$f_c$ the mode carries energy, but dispersively: the phase velocity
$c_{\mathrm{ph}}=c/\sqrt{1-(f_c/f)^2}$ exceeds $c$ and diverges at
cutoff, while the group velocity $c_{\mathrm{gr}}=c\sqrt{1-(f_c/f)^2}$
carries the energy and stays below $c$, rising from zero at cutoff toward
$c$ at high frequency. Their product is $c^2$ at every frequency. Only
the plane-wave mode ($f_c=0$) is non-dispersive; the higher modes smear
a pulse and set the frequency above which a duct silencer or an
intake system can no longer be treated by one-dimensional plane-wave
acoustics.}
\label{fig:vol03ch07:duct-cutoff}
\end{figure}
