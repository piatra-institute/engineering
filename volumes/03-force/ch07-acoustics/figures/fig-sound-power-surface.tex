% Figure: schematic of a hemispherical measurement surface over a source
% on a reflecting plane, with microphone positions, for the sound-power
% from SPL worked example. Pure tikzpicture (no axis environment).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% reflecting ground plane
\draw[enggray, thick] (-4.2,0) -- (4.2,0);
\foreach \x in {-4,-3.2,-2.4,-1.6,-0.8,0,0.8,1.6,2.4,3.2,4} {
  \draw[enggray] (\x,0) -- ++(-0.25,-0.25);
}
\node[enggray, font=\scriptsize, anchor=north east] at (4.2,-0.05) {reflecting floor};
% measurement hemisphere of radius r
\draw[engblue, very thick] (-3.2,0) arc[start angle=180, end angle=0, radius=3.2];
\node[engblue, font=\small, anchor=south] at (0,3.3) {measurement hemisphere, radius $r$};
% radius marker
\draw[engblack, ->, thin] (0,0) -- ({3.2*cos(55)},{3.2*sin(55)});
\node[engblack, font=\scriptsize, anchor=south west] at ({1.6*cos(55)},{1.6*sin(55)}) {$r$};
% the source: a small machine box at the centre on the floor
\fill[engorange!40, draw=engorange, thick] (-0.55,0) rectangle (0.55,0.7);
\node[engorange!70!black, font=\scriptsize] at (0,0.35) {source};
% microphone positions on the hemisphere (N points)
\foreach \a in {18,50,90,130,162} {
  \fill[engred] ({3.2*cos(\a)},{3.2*sin(\a)}) circle (2.2pt);
}
\node[engred, font=\scriptsize, anchor=west] at ({3.2*cos(18)+0.1},{3.2*sin(18)}) {mic $i$};
% area element annotation
\node[engblack, font=\scriptsize, anchor=west] at (3.3,1.6)
  {$S=2\pi r^2$};
\node[engblack, font=\scriptsize, anchor=west] at (3.3,1.1)
  {$L_W = \overline{L_p} + 10\log_{10}\dfrac{S}{S_0}$};
\end{tikzpicture}
\caption[Hemispherical sound-power measurement surface]{Sound-power
measurement of a source resting on a hard reflecting floor. The source
radiates into a half-space, so the enveloping measurement surface is a
hemisphere of area $S=2\pi r^2$ rather than a full sphere. Microphones
sample the sound pressure level at positions distributed over the
hemisphere; the surface-averaged level $\overline{L_p}$ is the energy
average of the individual readings. The sound power level follows from
the area as $L_W=\overline{L_p}+10\log_{10}(S/S_0)$ with the reference
area $S_0=1\,\si{\meter\squared}$. The construction turns a set of
position-dependent pressure measurements into a single
source-intrinsic quantity, the radiated power, which is what enclosure
and barrier design needs because it does not change when the room around
the source changes.}
\label{fig:vol03ch07:sound-power-surface}
\end{figure}
