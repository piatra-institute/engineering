% Figure: the first three standing-wave modes in a closed-closed tube.
% Pressure antinodes at the rigid walls, nodes in the interior.
% p_n(x) = cos(n pi x / L), n = 1, 2, 3.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={position $x/L$}, ylabel={pressure amplitude (normalised)},
  xmin=0, xmax=1, ymin=-1.4, ymax=1.4,
  axis lines=left,
  domain=0:1, samples=300,
  legend pos=north east,

% n = 1: half a wavelength fits, antinodes at both ends
\addplot[engred, very thick] {cos(deg(pi*x))};
\addlegendentry{$n=1$, $f_1=c/2L$}
% n = 2
\addplot[engblue, very thick] {cos(deg(2*pi*x))};
\addlegendentry{$n=2$, $f_2=c/L$}
% n = 3
\addplot[engorange, very thick] {cos(deg(3*pi*x))};
\addlegendentry{$n=3$, $f_3=3c/2L$}
% zero line
\addplot[enggray, thin] {0};
\end{axis}
\end{tikzpicture}
\caption[Standing-wave modes in a closed tube]{The first three
acoustic pressure modes of a rigid-walled tube of length $L$ closed at
both ends. Each rigid wall forces a pressure antinode (a maximum of the
standing pressure), and the mode shapes are $p_n(x)=\cos(n\pi x/L)$ with
natural frequencies $f_n = n\,c/(2L)$. The fundamental ($n=1$, red)
fits half a wavelength across the tube; each higher mode adds one
interior pressure node. A particle-velocity description is the
complement of this picture: velocity nodes sit where the pressure
antinodes are, because the rigid wall permits no normal velocity. The
same harmonic ladder governs an organ pipe, an exhaust resonator, and
the lowest axial room mode of section~7.5.}
\label{fig:vol03ch07:standing-wave}
\end{figure}
