% Figure: pressure amplitude of a spherical wave decaying as 1/r, and the
% corresponding sound pressure level falling 6 dB per doubling of distance.
% Two y-axes shown as two panels stacked is avoided; we plot the level (dB)
% relative to 1 m on a log-distance axis so the 6 dB/doubling slope is a line.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={distance from source $r$ (m, log scale)},
  ylabel={level relative to $1\,$m (dB)},
  xmode=log, log basis x=10,
  xmin=1, xmax=100, ymin=-42, ymax=2,
  axis lines=left,
  domain=1:100, samples=200,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% spherical (point) source: L(r) - L(1) = -20 log10(r)
\addplot[engblue, very thick] {-20*log10(x)};
\addlegendentry{point source, $-6\,$dB/doubling}
% cylindrical (line) source: -10 log10(r)
\addplot[engred, very thick] {-10*log10(x)};
\addlegendentry{line source, $-3\,$dB/doubling}
% mark a doubling 10 -> 20 m on the point-source curve
\addplot[engblack, only marks, mark=*, mark size=1.6pt, forget plot]
  coordinates {(10,-20) (20,-26.02)};
\node[engblack, font=\scriptsize, anchor=west] at (axis cs:21,-24)
  {$-6\,$dB, $10\to20\,$m};
\end{axis}
\end{tikzpicture}
\caption[Geometric spreading of point and line sources]{Sound pressure
level relative to the one-metre value as a function of distance for two
free-field geometries. A compact point source spreads its energy over the
expanding sphere of area $4\pi r^2$, so the intensity falls as $1/r^2$, the
pressure as $1/r$, and the level as $-20\log_{10}r$, which is six decibels
for every doubling of distance (blue, with the $10\to20\,\si{\meter}$
doubling marked). An infinite line source, such as a stream of traffic,
spreads over the cylinder of area $2\pi r$, so the intensity falls only as
$1/r$ and the level as $-10\log_{10}r$, three decibels per doubling (red).
The halved slope of the line source is why a motorway stays loud much
farther out than a single vehicle, and why a setback distance sized for one
machine underperforms against a road.}
\label{fig:vol03ch07:spherical-spreading}
\end{figure}
