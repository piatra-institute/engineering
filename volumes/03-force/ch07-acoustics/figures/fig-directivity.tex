% Figure: directivity of a circular piston as response versus angle off-axis,
% for two values of ka. Cartesian axis (no polar library needed).
% Smooth lobe models capturing the broad-to-narrow trend with size parameter.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={angle off-axis (degrees)},
  ylabel={normalised response $|p|/|p_{\max}|$},
  xmin=0, xmax=90, ymin=0, ymax=1.05,
  axis lines=left,
  domain=0:90, samples=200,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% small radiator ka ~ 2: broad, gentle roll-off
\addplot[engblue, very thick] {cos(x)^2};
\addlegendentry{$ka=2$ (broad beam)}
% large radiator ka ~ 8: narrow forward lobe
\addplot[engred, very thick] {max(0,cos(x))^16};
\addlegendentry{$ka=8$ (narrow beam)}
% the -3 dB (0.707) line marking half-power beamwidth
\addplot[enggray, thin, dashed] {0.707};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:55,0.72)
  {half-power ($-3$ dB)};
\end{axis}
\end{tikzpicture}
\caption[Directivity of a circular piston]{Normalised on-axis-referenced
response of a flat circular piston radiator of radius $a$ versus angle
off the beam axis, at two values of the size parameter $ka=2\pi
a/\lambda$. When the radiator is small compared to the wavelength
($ka\approx2$, blue) the response falls off gently and the beam is
broad, approaching omnidirectional. When the radiator is large compared
to the wavelength ($ka\approx8$, red) the energy concentrates into a
narrow forward lobe whose half-power width scales as $\sim\lambda/(2a)$.
The exact pattern is the piston directivity $|2J_1(ka\sin\theta)/(ka
\sin\theta)|$ with its side lobes; the curves here are smooth lobe
models that capture the broad-to-narrow trend that governs loudspeaker
dispersion, sonar beamwidth, and ultrasonic-probe focusing. Beamwidth
and radiator size trade against each other at fixed frequency.}
\label{fig:vol03ch07:directivity}
\end{figure}
