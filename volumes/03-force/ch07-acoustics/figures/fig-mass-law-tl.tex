% Figure: mass-law transmission loss versus frequency for three panel
% surface masses, with the +6 dB/octave slope and the coincidence dip annotated.
% TL = 20 log10(m'' f) - 47 dB (field-incidence mass law).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={frequency (Hz, log scale)},
  ylabel={transmission loss TL (dB)},
  xmode=log, log basis x=10,
  xmin=50, xmax=8000, ymin=0, ymax=70,
  axis lines=left,
  domain=50:8000, samples=200,
  legend pos=north west,
  legend cell align=left,
  legend style={font=\scriptsize},
  tick label style={font=\scriptsize},
  label style={font=\small},
  grid=major, grid style={enggray!20},
]
% TL = 20 log10(m'' f) - 47, clamped at 0 for readability
\addplot[engblue, very thick]  {max(0, 20*log10(5*x)  - 47)};
\addlegendentry{$m''=5\,$kg/m$^2$ (plywood)}
\addplot[engorange, very thick] {max(0, 20*log10(14*x) - 47)};
\addlegendentry{$m''=14\,$kg/m$^2$ (steel 1.8\,mm)}
\addplot[engred, very thick]   {max(0, 20*log10(40*x) - 47)};
\addlegendentry{$m''=40\,$kg/m$^2$ (steel 5\,mm)}
% required TL target line
\addplot[enggray, thin, dashed] {30};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:55,30.5) {required TL $=30$\,dB};
% mark 500 Hz design point on the 14 kg/m2 curve: 20 log10(14*500)-47 = 29.9
\addplot[engblack, only marks, mark=*, mark size=2pt] coordinates {(500,29.9)};
\node[engblack, font=\scriptsize, anchor=north west] at (axis cs:520,29) {design point};
\end{axis}
\end{tikzpicture}
\caption[Mass-law transmission loss of an enclosure panel]{Field-incidence
mass-law transmission loss, $\text{TL}\approx 20\log_{10}(m''f)-47\,\si{\decibel}$,
for three panel surface masses. The loss climbs about $6\,\si{\decibel}$
per doubling of either frequency or surface mass, so heavier panels and
higher frequencies are easier to block. Doubling the surface mass buys
the same $6\,\si{\decibel}$ as doubling the frequency, which is why a
low-frequency requirement drives the panel mass up fast. The dashed line
marks the $30\,\si{\decibel}$ target of the worked enclosure; the
$14\,\si{\kilogram\per\meter\squared}$ panel (about $1.8\,\si{\milli\meter}$
of steel) meets it at the $500\,\si{\hertz}$ design point. The mass law
ignores the coincidence dip near the panel's critical frequency and the
stiffness-controlled region well below the first panel resonance; both
narrow the usable band of a real panel.}
\label{fig:vol03ch07:mass-law-tl}
\end{figure}
