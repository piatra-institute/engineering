% Figure: optimum mid-band reverberation time versus room volume for three
% use classes (speech, chamber music, symphonic), with the auditorium design
% point of the case study marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={room volume $V$ (m$^3$, log scale)},
  ylabel={optimum mid-band $T_{60}$ (s)},
  xmode=log, log basis x=10,
  xmin=100, xmax=30000, ymin=0, ymax=2.6,
  axis lines=left,
  domain=100:30000, samples=80,
  legend pos=north west,
  legend cell align=left,
  legend style={font=\scriptsize},
  tick label style={font=\scriptsize},
  label style={font=\small},
  grid=major, grid style={enggray!20},
]
% speech: roughly 0.3 + 0.07*log10(V/100)
\addplot[engblue, very thick]   {0.32 + 0.10*log10(x/100)};
\addlegendentry{speech / lecture}
% chamber music
\addplot[englime, very thick]   {0.9 + 0.20*log10(x/100)};
\addlegendentry{chamber music}
% symphonic
\addplot[engred, very thick]    {1.3 + 0.28*log10(x/100)};
\addlegendentry{symphonic / opera}
% design point of the case study: V = 4200 m^3, target T60 = 1.0 s (lecture+light music hall)
\addplot[engblack, only marks, mark=*, mark size=2pt] coordinates {(4200,1.0)};
\node[engblack, font=\scriptsize, anchor=west] at (axis cs:4600,1.0) {design point};
\end{axis}
\end{tikzpicture}
\caption[Optimum reverberation time versus room volume by use]{Optimum
mid-band reverberation time as a function of room volume for three use
classes. Larger halls tolerate and need longer reverberation, and music
wants more than speech; the bands widen with volume because a large hall
serves more than one programme. The multi-purpose hall of the case study,
$V\approx 4200\,\si{\meter\cubed}$, is designed to a mid-band target of
$1.0\,\si{\second}$, marked on the figure, a compromise that keeps speech
intelligible while giving amplified and small-ensemble music enough
liveness. The curves are design guides, not laws; the programme committee,
not the formula, sets the final target.}
\label{fig:vol03ch07:rt60-targets}
\end{figure}
