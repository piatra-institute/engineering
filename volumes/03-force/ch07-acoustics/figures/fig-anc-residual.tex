% Figure: residual level after active cancellation versus phase-match
% error, from 20 log10( 2 sin(dphi/2) ), showing how tight the phase
% budget must be to reach a given amount of cancellation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={phase-match error $\Delta\phi$ (degrees)},
  ylabel={residual level re uncancelled (dB)},
  xmin=0, xmax=60, ymin=-40, ymax=6,
  axis lines=left,
  domain=0.5:60, samples=200,
  legend pos=south east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% residual/pn = 2 sin(dphi/2), level = 20 log10 of that. dphi in degrees.
\addplot[engblue, very thick] {20*log10( 2*sin(x/2) )};
\addlegendentry{residual $=20\log_{10}\!\big(2\sin\tfrac{\Delta\phi}{2}\big)$}
% -20 dB target line and its 5.7 deg crossing
\addplot[enggray, thin, dashed] coordinates {(0,-20) (60,-20)};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:30,-19.5) {$-20$ dB target};
\addplot[engred, thin, dotted] coordinates {(5.7,-40) (5.7,-20)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:6.5,-32) {$5.7^{\circ}$};
% 0 dB (no cancellation) crossing at dphi = 60 deg
\addplot[enggray, thin, dotted] coordinates {(0,0) (60,0)};
\node[enggray, font=\scriptsize, anchor=north east] at (axis cs:59,-0.5) {no benefit};
\end{axis}
\end{tikzpicture}
\caption[Active-cancellation residual versus phase error]{Residual sound
level after ideal amplitude-matched active cancellation, plotted against
the phase-match error between the anti-noise and the noise. The residual
follows $2\sin(\Delta\phi/2)$, so a perfect phase match cancels
completely, a $5.7^{\circ}$ error leaves a tenth of the amplitude
($-20\,\si{\decibel}$), and at $60^{\circ}$ the sum equals the original
and the system yields no benefit. Beyond $60^{\circ}$ the added sound
makes matters worse. The curve is why active control is a low-frequency
tool: at low frequency a wavelength spans metres, so the few millimetres
of path and processing error translate to a small phase error and deep
cancellation, while at high frequency the same fixed timing error is a
large fraction of a cycle and the residual climbs toward the no-benefit
line. Amplitude mismatch adds a second, smaller term, so a practical
system budgets both.}
\label{fig:vol03ch07:anc-residual}
\end{figure}
