% Figure: transmission loss of a simple expansion-chamber muffler versus
% frequency, the periodic lobed curve TL = 10 log10[1 + (1/4)(m - 1/m)^2 sin^2(kL)],
% with the firing-order harmonics of an idling four-cylinder engine marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12.5cm, height=7.5cm,
  xlabel={frequency (Hz)},
  ylabel={transmission loss TL (dB)},
  xmin=0, xmax=900, ymin=0, ymax=22,
  axis lines=left,
  domain=1:900, samples=400,
  legend pos=north east,
  legend cell align=left,
  legend style={font=\scriptsize},
  tick label style={font=\scriptsize},
  label style={font=\small},
  grid=major, grid style={enggray!20},
]
% TL = 10 log10[1 + (1/4)(m - 1/m)^2 sin^2(kL)], m = area ratio = 16, L = 0.30 m, c = 343
% k L = 2 pi f L / c = 2 pi f (0.30) / 343
% (1/4)(16 - 1/16)^2 = (1/4)(15.9375)^2 = 63.5
\addplot[engblue, very thick] {10*log10( 1 + 63.5*( sin(deg(2*pi*x*0.30/343)) )^2 )};
\addlegendentry{expansion chamber, $m=16$, $L=0.30\,$m}
% firing-order harmonics: idle 800 rpm, 4-cylinder 4-stroke -> 2 firing events/rev
% fundamental = (800/60)*2 = 26.7 Hz; mark 2nd-order at engine orders n*26.7 up to band
\foreach \f in {27, 53, 80, 107, 160, 213, 320, 427} {
  \addplot[engred, dashed, thin, forget plot] coordinates {(\f,0) (\f,22)};
}
\node[engred, font=\scriptsize, anchor=south, rotate=90] at (axis cs:80,2) {firing harmonics};
% the troughs where sin(kL)=0: f = n c / (2L) = n*343/0.6 = 572 n -> first pass-through near 572 Hz
\node[engblack, font=\scriptsize, anchor=north] at (axis cs:572,3.5) {pass-through $f=c/2L$};
\addplot[engblack, only marks, mark=*, mark size=1.6pt, forget plot] coordinates {(572,0)};
\end{axis}
\end{tikzpicture}
\caption[Expansion-chamber muffler transmission loss]{Transmission loss of
a simple expansion-chamber muffler of area ratio $m=16$ and length
$L=0.30\,\si{\meter}$, from
$\text{TL}=10\log_{10}\!\big[1+\tfrac14(m-1/m)^2\sin^2(kL)\big]$. The loss
is a comb: it peaks where the chamber is an odd number of quarter
wavelengths long ($kL=\pi/2,3\pi/2,\dots$) and falls to zero at the
pass-through frequencies where the chamber is a whole number of half
wavelengths ($f=nc/2L$), at which the chamber is acoustically invisible.
The dashed lines mark the firing-order harmonics of an idling
four-cylinder four-stroke engine; a single chamber tuned to cover the
low orders leaves the pass-through gap near $572\,\si{\hertz}$, the reason
a real exhaust chains chambers of different lengths so that one chamber's
trough falls on another's peak.}
\label{fig:vol03ch07:muffler-tl}
\end{figure}
