% Figure: low-frequency on-axis response of a sealed-box loudspeaker, a
% second-order high-pass whose corner is the system resonance f_c and whose
% sharpness is the total Q. Three Q values show the maximally-flat and the
% peaked alignments. Response in dB versus normalised frequency f/f_c.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={normalised frequency $f/f_c$ (log scale)},
  ylabel={on-axis response (dB)},
  xmode=log, log basis x=10,
  xmin=0.2, xmax=5, ymin=-24, ymax=8,
  axis lines=left,
  domain=0.2:5, samples=300,
  legend pos=south east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% second-order high-pass magnitude in dB:
% |H| = (f/fc)^2 / sqrt( (1-(f/fc)^2)^2 + ((f/fc)/Q)^2 )
% over-damped, Q = 0.5
\addplot[engblue, very thick]
  {10*log10( (x^4) / ( (1-x^2)^2 + (x/0.5)^2 ) )};
\addlegendentry{$Q=0.5$ (over-damped)}
% maximally flat, Q = 0.707
\addplot[englime, very thick]
  {10*log10( (x^4) / ( (1-x^2)^2 + (x/0.707)^2 ) )};
\addlegendentry{$Q=0.707$ (maximally flat)}
% peaked, Q = 1.2
\addplot[engred, very thick]
  {10*log10( (x^4) / ( (1-x^2)^2 + (x/1.2)^2 ) )};
\addlegendentry{$Q=1.2$ (peaked)}
% the -3 dB line and the corner at f/fc = 1
\addplot[enggray, thin, dashed] {-3};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:0.21,-2.6)
  {$-3\,$dB};
\addplot[enggray, thin, dashed, forget plot] coordinates {(1,-24) (1,8)};
\node[enggray, font=\scriptsize, anchor=south, rotate=90]
  at (axis cs:1,-20) {corner $f_c$};
\end{axis}
\end{tikzpicture}
\caption[Sealed-box loudspeaker low-frequency response]{On-axis
low-frequency response of a sealed-box (closed-box) loudspeaker, a
second-order high-pass filter whose corner frequency is the in-box system
resonance $f_c$ and whose shape near the corner is set by the total quality
factor $Q_{tc}$. The over-damped alignment ($Q=0.5$, blue) rolls off
gently and early; the maximally-flat alignment ($Q=0.707$, green) holds
level closest to the corner before falling at twelve decibels per octave;
the peaked alignment ($Q=1.2$, red) lifts the response by a few decibels
just above $f_c$ at the price of a less controlled transient. The sealed
box trades low-frequency extension against box size and damping, the design
choice the loudspeaker-in-a-box vignette carries through with numbers.}
\label{fig:vol03ch07:sealed-box}
\end{figure}
