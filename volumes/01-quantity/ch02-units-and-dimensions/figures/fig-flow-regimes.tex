% Figure: the four flow regimes selected by Re and Ma, with the
% characteristic dimensionless groups that dominate each regime.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  regime/.style={draw=engblack, rounded corners=2pt,
    minimum width=4cm, minimum height=2cm, font=\small,
    align=center, inner sep=4pt},
  axislab/.style={font=\small\bfseries},
  groupbox/.style={font=\scriptsize, align=left, anchor=north west},
]

% Axis arrows
\draw[-{Stealth}, thick] (0, 0) -- (10.5, 0)
  node[axislab, right] {$\mathrm{Re}$};
\draw[-{Stealth}, thick] (0, 0) -- (0, 6.5)
  node[axislab, above] {$\mathrm{Ma}$};

% Axis labels for thresholds
\node[engblack, font=\scriptsize, anchor=north] at (3.0, 0)
  {$\mathrm{Re} \approx 2300$};
\node[engblack, font=\scriptsize, anchor=east] at (0, 3.0)
  {$\mathrm{Ma} \approx 0.3$};

% Threshold dashed lines
\draw[engblack!50, dashed] (3.0, 0) -- (3.0, 6.0);
\draw[engblack!50, dashed] (0, 3.0) -- (10.0, 3.0);

% Four regime boxes
\node[regime, fill=engblue!10] at (1.5, 1.5)
  {laminar incompressible\\$\mathrm{Re} < 2300$,
   $\mathrm{Ma} < 0.3$\\dominant: $\mathrm{Re}$};

\node[regime, fill=engred!10] at (6.5, 1.5)
  {turbulent incompressible\\$\mathrm{Re} > 4000$,
   $\mathrm{Ma} < 0.3$\\dominant: $\mathrm{Re}$, $e/D$};

\node[regime, fill=englime!20] at (1.5, 5.0)
  {laminar compressible\\(rare in practice)\\dominant: $\mathrm{Ma}$};

\node[regime, fill=engblack!10] at (6.5, 5.0)
  {turbulent compressible\\$\mathrm{Re} > 4000$,
   $\mathrm{Ma} > 0.3$\\dominant: $\mathrm{Re}$, $\mathrm{Ma}$};

% Secondary-group call-outs
\node[groupbox] at (0.2, -0.4)
  {also: $\mathrm{Fr}$ (free surfaces),
   $\mathrm{We}$ (drops),\\
   $\mathrm{Gr}$/$\mathrm{Ra}$ (buoyancy),
   $\mathrm{St}$ (vortex shed.),\\
   $\mathrm{Pr}$ (thermal), $\mathrm{Sc}$ (mass),
   $\mathrm{Kn}$ (rarefied)};

\end{tikzpicture}
\caption[Flow regimes by Re and Ma]{The first-order regime map for
isothermal Newtonian flow: Reynolds number distinguishes laminar
from turbulent, Mach number distinguishes incompressible from
compressible. The four boxes select different dominant dimensionless
groups; the engineer chooses the working correlations from the
literature based on which box the design point sits in. Off-diagonal
groups (Froude, Weber, Grashof and Rayleigh, Strouhal, Prandtl,
Schmidt, Knudsen) enter when free surfaces, drops, buoyancy, vortex
shedding, heat or mass transport, or rarefaction matters.}
\label{fig:vol01:ch02:flow-regimes}
\end{figure}
