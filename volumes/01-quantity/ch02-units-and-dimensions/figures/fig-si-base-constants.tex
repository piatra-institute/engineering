% Figure: the seven SI base units and their defining constants, shown
% as a star diagram with each unit anchored to its constant.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  unit/.style={draw=engblack, fill=englime!20, rounded corners=2pt,
    minimum width=2cm, minimum height=0.8cm, font=\small\bfseries,
    align=center},
  constant/.style={draw=engblack, fill=engblue!10, rounded corners=2pt,
    minimum width=2.6cm, minimum height=0.9cm, font=\scriptsize,
    align=center},
  arrow/.style={-{Stealth}, thick, draw=engblack!70},
]

% Center: SI label
\node[font=\bfseries, fill=engred!10, rounded corners=2pt,
  minimum width=1.4cm, minimum height=0.8cm, draw=engblack]
  (si) at (0,0) {SI 2019};

% Seven units around center
\node[unit] (s)   at ( 0.0,  3.2) {second\\(s)};
\node[unit] (m)   at ( 3.0,  2.0) {metre\\(m)};
\node[unit] (kg)  at ( 3.2, -1.0) {kilogram\\(kg)};
\node[unit] (A)   at ( 1.5, -2.8) {ampere\\(A)};
\node[unit] (K)   at (-1.5, -2.8) {kelvin\\(K)};
\node[unit] (mol) at (-3.2, -1.0) {mole\\(mol)};
\node[unit] (cd)  at (-3.0,  2.0) {candela\\(cd)};

% Defining constants on outer ring
\node[constant] (cs)  at ( 0.0,  4.6) {$\Delta\nu_\text{Cs}$\\hyperfine freq.};
\node[constant] (c)   at ( 4.6,  2.8) {$c$\\speed of light};
\node[constant] (h)   at ( 4.8, -1.4) {$h$\\Planck constant};
\node[constant] (e)   at ( 2.3, -4.0) {$e$\\elementary charge};
\node[constant] (kB)  at (-2.3, -4.0) {$k_\text{B}$\\Boltzmann constant};
\node[constant] (NA)  at (-4.8, -1.4) {$N_\text{A}$\\Avogadro constant};
\node[constant] (Kcd) at (-4.6,  2.8) {$K_\text{cd}$\\luminous efficacy};

% Connections: each unit to its defining constant
\draw[arrow] (cs)  -- (s);
\draw[arrow] (c)   -- (m);
\draw[arrow] (h)   -- (kg);
\draw[arrow] (e)   -- (A);
\draw[arrow] (kB)  -- (K);
\draw[arrow] (NA)  -- (mol);
\draw[arrow] (Kcd) -- (cd);

% Each unit to SI center
\foreach \u in {s, m, kg, A, K, mol, cd} {
  \draw[draw=engblack!30, thin] (\u) -- (si);
}

\end{tikzpicture}
\caption[SI 2019: base units and defining constants]{The 2019 SI
redefinition: each of the seven base units is anchored to a fixed
numerical value of a defining physical constant. Realising a unit
in the laboratory becomes a measurement of that constant under
laboratory conditions. The redefinition cut the dependence on
artifacts (the old platinum-iridium kilogram in particular) and
made every unit invariant to artifact drift \cite{std:bipm-si2019}.}
\label{fig:vol01:ch02:si-constants}
\end{figure}
