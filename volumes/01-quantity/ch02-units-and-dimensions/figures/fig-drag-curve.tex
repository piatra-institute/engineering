% Figure: the drag-coefficient curve for a sphere as a function of
% Reynolds number, showing the three regimes (Stokes, constant,
% drag crisis).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=11cm, height=7cm,
  xlabel={Reynolds number $\mathrm{Re} = \rho v D / \mu$},
  ylabel={Drag coefficient $C_D$},
  xmin=1e-1, xmax=1e6,
  ymin=0.1, ymax=300,
  grid=both,
  major grid style={engblack!20},
  minor grid style={engblack!10},
  legend pos=north east,
  legend cell align=left,
  font=\small,
]

% Stokes regime: Cd = 24/Re
\addplot[engblue, very thick, domain=0.1:1, samples=20]
  {24/x};
\addlegendentry{Stokes: $C_D = 24/\mathrm{Re}$};

% Morsi-Alexander piecewise approximation across the full range
\addplot[engred, very thick, domain=1:1e5, samples=120]
  {24/x + 6/(1 + sqrt(x)) + 0.4};
\addlegendentry{Schiller-Naumann/Morsi-Alexander};

% Constant Cd regime
\addplot[engblack, dashed, very thick, domain=1e3:1e5] {0.44};
\addlegendentry{Newton: $C_D \approx 0.44$};

% Drag crisis at ~3e5 (shown schematically)
\addplot[engred, thick, dotted, domain=3e5:1e6, samples=40]
  {0.1 + 0.05*log10(x/3e5)};

% Annotations
\node[engblue, font=\scriptsize] at (axis cs:0.3, 70) {Stokes flow};
\node[engblack, font=\scriptsize] at (axis cs:1e4, 0.7) {Newton regime};
\node[engred, font=\scriptsize, align=center] at (axis cs:3e5, 0.18)
  {drag\\crisis};

\end{loglogaxis}
\end{tikzpicture}
\caption[Drag coefficient of a sphere]{The drag coefficient of a
smooth sphere as a function of Reynolds number, spanning seven
orders of magnitude. Three regimes: Stokes flow ($\mathrm{Re} \ll 1$,
$C_D = 24/\mathrm{Re}$); Newton regime ($10^3 \lesssim \mathrm{Re}
\lesssim 10^5$, $C_D$ roughly constant near 0.44); the drag crisis
($\mathrm{Re} \sim 3 \times 10^5$, where laminar-to-turbulent
boundary-layer transition delays separation and reduces $C_D$
sharply). The collapse onto a single curve over seven decades of
$\mathrm{Re}$ is what makes the dimensional-similarity argument
economically valuable. See \texttt{code/drag\_curve.py} for the
Morsi-Alexander piecewise correlation
\cite{paper:morsi-alexander-1972}.}
\label{fig:vol01:ch02:drag-curve}
\end{figure}
