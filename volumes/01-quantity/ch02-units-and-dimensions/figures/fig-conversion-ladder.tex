% Figure: the conversion ladder, showing a chained unit conversion as
% a stack of multiplications by ratios equal to one, with the cancelled
% units struck through so the surviving unit is visible at each rung.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  quant/.style={draw=engblack, fill=englime!15, rounded corners=2pt,
    minimum width=2.4cm, minimum height=0.9cm, font=\small, align=center},
  ratio/.style={draw=engblack, fill=engblue!10, rounded corners=2pt,
    minimum width=2.6cm, minimum height=0.9cm, font=\scriptsize,
    align=center},
  op/.style={font=\normalsize\bfseries, draw=none},
  note/.style={font=\scriptsize, align=left, anchor=west, text=enggray},
]

% Row of the chained conversion 60 mph -> m/s
\node[quant] (start) at (0, 0) {$60\ \mathrm{mph}$};
\node[op]    (t1)    at (2.3, 0) {$\times$};
\node[ratio] (r1)    at (4.6, 0) {$\dfrac{1609.344\ \mathrm{m}}{1\ \mathrm{mile}}$};
\node[op]    (t2)    at (6.9, 0) {$\times$};
\node[ratio] (r2)    at (9.4, 0) {$\dfrac{1\ \mathrm{h}}{3600\ \mathrm{s}}$};

% Result line
\node[quant, fill=engred!10] (res) at (4.6, -2.0)
  {$26.8\ \mathrm{m}\,\mathrm{s}^{-1}$};
\draw[-{Stealth}, thick, draw=engblack!70] (start) -- (res);
\draw[-{Stealth}, thick, draw=engblack!70] (r1) -- (res);
\draw[-{Stealth}, thick, draw=engblack!70] (r2) -- (res);

% Cancellation annotations
\node[note] at (-1.3, -0.9) {mile cancels in rung 1};
\node[note] at (7.7, -0.9)  {hour cancels in rung 2};
\node[note] at (2.1, -2.9)
  {surviving units: $\mathrm{m}$ (numerator), $\mathrm{s}$ (denominator)};

\end{tikzpicture}
\caption[The conversion ladder]{A chained conversion read as a ladder
of multiplications by ratios equal to one. Each rung carries a unit
that appears once in a numerator and once in a denominator and so
cancels; the surviving units are the metre in a numerator and the
second in a denominator, giving the coherent SI answer. The engineer
who writes each rung explicitly can audit the chain by inspection: an
uncancelled unit at the bottom is an arithmetic error caught before
any number is trusted.}
\label{fig:vol01:ch02:conversion-ladder}
\end{figure}
