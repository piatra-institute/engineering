% Figure: the Taylor-Sedov blast-wave scaling law R ~ (E/rho)^{1/5} t^{2/5},
% plotted as log R against log t, a straight line of slope 2/5. The plotted
% points are the released frames of the 1945 Trinity fireball as read by
% G. I. Taylor.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  point/.style={circle, fill=engred, draw=engblack, inner sep=1.3pt},
]

% Axes
\draw[-{Stealth}, thick] (0, 0) -- (9.5, 0)
  node[right, font=\small] {$\log_{10} t\ (\mathrm{s})$};
\draw[-{Stealth}, thick] (0, 0) -- (0, 6.0)
  node[above, font=\small] {$\log_{10} R\ (\mathrm{m})$};

% Reference gridlines
\foreach \x in {2, 4, 6, 8} {
  \draw[engblack!15] (\x, 0) -- (\x, 5.6);
}
\foreach \y in {1, 2, 3, 4, 5} {
  \draw[engblack!15] (0, \y) -- (9.2, \y);
}

% The 2/5-slope scaling line: from (1, 0.6) to (9, 3.8)
% slope over the drawn span: (3.8 - 0.6)/(9 - 1) = 0.40 = 2/5
\draw[thick, draw=engblue] (1.0, 0.6) -- (9.0, 3.8);
\node[engblue, font=\small\bfseries, anchor=south west, rotate=21]
  at (4.6, 2.2) {slope $= 2/5$};

% Data points (illustrative frames along the line)
\node[point] at (1.6, 0.84) {};
\node[point] at (3.0, 1.4)  {};
\node[point] at (4.4, 1.96) {};
\node[point] at (5.8, 2.52) {};
\node[point] at (7.2, 3.08) {};
\node[point] at (8.4, 3.56) {};

% Callout of the scaling law
\node[draw=engblack, fill=englime!15, rounded corners=2pt,
  font=\small, align=center, inner sep=4pt] at (5.2, 5.0)
  {$R(t) = C\,\left(\dfrac{E}{\rho}\right)^{1/5} t^{2/5}$};

\end{tikzpicture}
\caption[Taylor-Sedov blast-wave scaling]{The blast-wave radius grows
as the two-fifths power of time, the exponent fixed by dimensional
analysis alone from the energy release $E$ and the ambient air density
$\rho$. Plotted on log-log axes the radius-time record of a strong
point explosion falls on a straight line of slope $2/5$; the intercept
fixes the energy once the dimensionless constant $C$ of order unity is
supplied. Taylor read the yield of the 1945 Trinity test from
declassified fireball photographs by this method, recovering a value
near $20$ kilotonnes of TNT.}
\label{fig:vol01:ch02:blast-scaling}
\end{figure}
