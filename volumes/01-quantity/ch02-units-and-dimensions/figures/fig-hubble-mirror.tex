% Figure: schematic of the Hubble primary-mirror metrology chain that
% produced the 2-um figure error. Shows the metering rod, the
% reflective null corrector, and the auxiliary tests that were
% overruled.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  box/.style={draw=engblack, rounded corners=2pt, minimum height=0.9cm,
    align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblack!80},
  warn/.style={draw=engred, thick, fill=engred!10, rounded corners=2pt,
    minimum height=0.9cm, align=center, font=\small},
]

% Top: metering rod with the misread feature
\node[box, fill=engblue!10, minimum width=4cm] (rod) at (0, 3.5)
  {Metering rod\\(end cap vs internal feature)};

% Center: reflective null corrector with wrong spacing
\node[warn, minimum width=4.5cm] (rnc) at (0, 1.8)
  {Reflective null corrector\\spacing off by $1.3 \,\si{\milli\meter}$};

% Below: the primary mirror, figured to wrong shape
\node[warn, minimum width=4.5cm] (mirror) at (0, 0)
  {Primary mirror figured\\$2\,\si{\micro\meter}$ too flat at the edge};

% Auxiliary tests on the side
\node[box, fill=englime!15, minimum width=3.5cm] (refn) at (-5, 2.7)
  {Refractive null test\\(disagrees)};
\node[box, fill=englime!15, minimum width=3.5cm] (inv) at (-5, 0.9)
  {Inverse null test\\(disagrees)};

% Both auxiliary tests dismissed
\node[font=\scriptsize\itshape, engred, align=center] at (-5, -0.2)
  {set aside under\\cost/schedule pressure};

% Connections
\draw[arrow] (rod) -- (rnc);
\draw[arrow] (rnc) -- (mirror);
\draw[arrow, draw=engred] (refn) -- (rnc);
\draw[arrow, draw=engred] (inv) -- (rnc);

% On-orbit consequence
\node[box, fill=engred!20, minimum width=4.5cm] (orbit) at (0, -1.6)
  {On-orbit blur (1990-1993)\\COSTAR + later instrument optics};
\draw[arrow] (mirror) -- (orbit);

\end{tikzpicture}
\caption[Hubble primary-mirror metrology chain]{The metrology chain
that produced the Hubble Space Telescope's $2\,\si{\micro\meter}$
figure error. A metering rod's end cap and an internal target feature
were not distinguished during the reflective null corrector's setup,
producing a $1.3 \,\si{\milli\meter}$ spacing error. The resulting
corrector misled the polishing operation. Two auxiliary tests (a
refractive null and an inverse null) flagged the figure error before
launch but were set aside under cost and schedule pressure. The 1993
Servicing Mission 1 installed COSTAR to apply optical correction;
subsequent instruments incorporated the correction directly. Adapted
from the NASA Optical Systems Failure Report
\cite{acc:nasa-hubble-optical-systems-1990}.}
\label{fig:vol01:ch02:hubble-mirror}
\end{figure}
