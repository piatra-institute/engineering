\begin{figure}[ht]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Reference cosine and three phase-shifted companions
\draw[->, thick] (-0.1, 0) -- (8.4, 0) node[right, font=\footnotesize] {$t$};
\draw[->, thick] (0, -1.3) -- (0, 1.4);

% Reference cosine (period 4)
\draw[domain=0:8, samples=200, smooth, engblack, thick]
  plot (\x, {cos(deg(2*pi*\x/4))});
\node[engblack, anchor=west, font=\footnotesize] at (8.4, 1.0)
  {reference};

% In quadrature: lead by pi/2 (shift left by one quarter period = 1)
\draw[domain=0:8, samples=200, smooth, engblue, thick, dashed]
  plot (\x, {cos(deg(2*pi*\x/4) + 90)});
\node[engblue, anchor=west, font=\footnotesize] at (8.4, 0.5)
  {quadrature ($+\pi/2$)};

% Out of phase: shift by pi (half period = 2)
\draw[domain=0:8, samples=200, smooth, engred, thick, densely dotted]
  plot (\x, {cos(deg(2*pi*\x/4) + 180)});
\node[engred, anchor=west, font=\footnotesize] at (8.4, 0.0)
  {out of phase ($\pi$)};

% Mark one period and the quarter-period lead
\draw[<->, enggray, thin] (0, -1.15) -- (4, -1.15)
  node[midway, below, font=\footnotesize] {one period $T$};
\draw[<->, enggray, thin] (0, -0.75) -- (1, -0.75)
  node[midway, below, font=\footnotesize, yshift=-1pt] {$T/4$};

\end{tikzpicture}
\caption[Phase relationships between same-frequency
sinusoids]{Three same-frequency sinusoids against a reference
cosine (black). The quadrature signal (blue dashed) leads by
$\pi/2$, one quarter period. The out-of-phase signal (red dotted)
is shifted by $\pi$, half a period, and is the negative of the
reference. All three share the period $T$; only the phase offset
distinguishes them. A relative phase is meaningful only between
signals of the same frequency.}
\label{fig:vol01:ch06:phase-relationships}
\end{figure}
