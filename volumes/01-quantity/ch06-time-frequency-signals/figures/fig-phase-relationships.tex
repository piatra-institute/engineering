\begin{figure}[ht]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Axes
\draw[->, thick] (-0.1, 0) -- (8.4, 0) node[right, font=\footnotesize] {$t$};
\draw[->, thick] (0, -1.4) -- (0, 1.5);
\node[anchor=east, font=\footnotesize] at (-0.1, 1.0) {$+A$};
\node[anchor=east, font=\footnotesize] at (-0.1, -1.0) {$-A$};

% Reference cosine (blue)
\draw[domain=0:8, samples=200, smooth, engblue, very thick]
  plot (\x, {cos(deg(2*pi*\x/4))});

% In quadrature: 90 degrees (pi/2) lag, red dashed
\draw[domain=0:8, samples=200, smooth, engred, thick, dashed]
  plot (\x, {cos(deg(2*pi*\x/4 - pi/2))});

% Out of phase: 180 degrees (pi) lag, amber dotted
\draw[domain=0:8, samples=200, smooth, engamber, thick, dotted]
  plot (\x, {cos(deg(2*pi*\x/4 - pi))});

% Legend
\node[engblue, anchor=west, font=\footnotesize] at (8.4, 1.15) {reference};
\node[engred, anchor=west, font=\footnotesize] at (8.4, 0.70) {quadrature ($\pi/2$)};
\node[engamber, anchor=west, font=\footnotesize] at (8.4, 0.25) {out of phase ($\pi$)};

% Quarter-period marker for quadrature offset
\pgfmathsetmacro{\qx}{1.0}
\draw[gray, dashed] (\qx, -1.4) -- (\qx, 1.2);
\node[gray, anchor=south, font=\footnotesize] at (\qx, 1.2) {$T/4$};

\end{tikzpicture}
\caption[Phase relationships between sinusoids of equal frequency]{Three
sinusoids of equal amplitude and frequency at three relative phases.
The reference cosine (solid) leads the quadrature signal (dashed) by
$\pi/2$, a quarter period; the out-of-phase signal (dotted) is the
negative of the reference, offset by $\pi$. A relative phase of $0$
would place a fourth curve exactly on top of the reference. Engineers
read these offsets directly off a two-channel oscilloscope by aligning
zero crossings.}
\label{fig:vol01:ch06:phase-relationships}
\end{figure}
