\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Vertical power ladder in watts, log scale: one decade per 0.85 cm
% map exponent e to y = (e + 6) * 0.85, spanning 10^-6 to 10^16
\def\ymap#1{(#1 + 6) * 0.85}

% Main spine
\draw[->, thick] (0, {\ymap{-6.5}}) -- (0, {\ymap{16.5}})
  node[anchor=south] {power (W)};

% Decade ticks
\foreach \e in {-6, -3, 0, 3, 6, 9, 12, 15} {
  \draw (-0.08, {\ymap{\e}}) -- (0.08, {\ymap{\e}});
  \node[anchor=east, font=\scriptsize] at (-0.12, {\ymap{\e}}) {$10^{\e}$};
}

% Anchor markers on the right of the spine
\foreach \e/\lbl in {
  -5/{single neuron firing},
  -3/{quartz wristwatch},
  0/{LED indicator lamp},
  2/{resting adult ($10^2$)},
  2.5/{active adult},
  3/{domestic kettle},
  4.7/{family car at cruise},
  6/{diesel locomotive},
  8/{large wind turbine},
  9.5/{jet airliner at take-off},
  12/{large power station},
  13.2/{world mean power draw},
  16/{sunlight on Earth's disk}} {
  \fill[engblue] (0, {\ymap{\e}}) circle (0.06);
  \draw[engblue, opacity=0.5] (0, {\ymap{\e}}) -- (0.5, {\ymap{\e}});
  \node[anchor=west, font=\scriptsize] at (0.55, {\ymap{\e}}) {\lbl};
}

% Human-scale band highlight
\fill[englime, opacity=0.12] (-2.6, {\ymap{2}}) rectangle (5.4, {\ymap{3}});
\node[anchor=east, enggray, font=\scriptsize\itshape] at (-1.0, {\ymap{2.5}}) {human};

\end{tikzpicture}
\caption[The power anchor ladder]{The working power ladder, in
watts on a logarithmic axis. Twenty-one decades separate a single
firing neuron from the sunlight intercepted by the Earth's disk. The
resting adult at $10^{2}\,\si{\watt}$ and the world's mean per-person
power draw at a few kilowatts are the two anchors an engineer reaches
for most often; the shaded band marks the narrow human range from
which many bioenergetic and infrastructural estimates start. An
engineer who carries this column places any new power figure within a
decade before reading its exact value.}
\label{fig:vol01:ch09:power-anchor-ladder}
\end{figure}
