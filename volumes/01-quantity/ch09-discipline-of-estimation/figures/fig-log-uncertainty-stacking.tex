\begin{figure}[ht]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Axes
\draw[->, thick] (0, 0) -- (8.5, 0) node[anchor=north east] {number of factors $k$};
\draw[->, thick] (0, 0) -- (0, 5.5) node[anchor=south east] {combined factor uncertainty};

% Tick marks on x: 1 to 8
\foreach \k in {1, 2, 3, 4, 5, 6, 7, 8} {
  \draw (\k, -0.05) -- (\k, 0.05);
  \node[anchor=north, font=\scriptsize] at (\k, -0.05) {\k};
}

% Tick marks on y: factor 1, 2, 3, 5, 10, 30
\foreach \y/\lbl in {0/1, 1.0/2, 1.58/3, 2.32/5, 3.32/10, 4.91/30} {
  \draw (-0.05, \y) -- (0.05, \y);
  \node[anchor=east, font=\scriptsize] at (-0.05, \y) {\lbl$\times$};
}

% Per-factor 1.5x, combine in quadrature: combined = 1.5^sqrt(k)
% In log10: log10(combined) = sqrt(k) * log10(1.5) ~ sqrt(k) * 0.176
% In our y units: y = sqrt(k) * 0.176 / log10(2) * 1 = sqrt(k) * 0.585
\draw[thick, engblue]
  plot[domain=1:8, samples=40] (\x, {sqrt(\x) * 0.585});
\node[anchor=west, engblue, font=\footnotesize] at (8, {sqrt(8) * 0.585}) {
  per-factor $1.5\times$};

% Per-factor 2x (sqrt(k) * log10(2) = sqrt(k)); y in our units
\draw[thick, engred]
  plot[domain=1:8, samples=40] (\x, {sqrt(\x) * 1.0});
\node[anchor=west, engred, font=\footnotesize] at (8, {sqrt(8) * 1.0}) {
  per-factor $2\times$};

% Worst-case stacking, per-factor 1.5x: combined = 1.5^k
% y = k * log10(1.5) / log10(2) = k * 0.585
\draw[thick, dashed, engblue]
  plot[domain=1:8, samples=40] (\x, {\x * 0.585});
\node[anchor=west, engblue, font=\footnotesize, opacity=0.7] at (6.5, {6.5 * 0.585}) {
  worst-case ($1.5\times$)};

% Horizontal target lines
\draw[dotted] (0, 1.58) -- (8, 1.58);
\node[anchor=east, font=\scriptsize] at (8, 1.85) {factor-of-three target};

\draw[dotted] (0, 3.32) -- (8, 3.32);
\node[anchor=east, font=\scriptsize] at (8, 3.55) {order-of-magnitude};

\end{tikzpicture}
\caption[How estimation uncertainty stacks with the number of
factors]{Combined multiplicative uncertainty as a function of the
number of independent factors in the decomposition, for two
per-factor widths ($1.5\times$ and $2\times$) and assuming
independent log-Gaussian errors that combine in quadrature. The
solid curves are the working approximation; the dashed line shows
worst-case stacking for the $1.5\times$ case. The
factor-of-three working target is reached at $k \approx 8$ for
per-factor $1.5\times$ errors and at $k \approx 3$ for per-factor
$2\times$ errors; order-of-magnitude precision sets a hard ceiling
on the number of factors a useful decomposition can carry.}
\label{fig:vol01:ch09:log-uncertainty}
\end{figure}
