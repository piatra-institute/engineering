\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Axes
\draw[->, thick] (0, 0) -- (10.5, 0) node[anchor=north east] {sample size $n$};
\draw[->, thick] (0, 0) -- (0, 5) node[anchor=south east] {estimate};

% Truth line
\draw[dashed, thick, engblue] (0, 3) -- (10, 3);
\node[anchor=west, engblue, font=\footnotesize] at (10, 3) {truth};

% Individual guesses scattered (log-spaced)
\foreach \x/\y in
  {0.3/1.5, 0.5/4.6, 0.8/0.8, 1.2/4.2, 1.6/1.9, 2.1/3.6, 2.6/2.4, 3.2/3.8,
   3.8/2.6, 4.5/3.4, 5.2/2.7, 6.0/3.3, 6.8/2.85, 7.6/3.15, 8.5/2.95, 9.4/3.05}
  \fill[engred, opacity=0.55] (\x, \y) circle (0.06);

% Running geometric mean trace (smoothed convergence to truth)
\draw[thick, engred]
  plot[smooth] coordinates
   {(0.3, 1.5) (0.5, 2.4) (0.8, 1.9) (1.2, 2.55) (1.6, 2.4)
    (2.1, 2.65) (2.6, 2.6) (3.2, 2.8) (3.8, 2.78) (4.5, 2.85)
    (5.2, 2.83) (6.0, 2.92) (6.8, 2.94) (7.6, 2.99) (8.5, 2.99) (9.4, 3.01)};

% Confidence band shrinking as 1/sqrt(n)
\draw[engred, opacity=0.4, dashed]
  plot[domain=0.3:10, samples=40] (\x, {3 + 1.5/sqrt(\x)});
\draw[engred, opacity=0.4, dashed]
  plot[domain=0.3:10, samples=40] (\x, {3 - 1.5/sqrt(\x)});

% Label
\node[anchor=south, engred, font=\footnotesize] at (4, 4.6) {
  running geometric mean};
\node[anchor=north, engred, font=\footnotesize, opacity=0.7] at (8, 0.4) {
  $\pm 1\sigma / \sqrt{n}$ band};

\end{tikzpicture}
\caption[Convergence of crowdsourced Fermi estimates toward
truth]{Individual estimates (filled circles) scatter widely; the
geometric mean (running curve) converges to the truth as
$1/\sqrt{n}$, reproducing the Galton ox-weight result. The
discipline that works for crowds, take many independent estimates
and aggregate, also works for the individual engineer with several
independent decompositions of the same problem. The aggregation has
to be in log space because Fermi factors compound multiplicatively.}
\label{fig:vol01:ch09:crowd-convergence}
\end{figure}
