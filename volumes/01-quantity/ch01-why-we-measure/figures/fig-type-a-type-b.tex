% Figure: schematic of Type A and Type B uncertainty evaluation as
% defined by the GUM. Type A is statistical; Type B is from other
% sources (calibration certificates, instrument specs, expert
% judgement).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  box/.style={draw=engblack, rounded corners=3pt, minimum height=0.9cm,
    align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblack!70},
]

% Top-level node
\node[box, fill=englime!20, minimum width=4.5cm] (combined)
  at (5, 4) {Combined standard uncertainty\\$u_c$};

% Type A branch
\node[box, fill=engblue!15, minimum width=3.5cm] (typea)
  at (1.5, 2) {\textbf{Type A}\\statistical analysis of repeated\\measurements};
% Type B branch
\node[box, fill=engblue!15, minimum width=3.5cm] (typeb)
  at (8.5, 2) {\textbf{Type B}\\other means: certificates, specs,\\expert judgement};

% Type A inputs
\node[box, minimum width=3.5cm] (a1) at (1.5, 0.3) {repeated readings};
% Type B inputs
\node[box, minimum width=3.5cm] (b1) at (8.5, 0.8) {calibration certificate};
\node[box, minimum width=3.5cm] (b2) at (8.5, -0.3) {instrument specifications};
\node[box, minimum width=3.5cm] (b3) at (8.5, -1.4) {prior measurement history};

% Arrows
\draw[arrow] (typea.north) -- (combined.south west);
\draw[arrow] (typeb.north) -- (combined.south east);
\draw[arrow] (a1) -- (typea);
\draw[arrow] (b1) -- (typeb);
\draw[arrow] (b2) -- (typeb);
\draw[arrow] (b3) -- (typeb);

% Expansion arrow
\node[box, fill=engred!15, minimum width=4.5cm] (expanded)
  at (5, 5.5) {Expanded uncertainty\\$U = k \, u_c$};
\draw[arrow] (combined) -- (expanded);

\end{tikzpicture}
\caption[Type A and Type B uncertainty]{The GUM's distinction
between Type A and Type B uncertainty evaluation
\cite{std:iso-iec-guide98}. Type A is statistical analysis of
repeated measurements; Type B is from other means. Both contribute
to the combined standard uncertainty $u_c$, which is then expanded
by a coverage factor (often $k = 2$) to produce the reported
expanded uncertainty $U$. The labels A and B refer to how the
uncertainty was evaluated, not to whether the underlying effect is
random or systematic.}
\label{fig:vol01:ch01:type-a-b}
\end{figure}
