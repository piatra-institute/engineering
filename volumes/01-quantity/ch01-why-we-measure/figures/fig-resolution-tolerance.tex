% Figure: the resolution-below-tolerance rule. A dispensed value sits
% inside a tolerance band; an instrument whose resolution is coarser
% than the band cannot certify the value, one whose resolution is
% finer can.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  tol/.style={draw=englime!60!engblack, very thick},
  tick/.style={draw=enggray, thin},
  label/.style={font=\small},
]
% Number line for the measured quantity. Target 8 uL sits at x=6.5;
% one uL = 1.3 units. Band 7.6..8.4 -> x = 6.02..6.98.
\draw[axis] (2.6,0) -- (11.6,0) node[anchor=west, label] {dispensed volume ($\si{\micro\liter}$)};
\foreach \x/\v in {3.9/6, 5.2/7, 6.5/8, 7.8/9, 9.1/10, 10.4/11} {
  \draw (\x,-0.1) -- (\x,0.1);
  \node[font=\scriptsize, below] at (\x,-0.1) {\v};
}

% Tolerance band around target 8 uL, +/- 0.4 uL.
\fill[englime, opacity=0.20] (6.02,0.3) rectangle (6.98,4.2);
\draw[tol] (6.02,0.3) -- (6.02,4.2);
\draw[tol] (6.98,0.3) -- (6.98,4.2);
\node[englime!55!engblack, font=\small, anchor=south] at (6.5,4.25)
  {tolerance $\pm 0.4\,\si{\micro\liter}$};
\draw[draw=engblack, thick, dotted] (6.5,0.5) -- (6.5,4.2);

% Coarse instrument: half-division reach of +/-5 uL, deliberately
% overrunning the band at both edges (arrowheads mark it continues).
\node[label, anchor=east] at (2.4,3.2) {\parbox{2.4cm}{\raggedleft
  coarse:\\ $\pm 5\,\si{\micro\liter}$}};
\draw[engred, very thick, {Stealth}-{Stealth}] (2.7,3.2) -- (10.3,3.2);
\fill[engred] (6.4,3.1) rectangle (6.6,3.3);

% Fine instrument: micropipette, +/- 0.1 uL.
\node[label, anchor=east] at (2.4,1.6) {\parbox{2.4cm}{\raggedleft
  fine:\\ $\pm 0.1\,\si{\micro\liter}$}};
\draw[engblue, very thick, |-|] (6.37,1.6) -- (6.63,1.6);
\fill[engblue] (6.4,1.5) rectangle (6.6,1.7);

\end{tikzpicture}
\caption[Instrument resolution against required tolerance]{Resolution
must be finer than tolerance. The green band is the allowed window for
a dispensed dose, $\pm 0.4\,\si{\micro\liter}$ around an
$8\,\si{\micro\liter}$ target. The coarse instrument (red) reads only
to $\pm 5\,\si{\micro\liter}$; its reach spills far past both edges of
the window, so a reading of ``on target'' cannot distinguish an
in-tolerance dose from one an order of magnitude out. The fine
instrument (blue) reads to $\pm 0.1\,\si{\micro\liter}$, sits inside
the window, and can certify the dose. The band is a property of the
requirement; the bracket is a property of the instrument; the two are
matched at the procedure-writing step, not at the bench.}
\label{fig:vol01:ch01:resolution-tolerance}
\end{figure}
