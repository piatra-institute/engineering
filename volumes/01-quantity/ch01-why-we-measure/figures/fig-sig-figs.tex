% Figure: comparison of significant-figures convention vs explicit
% uncertainty for the same reported value.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  bar/.style={very thick},
  label/.style={font=\small},
]

% Axis
\draw[axis] (0,0) -- (12,0) node[anchor=west, label] {voltage (V)};
\foreach \x/\v in {0/4.80, 2/4.90, 4/5.00, 6/5.10, 8/5.20, 10/5.30, 12/5.40} {
  \draw (\x,-0.1) -- (\x,0.1);
  \node[font=\scriptsize, below] at (\x,-0.1) {\v};
}

% Case 1: "5.32 V" by sig figs convention. Implied uncertainty +/- 0.005
\node[label, anchor=west] at (-2.5, 3.5) {\textbf{``5.32 V'' (3 sig figs):}};
\node[label, anchor=west, font=\scriptsize] at (-2.5, 3.0)
  {implied $\pm 0.005$};
\draw[bar, engblue] (10.4, 3.5) -- (10.4, 3.5);
\fill[engblue] (10.4, 3.4) rectangle (10.45, 3.6);
\draw[bar, engblue!50] (10.39, 3.3) -- (10.41, 3.7);
\draw[bar, engblue!50] (10.49, 3.3) -- (10.51, 3.7);

% Case 2: explicit uncertainty "5.32 +/- 0.05 V" (Type B from spec)
\node[label, anchor=west] at (-2.5, 2.2) {\textbf{``$5.32 \pm 0.05$ V'':}};
\node[label, anchor=west, font=\scriptsize] at (-2.5, 1.7)
  {explicit $u = 0.05$};
\draw[bar, engred] (9.4, 2.0) -- (11.4, 2.0);
\fill[engred] (10.4, 1.95) rectangle (10.45, 2.05);
\draw[bar, engred, |-|] (9.4, 1.7) -- (11.4, 1.7);

% Case 3: explicit expanded uncertainty "5.32 +/- 0.10 V (k=2)"
\node[label, anchor=west] at (-2.5, 0.9) {\textbf{``$5.32 \pm 0.10$ V $(k{=}2)$'':}};
\node[label, anchor=west, font=\scriptsize] at (-2.5, 0.4)
  {expanded $U = 0.10$};
\draw[bar, engblack] (8.4, 0.7) -- (12.4, 0.7);
\fill[engblack] (10.4, 0.65) rectangle (10.45, 0.75);
\draw[bar, engblack, |-|] (8.4, 0.4) -- (12.4, 0.4);

\end{tikzpicture}
\caption[Significant figures vs explicit uncertainty]{Three
conventions for reporting the same nominal value of $5.32 \,\si{\volt}$.
The significant-figures convention implies an uncertainty of order
$\pm 0.005 \,\si{\volt}$ (the third decimal place); explicit
uncertainty notation states it. The convention is sometimes sharper
than the underlying measurement supports and sometimes laxer; the
practice this book recommends is the explicit form when the number
matters.}
\label{fig:vol01:ch01:sig-figs}
\end{figure}
