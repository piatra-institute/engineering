% specification limit, showing accept, reject, and guard-band regions
% as a function of measured value and its uncertainty.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={Measured value (relative to tolerance limits)},
  ylabel={},
  xmin=-1.6, xmax=1.6,
  ymin=0, ymax=1.2,
  ytick=\empty,

  font=\small,
  axis lines=middle,
  clip=false,
]

% Reject zones (outside tolerance minus guard band)
\addplot[draw=none, fill=engred!12] coordinates
  {(-1.6,0) (-0.8,0) (-0.8,1) (-1.6,1)} \closedcycle;
\addplot[draw=none, fill=engred!12] coordinates
  {(0.8,0) (1.6,0) (1.6,1) (0.8,1)} \closedcycle;

% Guard-band zones (between tolerance limit and accept limit)
\addplot[draw=none, fill=engamber!18] coordinates
  {(-1.0,0) (-0.8,0) (-0.8,1) (-1.0,1)} \closedcycle;
\addplot[draw=none, fill=engamber!18] coordinates
  {(0.8,0) (1.0,0) (1.0,1) (0.8,1)} \closedcycle;

% Accept zone
\addplot[draw=none, fill=englime!20] coordinates
  {(-0.8,0) (0.8,0) (0.8,1) (-0.8,1)} \closedcycle;

% Tolerance limits
\addplot[engblack, thick, dashed] coordinates {(-1.0,0) (-1.0,1.05)};
\addplot[engblack, thick, dashed] coordinates {(1.0,0) (1.0,1.05)};

% Guarded acceptance limits
\addplot[engblue, very thick] coordinates {(-0.8,0) (-0.8,1.05)};
\addplot[engblue, very thick] coordinates {(0.8,0) (0.8,1.05)};

% Labels

% Guard-band width arrow
\draw[<->, engblue] (axis cs:0.8, 0.28) -- (axis cs:1.0, 0.28)

\end{axis}

\caption[Guard-banded conformity decision]{Guard-banded acceptance
around a two-sided tolerance $\pm T$. The accepted-value limit $A =
T - g$ is pulled inward from the tolerance limit by a guard band of
width $g$, chosen so that a measured value inside $\pm A$ conforms to
$\pm T$ at the required confidence given the measurement uncertainty.
A measured value in the amber guard band lies inside the tolerance
but too close to the limit for the measurement uncertainty to prove
conformity; the value is not accepted under a guarded rule. Widening
$g$ trades a lower false-accept probability for a higher false-reject
probability. The guard-band width follows from the test uncertainty
ratio and the accepted consumer-risk level per ILAC G8 and JCGM 106
\cite{std:ch03t3b-ilac-p14, std:ch03t3-jcgm-100-2008}.}
\label{fig:vol01:ch03:guard-band}
\end{figure}
