% Figure: distinguishing calibration, adjustment, verification,
% validation as four operations that work on different parts of the
% measurement-and-use system.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  op/.style={draw=engblack, rounded corners=2pt, minimum width=3.5cm,
    minimum height=1.4cm, align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblack!70},
]

\node[op, fill=engblue!10] (cal) at (-5, 1.5)
  {\textbf{Calibration}\\establishes relation\\between indication\\and reference};

\node[op, fill=englime!15] (adj) at (-1.5, 1.5)
  {\textbf{Adjustment}\\changes the\\instrument};

\node[op, fill=engred!10] (ver) at (2, 1.5)
  {\textbf{Verification}\\checks against\\specified requirements};

\node[op, fill=engred!15] (val) at (5.5, 1.5)
  {\textbf{Validation}\\fitness for\\intended use};

% Inputs
\node[font=\scriptsize, align=center, anchor=north] at (-5, 0.5)
  {reference\\+ indication};
\node[font=\scriptsize, align=center, anchor=north] at (-1.5, 0.5)
  {calibration result\\+ adjustment range};
\node[font=\scriptsize, align=center, anchor=north] at (2, 0.5)
  {calibration result\\+ specification};
\node[font=\scriptsize, align=center, anchor=north] at (5.5, 0.5)
  {verified instrument\\+ application};

% Outputs
\node[font=\scriptsize\itshape, align=center, anchor=south] at (-5, 2.9)
  {certificate};
\node[font=\scriptsize\itshape, align=center, anchor=south] at (-1.5, 2.9)
  {new setting};
\node[font=\scriptsize\itshape, align=center, anchor=south] at (2, 2.9)
  {pass / fail};
\node[font=\scriptsize\itshape, align=center, anchor=south] at (5.5, 2.9)
  {fitness assertion};

% Connecting arrows
\draw[arrow, draw=engblack!40] (cal) -- (adj)
  node[midway, above, font=\scriptsize] {if needed};
\draw[arrow, draw=engblack!40] (cal) -- (ver);
\draw[arrow, draw=engblack!40] (ver) -- (val);

\end{tikzpicture}
\caption[Calibration, adjustment, verification, validation]{Four
distinct operations that are routinely confused. Calibration
establishes a relation; adjustment changes the instrument;
verification produces a pass/fail decision against a specification;
validation asserts fitness for an intended use. Each takes
different inputs and produces different outputs. The discipline
of distinguishing them is the discipline of being honest about
what a calibration certificate does and does not say.}
\label{fig:vol01:ch03:cal-vs-adj}
\end{figure}
