% Figure: the traceability pyramid showing defining constants ->
% primary -> secondary -> working standards -> user instruments.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  level/.style={draw=engblack, minimum height=0.9cm, align=center,
    font=\small},
  arrow/.style={-{Stealth}, very thick, draw=engblack!70},
]

% Top: defining constants
\node[level, fill=engred!15, minimum width=4cm] (constants)
  at (0, 4.5) {\textbf{SI defining constants}\\(2019 redefinition: exact values)};

% Primary realisations
\node[level, fill=engblue!15, minimum width=6cm] (primary)
  at (0, 3.0) {\textbf{Primary realisations} (BIPM, NMIs)\\
  Josephson, quantum Hall, Kibble, XRCD, cesium clock};

% Secondary (transfer) standards
\node[level, fill=engblue!10, minimum width=8cm] (secondary)
  at (0, 1.5) {\textbf{Secondary / transfer standards}\\
  Calibrated by primary; held at NMIs and accredited labs};

% Working standards
\node[level, fill=englime!20, minimum width=10cm] (working)
  at (0, 0) {\textbf{Working standards}\\
  Day-to-day reference at calibration labs and industry};

% User instruments
\node[level, fill=englime!10, minimum width=12cm] (user)
  at (0, -1.5) {\textbf{User instruments}\\
  Field, laboratory, production, clinical, household};

% Arrows downward with growing uncertainty
\draw[arrow] (constants) -- (primary);
\draw[arrow] (primary) -- (secondary);
\draw[arrow] (secondary) -- (working);
\draw[arrow] (working) -- (user);

% Uncertainty annotations on the right
\node[font=\scriptsize\itshape, anchor=west] at (7, 4.5) {$u \approx 0$ (exact)};
\node[font=\scriptsize\itshape, anchor=west] at (7, 3.0) {$u \sim 10^{-9}$ to $10^{-11}$};
\node[font=\scriptsize\itshape, anchor=west] at (7, 1.5) {$u \sim 10^{-7}$ to $10^{-9}$};
\node[font=\scriptsize\itshape, anchor=west] at (7, 0) {$u \sim 10^{-5}$ to $10^{-7}$};
\node[font=\scriptsize\itshape, anchor=west] at (7, -1.5) {$u \sim 10^{-4}$ to $10^{-2}$};

% Left annotation
\node[font=\bfseries, rotate=90, anchor=south] at (-7, 1.5)
  {uncertainty grows};

\end{tikzpicture}
\caption[Traceability pyramid]{The traceability pyramid: SI
defining constants (exact by definition) anchor primary realisations
at national metrology institutes; secondary or transfer standards
disseminate the realisations to accredited calibration laboratories;
working standards at those laboratories calibrate user instruments
in field, laboratory, production, clinical, and household settings.
Each downward step adds uncertainty; the documentation of the chain
is what makes the user instrument's reading defensible. The
uncertainty figures shown are fractional and illustrative; specific
quantities and instruments have their own budgets
\cite{std:bipm-si2019, std:bipm-cipm-mra}.}
\label{fig:vol01:ch03:traceability-pyramid}
\end{figure}
