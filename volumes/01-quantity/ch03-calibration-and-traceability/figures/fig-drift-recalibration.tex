% Figure: instrument drift between calibrations, with tolerance
% band and recalibration intervals.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xlabel={Time (years)},
  ylabel={Indication error (relative units)},
  xmin=0, xmax=5,
  ymin=-1.5, ymax=1.5,
  grid=both,
  major grid style={engblack!20},
  font=\small,
  legend pos=south east,
  legend cell align=left,
]

% Tolerance band
\addplot[draw=none, fill=engred!10] coordinates
  {(0,1) (5,1) (5,-1) (0,-1)} \closedcycle;

% Linear drift (no recalibration)
\addplot[engred, very thick, domain=0:5, samples=50]
  {0.35*x};
\addlegendentry{drift without recalibration};

% Recalibration at year 2: reset to zero, drift resumes
\addplot[engblue, very thick] coordinates
  {(0,0) (2,0.7) (2,0) (4,0.7) (4,0) (5,0.35)};
\addlegendentry{recalibrate every 2 yr};

% Recalibration at year 1: reset to zero, drift resumes
\addplot[engblack, very thick, dashed] coordinates
  {(0,0) (1,0.35) (1,0) (2,0.35) (2,0) (3,0.35)
   (3,0) (4,0.35) (4,0) (5,0.35)};
\addlegendentry{recalibrate every 1 yr};

% Annotations
\node[engred, font=\scriptsize, anchor=east] at (axis cs:4.8, 1.1)
  {tolerance};
\node[engred, font=\scriptsize, anchor=east] at (axis cs:4.8, -1.1)
  {limit};

\end{axis}
\end{tikzpicture}
\caption[Drift and recalibration]{Linear drift in an instrument's
indication error, plotted against time. Without recalibration, the
error exceeds the tolerance band ($\pm 1$) at approximately year
$2.85$. Recalibration every $2$ years keeps the worst-case error
just under the tolerance throughout; recalibration every $1$ year
keeps it comfortably within. The recalibration-interval choice is
a trade-off between calibration cost (more frequent is more
expensive) and out-of-tolerance risk (less frequent is more risky).
The optimum depends on the drift rate, the tolerance, and the
relative cost of calibration vs out-of-tolerance discovery (see
\texttt{code/drift\_recalibration.py}).}
\label{fig:vol01:ch03:drift}
\end{figure}
