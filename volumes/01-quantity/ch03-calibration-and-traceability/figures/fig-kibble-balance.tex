% Figure: schematic of a Kibble balance showing the weighing-mode and
% moving-mode operations that together link mass to the Planck
% constant.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  comp/.style={draw=engblack, rounded corners=2pt, minimum height=0.8cm,
    align=center, font=\small},
  arrow/.style={-{Stealth}, thick, draw=engblack!70},
  measurement/.style={font=\scriptsize, anchor=west, engblue},
]

% Title
\node[font=\bfseries] at (0, 4.5) {Kibble balance: mass from $h$};

% Weighing mode (left side)
\node[font=\bfseries] at (-4, 3.5) {Weighing mode};
\node[comp, fill=englime!20] (mass) at (-4, 2.8) {Test mass $m$};
\node[comp, fill=engblue!10] (coilw) at (-4, 1.8) {Coil in field $B$};
\node[comp, fill=engred!10] (currw) at (-4, 0.8) {Current $I$};

\draw[arrow] (mass) -- (coilw)
  node[midway, right, font=\scriptsize] {$mg$};
\draw[arrow] (currw) -- (coilw)
  node[midway, right, font=\scriptsize] {$BIL$};

\node[measurement] at (-5.5, -0.2) {$mg = BIL$};

% Moving mode (right side)
\node[font=\bfseries] at (4, 3.5) {Moving mode};
\node[comp, fill=englime!20] (vel) at (4, 2.8) {Coil moves at $v$};
\node[comp, fill=engblue!10] (coilm) at (4, 1.8) {Same coil, same $B$};
\node[comp, fill=engred!10] (volt) at (4, 0.8) {Induced $U$};

\draw[arrow] (vel) -- (coilm);
\draw[arrow] (coilm) -- (volt)
  node[midway, right, font=\scriptsize] {$BLv$};

\node[measurement] at (3, -0.2) {$U = BLv$};

% Bottom: combine
\node[comp, fill=engred!20, minimum width=8cm] (result) at (0, -1.5)
  {Combine: $\boxed{m = \dfrac{UI}{gv}}$\\
  $U$, $I$ traceable via Josephson and quantum Hall to $h$};

\draw[arrow] (-4, -0.4) |- (result.west);
\draw[arrow] (4, -0.4) |- (result.east);

\end{tikzpicture}
\caption[Kibble balance schematic]{The Kibble balance realises the
kilogram by combining a weighing-mode measurement ($mg = BIL$) and
a moving-mode measurement ($U = BLv$). The geometric-flux product
$BL$ cancels between the two modes, yielding $m = UI/(gv)$. Because
the voltage $U$ and the current $I$ are realised through Josephson
junctions and the quantum-Hall resistance, both of which depend on
the Planck constant $h$, the realised mass traces to the fixed
$h$ of the post-2019 SI \cite{std:bipm-si2019, web:nist-kibble-balance}.}
\label{fig:vol01:ch03:kibble}
\end{figure}
