\begin{tikzpicture}[

  bucket/.style={rectangle, draw=engblack, fill=englime!20,
    align=center, font=\scriptsize\bfseries, minimum width=2.0cm,

  arrow/.style={-{Stealth}, draw=engblack!70, thick},
  edge/.style={font=\scriptsize\itshape, fill=white, inner sep=1pt},
]

\node[decision] (crit) at (0, 6) {Criticality\\(safety / regulatory)};

\node[decision] (drift-hi) at (-4, 4) {Drift\\$> 0.4\Delta$/yr?};
\node[decision] (drift-mid) at (4, 4) {Drift\\$> 0.4\Delta$/yr?};

\node[decision] (freq-hi-hi) at (-6, 1.8) {Freq\\$> 1000$/yr?};
\node[decision] (freq-hi-lo) at (-2, 1.8) {Freq\\$> 1000$/yr?};
\node[decision] (freq-lo-hi) at (2, 1.8) {Freq\\$> 1000$/yr?};
\node[decision] (freq-lo-lo) at (6, 1.8) {Freq\\$> 1000$/yr?};

\node[bucket] (b3mo) at (-7, -0.4) {$\leq 3$ mo or in-svc};
\node[bucket] (b6mo) at (-5, -0.4) {$6$ mo};
\node[bucket] (b12mo-a) at (-3, -0.4) {$12$ mo};
\node[bucket] (b12mo-b) at (-1, -0.4) {$12$ mo};
\node[bucket] (b18mo) at (1, -0.4) {$12$--$18$ mo};
\node[bucket] (b24mo) at (3, -0.4) {$24$ mo};
\node[bucket] (b24mo-b) at (5, -0.4) {$24$ mo};
\node[bucket] (b36mo) at (7, -0.4) {$36$ mo};

% Arrows from criticality
\draw[arrow] (crit) -| (drift-hi)
  node[edge, pos=0.25] {high};
\draw[arrow] (crit) -| (drift-mid)
  node[edge, pos=0.25] {low};

% Arrows from drift
\draw[arrow] (drift-hi) -| (freq-hi-hi) node[edge, pos=0.25] {yes};
\draw[arrow] (drift-hi) -| (freq-hi-lo) node[edge, pos=0.25] {no};
\draw[arrow] (drift-mid) -| (freq-lo-hi) node[edge, pos=0.25] {yes};
\draw[arrow] (drift-mid) -| (freq-lo-lo) node[edge, pos=0.25] {no};

% Arrows from frequency to buckets
\draw[arrow] (freq-hi-hi) -- (b3mo) node[edge, pos=0.5] {yes};
\draw[arrow] (freq-hi-hi.south) -- (b6mo.north);
\draw[arrow] (freq-hi-lo) -- (b12mo-a) node[edge, pos=0.5] {yes};
\draw[arrow] (freq-hi-lo.south) -- (b12mo-b.north);
\draw[arrow] (freq-lo-hi) -- (b18mo) node[edge, pos=0.5] {yes};
\draw[arrow] (freq-lo-hi.south) -- (b24mo.north);
\draw[arrow] (freq-lo-lo) -- (b24mo-b) node[edge, pos=0.5] {yes};
\draw[arrow] (freq-lo-lo.south) -- (b36mo.north);

\caption[Recalibration interval decision tree]{Decision tree for
setting an instrument's recalibration interval. The three inputs
(criticality, drift rate relative to tolerance, measurement
frequency) map to an interval bucket from $\leq 3 \,\text{months}$
with in-service checks (safety-critical, high-drift,
high-frequency) to $36 \,\text{months}$ (low-criticality, low-drift,
low-frequency). The buckets correspond to the risk-scoring
matrix in the chapter's recalibration-interval section; ILAC G24
\cite{std:ch03t3-ilac-g24-2022} and NCSLI RP-1
\cite{web:ch03t3-ncsli-rp-1-2010} are the source references for the
underlying interval-setting methods. Adjustment of the interval per
the as-found data of the prior cycle (NCSLI RP-1 geometric
adjustment rule, $1.2 \times$ for passes, $0.7 \times$ for fails)
shifts the bucket up or down at the next decision.}
\label{fig:vol01:ch03:interval-tree}
\end{figure}
