% Figure: apparent length error of a 100 mm gauge block versus
% temperature departure from the 20 C reference, for several
% materials, illustrating why length metrology is temperature-bound.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[width=11cm, height=6.4cm,
  xlabel={Temperature departure from $20\,^{\circ}\mathrm{C}$ (K)},
  ylabel={Apparent length error ($\mu\mathrm{m}$)},
  xmin=-5, xmax=5, ymin=-12, ymax=12, grid=both,
  major grid style={engblack!20}, font=\small,
  legend pos=north west, legend cell align=left]
\addplot[draw=none, fill=englime!15] coordinates
  {(-5,-1) (5,-1) (5,1) (-5,1)} \closedcycle;
\addplot[engred, very thick, domain=-5:5, samples=2] {2.3*x};
\addlegendentry{aluminium ($23 \times 10^{-6}\,\mathrm{K}^{-1}$)}
\addplot[engblue, very thick, domain=-5:5, samples=2] {1.15*x};
\addlegendentry{steel ($11.5 \times 10^{-6}\,\mathrm{K}^{-1}$)}
\addplot[engorange, very thick, domain=-5:5, samples=2] {0.86*x};
\addlegendentry{titanium ($8.6 \times 10^{-6}\,\mathrm{K}^{-1}$)}
\addplot[engblack, very thick, dashed, domain=-5:5, samples=2] {0.12*x};
\addlegendentry{Invar ($1.2 \times 10^{-6}\,\mathrm{K}^{-1}$)}
\end{axis}
\end{tikzpicture}
\caption[Thermal expansion of gauge blocks]{Apparent length error of
a $100\,\si{\milli\meter}$ gauge block against temperature departure
from the $20\,^{\circ}\mathrm{C}$ standard reference temperature, for
four materials. A one-kelvin departure moves a steel block by
$1.15\,\si{\micro\meter}$, ten times the $\sim 0.1\,\si{\micro\meter}$
calibration uncertainty of a class-0 block. The green band marks a
$\pm 1\,\si{\micro\meter}$ working corridor: aluminium leaves it
within half a kelvin, steel within one kelvin, Invar not until eight
kelvin. Precision length work either controls temperature to
$\pm 0.1\,\si{\kelvin}$ or applies a documented thermal correction
with its own uncertainty contribution. The slopes are $\alpha L$ for
each material at $L = 100\,\si{\milli\meter}$.}
\label{fig:vol01:ch03:thermal-expansion}
\end{figure}
