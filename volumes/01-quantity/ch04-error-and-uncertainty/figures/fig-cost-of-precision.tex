\draw[->, thick] (0, 0) -- (0, 7) node[above] {cost per measurement};

\foreach \x/\lbl in {1/$10^{-2}$, 3/$10^{-4}$, 5/$10^{-6}$, 7/$10^{-8}$, 9/$10^{-10}$, 11/$10^{-12}$} {

\draw[thick, engred] (3, 0.8) .. controls (4, 1.5) and (5, 3.5) .. (7, 5);
\node[engred, anchor=south] at (5, 3.5) {procedural refinement};

% Wall (national-metrology-institute capabilities)
\draw[thick, engblack] (7, 5) -- (8, 6) -- (8.5, 6.8);

\caption[Cost-of-precision curve]{Cost per measurement as a function
of target precision (rightward means tighter precision). Three regions
are typical: a flat region where off-the-shelf instruments suffice;
a rising region where procedural refinement and better calibrations
are needed; and a wall at the frontier of the relevant national
metrology institute's capabilities, where each additional decade of
precision requires investment in new instrumentation or methods.
Curve is illustrative; the position of the wall depends on the
quantity (frequency reaches $10^{-18}$ routinely; mass at the
$10^{-8}$ level remains close to the metrology frontier as of 2026).}
