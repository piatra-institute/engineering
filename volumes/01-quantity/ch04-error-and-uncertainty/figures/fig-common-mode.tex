  \node[anchor=south west] at (0, 3.3) {\textbf{Two absolute readings}};
  \draw[->, thick] (0, 0) -- (7.2, 0) node[right] {time};
  \draw[->, thick] (0, 0) -- (0, 3.2);

  % shared drifting baseline
  \draw[thick, enggray, dashed] (0, 0.6) .. controls (2.5, 1.4) and (4.5, 2.0) .. (7, 2.6);
  \node[enggray, anchor=south] at (5.6, 2.3) {shared drift};

  % reading x1 (upper trace) and x2 (lower trace), both drifting together
  \draw[thick, engblue] (0, 1.5) .. controls (2.5, 2.3) and (4.5, 2.9) .. (7, 3.0);
  \node[engblue, anchor=west] at (7.05, 3.0) {$x_1$};
  \draw[thick, engred] (0, 1.0) .. controls (2.5, 1.8) and (4.5, 2.4) .. (7, 2.5);
  \node[engred, anchor=west] at (7.05, 2.5) {$x_2$};
\end{scope}

% Bottom panel: the difference, with the shared drift removed
\begin{scope}[shift={(0,-4.2)}]

  \draw[->, thick] (0, 0) -- (7.2, 0) node[right] {time};
  \draw[->, thick] (0, 0) -- (0, 3.2);

  % flat difference near a constant value: drift cancels, only residual noise
  \draw[thick, engblack] (0, 1.5) -- (1.2, 1.55) -- (2.4, 1.45) -- (3.6, 1.52)

  \node[engblack, anchor=west] at (7.05, 1.49) {$x_1 - x_2$};

\end{scope}

\caption[Common-mode rejection in a difference measurement]{Two
readings $x_1$ and $x_2$ taken from the same drifting instrument
(top) each carry a large shared drift plus a small independent
residual. Their difference (bottom) removes the shared drift, leaving
only the residual: with correlation coefficient $\rho \to 1$ and equal
standard uncertainties, the common component cancels and $u(x_1 - x_2)$
falls far below either $u(x_1)$ or $u(x_2)$. Comparator measurements in
metrology are built to exploit this cancellation.}
\label{fig:vol01:ch04:common-mode}
