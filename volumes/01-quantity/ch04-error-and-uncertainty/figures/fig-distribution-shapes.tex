% Common baseline axis

\draw[->, thick] (0, -0.2) -- (0, 3.4) node[above] {density};

% Uniform on [-a, a] with a = 2: height 1/(2a) scaled for display
% Standard deviation a/sqrt(3) = 1.155; drawn at display height 1.0
\draw[thick, engblue] (-2, 0.05) -- (-2, 1.0) -- (2, 1.0) -- (2, 0.05);
\node[engblue, anchor=west] at (2.15, 1.0) {uniform};

% Triangular on [-a, a] with a = 2: peak at 0, height 1/a scaled
% Standard deviation a/sqrt(6) = 0.816; drawn peak at display height 2.0
\draw[thick, engred] (-2, 0.05) -- (0, 2.0) -- (2, 0.05);
\node[engred, anchor=south west] at (0.9, 1.15) {triangular};

% Gaussian with matched-ish spread: peak height 2.6, sigma ~ 1.0
\draw[thick, engblack, domain=-3.6:3.6, samples=80, smooth]
  plot ({\x}, {2.6*exp(-(\x*\x)/(2*1.0*1.0))});
\node[engblack, anchor=south east] at (-0.9, 1.3) {Gaussian};

% half-width markers for uniform and triangular
\draw[dashed, enggray] (-2, 0.05) -- (-2, 2.0);
\draw[dashed, enggray] (2, 0.05) -- (2, 2.0);
\node[enggray, anchor=north] at (-2, -0.15) {$-a$};
\node[enggray, anchor=north] at (2, -0.15) {$+a$};

\end{tikzpicture}
\caption[Three bounded distribution shapes and their divisors]{Three
distribution shapes assigned in Type B evaluation, each on the same
half-width $[-a, a]$. The uniform (rectangular) distribution has
standard deviation $a/\sqrt{3}$; the symmetric triangular distribution,
appropriate when the central value is more likely than the extremes,
has standard deviation $a/\sqrt{6}$; the Gaussian is unbounded and its
standard deviation is set separately. The divisor that converts a
stated half-width into a standard uncertainty is $\sqrt{3}$ for the
uniform, $\sqrt{6}$ for the triangular, and the coverage factor for a
stated Gaussian expanded uncertainty. Vertical heights are drawn for
legibility and do not represent equal area.}
