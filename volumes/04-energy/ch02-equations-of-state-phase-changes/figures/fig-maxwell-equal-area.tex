% Figure: the Maxwell equal-area construction on a subcritical van der Waals
% isotherm in reduced variables (T_r = 0.90). The reduced equation
% p_r = 8 T_r / (3 v_r - 1) - 3/v_r^2 develops a non-monotonic loop below
% the critical temperature. The physical (horizontal) coexistence line at
% p_r = p_sat is drawn so that the two shaded lobes between the loop and the
% line have equal area. The reduced saturation pressure at T_r = 0.90 is
% about 0.647 (the value the chapter quotes). Coordinates are sampled from
% the reduced vdW equation at T_r = 0.90; v_r ranges from just above 1/3.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={reduced volume $v_r = v/v_c$},
  ylabel={reduced pressure $p_r = p/p_c$},
  xmin=0.3, xmax=4.0, ymin=0.3, ymax=1.1,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
  clip=true,
]
% van der Waals isotherm at T_r = 0.90: p_r = 8(0.90)/(3 v_r - 1) - 3/v_r^2
% = 7.2/(3 v_r - 1) - 3/v_r^2. Domain kept above v_r = 1/3 to avoid the pole.
\addplot[engblue, very thick, smooth, samples=200, domain=0.40:3.9] {7.2/(3*x - 1) - 3/(x^2)};
\addlegendentry{vdW isotherm, $T_r=0.90$}
% Maxwell coexistence line at p_sat,r = 0.647, drawn between the liquid and
% vapour saturation volumes (v_f,r ~ 0.60 and v_g,r ~ 2.35 at this T_r).
\addplot[engred, very thick] coordinates {(0.60,0.647) (2.35,0.647)};
\addlegendentry{Maxwell line, $p_{\text{sat},r}=0.647$}
% endpoint markers at the saturated liquid and saturated vapour states
\addplot[only marks, mark=*, engblack, mark size=1.8pt] coordinates {(0.60,0.647) (2.35,0.647)};
\node[engblack, font=\scriptsize, anchor=east] at (axis cs:0.58,0.66) {$v_{f}$};
\node[engblack, font=\scriptsize, anchor=west] at (axis cs:2.40,0.66) {$v_{g}$};
% labels for the two equal lobes
\node[enggray, font=\scriptsize] at (axis cs:0.80,0.78) {area A};
\node[enggray, font=\scriptsize] at (axis cs:1.55,0.58) {area B};
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:0.62,0.45) {unstable branch $(\partial p/\partial v)_T>0$};
\end{axis}
\end{tikzpicture}
\caption[Maxwell equal-area construction on a van der Waals
isotherm]{The Maxwell equal-area construction on a subcritical van der Waals
isotherm in reduced variables at $T_r=0.90$. The reduced equation
$p_r = 8T_r/(3v_r-1) - 3/v_r^2$ develops a loop with a rising middle branch
where $(\partial p/\partial v)_T>0$, which is mechanically unstable and
cannot be a physical equilibrium. The physical isotherm replaces the loop
with the horizontal coexistence line at the saturation pressure
$p_{\text{sat},r}$, drawn so that the upper lobe (area A, between the line
and the loop on the liquid side) equals the lower lobe (area B, on the
vapour side). Equal areas are the graphical statement of equal Gibbs free
energy of the two phases, the same equilibrium condition the
Clausius-Clapeyron derivation uses. The construction locates the
saturated-liquid volume $v_f$ and the saturated-vapour volume $v_g$ as the
endpoints of the line, and at $T_r=0.90$ it returns $p_{\text{sat},r}\approx
0.647$, the value the numerical routine reports.}
\label{fig:vol04ch02:maxwell-equal-area}
\end{figure}
