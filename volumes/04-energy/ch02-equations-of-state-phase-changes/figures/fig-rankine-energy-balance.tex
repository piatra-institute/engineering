% Figure: energy accounting for the worked Rankine cycle, drawn as a bar
% chart of the per-kilogram energy flows. Boiler heat in (q_in) splits into
% net work out and condenser heat rejected; the pump work is a small input.
% Numbers are the case-study values: q_in = 3160 kJ/kg, w_turbine = 1273,
% w_pump = 8.1, w_net = 1265, q_out = 1895 kJ/kg, eta = 0.40.
\begin{figure}[htbp]
\centering
\definecolor{engteal}{RGB}{30,120,120}
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  ybar,
  bar width=22pt,
  xlabel={energy flow},
  ylabel={specific energy (kJ/kg)},
  ymin=0, ymax=3400,
  symbolic x coords={boiler in,turbine out,pump in,net work,condenser out},
  xtick=data,
  x tick label style={font=\scriptsize, rotate=20, anchor=east},
  tick label style={font=\scriptsize},
  label style={font=\small},
  nodes near coords,
  nodes near coords style={font=\tiny},
  clip=false,
]
\addplot[draw=engblack, fill=engorange] coordinates {
  (boiler in,3160)
  (turbine out,1273)
  (pump in,8.1)
  (net work,1265)
  (condenser out,1895)
};
\end{axis}
\end{tikzpicture}
\caption[Energy accounting for the worked Rankine cycle]{The per-kilogram
energy flows of the worked steam Rankine cycle. The boiler delivers
$q_{\text{in}}=3160\,\text{kJ}/\text{kg}$, the largest flow in the cycle.
Of that, the turbine returns $w_{\text{turbine}}=1273\,\text{kJ}/\text{kg}$
of work, against which the feed pump consumes
$w_{\text{pump}}=8.1\,\text{kJ}/\text{kg}$, leaving a net work of
$w_{\text{net}}=1265\,\text{kJ}/\text{kg}$. The condenser rejects the
balance, $q_{\text{out}}=1895\,\text{kJ}/\text{kg}$, to the cooling water.
The thermal efficiency is the net work divided by the boiler heat,
$\eta = 1265/3160 = 0.40$. The figure makes the dominant fact of a steam
plant visible at a glance: most of the heat added at the boiler leaves at
the condenser, and the pump work is a rounding error against the turbine
work, which is why the pump term is so often neglected in a first pass.}
\label{fig:vol04ch02:rankine-energy-balance}
\end{figure}
