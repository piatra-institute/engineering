% Figure: p-v diagram with the vapour dome and three isotherms. One
% supercritical isotherm (monotonic), one critical isotherm (inflexion
% at the critical point), and one subcritical isotherm that runs flat
% across the two-phase region between the saturated-liquid and
% saturated-vapour limbs. Schematic, log-like horizontal spacing.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={specific volume $v$ (arb.\ units)},
  ylabel={pressure $p$ (arb.\ units)},
  xmin=0.3, xmax=10, ymin=0, ymax=10,
  axis lines=left,
  xtick=\empty, ytick=\empty,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=true,
]
% vapour dome: left limb (saturated liquid) and right limb (saturated vapour),
% peak at the critical point (vc=2.2, pc=6.5)
\addplot[engblue, very thick, smooth] coordinates {
  (0.9,0.6) (1.05,1.8) (1.25,3.2) (1.55,4.6) (1.9,5.8) (2.2,6.5)
};
\addplot[engblue, very thick, smooth] coordinates {
  (2.2,6.5) (2.8,5.9) (3.6,5.0) (4.8,3.8) (6.5,2.4) (8.8,1.0)
};
% critical point marker
\addplot[only marks, mark=*, engblack, mark size=2.2pt] coordinates {(2.2,6.5)};
\node[engblack, font=\scriptsize, anchor=south] at (axis cs:2.2,6.7) {critical point};
% subcritical isotherm: high p in liquid region, flat across dome, falls in vapour
\addplot[engred, thick, smooth] coordinates {
  (0.85,9.4) (0.95,7.0) (1.12,4.2)
};
\addplot[engred, thick] coordinates {(1.12,4.2) (5.4,4.2)};

  (5.4,4.2) (6.6,3.0) (8.2,2.0) (9.6,1.4)
};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:9.6,1.4) {$T<T_c$};

\addplot[enggray, thick, smooth] coordinates {
  (1.0,9.8) (1.8,8.2) (3.0,6.6) (5.0,5.0) (7.5,3.8) (9.6,3.2)

\node[enggray, font=\scriptsize, anchor=west] at (axis cs:9.6,3.4) {$T>T_c$};

\end{axis}

\caption[Pressure-volume isotherms and the vapour dome]{Pressure-volume
diagram with the vapour dome (blue) and three isotherms. The dome is the
locus of saturated states: its left limb is the saturated liquid, its
right limb the saturated vapour, and its peak the critical point. A
subcritical isotherm ($T<T_c$, red) drops steeply through the
compressed-liquid region, runs horizontal across the two-phase region at
the saturation pressure (between the saturated-liquid volume $v_f$ and
the saturated-vapour volume $v_g$), then falls through the
superheated-vapour region. The critical isotherm ($T=T_c$, orange) has a
horizontal inflexion at the critical point. The supercritical isotherm
($T>T_c$, grey) is monotonic and never touches the dome. Shapes are
schematic.}
\label{fig:vol04ch02:pv-isotherms-dome}
\end{figure}
