% Figure: saturation (vapour) pressure of propane against temperature over
% the residential storage range. The curve is the propane saturation line;
% the marked points are the design temperatures used in the LPG-tank case
% study (winter -30 C, design cold -10 C, ambient 25 C, hot-day 50 C,
% fire-relief set point ~ 65 C). Absolute pressures in kPa.
% Coordinates are handbook propane saturation values; the curve is drawn
% smooth through them, the rapid rise being the Clausius-Clapeyron signature.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={temperature $T$ ($^{\circ}$C)},
  ylabel={saturation pressure $p_{\text{sat}}$ (kPa, absolute)},
  xmin=-40, xmax=70, ymin=0, ymax=2400,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
]
% handbook propane saturation pressures (T in C, p_sat absolute kPa):
% -40 -> 110, -30 -> 168, -20 -> 245, -10 -> 345, 0 -> 474,
% 10 -> 637, 25 -> 952, 40 -> 1370, 50 -> 1713, 65 -> 2340
\addplot[engblue, very thick, smooth] coordinates {
  (-40,110) (-30,168) (-20,245) (-10,345) (0,474)
  (10,637) (25,952) (40,1370) (50,1713) (65,2340)
};
\addlegendentry{propane saturation curve}
\addplot[only marks, mark=*, engred, mark size=2.2pt] coordinates {
  (-10,345) (25,952) (50,1713)
};
\addlegendentry{case-study design points}
% horizontal reference at the tank pressure rating ~ 2.5 MPa is off-scale;
% mark instead the relief-valve set band near 2.07 MPa (UL/handbook)
\addplot[engorange, very thick, dashed, domain=-40:70] {2070};
\node[engorange, font=\scriptsize, anchor=south east] at (axis cs:68,2080)
  {relief set $\approx 2.07$ MPa};
\node[engblack, font=\scriptsize, anchor=north west] at (axis cs:-8,360)
  {$-10\,^{\circ}$C: 345 kPa};
\node[engblack, font=\scriptsize, anchor=east] at (axis cs:24,1000)
  {25\,$^{\circ}$C: 952 kPa};
\node[engblack, font=\scriptsize, anchor=east] at (axis cs:49,1760)
  {50\,$^{\circ}$C: 1713 kPa};
\end{axis}
\end{tikzpicture}
\caption[Propane saturation pressure against temperature]{Saturation
pressure of propane over the residential storage range. The pressure a
closed tank holds is set by the saturation curve at the liquid
temperature, so the tank's pressure rating follows the highest
foreseeable temperature rather than any filling choice. The three marked
points are the design temperatures carried through the LPG-tank case
study: the cold design point at $-10\,^{\circ}$C ($345$ kPa), the
mild-ambient point at $25\,^{\circ}$C ($952$ kPa), and the hot-day point
at $50\,^{\circ}$C ($1713$ kPa). The dashed line is a representative
relief-valve set pressure near $2.07$ MPa; the curve reaches it only at a
fire-driven wall temperature well above any weather, which is why relief
sizing is governed by the fire case rather than by ambient warming.}
\label{fig:vol04ch02:propane-vapour-pressure}
\end{figure}
