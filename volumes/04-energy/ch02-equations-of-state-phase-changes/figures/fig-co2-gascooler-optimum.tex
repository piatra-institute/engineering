% Figure: COP of a CO2 transcritical cycle as a function of gas-cooler
% (high-side) pressure, at a fixed gas-cooler outlet temperature. The curve
% has a flat-topped maximum: too low a high-side pressure leaves the
% supercritical fluid warm and enthalpy-rich at the gas-cooler exit (poor
% cooling per unit work), too high a pressure spends extra compressor work
% for diminishing enthalpy reduction. The optimum near 9-10 MPa for a 35 C
% gas-cooler outlet is the working result. Values are illustrative of the
% trade and are consistent with the worked example.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={gas-cooler (high-side) pressure $p_{\text{gc}}$ (MPa)},
  ylabel={coefficient of performance, COP},
  xmin=7.5, xmax=13, ymin=1.5, ymax=3.2,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=south east,
  legend style={font=\scriptsize},
]
% COP(p_gc) at a 35 C gas-cooler outlet: rises steeply, peaks near 9.5 MPa,
% then declines slowly. Points are illustrative of the documented shape.
\addplot[engblue, very thick, smooth, mark=*, mark size=1.6pt] coordinates {
  (8.0,1.95) (8.5,2.45) (9.0,2.78) (9.5,2.90) (10.0,2.86)
  (10.5,2.74) (11.0,2.60) (12.0,2.34) (13.0,2.10)
};
\addlegendentry{$T_{\text{gc,out}}=35\,^{\circ}\mathrm{C}$}
% mark the optimum
\addplot[only marks, mark=*, engred, mark size=2.4pt] coordinates {(9.5,2.90)};
\node[engred, font=\scriptsize, anchor=south] at (axis cs:9.5,2.95) {optimum};
% vertical dropline at the optimum
\addplot[engred, dashed] coordinates {(9.5,1.5) (9.5,2.90)};
\end{axis}
\end{tikzpicture}
\caption[Gas-cooler pressure optimum for a CO\textsubscript{2} transcritical
cycle]{Coefficient of performance of a carbon-dioxide transcritical cycle
against the gas-cooler (high-side) pressure, at a fixed gas-cooler outlet
temperature of $35\,\degC$. The curve has a single broad maximum. Below the
optimum the supercritical fluid leaves the gas cooler warm and
enthalpy-rich, so the throttle delivers a small refrigerating effect per
unit of compressor work and the COP falls steeply. Above the optimum the
gas-cooler exit enthalpy keeps dropping, which helps the cooling, but the
extra compressor work needed to reach the higher pressure outweighs the
gain, and the COP declines slowly. The maximum near $9.5\,\text{MPa}$ for a
$35\,\degC$ outlet is the discharge-pressure trade unique to a transcritical
cycle: the high-side pressure is a control variable a CO\textsubscript{2}
unit must actively manage, because no condensing saturation curve fixes it.}
\label{fig:vol04ch02:co2-gascooler-optimum}
\end{figure}
