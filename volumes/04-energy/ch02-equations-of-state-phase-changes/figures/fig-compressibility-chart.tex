% Figure: generalised compressibility chart, Z against reduced pressure
% for several reduced-temperature isotherms. Curves are schematic but
% qualitatively correct: low-T_r isotherms dip well below Z=1 at moderate
% p_r and recover; high-T_r isotherms stay near 1. Hand-placed control
% points based on the Nelson-Obert chart shape.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={reduced pressure $p_r=p/p_c$},
  ylabel={compressibility factor $Z=pv/(R_sT)$},

  axis lines=left,

\addplot[enggray, dashed, thick] coordinates {(0,1) (7,1)};
\addlegendentry{ideal gas $Z=1$}

\addplot[engred, very thick, smooth] coordinates {

};

\caption[Generalised compressibility chart]{The generalised
compressibility chart: the compressibility factor $Z=pv/(R_sT)$ plotted
against reduced pressure $p_r=p/p_c$ for several reduced-temperature
isotherms $T_r=T/T_c$. At low reduced pressure ($p_r<0.1$) or high
reduced temperature ($T_r>2$) the curves hug $Z=1$ and the ideal-gas
approximation is safe. Near the critical region ($T_r\approx 1$, $p_r$ of
order unity) $Z$ dips toward $0.6$ as attractive forces dominate, then
rises above unity at high pressure as the excluded volume of the
molecules takes over. The chart is read the same way for most non-polar
substances, the working statement of the principle of corresponding
states. Curve positions are schematic, of the Nelson-Obert form.}
\label{fig:vol04ch02:compressibility-chart}
\end{figure}
