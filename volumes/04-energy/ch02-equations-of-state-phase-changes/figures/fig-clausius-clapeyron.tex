% Figure: Clausius-Clapeyron plot of ln(p_sat) against 1/T for water,
% which is nearly a straight line of slope -h_fg/R_s. Data points are
% steam-table saturation values; the fitted line uses h_fg = 2257 kJ/kg
% (the 100 C latent heat) anchored at the normal boiling point.
% Computed coordinates: x = 1000/T (1/kK), y = ln(p in kPa).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={$1000/T$ (1/K)},
  ylabel={$\ln(p_{\text{sat}}/\text{kPa})$},
  xmin=1.6, xmax=3.8, ymin=-1, ymax=10,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north east,
  legend style={font=\scriptsize},
]
% steam-table saturation points (T in K, p_sat in kPa):
%  273.15 -> 0.611,  323.15 -> 12.35, 373.15 -> 101.3,
%  423.15 -> 476.2, 473.15 -> 1555,  523.15 -> 3976, 573.15 -> 8588
% mapped to x=1000/T, y=ln(p)
\addplot[only marks, mark=*, engblue, mark size=2.2pt] coordinates {
  (3.661,-0.493)
  (3.095,2.514)
  (2.680,4.618)
  (2.363,6.166)
  (2.113,7.349)
  (1.912,8.288)
  (1.745,9.058)
};
\addlegendentry{steam-table data}
% Clausius-Clapeyron straight-line fit. Two-point line through
% the boiling point and a near-critical point gives slope ~ -5.05 per (1/kK)
% intercept ~ 18.15 in y when x is in 1/kK units. We draw across the range.
\addplot[engred, very thick, domain=1.7:3.7] {-5.05*x + 23.13};
\addlegendentry{straight-line fit}
\end{axis}
\end{tikzpicture}
\caption[Clausius-Clapeyron plot for water]{Saturation pressure of water
plotted as $\ln(p_{\text{sat}})$ against $1/T$. The steam-table data
(blue points) fall close to a straight line, which is the integrated
Clausius-Clapeyron prediction $\ln p \approx -h_{fg}/(R_s T) + C$ under
the approximations that the vapour is ideal and that $v_g\gg v_f$. The
slope of the line is $-h_{fg}/R_s$; reading it off gives the latent heat
directly. The mild curvature, visible at the high-temperature (low
$1/T$) end, is the signature of the temperature dependence of $h_{fg}$
and of the vapour's departure from ideal-gas behaviour as the critical
point is approached.}
\label{fig:vol04ch02:clausius-clapeyron}
\end{figure}
