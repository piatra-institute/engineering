% Figure: the dew-point construction for moist air. The curve is the
% saturation pressure of water vapour against temperature (the same
% Clausius-Clapeyron curve seen earlier, here on linear axes over the
% comfort range). A horizontal line at the actual water-vapour partial
% pressure cuts the saturation curve at the dew-point temperature: cooling
% the air at constant pressure moves left along the horizontal line until
% it reaches saturation, where the first droplet forms.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.4cm,
  xlabel={temperature $T$ ($^{\circ}$C)},
  ylabel={water-vapour pressure (kPa)},
  xmin=0, xmax=35, ymin=0, ymax=5.6,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=true,
]
% saturation pressure of water (kPa) vs temperature (C), anchored to
% steam-table values: 0.611 at 0, 1.228 at 10, 1.706 at 15, 2.339 at 20,
% 3.169 at 25, 4.246 at 30, 5.628 at 35.
\addplot[engblue, very thick, smooth] coordinates {
  (0,0.611) (5,0.872) (10,1.228) (15,1.706) (20,2.339)
  (25,3.169) (30,4.246) (35,5.628)
};
\node[engblue, font=\scriptsize, anchor=south east] at (axis cs:33,5.2)
  {$p_{\text{sat}}(T)$};
% actual partial pressure of water vapour: 1.706 kPa (the 20 C air at 73%
% RH example, p_v = 0.73 * 2.339 = 1.706). Horizontal line.
\addplot[engred, very thick, dashed] coordinates {(0,1.706) (20,1.706)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:2.4,1.95)
  {$p_v = 1.71$ kPa};
% the state point at 20 C
\addplot[only marks, mark=*, engblack, mark size=1.6pt] coordinates {(20,1.706)};
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:20,1.78)
  {air at $20\,^{\circ}$C};
% dew point where the horizontal line meets the saturation curve: 15 C
\addplot[only marks, mark=*, engorange, mark size=1.8pt] coordinates {(15,1.706)};
\addplot[enggray, thick, dotted] coordinates {(15,0) (15,1.706)};
\node[engorange, font=\scriptsize, anchor=north] at (axis cs:15,1.55)
  {dew point $15\,^{\circ}$C};
% cooling arrow along the constant-partial-pressure line
\draw[-{Stealth}, engblack, thick] (axis cs:19,1.706) -- (axis cs:15.6,1.706);
\node[engblack, font=\scriptsize, anchor=south] at (axis cs:17.3,1.706)
  {cool};
\end{axis}
\end{tikzpicture}
\caption[The dew-point construction for moist air]{The dew-point
construction. The rising curve is the saturation pressure of water vapour
against temperature, the engineering range of the same Clausius-Clapeyron
relation drawn earlier. Air at $20\,\degC$ holding water vapour at a
partial pressure of $1.71\,\text{kPa}$ (a relative humidity of
$1.71/2.34 = 73\,\%$) sits on the dashed horizontal line, below the
saturation curve. Cooling the air at constant total pressure holds the
water-vapour partial pressure fixed and moves the state leftward along the
dashed line until it meets the saturation curve, here at the dew-point
temperature of $15\,\degC$. At that point the air is saturated and any
further cooling condenses water: the first droplet on a cold-drink glass,
the fog on a window, the cloud base above a rising air parcel. The dew
point is read directly off the chart as the temperature at which the
actual vapour pressure equals the saturation pressure.}
\label{fig:vol04ch02:dewpoint-construction}
\end{figure}
