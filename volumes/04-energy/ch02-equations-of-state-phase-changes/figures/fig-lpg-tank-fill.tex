% Figure: liquid-fill fraction of a fixed-mass LPG charge against
% temperature, showing why the 80 percent fill limit exists. A tank
% filled to 80 percent liquid by volume at the fill temperature has its
% liquid expand as it warms; the curve plots the liquid volume fraction
% of the SAME mass charge against temperature, using the propane liquid
% density. The tank goes liquid-full (no ullage) at the temperature where
% the curve reaches 100 percent, beyond which the tank is hydraulically
% solid and any further warming is resisted by the liquid bulk modulus.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={temperature $T$ ($^{\circ}$C)},
  ylabel={liquid volume fraction of tank (\%)},
  xmin=0, xmax=70, ymin=75, ymax=105,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
]
% Charge fixed at 80 percent fill at T_fill = 15 C. Liquid density of
% propane (kg/m^3): 15 -> 507, 25 -> 494, 40 -> 472, 50 -> 457, 58 -> 444.
% Liquid volume fraction = 0.80 * rho(15)/rho(T).
% 15 -> 80.0, 25 -> 82.1, 40 -> 85.9, 50 -> 88.8, 58 -> 91.4
% extrapolated to the liquid-full point near 96 C is off-scale; within
% the residential band the fraction climbs but stays below 100.
\addplot[engblue, very thick, smooth] coordinates {
  (0,77.8) (15,80.0) (25,82.1) (40,85.9) (50,88.8) (58,91.4) (70,95.6)
};
\addlegendentry{liquid fraction (80\% fill at 15\,$^{\circ}$C)}
% over-filled charge: 95 percent at 15 C reaches liquid-full much sooner
\addplot[engred, very thick, smooth, dashed] coordinates {
  (0,92.4) (15,95.0) (25,97.5) (33,100.0)
};
\addlegendentry{over-filled (95\% at 15\,$^{\circ}$C)}
\addplot[engblack, very thick, dotted, domain=0:70] {100};
\node[engblack, font=\scriptsize, anchor=south east] at (axis cs:69,100.3)
  {liquid-full (no ullage)};
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:25,98.0)
  {hydraulically solid at $\approx 33\,^{\circ}$C};
\end{axis}
\end{tikzpicture}
\caption[Liquid-fill fraction of an LPG tank against temperature]{Liquid
volume fraction of a fixed propane charge as the tank warms, the basis of
the $80\,\%$ fill limit. A correctly filled tank (solid curve, $80\,\%$
liquid at the $15\,^{\circ}$C fill temperature) keeps a vapour ullage
across the whole residential temperature band: the liquid expands as it
warms, but reaches only about $96\,\%$ of the tank volume at
$70\,^{\circ}$C. An over-filled tank (dashed curve, $95\,\%$ liquid at
fill) becomes liquid-full near $33\,^{\circ}$C, after which there is no
vapour space to absorb further thermal expansion and the pressure rises
along the steep liquid-compressibility slope rather than the gentle
saturation curve. The ullage is the engineering margin that keeps tank
pressure on the saturation curve of figure
\ref{fig:vol04ch02:propane-vapour-pressure} rather than on the bulk
modulus of the liquid.}
\label{fig:vol04ch02:lpg-tank-fill}
\end{figure}
