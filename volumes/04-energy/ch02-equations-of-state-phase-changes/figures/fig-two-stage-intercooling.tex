% Figure: two-stage compression with intercooling on the p-v plane. A single
% adiabatic compression from p1 to p3 follows the steep upper adiabat and its
% area (work) is large; splitting the compression at an intermediate pressure
% p2 and cooling the gas back to the inlet temperature at constant pressure
% before the second stage lets the second stage start from a smaller volume,
% so the two-stage path hugs closer to the isotherm and the shaded area saved
% is the intercooling work saving.

]

\addplot[engred, very thick, smooth] coordinates {
  (1.00,1.00) (0.90,1.17) (0.80,1.40) (0.70,1.72) (0.60,2.16) (0.50,2.83)
  (0.40,3.90) (0.35,4.67) (0.30,5.75) (0.25,7.36) (0.215,8.00)
};
\addlegendentry{single-stage adiabat}

};

\caption[Two-stage compression with intercooling on the $p$-$v$ plane]{A gas
compressed in one adiabatic stage follows the steep upper path (red) whose area
to the pressure axis is the compression work. Splitting the compression at an
intermediate pressure $p_2$ and cooling the gas at constant pressure back to
its inlet temperature before the second stage (blue) restarts the second stage
from a smaller specific volume, so the two-stage path hugs closer to the
reference isotherm (grey dashed), the lowest-work path a perfectly cooled
compressor could follow. The area between the single adiabat and the two-stage
path is the work the intercooler saves; adding stages drives the path toward
the isotherm at the cost of more heat exchangers, which is why large air and
refrigeration compressors are multi-stage with intercooling between stages.}
\label{fig:vol04ch02:two-stage-intercooling}
\end{figure}
