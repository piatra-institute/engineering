% Figure: dimensionless enthalpy departure (h^ideal - h)/(R T_c) plotted
% against reduced pressure for two reduced temperatures, the generalised
% enthalpy-departure chart used to correct an ideal-gas enthalpy to the real
% value. The departure grows with pressure and is largest near the critical
% temperature, where the attractive interaction is strongest. Curves are
% illustrative of the documented shape from the corresponding-states
% (Lee-Kesler) departure correlation, with the residual plotted as a
% positive quantity (enthalpy deficit relative to the ideal gas).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={reduced pressure $p_r = p/p_c$},
  ylabel={$(h^{\text{ideal}} - h)/(R\,T_c)$},
  xmin=0, xmax=4, ymin=0, ymax=4.5,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
]
% T_r = 1.0 : steepest rise; large enthalpy deficit near critical
\addplot[engred, very thick, smooth] coordinates {
  (0,0) (0.25,0.45) (0.5,0.95) (1.0,1.95) (1.5,2.75)
  (2.0,3.30) (2.5,3.70) (3.0,3.95) (3.5,4.10) (4.0,4.20)
};
\addlegendentry{$T_r = 1.0$}
% T_r = 1.5 : milder
\addplot[engblue, very thick, smooth] coordinates {
  (0,0) (0.5,0.30) (1.0,0.62) (1.5,0.95) (2.0,1.28)
  (2.5,1.58) (3.0,1.85) (3.5,2.08) (4.0,2.28)
};
\addlegendentry{$T_r = 1.5$}
% T_r = 2.0 : smallest departure
\addplot[engamber, very thick, smooth] coordinates {
  (0,0) (0.5,0.16) (1.0,0.33) (1.5,0.50) (2.0,0.67)
  (2.5,0.83) (3.0,0.98) (3.5,1.12) (4.0,1.24)
};
\addlegendentry{$T_r = 2.0$}
\end{axis}
\end{tikzpicture}
\caption[Generalised enthalpy-departure chart]{Dimensionless enthalpy
departure $(h^{\text{ideal}} - h)/(R\,T_c)$ against reduced pressure for
three reduced temperatures. The departure is the amount by which a real
gas falls short of the ideal-gas enthalpy at the same temperature and
pressure; it is read off this chart and subtracted from the easily
computed ideal-gas enthalpy to recover the real value. The departure
grows with pressure as the molecules crowd, and it is largest near the
critical temperature (red), where the attractive interaction is deepest,
exactly where the van der Waals worked example placed a carbon-dioxide
deficit of $87\,\text{kJ}/\text{kg}$. At high reduced temperature (amber)
the departure stays small and the ideal-gas enthalpy is a usable estimate
even at elevated pressure.}
\label{fig:vol04ch02:departure-functions}
\end{figure}
