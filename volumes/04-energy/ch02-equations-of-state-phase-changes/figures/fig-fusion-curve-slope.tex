% Figure: schematic p-T diagram comparing the fusion (melting) curve of a
% normal substance (positive slope, solid denser than liquid) with the
% anomalous fusion curve of water (negative slope, solid less dense than
% liquid). The sublimation and vapourisation curves are shown for context.
% Axes are schematic (no numbers on the pressure axis); the point of the
% figure is the sign of the fusion-curve slope, which the Clapeyron
% relation ties to the sign of the volume change on melting.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm, height=8cm,
  xlabel={temperature $T$},
  ylabel={pressure $p$},
  xmin=0, xmax=10, ymin=0, ymax=10,
  axis lines=left,
  xtick=\empty, ytick=\empty,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=north west,
  legend style={font=\scriptsize},
  clip=false,
]
% Triple point common to both, placed at (4,3)
% Sublimation curve (solid-vapour): rises from origin to triple point
\addplot[enggray, thick, domain=0.6:4, samples=40] {3*exp(0.55*(x-4))};
\addlegendentry{sublimation}
% Vapourisation curve (liquid-vapour): from triple point up to a critical point
\addplot[engblue, thick, domain=4:8.5, samples=40] {3*exp(0.32*(x-4))};
\addlegendentry{vapourisation}
% Fusion curve, normal substance: steep positive slope from triple point
\addplot[engred, very thick, domain=4:4.9, samples=20] {3 + 9*(x-4)};
\addlegendentry{fusion (normal)}
% Fusion curve, water: steep negative slope from triple point
\addplot[engorange, very thick, dashed, domain=3.1:4, samples=20] {3 + 9*(x-4)};
\addlegendentry{fusion (water)}
% Triple point marker
\addplot[only marks, mark=*, engblack, mark size=2pt] coordinates {(4,3)};
\node[anchor=west, font=\scriptsize] at (axis cs:4.15,2.6) {triple point};
% Critical point marker at end of vapourisation curve
\addplot[only marks, mark=*, engblack, mark size=2pt] coordinates {(8.5,7.86)};
\node[anchor=east, font=\scriptsize] at (axis cs:8.35,8.2) {critical point};
% Region labels
\node[font=\small\itshape] at (axis cs:2.2,6.5) {solid};
\node[font=\small\itshape] at (axis cs:5.2,7.5) {liquid};
\node[font=\small\itshape] at (axis cs:7.5,2.2) {vapour};
\end{axis}
\end{tikzpicture}
\caption[Fusion-curve slope and the water anomaly]{The three coexistence
curves of a pure substance meeting at the triple point. For a normal
substance (solid red) the fusion curve leans forward with a steep positive
slope, because the solid is denser than the liquid and $\Delta v > 0$ on
melting drives $dp/dT = h_{sf}/(T\,\Delta v) > 0$. Water (dashed orange) is
the exception: ice is less dense than liquid water, so $\Delta v < 0$ on
melting and the fusion curve leans backward with a negative slope, which is
why raising the pressure lowers the melting point of ice. The fusion curves
are drawn far steeper than the vapourisation curve because the volume change
on melting is a thousand times smaller than the volume change on boiling, so
the Clapeyron slope is a thousand times larger.}
\label{fig:vol04ch02:fusion-curve-slope}
\end{figure}
