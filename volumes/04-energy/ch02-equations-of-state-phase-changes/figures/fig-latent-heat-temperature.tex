% Figure: latent heat of vapourisation of water against temperature, falling
% from 2501 kJ/kg at 0 C toward zero at the critical point (374 C). The curve
% is a Watson-type fit to the steam-table h_fg values; it shows the vanishing
% of the latent heat at the critical point where the two limbs of the dome
% meet. A few marked table points anchor the curve.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={temperature $T$ (\si{\degreeCelsius})},
  ylabel={latent heat $h_{fg}$ (\si{\kilo\joule\per\kilogram})},
  xmin=0, xmax=374, ymin=0, ymax=2700,
  axis lines=left,
  xtick={0,100,200,300,374},
  xticklabels={$0$,$100$,$200$,$300$,$T_c$},
  ytick={0,500,1000,1500,2000,2500},
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=true,
]
% Watson-type curve h_fg = 2501 * ((374.15 - T)/374.15)^0.38, sampled.
\addplot[engblue, very thick, smooth] coordinates {
  (0,2501) (20,2454) (40,2406) (60,2358) (80,2308) (100,2257)

\addplot[enggray, dashed, thick] coordinates {(374,0) (374,2700)};

  {$h_{fg}\to 0$ at $T_c$};

\end{tikzpicture}
\caption[Latent heat of vapourisation of water against temperature]{The latent
heat of vapourisation of water falls with temperature, from
$2501\,\si{\kilo\joule\per\kilogram}$ at the ice point to zero at the critical
temperature $T_c=374\,\si{\degreeCelsius}$, where the saturated-liquid and
saturated-vapour states merge and no heat is needed to cross between them. The
marked points are steam-table values; the curve is a Watson-type fit. The steep
drop past $300\,\si{\degreeCelsius}$ is the thermodynamic reason a
high-temperature boiler extracts progressively less latent heat per kilogram as
it approaches the critical pressure, and the reason the vapour dome narrows
toward its peak, since the width of the dome on a $T$-$s$ diagram is
$s_{fg}=h_{fg}/T$.}
\label{fig:vol04ch02:latent-heat-temperature}
\end{figure}
