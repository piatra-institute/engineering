% Figure: fugacity coefficient phi = f/p plotted against reduced pressure
% for a family of reduced temperatures, the generalised fugacity-coefficient
% chart that mirrors the compressibility chart. At low reduced pressure phi
% approaches 1 (the ideal-gas limit f -> p). Below the critical temperature
% phi dips well below 1 because the attractive forces lower the escaping
% tendency; at high reduced temperature phi can rise above 1 as the
% excluded-volume repulsion dominates. Curves are illustrative of the
% documented shape from the Lee-Kesler / corresponding-states correlation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={reduced pressure $p_r = p/p_c$},
  ylabel={fugacity coefficient $\phi = f/p$},
  xmin=0, xmax=4, ymin=0.3, ymax=1.15,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=south west,
  legend style={font=\scriptsize},
]
% T_r = 1.0 : dips deepest, attraction dominates near critical
\addplot[engred, very thick, smooth] coordinates {
  (0,1.00) (0.25,0.93) (0.5,0.86) (1.0,0.72) (1.5,0.64)
  (2.0,0.59) (2.5,0.57) (3.0,0.57) (3.5,0.59) (4.0,0.62)
};
\addlegendentry{$T_r = 1.0$}
% T_r = 1.5 : shallower dip
\addplot[engblue, very thick, smooth] coordinates {
  (0,1.00) (0.5,0.95) (1.0,0.90) (1.5,0.87) (2.0,0.85)
  (2.5,0.85) (3.0,0.86) (3.5,0.88) (4.0,0.91)
};
\addlegendentry{$T_r = 1.5$}
% T_r = 2.0 : nearly flat, rises above 1 at high p_r
\addplot[engamber, very thick, smooth] coordinates {
  (0,1.00) (0.5,0.98) (1.0,0.97) (1.5,0.97) (2.0,0.98)
  (2.5,0.99) (3.0,1.01) (3.5,1.04) (4.0,1.07)
};
\addlegendentry{$T_r = 2.0$}
% ideal-gas reference line phi = 1
\addplot[enggray, dashed, thick, domain=0:4] {1.0};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:4,1.0) {$\phi = 1$ (ideal)};
\end{axis}
\end{tikzpicture}
\caption[Generalised fugacity-coefficient chart]{Fugacity coefficient
$\phi = f/p$ against reduced pressure for three reduced temperatures, the
companion of the compressibility chart. As $p_r \to 0$ every curve returns
to $\phi = 1$, the ideal-gas limit in which the fugacity equals the
pressure. The $T_r = 1.0$ curve (red) dips deepest, because near the
critical temperature the attractive forces lower the escaping tendency of
the gas far below its ideal value; the $T_r = 2.0$ curve (amber) stays near
unity and rises above it at high pressure, where the excluded-volume
repulsion makes the gas harder to compress than ideal and raises its
escaping tendency. The fugacity coefficient is the multiplicative
correction that carries the ideal-gas chemical-potential expression to a
real gas, and it is the quantity a phase-equilibrium calculation equates
between coexisting phases.}
\label{fig:vol04ch02:fugacity-coefficient}
\end{figure}
