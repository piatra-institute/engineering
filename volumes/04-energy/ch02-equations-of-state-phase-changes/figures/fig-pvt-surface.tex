% Figure: schematic of the p-v-T surface of a pure substance, drawn in a
% simple 3D axis. The full surface is not rendered; we show the three
% axes and the projection idea, with the single-phase and two-phase
% regions sketched as patches. Kept deliberately simple to compile
% reliably: a 3D box with labelled faces and a sketched ridge.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % simple isometric axes
  \coordinate (O) at (0,0);
  \draw[->] (O) -- (6.2,0) node[right, font=\small] {$v$};
  \draw[->] (O) -- (0,5.2) node[above, font=\small] {$p$};
  \draw[->] (O) -- (-2.6,-1.6) node[below left, font=\small] {$T$};
  % a sketched surface ridge: the saturation dome projected into the v-p plane
  % drawn as a curved patch boundary
  \draw[engblue, very thick] (1.0,0.6) .. controls (1.4,2.6) and (2.2,3.6) .. (3.0,3.8)
        .. controls (3.9,3.6) and (4.8,2.4) .. (5.4,0.8);
  % depth offsets for a back edge (toward higher T)
  \draw[engblue, thick, dashed] (0.2,0.0) .. controls (0.6,1.8) and (1.4,2.8) .. (2.2,3.0)
        .. controls (3.1,2.8) and (4.0,1.7) .. (4.6,0.2);
  % connect front and back domes to suggest the two-phase ridge surface
  \draw[enggray] (1.0,0.6) -- (0.2,0.0);
  \draw[enggray] (3.0,3.8) -- (2.2,3.0);
  \draw[enggray] (5.4,0.8) -- (4.6,0.2);
  % region labels
  \node[font=\small] at (0.9,4.4) {liquid};
  \node[font=\small] at (4.7,4.2) {vapour};
  \node[engblue, font=\scriptsize, align=center] at (3.0,2.0) {two-phase\\ridge};
  \node[font=\scriptsize, align=center, anchor=west] at (5.6,4.6) {supercritical};
\end{tikzpicture}
\caption[The p-v-T surface of a pure substance]{Schematic of the
$p$-$v$-$T$ surface of a pure substance. Every equilibrium state of a
simple compressible substance is one point on this single surface; the
flat working diagrams of the chapter are its projections. Projecting
along the temperature axis gives the $p$-$v$ diagram with its vapour
dome; projecting along the volume axis gives the $p$-$T$ phase diagram
with its coexistence curves. The ridge running across the surface (blue)
is the two-phase region, where liquid and vapour coexist and the surface
is ruled by horizontal tie lines. The single equation of state, or the
single tabulated surface, is what lets two independent properties fix all
the rest. The drawing is schematic.}
\label{fig:vol04ch02:pvt-surface}
\end{figure}
