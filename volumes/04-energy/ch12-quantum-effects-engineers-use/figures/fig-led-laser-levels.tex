% Figure: energy-level / recombination diagram contrasting spontaneous
% emission in an LED with stimulated emission and feedback in a laser diode.
% Two side-by-side panels using plain TikZ.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
% ==== LED panel: spontaneous emission ====
\begin{scope}[xshift=0cm]
  \draw[engblue, very thick] (0,3.2) -- (3,3.2);   % conduction band
  \draw[engblue, very thick] (0,0.6) -- (3,0.6);   % valence band
  \node[anchor=east] at (-0.05,3.2) {$E_c$};
  \node[anchor=east] at (-0.05,0.6) {$E_v$};
  % electron drops, emits photon
  \filldraw[engred] (1.0,3.2) circle (2.2pt);
  \draw[->, engred, thick] (1.0,3.1) -- (1.0,0.75);
  % photon emitted in random direction
  \draw[->, engorange, thick] (1.0,1.9) -- (2.4,2.7);
  \node[engorange, font=\tiny, anchor=south west] at (2.0,2.4) {photon};
  \node[anchor=south] at (1.5,3.3) {$\hbar\omega \approx E_g$};
  \node[align=center] at (1.5,-0.4) {LED:\\spontaneous emission};
\end{scope}

% ==== Laser panel: stimulated emission + cavity ====
\begin{scope}[xshift=6cm]
  \draw[engblue, very thick] (0,3.2) -- (3,3.2);
  \draw[engblue, very thick] (0,0.6) -- (3,0.6);
  \node[anchor=east] at (-0.05,3.2) {$E_c$};
  \node[anchor=east] at (-0.05,0.6) {$E_v$};
  % cavity mirrors (facets)
  \draw[engblack, very thick] (-0.25,0.2) -- (-0.25,3.6);
  \draw[engblack, very thick] (3.25,0.2) -- (3.25,3.6);
  \node[anchor=south, font=\tiny] at (-0.25,3.6) {facet};
  \node[anchor=south, font=\tiny] at (3.25,3.6) {facet};
  % incoming photon stimulates a second identical photon
  \filldraw[engred] (1.4,3.2) circle (2.2pt);
  \draw[->, engred, thick] (1.4,3.1) -- (1.4,0.75);
  \draw[->, engorange, thick] (0.2,2.05) -- (1.2,2.05);
  \draw[->, engorange, thick] (1.6,2.05) -- (2.8,2.05);
  \draw[->, engorange, thick] (0.2,1.75) -- (1.2,1.75);
  \draw[->, engorange, thick] (1.6,1.75) -- (2.8,1.75);
  \node[engorange, font=\tiny, anchor=north] at (2.2,1.7) {two in phase};
  \node[align=center] at (1.5,-0.4) {laser diode:\\stimulated emission};
\end{scope}
\end{tikzpicture}
\caption[LED versus laser diode emission]{Recombination in a forward-biased
junction. In the LED (left) an injected electron drops from the conduction
band to a hole in the valence band and emits a photon of energy near the gap,
in a random direction and with random phase: spontaneous emission. In the
laser diode (right) a photon already in the cavity stimulates the emission of
a second photon identical in direction, frequency, and phase; cleaved-facet
mirrors feed the light back through the gain region, and above a threshold
current the output becomes coherent and monochromatic. Same junction, same
band gap, different photon statistics.}
\label{fig:vol04ch12:led-laser-levels}
\end{figure}
