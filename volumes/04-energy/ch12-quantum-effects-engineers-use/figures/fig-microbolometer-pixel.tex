% Figure: cross-section schematic of an uncooled microbolometer pixel. An
% infrared-absorbing membrane carrying a temperature-sensitive resistor is
% suspended on thin legs above a silicon readout circuit, with a vacuum gap
% and a reflector beneath tuned to a quarter wavelength. Pure schematic, no
% plotted data, so no computed coordinates beyond fixed positions.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize, scale=1.0]
% incoming infrared radiation arrows
\foreach \x in {1.5,3.0,4.5} {
  \draw[engred, ->, thick] (\x,5.0) -- (\x,4.05);
}
\node[engred, anchor=west, font=\tiny] at (5.0,4.7) {incident IR ($8$--$14\,\si{\micro\meter}$)};

% suspended absorber membrane
\draw[engblue, very thick, fill=engblue!12]
  (1.0,3.8) rectangle (5.0,4.0);
\node[engblue, anchor=south, font=\tiny] at (3.0,4.02)
  {absorber + thermistor ($R(T)$)};

% support legs (thin, high thermal resistance)
\draw[engblack, thick] (1.2,3.8) -- (1.2,3.0);
\draw[engblack, thick] (4.8,3.8) -- (4.8,3.0);
\node[enggray, anchor=east, font=\tiny] at (1.15,3.4) {leg};
\node[enggray, anchor=west, font=\tiny] at (4.85,3.4) {leg};

% vacuum gap label
\draw[<->, enggray] (3.0,3.78) -- (3.0,3.02);
\node[enggray, anchor=west, font=\tiny] at (3.05,3.4)
  {vacuum gap $\approx \lambda/4$};

% reflector and readout substrate
\draw[engorange, very thick] (0.6,3.0) -- (5.4,3.0);
\node[engorange, anchor=south east, font=\tiny] at (5.4,3.02) {reflector};
\draw[engblack, thick, fill=enggray!12]
  (0.6,2.2) rectangle (5.4,3.0);
\node[anchor=center, font=\tiny] at (3.0,2.6)
  {silicon readout circuit (ROIC)};

% reflected wave returning to membrane (resonant cavity)
\draw[engred, ->, dashed] (2.0,3.05) -- (2.0,3.78);
\end{tikzpicture}
\caption[Uncooled microbolometer pixel]{Cross-section of one pixel of an
uncooled microbolometer array. Long-wave infrared from a scene at room
temperature is absorbed by a thin membrane whose temperature rises by a small
fraction of a kelvin; the membrane carries a thermistor whose resistance
$R(T)$ the silicon readout circuit beneath measures. The membrane is suspended
on two thin legs that give it a high thermal resistance to the substrate, so a
small absorbed power produces a usable temperature rise, and it sits a quarter
wavelength above a reflector so the cavity resonance returns unabsorbed
radiation for a second pass. The device needs no cryogenic cooling because it
senses heat, not individual band-gap photons; its limit is the thermal
fluctuation of the tiny membrane, not the photon shot noise of a cooled
photodiode.}
\label{fig:vol04ch12:microbolometer-pixel}
\end{figure}
