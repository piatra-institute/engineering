% Figure: the Planck spectral energy density against the classical
% Rayleigh-Jeans result, plotted in a dimensionless variable x = h nu / kT.
% Rayleigh-Jeans rises as x^2 without bound (the ultraviolet catastrophe);
% Planck follows it at small x then turns over and falls. Curves from
% explicit sample points so no in-figure division by (exp(x)-1) is performed.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10cm, height=6.2cm,
    xlabel={dimensionless frequency $x = h\nu/k_B T$},
    ylabel={spectral density (arb.\ units)},
    xmin=0, xmax=8,
    ymin=0, ymax=6,
    xtick={0,2,4,6,8},
    ytick={0,1,2,3,4,5,6},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},

    legend style={font=\tiny, at={(0.97,0.97)}, anchor=north east,
                  draw=enggray!40},
    every axis plot/.append style={very thick},
]
% Rayleigh-Jeans: u_RJ proportional to x^2 (mode count x^2 times kT per mode)
\addplot[engred, domain=0:2.45, samples=40] {\x*\x};
\addlegendentry{Rayleigh-Jeans $\propto x^2$}
% Planck: u proportional to x^3 / (exp(x)-1); sampled values (normalised so
% small-x matches the RJ x^2 line: x^3/(e^x-1) -> x^2 as x->0)
\addplot[engblue] coordinates {

  (6.5,0.733) (7.0,0.549) (8.0,0.298)
};
\addlegendentry{Planck $\propto x^3/(e^x-1)$}
% mark the Planck peak at x = 2.821
\addplot[engblack, only marks, mark=*, mark size=1.4pt]
  coordinates {(2.821,2.493)};
\node[engblack, anchor=west, font=\tiny] at (axis cs:2.95,2.62)
  {peak $x = 2.82$ (Wien)};
\end{axis}
\end{tikzpicture}
\caption[Planck law against the Rayleigh-Jeans divergence]{Spectral energy
density against dimensionless frequency $x = h\nu/k_B T$. The classical
Rayleigh-Jeans result (counting field modes and giving each the equipartition
energy $k_B T$) rises without bound as $x^2$, the ultraviolet catastrophe. The
Planck result replaces the equipartition energy with the Bose-Einstein mean
energy $h\nu/(e^{h\nu/k_B T}-1)$ of a quantised oscillator; it tracks
Rayleigh-Jeans at small $x$, where the quantum of energy is small against
$k_B T$, then turns over and falls exponentially once $h\nu$ exceeds $k_B T$.
The turnover fixes the spectral peak at $x \approx 2.82$, which is Wien's
displacement law in dimensionless form, and the area under the curve is the
Stefan-Boltzmann $T^4$ total. The single change from $k_B T$ per mode to the
Bose mean energy per mode is the whole of the quantum hypothesis as an engineer
meets it.}
\label{fig:vol04ch12:planck-vs-rj}
\end{figure}
