% Figure: density of states for a bulk 3D semiconductor near the band edges,
% with the Fermi-Dirac occupation overlaid, illustrating how carrier density
% is the product of available states and their occupation. Conduction-band
% DOS rises as sqrt(E - E_c); valence-band DOS rises as sqrt(E_v - E).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={energy $E$ (eV, relative to mid-gap)},
  ylabel={density of states $g(E)$ (relative)},
  xmin=-1.4, xmax=1.4, ymin=0, ymax=1.6,
  axis lines=left,
  legend pos=north west,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
]
% valence band: DOS ~ sqrt(E_v - E) for E < E_v = -0.56 (gap 1.12 eV centred)
\addplot[engblue, very thick, domain=-1.4:-0.56, samples=60]
  {1.2*sqrt(max(-0.56 - x, 0))};
\addlegendentry{valence-band DOS}
% conduction band: DOS ~ sqrt(E - E_c) for E > E_c = +0.56
\addplot[engred, very thick, domain=0.56:1.4, samples=60]
  {1.2*sqrt(max(x - 0.56, 0))};
\addlegendentry{conduction-band DOS}
% gap region: zero between E_v and E_c
\draw[enggray, dashed] (axis cs:-0.56,0) -- (axis cs:-0.56,1.55);
\draw[enggray, dashed] (axis cs:0.56,0) -- (axis cs:0.56,1.55);
\node[font=\scriptsize, anchor=south] at (axis cs:0,1.4) {gap $E_g$};
\draw[<->] (axis cs:-0.56,0.15) -- (axis cs:0.56,0.15);
\end{axis}
\end{tikzpicture}
\caption[Density of states near the semiconductor band edges]{Density of
available electronic states near the band edges of a bulk three-dimensional
semiconductor. In each band the density of states rises as the square root of
the distance from the edge, $g(E) \propto \sqrt{|E - E_{c,v}|}$, and is zero
inside the gap. The carrier density is the integral of this density of states
weighted by the Fermi-Dirac occupation: electrons in the conduction band,
holes (empty states) in the valence band. Reducing the device dimension
changes the square-root law to a staircase (two dimensions) or a set of
spikes (zero dimensions, the quantum dot), which is the design knob behind
quantum-well lasers and quantum-dot displays.}
\label{fig:vol04ch12:density-of-states}
\end{figure}
