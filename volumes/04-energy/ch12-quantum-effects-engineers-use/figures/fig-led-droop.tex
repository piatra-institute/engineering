% Figure: efficiency droop in an InGaN LED. Internal quantum efficiency rises
% to a peak at low current density then falls as drive current increases,
% the "droop" that forces large-area dice and limits luminaire power.
% A smooth curve drawn from explicit sample points (no computed sqrt).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10cm, height=6cm,
    xlabel={current density $J$ (\si{\ampere\per\centi\meter\squared})},
    ylabel={internal quantum efficiency (\%)},
    xmin=0, xmax=120,
    ymin=0, ymax=100,
    xtick={0,20,40,60,80,100,120},
    ytick={0,20,40,60,80,100},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    every axis plot/.append style={very thick},
]
% sample IQE(J): rises from low J to a peak near J=10 then droops
\addplot[engblue, mark=*, mark size=1.2pt] coordinates {
  (1,55) (2,68) (4,78) (6,83) (8,85) (10,86)
  (15,85) (20,83) (30,79) (40,75) (50,71)
  (60,67) (80,60) (100,54) (120,49)
};
% peak marker
\addplot[engred, only marks, mark=o, mark size=3pt] coordinates {(10,86)};

% typical luminaire operating point
\addplot[engorange, only marks, mark=square*, mark size=2.5pt]

\caption[Efficiency droop in an InGaN LED]{Internal quantum efficiency of a
blue InGaN LED against drive current density. The efficiency rises steeply at
low current as radiative recombination overtakes defect-trap recombination,
peaks near $10\,\si{\ampere\per\centi\meter\squared}$, then droops as the
current is pushed higher. The droop is attributed mainly to Auger
recombination, whose rate grows with the cube of carrier density, and to
carrier leakage out of the quantum wells. A designer who wants more light per
die operates well past the peak (square marker) and pays the droop penalty, or
spreads the current over more die area and pays in cost; the trade between flux,
efficiency, and die cost is set by this curve.}
\label{fig:vol04ch12:led-droop}
\end{figure}
