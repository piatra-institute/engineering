% Figure: energy-band diagram contrasting conductor, semiconductor, and
% insulator. Valence band (filled, blue), conduction band (empty, light),
% and the gap between them. The conductor has overlapping bands; the
% semiconductor a small gap; the insulator a large gap. Drawn with plain
% TikZ rectangles and a vertical energy axis.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
% vertical energy axis
\draw[->] (-0.6,0) -- (-0.6,6.2) node[above] {energy $E$};

% ==== Conductor ====
\begin{scope}[xshift=0cm]
  \fill[engblue!70] (0,0) rectangle (2,2.6);   % valence
  \fill[engblue!35] (0,2.4) rectangle (2,5.0);  % conduction overlapping
  \draw (0,0) rectangle (2,2.6);
  \draw (0,2.4) rectangle (2,5.0);
  \node at (1,1.3) {valence};
  \node at (1,3.9) {conduction};
  \node[align=center] at (1,-0.6) {conductor\\(bands overlap)};
\end{scope}

% ==== Semiconductor ====
\begin{scope}[xshift=4cm]
  \fill[engblue!70] (0,0) rectangle (2,2.4);   % valence
  \fill[engblue!12] (0,3.0) rectangle (2,5.0);  % conduction
  \draw (0,0) rectangle (2,2.4);
  \draw (0,3.0) rectangle (2,5.0);
  \node at (1,1.2) {valence};
  \node at (1,4.0) {conduction};
  \draw[<->] (1,2.4) -- (1,3.0);
  \node[anchor=west] at (1.05,2.7) {$E_g \approx 1$ eV};
  \node[align=center] at (1,-0.6) {semiconductor\\(small gap)};
\end{scope}

% ==== Insulator ====
\begin{scope}[xshift=8cm]
  \fill[engblue!70] (0,0) rectangle (2,1.8);   % valence
  \fill[engblue!12] (0,4.6) rectangle (2,5.6);  % conduction
  \draw (0,0) rectangle (2,1.8);
  \draw (0,4.6) rectangle (2,5.6);
  \node at (1,0.9) {valence};
  \node at (1,5.1) {conduction};
  \draw[<->] (1,1.8) -- (1,4.6);
  \node[anchor=west] at (1.05,3.2) {$E_g \gtrsim 5$ eV};
  \node[align=center] at (1,-0.6) {insulator\\(large gap)};
\end{scope}
\end{tikzpicture}
\caption[Energy bands of conductor, semiconductor, insulator]{The
band-occupation picture that separates the three electrical classes of
solid. A conductor has a partially filled band, or a filled valence band
overlapping an empty conduction band, so carriers move under any field. A
semiconductor has a small gap ($\sim 1$ eV for silicon), narrow enough that
thermal energy promotes a usable density of carriers across it and doping
controls that density. An insulator has a gap so wide ($\gtrsim 5$ eV) that
the conduction band stays empty at any ordinary temperature. The whole of
device electronics lives in the middle column.}
\label{fig:vol04ch12:energy-bands}
\end{figure}
