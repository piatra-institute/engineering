% Figure: measured-style diode I-V on a semilog axis, showing the three
% regions the project asks the reader to identify: noise-floor / sub-threshold
% at low V, the straight exponential (ideal Shockley) region, and the
% series-resistance roll-off at high V where the curve bends below the
% extrapolated line.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogyaxis}[
  width=12cm, height=7.5cm,
  xlabel={forward voltage $V$ (V)},
  ylabel={current $I$ (A)},
  xmin=0.2, xmax=0.95, ymin=1e-11, ymax=1e-1,
  axis lines=left,
  legend pos=north west,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
]
% ideal Shockley line: I = Is*exp(V/VT), Is=5e-12, VT=0.026.
% On semilog this is a straight line. Plot the ideal extrapolation (dashed).
\addplot[enggray, thick, dashed, domain=0.30:0.92, samples=40]
  {5e-12*exp(x/0.026)};
\addlegendentry{ideal Shockley extrapolation}
% measured: floor at low V (~5e-12 flat), exponential mid, roll-off high V.
\addplot[engblue, very thick, mark=*, mark size=1pt] coordinates {
  (0.20,5.5e-12) (0.25,6e-12) (0.30,7e-12) (0.35,2.5e-11)
  (0.40,1.5e-10) (0.45,1.0e-9) (0.50,7e-9) (0.55,5e-8)
  (0.60,3.5e-7) (0.65,2.3e-6) (0.70,1.4e-5) (0.75,7e-5)
  (0.78,1.6e-4) (0.82,5e-4) (0.86,1.4e-3) (0.90,3.0e-3)
};
\addlegendentry{measured diode}
% region annotations
\node[font=\scriptsize, anchor=south] at (axis cs:0.28,3e-11) {noise floor};
\node[font=\scriptsize, anchor=south, rotate=58] at (axis cs:0.56,5e-8) {exponential};
\node[font=\scriptsize, anchor=north west] at (axis cs:0.80,2e-3) {series-$R$ roll-off};
\end{semilogyaxis}
\end{tikzpicture}
\caption[Diode current-voltage curve on a semilog axis]{Current-voltage
characteristic of a small-signal silicon diode plotted as $\log I$ against
$V$. Three regions stand out. Below about $0.4$ V the current sits at the
instrument noise floor and the data are upper bounds. Between roughly $0.4$
and $0.7$ V the curve is the straight line of the Shockley equation, slope
$1/V_T$ (a decade of current per $\approx 60$ mV at room temperature), and
its extrapolation to $V = 0$ gives the saturation current $I_s$. Above about
$0.7$ V the series resistance of the diode and the test circuit drops a
growing share of the applied voltage, and the measured curve bends below the
extrapolated exponential. Identifying these three regions is the object of
the chapter project.}
\label{fig:vol04ch12:diode-iv-semilog}
\end{figure}
