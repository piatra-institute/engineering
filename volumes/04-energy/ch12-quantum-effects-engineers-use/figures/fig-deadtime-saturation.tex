% Figure: dead-time saturation of a single-photon counter. The measured count
% rate follows the true rate at low flux, then saturates because each detected
% photon blinds the detector for a fixed dead time. Non-paralysable model
% m = r/(1+r*tau); points computed offline and entered explicitly.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10cm, height=6cm,
    xlabel={true photon rate $r$ (Mcps)},
    ylabel={measured count rate $m$ (Mcps)},
    xmin=0, xmax=50,
    ymin=0, ymax=25,
    xtick={0,10,20,30,40,50},
    ytick={0,5,10,15,20,25},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    legend style={font=\tiny, at={(0.03,0.97)}, anchor=north west,
                  draw=enggray!50},
    every axis plot/.append style={very thick},
]
% ideal 1:1 line
\addplot[enggray, dashed] coordinates {(0,0) (25,25)};
\addlegendentry{ideal, $m=r$}
% non-paralysable with dead time tau = 40 ns: m = r/(1+r*tau), r in Mcps, tau in us=0.04
\addplot[engblue, mark=*, mark size=1.0pt] coordinates {
  (0,0.0) (2,1.85) (5,4.17) (8,6.06) (12,8.11) (16,9.76)
  (20,11.11) (25,12.50) (30,13.64) (35,14.58) (40,15.38)
  (45,16.07) (50,16.67)
};
\addlegendentry{$\tau=40\,\si{\nano\second}$}
% asymptote 1/tau = 25 Mcps
\addplot[engred, dotted] coordinates {(0,25) (50,25)};
\node[engred, anchor=south east, font=\tiny] at (axis cs:49,25)
  {saturation $1/\tau$};
% mark 10% loss point (where m/r = 0.9 -> r*tau=0.111 -> r~2.8 Mcps)
\addplot[engorange, only marks, mark=o, mark size=2.6pt]
  coordinates {(2.8,2.52)};
\node[engorange, anchor=north west, font=\tiny] at (axis cs:3.2,2.4)
  {$10\,\%$ loss};
\end{axis}
\end{tikzpicture}
\caption[Dead-time saturation of a single-photon counter]{Measured count rate of
a single-photon detector against the true photon arrival rate. Each detected
photon blinds the detector for a fixed dead time $\tau$, so at low flux the
measured rate tracks the true rate one for one (dashed) but at high flux it
saturates toward $1/\tau$ (dotted) because the detector spends most of its time
recovering. The non-paralysable model $m=r/(1+r\tau)$ passes through a
ten-percent loss when $r\tau\approx0.11$ (circle), which for a $40\,\si{\nano
\second}$ dead time is a true rate near $2.8$ million counts per second. The
correction is invertible, $r=m/(1-m\tau)$, so a counter is usable well into
saturation provided the dead time is known; the figure is why photon-counting
instruments quote both a maximum count rate and a dead-time correction.}
\label{fig:vol04ch12:deadtime-saturation}
\end{figure}
