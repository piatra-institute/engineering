% Figure: quantum tunnelling through a finite rectangular barrier. A particle
% of energy E below the barrier height phi approaches from the left; the
% wavefunction is oscillatory outside the barrier, decays exponentially
% inside it, and emerges with reduced amplitude. Schematic of the potential
% with a sketched wavefunction.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
% axes
\draw[->] (0,0) -- (11,0) node[right] {position $x$};
\draw[->] (0,0) -- (0,4.2) node[above] {energy};

% potential barrier: zero outside, height phi between x=4.5 and x=6.5
\draw[engblack, very thick] (0.2,0) -- (4.5,0) -- (4.5,3.0) -- (6.5,3.0) -- (6.5,0) -- (10.7,0);
\node[anchor=south] at (5.5,3.05) {barrier height $\phi$};
\node[anchor=west] at (5.6,1.5) {width $d$};
\draw[<->] (4.5,1.5) -- (6.5,1.5);

% particle energy level E < phi
\draw[engred, thick, dashed] (0.2,1.5) -- (10.7,1.5);
\node[engred, anchor=east] at (0.15,1.5) {$E$};

% incoming oscillatory wavefunction (left), centred on E level
\draw[engblue, thick, domain=0.3:4.5, samples=90, smooth]
  plot (\x, {1.5 + 0.5*sin(\x*327)});
% exponential decay inside barrier
\draw[engblue, thick, domain=4.5:6.5, samples=60, smooth]
  plot (\x, {1.5 + 0.5*exp(-1.6*(\x-4.5))});
% transmitted oscillation (right), small amplitude
\draw[engblue, thick, domain=6.5:10.6, samples=90, smooth]
  plot (\x, {1.5 + 0.10*sin(\x*327)});
\node[engblue, anchor=south] at (2.3,2.1) {incident $\psi$};
\node[engblue, anchor=south] at (8.6,1.62) {transmitted $\psi$};
\end{tikzpicture}
\caption[Quantum tunnelling through a finite barrier]{A particle of energy
$E$ below a barrier of height $\phi$ and width $d$. Classically the particle
cannot pass. Quantum mechanically the wavefunction decays exponentially
inside the barrier and emerges on the far side with reduced amplitude, so the
transmission probability is non-zero: $T \sim \exp(-2d\sqrt{2m\phi}/\hbar)$
in the thick-barrier limit. Exponential sensitivity to thickness is why
gate-oxide leakage doubles every time the oxide thins by a fixed fraction,
and why flash memory retains charge for years behind a barrier only a few
nanometres thick.}
\label{fig:vol04ch12:tunneling-well}
\end{figure}
