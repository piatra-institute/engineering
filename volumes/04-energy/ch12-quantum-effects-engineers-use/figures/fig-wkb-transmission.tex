% Figure: WKB transmission coefficient against barrier thickness for two
% barrier heights. The transmission falls exponentially with thickness; a
% taller barrier falls faster. Curves drawn from explicit sample points
% computed from T = exp(-2 kappa d) so no in-figure sqrt is evaluated.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{semilogyaxis}[
    width=10cm, height=6.4cm,
    xlabel={barrier thickness $d$ (\si{\nano\meter})},
    ylabel={transmission $T$},
    xmin=0, xmax=3,
    ymin=1e-9, ymax=1,
    xtick={0,0.5,1,1.5,2,2.5,3},
    ytick={1e-9,1e-7,1e-5,1e-3,1e-1},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    legend style={font=\tiny, at={(0.97,0.97)}, anchor=north east,
                  draw=enggray!40},
    every axis plot/.append style={very thick},
]
% phi = 3.1 eV, m* = 0.5 me: kappa = 1.8e10 /m = 18 /nm, so T = exp(-36 d_nm)
\addplot[engblue] coordinates {
  (0,1) (0.1,0.0273) (0.2,7.4e-4) (0.3,2.0e-5)
  (0.4,5.5e-7) (0.5,1.5e-8)
};
\addlegendentry{$\phi = 3.1\,\text{eV}$ (SiO$_2$)}
% phi = 1.0 eV, m* = 0.5 me: kappa scales as sqrt(phi) -> 18*sqrt(1/3.1)=10.2/nm,
% T = exp(-20.4 d_nm)
\addplot[engred] coordinates {
  (0,1) (0.2,0.0169) (0.4,2.8e-4) (0.6,4.8e-6)
  (0.8,8.0e-8) (0.9,1.0e-8)
};
\addlegendentry{$\phi = 1.0\,\text{eV}$}
% mark the flash-memory retention barrier and the gate-leakage floor regions
\draw[enggray, dashed] (axis cs:1.2,1e-9) -- (axis cs:1.2,1);
\node[enggray, anchor=south, font=\tiny, rotate=90]
  at (axis cs:1.2,1e-6) {SiO$_2$ gate floor $\approx 1.2$ nm};
\end{semilogyaxis}
\end{tikzpicture}
\caption[WKB transmission against barrier thickness]{Transmission coefficient
of a rectangular barrier in the thick-barrier (WKB) limit,
$T \approx \exp(-2\kappa d)$ with $\kappa = \sqrt{2m^{*}\phi}/\hbar$, against
barrier thickness for two barrier heights at fixed effective mass
$m^{*} = 0.5\,m_e$. The straight lines on the semilog axis are the signature
of exponential decay: each fixed increment of thickness multiplies the
transmission by a constant factor. The taller SiO$_2$ barrier
(upper line) falls faster than the shallower one because
$\kappa$ grows as the square root of the barrier height. The exponential
sensitivity is why the silicon gate oxide stalled near a physical thickness of
$1.2\,\si{\nano\meter}$, below which the leakage became unmanageable, and why
the same physics, run the other way, lets a flash-memory floating gate hold its
charge for years behind a barrier only a few nanometres thick.}
\label{fig:vol04ch12:wkb-transmission}
\end{figure}
