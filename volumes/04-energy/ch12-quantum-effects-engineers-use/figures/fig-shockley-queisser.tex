% Figure: Shockley-Queisser detailed-balance efficiency limit for a single-
% junction solar cell as a function of band gap under AM1.5G illumination.
% A broad peak near 1.1-1.4 eV at ~33%, with silicon and GaAs marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={band gap $E_g$ (eV)},
  ylabel={limiting efficiency $\eta$ (\%)},
  xmin=0.4, xmax=2.6, ymin=0, ymax=36,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% Schematic Shockley-Queisser curve: rises, broad peak ~33% near 1.34 eV,
% falls off both sides. Use sampled coordinates of the standard shape.
\addplot[engblue, very thick, smooth] coordinates {
  (0.40,8) (0.60,18) (0.80,26) (1.00,31) (1.12,32.5)
  (1.34,33.5) (1.50,32.8) (1.70,30) (1.90,26) (2.10,21)
  (2.30,16) (2.50,11)
};
% silicon marker
\filldraw[engred] (axis cs:1.12,32.5) circle (2.2pt);
\node[engred, font=\scriptsize, anchor=north] at (axis cs:1.12,31.0) {Si};
% GaAs marker
\filldraw[engorange] (axis cs:1.42,33.4) circle (2.2pt);
\node[engorange, font=\scriptsize, anchor=south] at (axis cs:1.42,34.0) {GaAs};
% peak guide
\draw[enggray, dashed] (axis cs:1.34,0) -- (axis cs:1.34,33.5);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:1.55,12) {optimum $\approx 1.34$ eV};
\end{axis}
\end{tikzpicture}
\caption[Shockley-Queisser efficiency limit versus band gap]{Detailed-balance
efficiency limit of a single-junction solar cell under the standard AM1.5G
spectrum, as a function of the absorber band gap. The curve is broad with a
peak near $33\,\%$ at a gap of about $1.34$ eV, close to gallium arsenide.
Silicon ($1.12$ eV) sits just below the peak. Too small a gap wastes the
excess photon energy as heat; too large a gap leaves the low-energy photons
unabsorbed. The single-junction ceiling is the reason high-efficiency cells
either tune the gap to the peak or stack several gaps in a multi-junction
stack.}
\label{fig:vol04ch12:shockley-queisser}
\end{figure}
