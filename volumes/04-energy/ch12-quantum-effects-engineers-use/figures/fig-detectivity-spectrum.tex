% Figure: specific detectivity D* against wavelength for three detector families.
% Each detector peaks near its band-gap cutoff and falls beyond it. The y-axis
% carries D* on a log scale; sample points entered explicitly as their base-ten
% logarithms so the figure carries no runtime computation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10.5cm, height=6.4cm,
    xlabel={wavelength $\lambda$ (\si{\micro\meter})},
    ylabel={$\log_{10} D^{*}$ (\si{\centi\meter}\,Hz$^{1/2}$/W)},
    xmin=0.4, xmax=14,
    ymin=9, ymax=13.5,
    xtick={0.4,2,4,6,8,10,12,14},
    ytick={9,10,11,12,13},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    legend style={font=\tiny, at={(0.97,0.03)}, anchor=south east,
                  draw=enggray!50},
    every axis plot/.append style={very thick},
]
% Silicon photodiode: peaks ~0.9 um, cutoff ~1.1 um
\addplot[engblue, mark=*, mark size=1.0pt] coordinates {
  (0.45,11.8) (0.60,12.4) (0.75,12.7) (0.90,12.9) (1.00,12.6)
  (1.05,12.0) (1.10,10.6)
};
\addlegendentry{Si photodiode}
% InGaAs: peaks ~1.6 um, cutoff ~1.7 um
\addplot[engred, dashed, mark=square, mark size=1.0pt] coordinates {
  (0.9,11.7) (1.1,12.3) (1.3,12.6) (1.55,12.7) (1.65,12.4)
  (1.70,11.6) (1.75,10.2)
};

  (3,11.6) (5,11.9) (7,11.8) (9,11.5) (10,11.2) (11,10.6) (12,9.6)
};
\addlegendentry{HgCdTe (77\,K)}

\addplot[enggray, densely dotted, no marks] coordinates {(0.4,13.0)(14,13.0)};

\caption[Specific detectivity of three detector families]{Specific detectivity
$D^{*}$ against wavelength for a silicon photodiode, an InGaAs photodiode, and a
cooled mercury-cadmium-telluride long-wave infrared detector. Each family rises
toward its band-gap cutoff, where it collects the most signal per unit dark
noise, then falls sharply beyond the cutoff where the photon carries less than
the gap. $D^{*}$ normalises the noise-equivalent power to unit area and unit
bandwidth, so it ranks detectors on a common footing regardless of size; the
dotted line marks the background-limited ceiling a detector reaches when the
shot noise of the ambient photon flux, not the device, sets the floor. The
choice of detector for a job is the choice of the curve whose peak sits at the
working wavelength.}
\label{fig:vol04ch12:detectivity-spectrum}
\end{figure}
