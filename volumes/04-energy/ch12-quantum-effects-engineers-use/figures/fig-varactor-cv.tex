% Figure: junction capacitance against reverse bias for an abrupt pn junction,
% the varactor curve. Capacitance falls as the square root of total junction
% voltage because the depletion width grows the same way. Sample points computed
% offline and entered explicitly so the figure carries no runtime sqrt.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10cm, height=6.2cm,
    xlabel={reverse bias $V_R$ (\si{\volt})},
    ylabel={junction capacitance $C_j$ (pF)},
    xmin=0, xmax=20,
    ymin=0, ymax=12,
    xtick={0,4,8,12,16,20},
    ytick={0,2,4,6,8,10,12},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    legend style={font=\tiny, at={(0.97,0.97)}, anchor=north east,
                  draw=enggray!50},
    every axis plot/.append style={very thick},
]
% C = C0 * (Vbi/(Vbi+VR))^(1/2), abrupt junction, C0=10 pF at zero bias, Vbi=0.83
\addplot[engblue, mark=*, mark size=1.1pt] coordinates {
  (0,10.00) (1,7.29) (2,6.11) (3,5.38) (4,4.87)
  (6,4.16) (8,3.69) (10,3.35) (12,3.09) (14,2.87)
  (16,2.70) (18,2.55) (20,2.42)
};
\addlegendentry{abrupt, $C\propto V^{-1/2}$}
% hyperabrupt profile falls faster: exponent ~ -1 (illustrative sample)
\addplot[engred, dashed, mark=square, mark size=1.1pt] coordinates {
  (0,10.00) (1,6.20) (2,4.55) (3,3.60) (4,2.98)
  (6,2.22) (8,1.77) (10,1.48) (12,1.27) (14,1.11)
  (16,0.99) (18,0.90) (20,0.82)
};
\addlegendentry{hyperabrupt, steeper}
% mark a tuning band
\draw[engorange, thick, <->] (axis cs:4,4.87) -- (axis cs:16,2.70);
\node[engorange, anchor=south west, font=\tiny] at (axis cs:8.2,3.9)
  {tuning range};
\end{axis}
\end{tikzpicture}
\caption[Varactor capacitance against reverse bias]{Depletion capacitance of a
reverse-biased pn junction against applied reverse voltage. For an abrupt
junction the depletion width grows as the square root of total junction
voltage, so the capacitance falls as $C_j\propto(V_{bi}+V_R)^{-1/2}$ (solid).
A hyperabrupt doping profile, graded so the lightly doped side thins toward the
metallurgical junction, gives a steeper fall (dashed) and a wider tuning ratio
for the same voltage swing. The bias-tunable capacitance is the varactor
exploited in voltage-controlled oscillators and RF tuners; the double arrow
marks the capacitance change a $4$ to $16\,\si{\volt}$ control swing delivers.}
\label{fig:vol04ch12:varactor-cv}
\end{figure}
