% Figure: optical-link power budget. Starting from the launched laser power, each
% loss term (connectors, fibre attenuation, splices) subtracts decibels until the
% received power meets the receiver sensitivity; the gap between the final level
% and the sensitivity is the link margin. All levels entered explicitly in dBm.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=11cm, height=6.2cm,
    ybar,
    bar width=13pt,
    xlabel={budget stage},
    ylabel={optical power (dBm)},
    ymin=-40, ymax=5,
    ytick={-40,-30,-20,-10,0},
    symbolic x coords={launch,conn1,fibre,splices,conn2,received,sens},
    xtick=data,
    x tick label style={rotate=35, anchor=east, font=\tiny},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    grid=both,
    grid style={enggray!25},
    every axis plot/.append style={draw=engblue, fill=engblue!35},
]
\addplot coordinates {
  (launch,0) (conn1,-0.5) (fibre,-25.5) (splices,-26.5)
  (conn2,-27.0) (received,-27.0)
};
% receiver sensitivity as a separate red bar
\addplot[draw=engred, fill=engred!30] coordinates {(sens,-31)};
% margin arrow between received and sensitivity
\draw[engorange, <->] (axis cs:received,-27.0) -- (axis cs:received,-31.0);
\node[engorange, anchor=west, font=\tiny, align=left]
  at (axis cs:received,-29) {margin\\$4$ dB};
\node[enggray, anchor=south, font=\tiny] at (axis cs:fibre,-25.0)
  {\SI{100}{\kilo\meter}};
% dBm axis: plain text label, dBm is not a siunitx unit
\end{axis}
\end{tikzpicture}
\caption[Optical-link power budget]{Power budget for a single-mode fibre link.
The laser launches near $0\,\text{dBm}$; a connector, the distributed fibre
attenuation over \SI{100}{\kilo\meter} at about \SI{0.25}{\decibel\per\kilo
\meter}, a pair of fusion splices, and a second connector each subtract their
share, leaving the received power at the far end. The receiver sensitivity, the
lowest power at which it meets the target error rate, sits below the received
level, and the gap between them is the link margin held against ageing, bends,
and temperature. A budget that closes with positive margin is the working
sign-off for an optical span, and the fibre attenuation is the term that fixes
how far a span may run before an amplifier is required.}
\label{fig:vol04ch12:link-budget}
\end{figure}
