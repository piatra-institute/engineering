% Figure: band bending across a pn junction at equilibrium. Conduction and
% valence band edges bend across the depletion region by the built-in
% potential eV_bi; the Fermi level is flat through the device. Drawn with
% plain TikZ paths.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
% axes
\draw[->] (0,0) -- (11,0) node[right] {position $x$};
\draw[->] (0,0) -- (0,5) node[above] {energy $E$};

% conduction band edge E_c (upper). n-side low (more filled), p-side high.
% n region flat low, transition bend, p region flat high.
\draw[engblue, very thick]
  (0.3,1.4) -- (3.5,1.4)
  .. controls (5.0,1.4) and (6.0,3.5) .. (7.5,3.5)
  -- (10.7,3.5);
\node[engblue, anchor=west] at (10.0,3.85) {$E_c$};

% valence band edge E_v (lower), parallel, gap below E_c.
\draw[engblue, very thick]
  (0.3,0.45) -- (3.5,0.45)
  .. controls (5.0,0.45) and (6.0,2.55) .. (7.5,2.55)
  -- (10.7,2.55);
\node[engblue, anchor=west] at (10.0,2.20) {$E_v$};

% Fermi level: flat dashed line through whole device.
\draw[engred, thick, dashed] (0.3,1.7) -- (10.7,1.7);
\node[engred, anchor=west] at (10.0,1.55) {$E_F$};

% depletion region shading
\fill[enggray, opacity=0.12] (3.5,0) rectangle (7.5,4.6);
\draw[enggray, dashed] (3.5,0) -- (3.5,4.6);
\draw[enggray, dashed] (7.5,0) -- (7.5,4.6);
\node[anchor=south] at (5.5,4.55) {depletion region};

% built-in potential bracket
\draw[<->] (8.6,1.4) -- (8.6,3.5);
\node[anchor=west] at (8.65,2.45) {$eV_{bi}$};

% region labels
\node at (1.8,4.2) {n-type};
\node at (9.3,0.9) {p-type};
\end{tikzpicture}
\caption[Band bending across a pn junction at equilibrium]{Conduction and
valence band edges across a pn junction in equilibrium. Diffusion of
electrons from the n side and holes from the p side leaves a charged
depletion region whose field bends the bands by the built-in potential
$eV_{bi}$. The Fermi level (red dashed) is flat through the whole device, the
condition for zero net current. Forward bias lowers the barrier and lets
diffusion current flow; reverse bias raises it and only the small
saturation current remains. The same bent-band picture, illuminated or
electrically driven, explains the LED, the laser diode, and the solar cell.}
\label{fig:vol04ch12:pn-junction-bands}
\end{figure}
