% Figure: modal gain versus injected current density in a laser diode, with the
% fixed cavity-loss line. Lasing begins where the modal gain rises to meet the
% total loss (mirror loss plus internal loss). Below that current the cavity
% cannot sustain oscillation; the crossing fixes the threshold current density.
% Sampled coordinates only.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\scriptsize]
\begin{axis}[
    width=10.5cm, height=6.2cm,
    xlabel={injected current density $J$ (\si{\ampere\per\centi\meter\squared})},
    ylabel={modal gain $g$ (\si{\per\centi\meter})},
    xmin=0, xmax=1200,
    ymin=0, ymax=60,
    xtick={0,200,400,600,800,1000,1200},
    ytick={0,10,20,30,40,50,60},
    grid=both,
    grid style={enggray!25},
    tick label style={font=\tiny},
    label style={font=\scriptsize},
    axis line style={enggray},
    legend style={font=\tiny, at={(0.98,0.02)}, anchor=south east,
      draw=enggray, fill=white},
    every axis plot/.append style={very thick},
]
% modal gain g(J): logarithmic-like rise that saturates (transparency near J0)
\addplot[engblue] coordinates {
  (0,0) (50,0) (100,3) (150,8) (200,13) (300,21) (400,27)
  (500,32) (600,36) (700,39) (800,42) (1000,46) (1200,49)
};
\addlegendentry{modal gain $g(J)$}
% total cavity loss line (mirror + internal), horizontal
\addplot[engred, dashed, domain=0:1200] {30};
\addlegendentry{cavity loss $\alpha_i + \alpha_m$}
% threshold marker where g meets loss (~J=550 at g=30)
\addplot[engorange, only marks, mark=*, mark size=2.5pt] coordinates {(550,30)};
\draw[engorange, densely dotted] (axis cs:550,0) -- (axis cs:550,30);
\node[engorange, anchor=south west, font=\tiny] at (axis cs:560,8)
  {$J_{th}$};
% transparency point
\addplot[enggray, only marks, mark=square, mark size=2pt] coordinates {(50,0)};
\node[enggray, anchor=south west, font=\tiny] at (axis cs:55,1.5)
  {transparency $J_{tr}$};
% region labels
\node[engblue, anchor=south, font=\tiny] at (axis cs:850,52) {lasing};
\node[enggray, anchor=south, font=\tiny] at (axis cs:250,40)
  {sub-threshold (LED-like)};
\end{axis}
\end{tikzpicture}
\caption[Gain and loss in a laser diode]{Modal gain of the active region as a
function of injected current density, against the fixed loss line of the cavity.
At low current the medium is absorbing; the square marker is the transparency
point where injection just makes the medium neither absorbing nor amplifying.
The gain then climbs with current but saturates as the bands fill. Oscillation
begins only where the modal gain rises to meet the total round-trip loss, the
sum of the distributed internal loss and the mirror loss of the cleaved facets;
the current density at that crossing is the threshold $J_{th}$. Lowering the
mirror loss with a coated rear facet, or concentrating the carriers with a
quantum well, drops the loss line and so the threshold.}
\label{fig:vol04ch12:laser-gain-loss}
\end{figure}
