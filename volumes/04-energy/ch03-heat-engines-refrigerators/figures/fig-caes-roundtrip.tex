% Figure: round-trip efficiency of compressed-air energy storage against the
% fraction of compression heat that is stored and returned to the expansion.
% A diabatic plant that throws away the heat of compression and reheats with
% fuel on expansion has a low electricity-to-electricity ratio; an adiabatic
% plant that stores the compression heat and returns it recovers far more.
% Illustrative linear-plus-offset relation, not a plant calculation.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={fraction of compression heat stored and returned},
  ylabel={round-trip efficiency (\%)},
  xmin=0, xmax=1, ymin=0, ymax=80,
  axis lines=left,
  legend pos=north west, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize}, clip=false,
]
% round-trip = base + slope*fraction. Diabatic (fraction=0) ~ 42% (with fuel),
% adiabatic (fraction=1) ~ 70%. eta = 42 + 28*f
\addplot[engblue, very thick, samples=40, domain=0:1]
  {42 + 28*x};
\addlegendentry{store-and-return round trip}
% mark diabatic and adiabatic endpoints
\addplot[engred, only marks, mark=*, mark size=1.8pt] coordinates {(0,42)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.03,45)
  {diabatic (heat discarded, fuel reheat)};
\addplot[engblack, only marks, mark=*, mark size=1.8pt] coordinates {(1,70)};
\node[engblack, font=\scriptsize, anchor=east] at (axis cs:0.97,67)
  {adiabatic (heat stored, no fuel)};
% battery reference band
\addplot[enggray, dashed, thick, samples=2, domain=0:1] {88};
\node[enggray, font=\scriptsize, anchor=north east] at (axis cs:1,86)
  {lithium battery $\approx 88\,\%$ (reference)};
\end{axis}
\end{tikzpicture}
\caption[Compressed-air storage round-trip efficiency]{Electricity-to-electricity
round-trip efficiency of a compressed-air energy store against the fraction of the
compression heat that is captured and returned to the air on expansion. A diabatic
plant cools the air after compression, discards that heat, and must burn fuel to
reheat the air before it expands, so its electricity-only round trip is low and its
true efficiency depends on how the reheat fuel is charged. An adiabatic plant
stores the heat of compression in a thermal reservoir and returns it to the air on
the way out, needing no combustion, and recovers a far larger share. The recovered
work rises with the stored-heat fraction because the heat of compression is exactly
the exergy the expansion needs back. A lithium battery reference band marks the
competitor the store must beat on cost, since it already beats it on round-trip
efficiency; the case for compressed air rests on storing energy for many hours at
low cost per unit of capacity, not on matching the battery's efficiency.}
\label{fig:vol04ch03:caes-roundtrip}
\end{figure}
