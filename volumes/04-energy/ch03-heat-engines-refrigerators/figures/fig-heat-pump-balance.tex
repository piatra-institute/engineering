% Figure: residential air-source heat pump seasonal performance. Three curves
% against outdoor temperature: the building heat demand (rising as it gets
% colder), the heat-pump delivered capacity (falling as it gets colder), and
% the field COP (also falling). Their intersection is the balance point, below
% which supplemental resistance heat must make up the shortfall.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8.5cm,
  xlabel={outdoor temperature $T_{\text{out}}$ ($^{\circ}$C)},
  ylabel={heating rate (kW)},
  xmin=-20, xmax=12, ymin=0, ymax=12,
  axis lines=left,
  legend pos=north east, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
  axis y line*=left,
]
% Building demand: linear, zero at 18 C balance, rising as it gets colder.
% Q_demand = k (T_indoor_set - T_out); take k = 0.30 kW/K, T_set = 20.
\addplot[engred, very thick, samples=2, domain=-20:12] {0.30*(20 - x)};
\addlegendentry{building heat demand}
% Heat-pump delivered capacity: falls roughly linearly as it gets colder
% (compressor lift grows, defrost intervenes). Capacity ~ 5 + 0.18*T_out.
\addplot[engblue, very thick, samples=2, domain=-20:12] {5 + 0.18*x};
\addlegendentry{heat-pump capacity}
% Balance point: where demand = capacity. 0.30(20 - T) = 5 + 0.18 T
% 6 - 0.30 T = 5 + 0.18 T -> 1 = 0.48 T -> T = 2.08 C, Q ~ 5.37 kW.
\addplot[engblack, only marks, mark=*, mark size=2pt] coordinates {(2.08,5.37)};
\node[font=\scriptsize, anchor=south west] at (axis cs:2.08,5.37)
  {balance point $\approx 2\,^{\circ}$C};
% Shade the supplemental-heat region conceptually with an annotation.
\node[font=\scriptsize, engred, anchor=west] at (axis cs:-19,9.5)
  {supplemental};
\node[font=\scriptsize, engred, anchor=west] at (axis cs:-19,8.7)
  {resistance heat};
\draw[-{Latex[length=2mm]}, engred] (axis cs:-15,8.4) -- (axis cs:-13,7.0);
\end{axis}
% === second axis for COP, overlaid ===
\begin{axis}[
  width=12cm, height=8.5cm,
  xmin=-20, xmax=12, ymin=0, ymax=6,
  axis y line*=right,
  axis x line=none,
  ylabel={field COP},
  ylabel style={font=\small}, tick label style={font=\scriptsize},
  ytick={1,2,3,4,5},
]
% Field COP: ~1.7 at -15 C rising to ~4.0 at +10 C. COP ~ 2.7 + 0.10*T_out,
% clamped so it stays above 1. Plot over the realistic operating band.
\addplot[englime!60!black, very thick, dashed, samples=2, domain=-15:10]
  {2.7 + 0.10*x};
\node[font=\scriptsize, englime!50!black, anchor=south east]
  at (axis cs:9,3.6) {field COP (right axis)};
\end{axis}
\end{tikzpicture}
\caption[Air-source heat-pump seasonal balance]{Seasonal performance of a
residential air-source heat pump against outdoor temperature. The building's
heat demand (red) rises as it gets colder, while the heat pump's delivered
capacity (blue) falls, because the compressor lift grows and defrost cycles
interrupt heating. The two cross at the \emph{balance point} near
$2\,^{\circ}$C; above it the heat pump alone carries the load, below it the
shortfall is met by supplemental resistance heat at $\text{COP}=1$, which
pulls the seasonal average performance down. The field COP (green, right axis)
falls from about $4$ in mild weather toward $2$ in the cold, so the coldest
hours are the ones where the heat pump both delivers least and costs most per
delivered joule. Sizing and the choice of balance point are the central design
decisions; numbers are representative of a single-stage unit, not a specific
product.}
\label{fig:vol04ch03:heat-pump-balance}
\end{figure}
