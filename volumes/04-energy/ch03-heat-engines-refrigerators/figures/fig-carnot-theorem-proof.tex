% Figure: the Carnot-theorem refutation diagram. A hypothetical engine X with
% efficiency above the Carnot value drives a reversible Carnot engine run
% backward as a refrigerator between the same two reservoirs. If X were more
% efficient, the pair would move heat from cold to hot with no net work input,
% violating the Clausius statement of the second law. The drawing is a block
% diagram of the two devices and the reservoirs, with heat arrows labelled.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  >={Latex[length=2.4mm]},
  res/.style={draw, thick, minimum width=5.2cm, minimum height=0.9cm,
    font=\small},
  dev/.style={draw, thick, minimum width=2.0cm, minimum height=1.6cm,
    font=\small, align=center},
]
% reservoirs
\node[res, fill=engred!12] (hot) at (0,3.2) {hot reservoir $T_h$};
\node[res, fill=engblue!12] (cold) at (0,-3.2) {cold reservoir $T_c$};
% devices
\node[dev, fill=engamber!12] (X) at (-1.7,0) {engine\\$X$};
\node[dev, fill=enggray!12] (C) at (1.7,0) {Carnot\\refrigerator};
% engine X: takes Q_h from hot, delivers work W, rejects Q_cX to cold
\draw[->, engred, thick] (hot.south -| X) -- (X.north)
  node[midway, left, font=\scriptsize] {$Q_h$};
\draw[->, engblue, thick] (X.south) -- (cold.north -| X)
  node[midway, left, font=\scriptsize] {$Q_h - W$};
% work from X to C
\draw[->, engblack, thick] (X.east) -- (C.west)
  node[midway, above, font=\scriptsize] {$W$};
% Carnot refrigerator: driven by W, lifts Q_cC from cold to hot, rejects Q_h
\draw[<-, engred, thick] (hot.south -| C) -- (C.north)
  node[midway, right, font=\scriptsize] {$Q_h$};
\draw[<-, engblue, thick] (C.south) -- (cold.north -| C)
  node[midway, right, font=\scriptsize] {$Q_h - W'$};
% net-effect annotation
\node[font=\scriptsize, engred, align=center, anchor=west] at (4.0,1.4)
  {if $\eta_X > \eta_{\text{Carnot}}$\\then $W > W'$};
\node[font=\scriptsize, engblue, align=center, anchor=west] at (4.0,-1.4)
  {net heat flows\\cold $\to$ hot\\with no input};
\end{tikzpicture}
\caption[Carnot-theorem refutation diagram]{The construction that proves no
engine can beat the Carnot bound. A hypothetical engine $X$ operating between
$T_h$ and $T_c$ produces work $W$ from a heat intake $Q_h$, and that work drives
a reversible Carnot engine run backward as a refrigerator between the same two
reservoirs. The Carnot device, being reversible, returns exactly $Q_h$ to the
hot reservoir when supplied with the work $W'$ a Carnot engine would have
produced from that same $Q_h$. If $X$ were more efficient than Carnot it would
make more work from the same intake, $W > W'$, so the pair could be sized to
return all of $Q_h$ to the hot reservoir while still drawing net heat out of the
cold reservoir, a self-acting transfer of heat from cold to hot with no external
work. That outcome is exactly what the Clausius statement of the second law
forbids, so the assumption $\eta_X > \eta_{\text{Carnot}}$ is impossible. The
bound holds for any working fluid and any cycle geometry, because the argument
never used either.}
\label{fig:vol04ch03:carnot-theorem-proof}
\end{figure}
