% Figure: combined-cycle energy-flow schematic. Brayton (gas turbine) on top,
% its exhaust feeding a heat-recovery steam generator that drives a Rankine
% (steam turbine) bottom cycle. Sankey-style proportional arrows.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  box/.style={draw, very thick, minimum width=2.8cm, minimum height=1.1cm, font=\small},
  arr/.style={-{Latex[length=2.6mm]}, very thick},
]
\node[box, fill=engred!12] (fuel) at (0,4) {fuel $100$};
\node[box, fill=englime!15] (gt) at (0,2) {gas turbine};
\node[box, fill=englime!15] (st) at (0,-1) {steam turbine};
\node[box, fill=engblue!12] (hrsg) at (4,0.5) {HRSG};
\node[box, fill=enggray!15] (stack) at (-4,0.5) {stack loss $\approx 8$};
\node[box, fill=enggray!15] (cond) at (-4,-2.5) {condenser $\approx 28$};
% flows
\draw[arr, engred] (fuel.south) -- node[right, font=\scriptsize]{$100$} (gt.north);
\draw[arr, engorange] (gt.east) -| node[pos=0.2, above, font=\scriptsize]{elec $\approx 38$} ++(2.2,0);
\draw[arr, engblue] (gt.south) .. controls +(0,-0.6) and +(0,0.8) .. (hrsg.north)
  node[pos=0.5, right, font=\scriptsize]{exhaust $\approx 62$};
\draw[arr, enggray] (gt.west) -- node[above, font=\scriptsize]{} (stack.east);
\draw[arr, engblue] (hrsg.south) .. controls +(0,-0.6) and +(0,0.8) .. (st.east)
  node[pos=0.5, left, font=\scriptsize]{steam};
\draw[arr, engorange] (st.west) -- ++(-2.2,0) node[above, font=\scriptsize]{elec $\approx 20$};
\draw[arr, enggray] (st.south) -- (cond.north);
\node[font=\scriptsize, anchor=north] at (0,-2.6) {total electricity $\approx 58$ of $100$ fuel};
\end{tikzpicture}
\caption[Combined-cycle energy flow]{Energy flow through a combined-cycle power
plant, with all figures as a percentage of the fuel energy input. The gas
turbine (Brayton top cycle) converts about $38$ to electricity and exhausts the
balance hot. The heat-recovery steam generator (HRSG) captures the exhaust and
raises steam for the steam turbine (Rankine bottom cycle), which converts about
another $20$ to electricity. The two cycles together reach roughly $58$,
against $38$ for the gas turbine alone, which is why the combined cycle is the
modern standard for large fossil-fuel generation. The stack and condenser
carry the unrecovered remainder.}
\label{fig:vol04ch03:combined-cycle-flow}
\end{figure}
