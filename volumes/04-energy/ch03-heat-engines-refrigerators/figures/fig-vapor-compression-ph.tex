% Figure: vapour-compression refrigeration cycle on the pressure-enthalpy
% (p-h) plane with a schematic saturation dome. Four states: 1 sat. vapour at
% evaporator pressure, 2 superheated after compressor, 3 sat. liquid at
% condenser, 4 wet mixture after throttle (h4 = h3).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=8cm,
  xlabel={specific enthalpy $h$ (kJ/kg)},
  ylabel={pressure $p$ (kPa, log)},
  ymode=log,
  xmin=150, xmax=460, ymin=80, ymax=1500,
  axis lines=left,
  ytick={200,1000}, yticklabels={$p_{\text{evap}}$,$p_{\text{cond}}$},
  xtick=\empty,
  tick label style={font=\scriptsize}, label style={font=\small},
  clip=false,
]
% saturation dome (schematic): liquid line on left, vapour line on right
\addplot[enggray, thick, smooth] coordinates {
  (180,80) (190,200) (210,1000) (235,1500)
};
\addplot[enggray, thick, smooth] coordinates {
  (235,1500) (270,1000) (398,200) (400,80)
};
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:235,1500) {critical};
% cycle path
% 1 sat vapour at p_evap: (h1=398, p=200)
% 1->2 compression (isentropic), to p_cond: superheated (h2=435, p=1000)
% 2->3 condenser: reject heat, h falls to sat liquid (h3=256, p=1000)
% 3->4 throttle isenthalpic: h const, p drops (h4=256, p=200)
% 4->1 evaporator: absorb heat along p_evap to sat vapour (h1=398)
\addplot[engblue, very thick] coordinates {(398,200) (435,1000)};      % compressor 1->2
\addplot[engblue, very thick] coordinates {(435,1000) (256,1000)};     % condenser 2->3
\addplot[engblue, very thick] coordinates {(256,1000) (256,200)};      % throttle 3->4
\addplot[engblue, very thick] coordinates {(256,200) (398,200)};       % evaporator 4->1
\filldraw[engred] (axis cs:398,200) circle (1.6pt) node[anchor=north west, font=\scriptsize]{1};
\filldraw[engred] (axis cs:435,1000) circle (1.6pt) node[anchor=south west, font=\scriptsize]{2};
\filldraw[engred] (axis cs:256,1000) circle (1.6pt) node[anchor=south east, font=\scriptsize]{3};
\filldraw[engred] (axis cs:256,200) circle (1.6pt) node[anchor=north east, font=\scriptsize]{4};
\node[font=\scriptsize, engorange, anchor=south] at (axis cs:327,205) {$q_c = h_1 - h_4$};
\node[font=\scriptsize, engorange, anchor=north] at (axis cs:345,1000) {$q_h = h_2 - h_3$};
\node[font=\scriptsize, engorange, anchor=west] at (axis cs:417,450) {$w_c$};
\end{axis}
\end{tikzpicture}
\caption[Vapour-compression cycle on the pressure-enthalpy plane]{The ideal
vapour-compression refrigeration cycle on the pressure-enthalpy plane. The
compressor raises the saturated vapour from evaporator to condenser pressure
(1 to 2). The condenser rejects $q_h = h_2 - h_3$ and the refrigerant leaves as
saturated liquid (2 to 3). The expansion valve throttles the liquid back to
evaporator pressure at constant enthalpy (3 to 4, the vertical line, $h_4 =
h_3$). The evaporator absorbs $q_c = h_1 - h_4$ from the cold space as the
mixture boils to saturated vapour (4 to 1). The cooling coefficient of
performance is $q_c/w_c = (h_1 - h_4)/(h_2 - h_1)$. The condenser and
evaporator processes are horizontal because they occur at constant pressure.}
\label{fig:vol04ch03:vapor-compression-ph}
\end{figure}
