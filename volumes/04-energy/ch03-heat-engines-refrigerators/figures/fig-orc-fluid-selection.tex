% Figure: organic Rankine cycle working-fluid selection. Cycle thermal
% efficiency against evaporator temperature for three candidate dry fluids
% across a fixed condenser temperature. Each fluid is usable only up to its
% critical temperature, so the curves end where the fluid runs out of subcritical
% headroom; the highest-critical fluid reaches the highest evaporator temperature
% and the highest efficiency, but a real selection also weighs flammability,
% volume flow, and cost, which the caption notes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={evaporator temperature $T_{\text{evap}}$ ($^{\circ}$C)},
  ylabel={cycle thermal efficiency (\%)},
  xmin=60, xmax=160, ymin=0, ymax=20,
  axis lines=left,
  legend pos=north west, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
]
% Each curve is the ideal Carnot-fraction efficiency at a 35 C condenser,
% eta = k * (1 - T_cond/T_evap) with T in kelvin, k ~ 0.55 (second-law fraction
% typical of a small ORC turbine), drawn only up to each fluid's usable ceiling.
% R-245fa: critical near 154 C, usable to ~130 C subcritical.
\addplot[engblue, very thick, samples=40, domain=60:130]
  {0.55*(1 - 308/(x+273))*100};
\addlegendentry{R-245fa (low GWP target, mild ceiling)}
% Toluene: critical near 319 C, usable far higher, here to 160 C.
\addplot[engred, very thick, samples=40, domain=60:160]
  {0.60*(1 - 308/(x+273))*100};
\addlegendentry{toluene (high ceiling, flammable)}
% Isopentane: critical near 187 C, usable to ~150 C.
\addplot[englime!70!black, very thick, dashed, samples=40, domain=60:150]
  {0.52*(1 - 308/(x+273))*100};
\addlegendentry{isopentane (moderate ceiling)}
% mark a representative design point on the toluene curve at 150 C
\addplot[engblack, only marks, mark=*, mark size=2pt] coordinates {(150,11.7)};
\node[font=\scriptsize, anchor=north west] at (axis cs:150,11.5)
  {design point};
\end{axis}
\end{tikzpicture}
\caption[Working-fluid selection for an organic Rankine cycle]{Ideal cycle
thermal efficiency against evaporator temperature for three candidate organic
Rankine working fluids, all against a $35\,^{\circ}$C condenser, each curve
scaled by a representative small-turbine second-law fraction. Efficiency rises
with evaporator temperature because the average temperature of heat addition
rises, so a fluid with a higher critical temperature can be pushed to a higher
evaporator temperature and a higher efficiency. The ceiling of each curve is the
subcritical limit the fluid allows: R-245fa, with a low critical temperature,
tops out early; toluene reaches much higher but is flammable and needs careful
handling; isopentane sits between them. The selection is never efficiency alone,
since the fluid also sets the turbine volume flow, the condenser pressure
relative to atmosphere, the flammability and toxicity class, and the global
warming potential the Kigali schedule now constrains. Curves are
representative, not fluid-property-exact.}
\label{fig:vol04ch03:orc-fluid-selection}
\end{figure}
