% Figure: thermal efficiency vs compression/pressure ratio for the Otto cycle
% (eta = 1 - r^-(gamma-1)) and the Brayton cycle (eta = 1 - r_p^-((g-1)/g)).
% gamma = 1.4.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={ratio ($r$ for Otto, $r_p$ for Brayton)},
  ylabel={air-standard thermal efficiency},
  xmin=1, xmax=30, ymin=0, ymax=0.75,
  axis lines=left,
  legend pos=south east, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize}, clip=false,
  ytick={0,0.2,0.4,0.6}, yticklabels={0,0.2,0.4,0.6},
]
% Otto: eta = 1 - r^-0.4
\addplot[engblue, very thick, samples=80, domain=1:30] {1 - x^(-0.4)};
\addlegendentry{Otto $\eta = 1 - r^{-(\gamma-1)}$}
% Brayton: eta = 1 - r_p^-(0.4/1.4) = 1 - r_p^-0.2857
\addplot[engred, very thick, samples=80, domain=1:30] {1 - x^(-0.285714)};
\addlegendentry{Brayton $\eta = 1 - r_p^{-(\gamma-1)/\gamma}$}
% typical operating bands
\addplot[englime!60!black, only marks, mark=*, mark size=1.6pt] coordinates {(9,0.585) (10,0.602)};
\node[font=\scriptsize, anchor=north west] at (axis cs:10,0.602) {gasoline $r{=}8$--$10$};
\addplot[engorange, only marks, mark=square*, mark size=1.6pt] coordinates {(12,0.508) (18,0.566)};
\node[font=\scriptsize, anchor=west] at (axis cs:18,0.45) {gas-turbine $r_p{=}12$--$20$};
\end{axis}
\end{tikzpicture}
\caption[Air-standard efficiency versus ratio, Otto and Brayton]{Air-standard
thermal efficiency against ratio for the Otto cycle (efficiency set by the
compression ratio $r$) and the Brayton cycle (efficiency set by the pressure
ratio $r_p$), both for $\gamma = 1.4$. The Otto curve rises faster because its
exponent $\gamma - 1 = 0.4$ exceeds the Brayton exponent
$(\gamma-1)/\gamma = 0.286$. Both curves show diminishing returns: the marginal
efficiency gain per unit of added ratio falls steadily, which is why real
engines settle at modest ratios where the mechanical and material penalties of
higher ratios outweigh the shrinking efficiency reward. The marked bands show
the operating ranges of automotive gasoline engines and industrial gas
turbines.}
\label{fig:vol04ch03:efficiency-vs-ratio}
\end{figure}
