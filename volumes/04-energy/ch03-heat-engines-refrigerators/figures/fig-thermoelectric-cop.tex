% Figure: maximum cooling COP of a thermoelectric (Peltier) module against the
% dimensionless figure of merit ZT, for a fixed temperature lift. The COP of a
% Peltier cooler equals the Carnot COP multiplied by a factor that rises from
% zero at ZT = 0 toward one as ZT grows without bound. Real modules sit near
% ZT = 1, far to the left, which is why thermoelectric cooling reaches only a
% small fraction of the Carnot bound.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11.5cm, height=7.5cm,
  xlabel={figure of merit $ZT$},
  ylabel={cooling COP},
  xmin=0, xmax=10, ymin=0, ymax=6,
  axis lines=left,
  tick label style={font=\small}, label style={font=\small},
  legend pos=north west, legend cell align=left, legend style={font=\scriptsize},
  clip=false,
]
% Carnot COP for T_c=273, T_h=300, lift 27: COP_C = 273/27 = 10.11
% Peltier optimum: COP = COP_C * (M - T_h/T_c)/(M + 1), M = sqrt(1 + ZT_mean)
% with T_mean = 286.5, ratio T_h/T_c = 300/273 = 1.0989.
% M(ZT) = sqrt(1 + ZT). Plot COP = 10.11*(sqrt(1+ZT) - 1.0989)/(sqrt(1+ZT)+1).
\addplot[engblue, very thick, samples=120, domain=0:10]
  {10.11*( (1+x)^0.5 - 1.0989 )/( (1+x)^0.5 + 1 )};
\addlegendentry{Peltier optimum, lift $27\,\mathrm{K}$}
% Carnot ceiling as a dashed line
\addplot[enggray, dashed, thick, samples=2, domain=0:10] {10.11};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:10,5.7)
  {Carnot COP $=10.1$ (off scale)};
% real module marker near ZT=1
\addplot[engred, only marks, mark=*, mark size=1.8pt] coordinates {(1,0.55)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:1.15,0.75)
  {real module, $ZT\approx1$};
\end{axis}
\end{tikzpicture}
\caption[Thermoelectric cooling COP versus figure of merit]{Maximum cooling
coefficient of performance of a thermoelectric module against its dimensionless
figure of merit $ZT$, for a fixed lift of $27\,\mathrm{K}$ ($0\,\degC$ cold side,
$27\,\degC$ hot side). The optimised COP is the Carnot value scaled by a factor
that climbs from zero and approaches the Carnot bound only as $ZT \to \infty$.
Commercial bismuth-telluride modules sit near $ZT \approx 1$, marked at the far
left, where the realised COP is a small fraction of the Carnot ceiling drawn
dashed above. The steep dependence on $ZT$ near the origin is why the search for
higher-$ZT$ materials is the central problem of solid-state cooling, and why
thermoelectrics remain confined to duties where their silence, absence of moving
parts, and small size outweigh their poor efficiency.}
\label{fig:vol04ch03:thermoelectric-cop}
\end{figure}
