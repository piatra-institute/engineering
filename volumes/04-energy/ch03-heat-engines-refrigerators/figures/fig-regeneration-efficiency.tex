% Figure: Brayton thermal efficiency with and without an ideal regenerator,
% against pressure ratio, for tau = T3/T1 = 4.67 (T3=1400, T1=300).
% Simple cycle:  eta = 1 - x^-1, x = r_p^0.285714.
% Ideal-regenerator cycle (effectiveness 1):  eta_reg = 1 - x/tau, valid only
% where x < tau (T2 < T4); beyond the crossover x = sqrt(tau) the regenerator
% would run backward, so the regenerated curve is drawn only to the crossover.
% The two curves cross at x = sqrt(tau), r_p = tau^(gamma/(2(gamma-1))).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={pressure ratio $r_p$},
  ylabel={air-standard thermal efficiency},
  xmin=1, xmax=40, ymin=0, ymax=0.75,
  axis lines=left,
  legend pos=south east, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize}, clip=false,
  ytick={0,0.2,0.4,0.6}, yticklabels={0,0.2,0.4,0.6},
]
% tau = 1400/300 = 4.667 ; crossover at r_p = tau^1.75 = 4.667^1.75 = 14.8
% simple cycle over full range
\addplot[engred, very thick, samples=100, domain=1:40] {1 - (x^0.285714)^(-1)};
\addlegendentry{simple cycle $\eta = 1 - r_p^{-(\gamma-1)/\gamma}$}
% ideal regenerator, only to crossover x=sqrt(tau)=2.160, r_p=14.8
\addplot[engblue, very thick, samples=100, domain=1:14.8]
  {1 - (x^0.285714)/4.667};
\addlegendentry{ideal regenerator $\eta = 1 - \tau^{-1} r_p^{(\gamma-1)/\gamma}$}
% crossover marker
\addplot[engblack, only marks, mark=*, mark size=1.7pt] coordinates {(14.8,0.537)};
\node[font=\scriptsize, anchor=south west] at (axis cs:14.8,0.537)
  {crossover};
\node[font=\scriptsize, engblue, anchor=east] at (axis cs:4.5,0.62)
  {regeneration pays};
\node[font=\scriptsize, engred, anchor=west] at (axis cs:22,0.43)
  {regeneration would hurt};
\end{axis}
\end{tikzpicture}
\caption[Brayton efficiency with and without regeneration]{Air-standard Brayton
efficiency with an ideal regenerator (blue) and without one (red), for a fixed
temperature ratio $\tau = T_3/T_1 = 4.67$. The regenerated efficiency falls with
pressure ratio, because a higher ratio leaves the compressor exit hotter and the
turbine exit cooler, shrinking the exhaust-to-inlet temperature gap the
regenerator feeds on. The two curves cross at the pressure ratio
$r_p = \tau^{\gamma/(2(\gamma-1))}$, where the compressor exit temperature equals
the turbine exit temperature and the regenerator has nothing to transfer. Below
the crossover a regenerator raises efficiency; above it the compressor exit is
hotter than the turbine exit, so a regenerator would carry heat the wrong way and
must be bypassed. Regeneration is therefore a low-pressure-ratio refinement, the
opposite design region from the high-ratio machines that chase peak simple-cycle
efficiency.}
\label{fig:vol04ch03:regeneration-efficiency}
\end{figure}
