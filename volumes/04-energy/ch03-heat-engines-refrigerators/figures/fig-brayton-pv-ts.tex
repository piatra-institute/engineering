% two adiabats. Pressure ratio r_p = 12, T1 = 300 K, T3 = 1400 K, gamma=1.4.
% T2 = 300*12^(0.4/1.4) = 610 K. T4 = 1400/12^(0.4/1.4) = 688 K.

\centering
\begin{tikzpicture}

\begin{axis}[

  axis lines=left, xtick=\empty, ytick=\empty,
  label style={font=\small}, clip=false,
]
% process 1->2 isentropic compression: p up, v down
\addplot[engblue, very thick, smooth] coordinates {(1.0,1) (0.6,3) (0.35,7) (0.25,12)};
% 2->3 const pressure heat addition: v up at p=12
\addplot[engred, very thick] coordinates {(0.25,12) (0.55,12)};
% 3->4 isentropic expansion
\addplot[engblue, very thick, smooth] coordinates {(0.55,12) (0.8,6) (1.1,2.5) (1.3,1)};
% 4->1 const pressure heat rejection at p=1 (truncate to axis)
\addplot[engblue, very thick] coordinates {(1.3,1) (1.0,1)};
\filldraw[engorange] (axis cs:1.0,1) circle (1.4pt) node[anchor=north east, font=\scriptsize]{1};

\node[font=\scriptsize, anchor=south] at (axis cs:0.55,7) {(a) $p$-$v$};
\end{axis}
% T-s diagram (right)
\begin{axis}[
  name=ts, at={($(pv.east)+(1.4cm,0)$)}, anchor=west,
  width=7.5cm, height=6.5cm,
  xlabel={specific entropy $s$}, ylabel={temperature $T$ (K)},
  xmin=0, xmax=2.6, ymin=200, ymax=1550,
  axis lines=left, xtick=\empty, ytick={300,610,688,1400},
  yticklabels={300,610,688,1400},
  tick label style={font=\scriptsize}, label style={font=\small}, clip=false,
]
% 1->2 isentropic compression: vertical (s const), T 300->610
\addplot[engblue, very thick] coordinates {(0.4,300) (0.4,610)};

\addplot[engred, very thick, smooth] coordinates {(0.4,610) (1.2,950) (2.0,1400)};

\addplot[engblue, very thick] coordinates {(2.0,1400) (2.0,688)};

\addplot[engblue, very thick, smooth] coordinates {(2.0,688) (1.2,460) (0.4,300)};
\filldraw[engorange] (axis cs:0.4,300) circle (1.4pt) node[anchor=north east, font=\scriptsize]{1};

\end{axis}

\caption[Brayton cycle on the pressure-volume and temperature-entropy
planes]{The air-standard Brayton cycle. (a) On the $p$-$v$ plane the cycle is
bounded by two adiabats (compression 1 to 2, expansion 3 to 4) and two isobars
(combustion 2 to 3, exhaust 4 to 1). (b) On the $T$-$s$ plane the adiabats are
vertical and the isobars diverge upward, so the heat-addition isobar (2 to 3)
runs hotter than the heat-rejection isobar (4 to 1). The numbers shown are for
pressure ratio $r_p = 12$, $T_1 = 300$ K, $T_3 = 1400$ K, $\gamma = 1.4$, giving
$T_2 = 610$ K and $T_4 = 688$ K.}
\label{fig:vol04ch03:brayton-pv-ts}
\end{figure}
