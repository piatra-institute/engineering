% Figure: Carnot heating and cooling COP versus temperature lift (T_h - T_c),
% with T_h fixed at 308 K (35 C). Shows the divergence as lift -> 0 and the
% decay as lift grows.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={temperature lift $T_h - T_c$ (K)},
  ylabel={Carnot coefficient of performance},
  xmin=5, xmax=60, ymin=0, ymax=20,
  axis lines=left,
  legend pos=north east, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize}, clip=false,
]
% T_h = 308 fixed. COP_HP = T_h / (lift) = 308 / x. COP_R = (T_h - x)/x.
\addplot[engred, very thick, samples=120, domain=5:60] {308/x};
\addlegendentry{heat pump $\text{COP}_{HP} = T_h/(T_h - T_c)$}
\addplot[engblue, very thick, samples=120, domain=5:60] {(308 - x)/x};
\addlegendentry{refrigerator $\text{COP}_R = T_c/(T_h - T_c)$}
% reference: typical realised fraction band ~0.5 of Carnot at lift=40
\addplot[englime!60!black, only marks, mark=*, mark size=1.6pt] coordinates {(40,7.7)};
\node[font=\scriptsize, anchor=south west] at (axis cs:40,7.7) {Carnot $7.7$ at lift $40$};
\addplot[engorange, only marks, mark=square*, mark size=1.6pt] coordinates {(40,3.85)};
\node[font=\scriptsize, anchor=north west] at (axis cs:40,3.85) {realised $\approx 0.5\times$};
\end{axis}
\end{tikzpicture}
\caption[Carnot coefficient of performance versus temperature lift]{Carnot
heating and cooling coefficients of performance against the temperature lift
$T_h - T_c$, with the hot reservoir fixed at $T_h = 308$ K ($35\,^{\circ}$C).
Both diverge as the lift shrinks toward zero (moving heat across a tiny
temperature gap costs almost no work) and fall as the lift grows. The heating
curve sits exactly one unit above the cooling curve at every lift. A real
air-source heat pump achieves roughly half the Carnot value, the marked drop
from $7.7$ to about $3.8$ at a $40$ K lift, which is why a heat pump can still
deliver several units of heat per unit of electricity.}
\label{fig:vol04ch03:cop-vs-temperature-lift}
\end{figure}
