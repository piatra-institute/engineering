% Figure: exergy (second-law) accounting of a combined-cycle plant, as a
% percentage of the fuel exergy in. Where the first law tracks where energy
% leaves, the exergy balance tracks where work potential is destroyed. The
% combustor dominates the destruction, a fact the first law hides entirely.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xbar,
  xlabel={share of fuel exergy (\%)},
  xmin=0, xmax=45,
  ytick=data,
  symbolic y coords={
    {stack + condenser loss},
    {steam-cycle irreversibility},
    {gas-turbine irreversibility},
    {combustion irreversibility},
    {net electricity (exergy out)}},
  y dir=reverse,
  nodes near coords, nodes near coords align={horizontal},
  nodes near coords style={font=\scriptsize},
  tick label style={font=\scriptsize}, label style={font=\small},
  bar width=10pt,
  enlarge y limits=0.18,
  axis lines=left,
]
\addplot[fill=englime!55!black, draw=englime!40!black] coordinates {
  (58,{net electricity (exergy out)})
  (28,{combustion irreversibility})
  (6,{gas-turbine irreversibility})
  (4,{steam-cycle irreversibility})
  (4,{stack + condenser loss})
};
\end{axis}
\end{tikzpicture}
\caption[Exergy destruction in a combined-cycle plant]{Second-law (exergy)
accounting of a $60\,\%$-class combined-cycle plant, each bar a share of the
fuel exergy supplied. About $58$ leaves as electricity, which is nearly pure
exergy. The largest single loss is the combustion itself, near $28$: burning
fuel at a flame temperature far above the working-gas temperature destroys
work potential before any turbine touches the gas, a loss the first law cannot
see because no energy crosses the boundary there. The turbine, compressor, and
steam-cycle irreversibilities are each a few percent, and the stack and
condenser, which carry most of the first-law energy loss, carry little exergy
because they reject heat at low temperature. The lesson for the designer is
that the place to look for improvement is the combustor and the
heat-addition path, not the visibly hot stack. Figures are representative,
following the second-law analyses in the standard texts.}
\label{fig:vol04ch03:exergy-destruction}
\end{figure}
