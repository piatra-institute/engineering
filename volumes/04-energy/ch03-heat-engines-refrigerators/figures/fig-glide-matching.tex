% Figure: temperature against cumulative heat transferred (a T-Q diagram) in the
% boiler of a waste-heat cycle. The hot source cools along a sloping line as it
% gives up heat; a pure working fluid boils at constant temperature, leaving a
% wide pinch, while a zeotropic mixture boils with a temperature glide that
% tracks the source and narrows the average gap. The shaded gap is where the
% exergy is destroyed.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12.5cm, height=8cm,
  xlabel={cumulative heat transferred $Q$ (\% of duty)},
  ylabel={temperature $T$ ($^{\circ}$C)},
  xmin=0, xmax=100, ymin=20, ymax=200,
  axis lines=left,
  legend pos=north east, legend cell align=left,
  legend style={font=\scriptsize},
  tick label style={font=\scriptsize}, label style={font=\small},
  clip=false,
]
% Source cools from 190 (at Q=0, the hot end) to 90 (at Q=100, cold end).
\addplot[engred, very thick, samples=2, domain=0:100] {190 - 1.0*x};
\addlegendentry{hot source (cooling)}
% Pure fluid: liquid preheat from 40 up to boiling at 150, then boil flat at 150.
% Preheat portion Q 0..25 (cold end of exchanger, low T), boiling 25..100 flat.
% Plot as the cold stream heating from the cold (right) end to hot (left) end;
% show it in the same left-to-right Q as the source for readability: pure fluid
% takes heat and its temperature rises from 40 (at Q=0 here means start of heating)
% To keep both on the same Q axis (heat given up = heat taken up), draw the cold
% stream rising left to right from 40 to its boiling plateau at 150.
\addplot[engblue, very thick, samples=2, domain=0:30] {40 + (110/30)*x};
\addplot[engblue, very thick, samples=2, domain=30:100] {150};
\addlegendentry{pure fluid (isothermal boil)}
% Zeotropic mixture: preheat then a gliding boil from 120 up to 165, tracking source.
\addplot[englime!55!black, very thick, dashed, samples=2, domain=0:22] {40 + (80/22)*x};
\addplot[englime!55!black, very thick, dashed, samples=2, domain=22:100]
  {120 + (60/78)*(x-22)};
\addlegendentry{zeotropic mixture (gliding boil)}
% pinch annotation for pure fluid
\draw[enggray, <->] (axis cs:70,150) -- (axis cs:70,120);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:71,135) {pinch};
\end{axis}
\end{tikzpicture}
\caption[Temperature-heat diagram of glide matching in a waste-heat boiler]{The
boiler of a waste-heat recovery cycle, drawn as temperature against the heat
already transferred. The hot source gives up its heat and cools along the upper
sloping line. A pure working fluid preheats as a liquid and then boils at one
fixed temperature, so its curve runs flat through the boiling region and opens a
wide temperature gap, the pinch, between itself and the still-cooling source; the
heat crossing that gap is transferred irreversibly and its work potential is
lost. A zeotropic mixture boils across a range of temperatures, a glide, and its
curve can be chosen to slope with the source, narrowing the average gap and
recovering more of the work potential. Matching the glide of the working fluid to
the slope of the source is the design idea behind the Kalina cycle and behind
mixture refrigerants, and it is the same average-temperature lever every Rankine
refinement pulls, applied to the shape of the heat-addition curve rather than to
its endpoints.}
\label{fig:vol04ch03:glide-matching}
\end{figure}
