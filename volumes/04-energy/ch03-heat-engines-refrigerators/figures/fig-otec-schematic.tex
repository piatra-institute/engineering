% Figure: ocean thermal energy conversion (OTEC), closed-cycle. A small ammonia
% Rankine loop between warm surface water (~26 C) and cold deep water (~5 C),
% drawn as a four-component loop with the two water streams entering the
% evaporator and condenser. The tiny temperature span is the whole story.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  box/.style={draw, engblack, thick, minimum width=2.2cm, minimum height=1.0cm,
    align=center, font=\scriptsize},
  wlabel/.style={font=\scriptsize, engblue},
  clabel/.style={font=\scriptsize, engred},
]
% four components of the closed loop
\node[box] (evap) at (0,0) {evaporator};
\node[box] (turb) at (4,0) {turbine \\ (net work)};
\node[box] (cond) at (4,-2.6) {condenser};
\node[box] (pump) at (0,-2.6) {pump};
% working-fluid loop (ammonia)
\draw[engamber, very thick, -{Latex}] (evap.east) -- (turb.west)
  node[midway, above, font=\scriptsize]{NH$_3$ vapour};
\draw[engamber, very thick, -{Latex}] (turb.south) -- (cond.north);
\draw[engamber, very thick, -{Latex}] (cond.west) -- (pump.east)
  node[midway, below, font=\scriptsize]{NH$_3$ liquid};
\draw[engamber, very thick, -{Latex}] (pump.north) -- (evap.south);
% warm surface water into evaporator
\draw[engblue, very thick, -{Latex}] (-3.0,0.35) -- (evap.west |- 0,0.35)
  node[pos=0, left, wlabel]{warm surface};
\draw[engblue, very thick, -{Latex}] (evap.west |- 0,-0.35) -- (-3.0,-0.35)
  node[pos=1, left, wlabel]{$26\,^{\circ}$C};
% cold deep water into condenser
\draw[engred, very thick, -{Latex}] (7.0,-2.25) -- (cond.east |- 0,-2.25)
  node[pos=0, right, clabel]{cold deep};
\draw[engred, very thick, -{Latex}] (cond.east |- 0,-2.95) -- (7.0,-2.95)
  node[pos=1, right, clabel]{$5\,^{\circ}$C};
% Carnot annotation
\node[font=\scriptsize, align=center, enggray] at (2,-4.4)
  {Carnot bound $1 - 278/299 = 7.0\%$; realised $\approx 3\%$};
\end{tikzpicture}
\caption[Closed-cycle ocean thermal energy conversion schematic]{A closed-cycle
ocean thermal energy conversion plant runs an ammonia Rankine loop between warm
surface seawater near $26\,^{\circ}$C and cold water pumped from a depth of
roughly a kilometre near $5\,^{\circ}$C. The warm water boils the ammonia in the
evaporator; the vapour drives the turbine; the cold water condenses the ammonia;
the pump returns it. The temperature span of about $21\,^{\circ}$C is the whole
resource, so the Carnot bound is only about seven percent and a real plant
reaches roughly three percent after the large pumping power for the two enormous
seawater flows is subtracted. The appeal is not efficiency but that the resource
is vast and steady across the tropical oceans, and that the same cold deep water
serves district cooling and aquaculture, which is what carries the economics that
the low cycle efficiency alone would sink.}
\label{fig:vol04ch03:otec-schematic}
\end{figure}
