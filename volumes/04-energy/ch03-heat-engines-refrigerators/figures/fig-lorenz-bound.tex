% Figure: the Lorenz cycle bound for a refrigerator working against sensible
% (gliding) streams rather than isothermal reservoirs. When the cold load and the
% heat sink are finite streams that change temperature as heat is exchanged, the
% relevant reversible bound uses the log-mean temperatures of the two streams, not
% their endpoint values. The plot compares the Carnot COP computed from stream
% inlet temperatures (optimistic) against the Lorenz COP from log-mean
% temperatures (the honest bound) as the sink glide widens.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={heat-sink temperature glide $\Delta T_h$ (K)},
  ylabel={reversible COP (cooling)},
  xmin=0, xmax=40, ymin=2, ymax=9,
  axis lines=left,
  legend pos=north east, legend cell align=left,
  tick label style={font=\scriptsize}, label style={font=\small},
  legend style={font=\scriptsize}, clip=false,
]
% Fixed cold stream: evaporator sees a load fixed at T_c = 273 K (isothermal here).
% Sink stream enters at 300 K and leaves at 300 + dTh; endpoint Carnot uses the
% inlet 300 K: COP_endpoint = 273/(300 - 273) = 10.1 ... too high; use a warmer
% cold side. Let cold load log-mean = 275, sink inlet = 305.
% Endpoint (optimistic) Carnot: uses sink INLET 305: COP = 275/(305-275)=9.17 flat.
\addplot[engblue, very thick, samples=2, domain=0:40] {275/(305-275)};
\addlegendentry{Carnot from sink inlet (optimistic)}
% Lorenz: uses sink LOG-MEAN temperature. As glide dTh grows, log-mean rises
% above the inlet, so the effective lift grows and the COP falls.
% sink log-mean T_lm = dTh / ln((305+dTh)/305), evaluated with inlet 305.
% For dTh->0, T_lm -> 305. COP_L = 275/(T_lm - 275).
\addplot[engred, very thick, samples=120, domain=0.5:40]
  {275/( (x/ln((305+x)/305)) - 275)};
\addlegendentry{Lorenz from sink log-mean (honest)}
% marker at dTh = 20
\addplot[engblack, only marks, mark=*, mark size=1.7pt]
  coordinates {(20, {275/((20/ln(325/305)) - 275)})};
\node[engblack, font=\scriptsize, anchor=west] at (axis cs:21,4.6)
  {$20\,\mathrm{K}$ glide};
\end{axis}
\end{tikzpicture}
\caption[Lorenz bound for a gliding heat sink]{Reversible cooling coefficient of
performance against the temperature glide of the heat sink, for a refrigerator
rejecting heat to a finite stream that warms as it carries the heat away. The flat
upper line is the Carnot value computed from the sink's inlet temperature, which a
designer reaching for the most favourable number would quote. The falling curve is
the Lorenz value, computed from the sink's log-mean temperature, which is the
honest reversible bound when the sink is a sensible stream rather than an infinite
reservoir. As the glide widens the log-mean temperature climbs above the inlet, the
effective temperature lift grows, and the true bound falls well below the endpoint
Carnot figure. The gap is the reason a chiller sized against its cooling-water
inlet temperature disappoints in service once the water has warmed across the
condenser, and it is the same average-temperature accounting the gliding-source
power cycles turned on, read for a refrigerator instead of an engine.}
\label{fig:vol04ch03:lorenz-bound}
\end{figure}
