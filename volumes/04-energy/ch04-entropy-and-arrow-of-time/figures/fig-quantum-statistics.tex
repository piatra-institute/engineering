% Figure: the three occupation statistics. Mean occupation number of a
% single-particle state of energy epsilon, measured from the chemical
% potential mu, plotted against (epsilon - mu)/(k_B T). Bose-Einstein diverges
% as the energy approaches the chemical potential; Fermi-Dirac saturates at
% one; Maxwell-Boltzmann is the common high-energy limit of both. The vertical
% line at zero marks the chemical potential.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={reduced energy $(\epsilon-\mu)/k_B T$},
  ylabel={mean occupation $\langle n\rangle$},
  xmin=-3, xmax=5, ymin=0, ymax=3,
  axis lines=left,
  legend pos=north east, legend cell align=left,
  legend style={font=\scriptsize, draw=enggray!40},
  tick label style={font=\scriptsize},
  label style={font=\small},
  smooth,
]
% Bose-Einstein: 1/(e^x - 1), valid for x > 0
\addplot[engblue, very thick, domain=0.12:5, samples=160]
  {1/(exp(x)-1)};
\addlegendentry{Bose-Einstein $1/(e^{x}-1)$}
% Fermi-Dirac: 1/(e^x + 1)
\addplot[engred, very thick, domain=-3:5, samples=160]
  {1/(exp(x)+1)};
\addlegendentry{Fermi-Dirac $1/(e^{x}+1)$}
% Maxwell-Boltzmann: e^{-x}
\addplot[engamber, very thick, domain=-3:5, samples=160]
  {exp(-x)};
\addlegendentry{Maxwell-Boltzmann $e^{-x}$}
% chemical potential
\draw[enggray, dotted] (axis cs:0,0) -- (axis cs:0,3);
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:0.1,2.6)
  {$\epsilon=\mu$};
\end{axis}
\end{tikzpicture}
\caption[The three occupation statistics: Bose-Einstein, Fermi-Dirac, and
Maxwell-Boltzmann]{Mean occupation of a single-particle state of energy
$\epsilon$ against its distance from the chemical potential $\mu$, in units of
$k_B T$. The Fermi-Dirac curve never exceeds one, the Pauli exclusion built
into the statistics, and at $\epsilon=\mu$ it passes through one half. The
Bose-Einstein curve climbs without bound as the energy approaches the chemical
potential from above, the crowding that condenses bosons into the ground state.
Far above the chemical potential, where the states are sparsely filled, both
quantum curves collapse onto the classical Maxwell-Boltzmann exponential
$e^{-(\epsilon-\mu)/k_B T}$, which is why the classical count of section~4.2
holds whenever the occupation is small. The three statistics differ only where
the occupation approaches one, the dilute-versus-dense boundary that separates
a classical gas from a quantum one.}
\label{fig:vol04ch04:quantum-statistics}
\end{figure}
