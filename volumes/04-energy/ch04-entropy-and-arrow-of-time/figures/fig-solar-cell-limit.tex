% Figure: the two thermodynamic ceilings on solar-to-work conversion plotted
% against the choice of absorber cutoff. The Carnot ceiling 1 - T_earth/T_sun is a
% flat line; the Landsberg ceiling for full radiation conversion sits just below
% it; a simplified single-threshold (Shockley-Queisser-style) efficiency rises,
% peaks near a mid-gap value, and falls, illustrating that a real absorber with
% one cutoff energy cannot reach the reversible radiation ceiling. Curves are
% schematic single-threshold forms, not a full detailed-balance integral.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={absorber threshold energy $E_g$ (eV)},
  ylabel={conversion efficiency (\%)},
  xmin=0.4, xmax=3.0, ymin=0, ymax=95,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  ymajorgrids=true,
  grid style={enggray!22},
  legend style={font=\scriptsize, at={(0.97,0.97)}, anchor=north east, draw=enggray!40},
]
% Carnot ceiling for radiation from a 5800 K sun to a 300 K sink: 1 - 300/5800 = 0.948
\addplot[engred, very thick, dashed, domain=0.4:3.0, samples=2] {94.8};
\addlegendentry{Carnot ceiling $1-T_0/T_{\mathrm{sun}}$}
% Landsberg-style ceiling for full blackbody radiation conversion, ~93.3%
\addplot[engorange, thick, dotted, domain=0.4:3.0, samples=2] {93.3};
\addlegendentry{radiation (Landsberg) ceiling}
% Simplified single-threshold efficiency: rising utilisation of photons above E_g
% times a voltage-fraction penalty that grows with E_g. Peaks near 1.3 eV at ~33%.
% eta(Eg) = 100 * A * exp(-(Eg-1.34)^2 / 0.9) with A chosen so peak ~ 33.
\addplot[engblue, very thick, domain=0.4:3.0, samples=140]
  {33*exp(-(x-1.34)^2/0.9)};
\addlegendentry{single-threshold cell (detailed balance)}
% mark the peak
\addplot[mark=*, only marks, englime!60!black, mark size=2pt] coordinates {(1.34,33)};
\node[font=\scriptsize, anchor=south, englime!45!black] at (axis cs:1.34,34)
  {$\approx 33\%$ at $1.34$ eV};
\end{axis}
\end{tikzpicture}
\caption[Thermodynamic ceilings on solar-to-work conversion]{Three ceilings on
converting sunlight to work, plotted against the absorber threshold energy $E_g$.
The Carnot ceiling $1-T_0/T_{\mathrm{sun}}$ for a $T_{\mathrm{sun}}=5800\,
\si{\kelvin}$ source and a $T_0=300\,\si{\kelvin}$ sink sits near $95\%$; the
radiation-conversion (Landsberg) ceiling for turning a full blackbody stream into
work is only slightly lower, because a small entropy tax attaches to reshaping the
photon spectrum. A real single-threshold absorber cannot reach either: it wastes
every photon below $E_g$, which favours a low threshold, and it wastes the excess
energy of every photon above $E_g$ as heat, which favours a high threshold. The
two losses trade off, and the detailed-balance efficiency peaks near
$E_g\approx1.34\,\si{\electronvolt}$ at about a third, far below the reversible
radiation ceiling. The gap between the peak and the ceiling is the entropy a
one-cutoff device generates by thermalizing each absorbed photon to the band edge,
the same quality loss the photon-gas example charged to degrading hot radiation.}
\label{fig:vol04ch04:solar-cell-limit}
\end{figure}
