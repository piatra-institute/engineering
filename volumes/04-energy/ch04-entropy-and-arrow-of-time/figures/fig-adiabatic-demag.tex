% Figure: entropy of a spin-1/2 paramagnet against temperature at two magnetic
% field strengths, and the two-step cooling cycle of adiabatic demagnetization.
% At high field the spins align, so the spin entropy is suppressed; at low field
% the spins are free and carry the full k_B ln 2 per spin. Isothermal
% magnetization (vertical drop at fixed T) sheds spin entropy to the bath; the
% following adiabatic demagnetization (horizontal move at fixed S) slides down
% the low-field curve to a much lower temperature. The staircase reaches toward
% but never touches T = 0, the third law drawn as geometry.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7.2cm,
  xlabel={temperature $T$ (arbitrary units, $\propto T/B_{\mathrm{ref}}$)},
  ylabel={spin entropy $S/(N k_B)$},
  xmin=0, xmax=4, ymin=0, ymax=0.8,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  ymajorgrids=true,
  grid style={enggray!22},
  legend style={font=\scriptsize, at={(0.97,0.05)}, anchor=south east, draw=enggray!40},
]
% two-level spin entropy per spin: s(x) = ln(1+e^{-1/x}) + (1/x) e^{-1/x}/(1+e^{-1/x})
% with x = k_B T /(mu B). Low field: small effective gap -> curve rises fast.
% High field: large gap -> curve rises slowly. Model by scaling x.
% Low field (B small): argument scaled so gap is small -> use y = ln2 * T/(T+0.35)
\addplot[engblue, very thick, domain=0.001:4, samples=200]
  {0.6931*x/(x+0.35)};
\addlegendentry{low field $B_{\mathrm{low}}$}
% High field (B large): gap larger -> entropy suppressed at same T
\addplot[engred, very thick, dashed, domain=0.001:4, samples=200]
  {0.6931*x/(x+1.4)};
\addlegendentry{high field $B_{\mathrm{high}}$}
% Step 1: isothermal magnetization at T = 2.6 (vertical drop from blue to red)
\draw[englime!60!black, very thick, -{Latex[length=2.2mm]}]
  (axis cs:2.6,{0.6931*2.6/(2.6+0.35)}) -- (axis cs:2.6,{0.6931*2.6/(2.6+1.4)});
\node[font=\scriptsize, anchor=west, englime!45!black] at (axis cs:2.62,0.55)
  {1. isothermal magnetize};
% Step 2: adiabatic demagnetization at fixed S (horizontal move from red back to blue)
\pgfmathsetmacro{\Sfixed}{0.6931*2.6/(2.6+1.4)}
% find T on low-field curve with same S: solve 0.6931*T/(T+0.35)=Sfixed
\pgfmathsetmacro{\Tcold}{0.35*\Sfixed/(0.6931-\Sfixed)}
\draw[engorange, very thick, -{Latex[length=2.2mm]}]
  (axis cs:2.6,{\Sfixed}) -- (axis cs:{\Tcold},{\Sfixed});
\node[font=\scriptsize, anchor=south, engamber] at (axis cs:1.4,{\Sfixed+0.01})
  {2. adiabatic demagnetize};
\addplot[mark=*, only marks, engorange, mark size=1.8pt] coordinates {({\Tcold},{\Sfixed})};
\node[font=\scriptsize, anchor=north, engorange] at (axis cs:{\Tcold},{\Sfixed-0.02})
  {$T_{\mathrm{cold}}$};
\end{axis}
\end{tikzpicture}
\caption[Adiabatic demagnetization on the spin entropy--temperature plane]{Spin
entropy of a paramagnet against temperature at a low and a high magnetic field.
A stronger field aligns the spins and pushes the entropy curve to the right, so
at a fixed temperature the high-field state carries less spin entropy than the
low-field state. The cooling cycle is two strokes. First, magnetize at constant
temperature (vertical arrow): the spins align, spin entropy drops, and the shed
entropy leaves as heat to a bath held at the starting temperature. Second,
demagnetize adiabatically (horizontal arrow): with no heat exchanged the total
entropy is fixed, so the state slides left along the constant-entropy line back
onto the low-field curve, and the temperature falls to $T_{\mathrm{cold}}$. Each
repetition of the staircase reaches a lower temperature, and because the two
curves converge as $T \to 0$ the steps shrink, so no finite number of them
reaches absolute zero, the operational third law of
section~\ref{vol04:ch04}.}
\label{fig:vol04ch04:adiabatic-demag}
\end{figure}
