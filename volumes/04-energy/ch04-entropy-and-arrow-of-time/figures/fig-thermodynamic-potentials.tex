% Figure: the four thermodynamic potentials and the Legendre transforms that
% link them. Internal energy U(S,V) sits at the centre; each Legendre transform
% swaps one natural variable for its conjugate, trading an extensive variable for
% the intensive force conjugate to it. Enthalpy H = U + pV swaps V for p;
% Helmholtz F = U - TS swaps S for T; Gibbs G = U - TS + pV swaps both. The
% arrows are labelled with the term added or subtracted in each transform.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  pot/.style={draw=engblue, fill=engblue!8, rounded corners,
    minimum width=3.0cm, minimum height=1.5cm, align=center, font=\small},
  trans/.style={->, >=stealth, engred, thick, font=\scriptsize},
]
% the four potentials at the corners of a square
\node[pot] (U) at (0,0)   {Internal energy\\$U(S,V)$};
\node[pot] (H) at (6,0)   {Enthalpy\\$H(S,p)$};
\node[pot] (F) at (0,-4)  {Helmholtz\\$F(T,V)$};
\node[pot] (G) at (6,-4)  {Gibbs\\$G(T,p)$};
% Legendre transforms along the edges
\draw[trans] (U) -- (H) node[midway, above]{$+\,pV$ \quad (swap $V\!\to\!p$)};
\draw[trans] (U) -- (F) node[midway, left, align=right]{$-\,TS$\\(swap $S\!\to\!T$)};
\draw[trans] (H) -- (G) node[midway, right, align=left]{$-\,TS$\\(swap $S\!\to\!T$)};
\draw[trans] (F) -- (G) node[midway, below]{$+\,pV$ \quad (swap $V\!\to\!p$)};
% differentials beneath each box
\node[enggray, font=\scriptsize, below=0.05cm of U] {$dU = T\,dS - p\,dV$};
\node[enggray, font=\scriptsize, below=0.05cm of H] {$dH = T\,dS + V\,dp$};
\node[enggray, font=\scriptsize, below=0.05cm of F] {$dF = -S\,dT - p\,dV$};
\node[enggray, font=\scriptsize, below=0.05cm of G] {$dG = -S\,dT + V\,dp$};
\end{tikzpicture}
\caption[The four thermodynamic potentials and the Legendre transforms linking
them]{The four thermodynamic potentials and the Legendre transforms that connect
them. The internal energy $U(S,V)$ carries the two extensive natural variables;
each Legendre transform trades one extensive variable for the intensive force
conjugate to it, adding the product of the conjugate pair. Adding $pV$ swaps the
volume for the pressure and gives the enthalpy $H(S,p)$; subtracting $TS$ swaps
the entropy for the temperature and gives the Helmholtz free energy $F(T,V)$;
doing both gives the Gibbs free energy $G(T,p)$. Each potential is the energy
that is minimised at equilibrium when its own natural variables are held fixed, so
the choice of potential follows the choice of what the experiment or the plant
holds constant: $U$ at fixed $S$ and $V$, $H$ at fixed $S$ and $p$, $F$ at fixed
$T$ and $V$, $G$ at fixed $T$ and $p$. The differential under each box reads off
the partial derivatives directly, and the equality of the mixed second
derivatives of each differential is one Maxwell relation.}
\label{fig:vol04ch04:thermodynamic-potentials}
\end{figure}
