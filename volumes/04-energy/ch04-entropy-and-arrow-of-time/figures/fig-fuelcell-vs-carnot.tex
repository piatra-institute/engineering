% Figure: the thermodynamic efficiency ceilings of a hydrogen fuel cell and a
% heat engine, plotted against operating temperature. The fuel cell's ideal
% efficiency is the ratio of Gibbs free energy to enthalpy of reaction, which
% falls slowly with temperature; the heat engine's Carnot ceiling rises with the
% hot-side temperature against a fixed cold sink. The two curves cross: below the
% crossover the fuel cell's electrochemical route beats the Carnot ceiling, which
% is the point of the case study. A real fuel cell runs well under its own
% ceiling, marked by the lower band.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.4cm,
  xlabel={operating temperature $T$ ($\si{\kelvin}$)},
  ylabel={ideal efficiency},
  xmin=300, xmax=1200, ymin=0, ymax=1.0,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.97,0.97)}, anchor=north east, draw=enggray!40},
  ymajorgrids=true,
  grid style={enggray!20},
  clip=false,
]
% fuel-cell ideal efficiency eta = Delta G / Delta H, with Delta G(T) approx
% -237 + (T-298)*0.163 kJ/mol (liquid-water basis) and Delta H approx -286 kJ/mol.
% eta = (237 - 0.163*(T-298))/286, a slowly falling line.
\addplot[engblue, very thick, domain=300:1200, samples=80]
  {(237 - 0.163*(x-298))/286};
\addlegendentry{fuel cell, $\Delta G/\Delta H$}
% Carnot ceiling against a fixed cold sink at 320 K
\addplot[engred, very thick, domain=320:1200, samples=120] {1 - 320/x};
\addlegendentry{heat engine, $1 - T_C/T$}
% real-cell band: about 0.6 of the ideal, flat near room temperature
\addplot[engorange, thick, dashed, domain=300:600, samples=40]
  {0.6*(237 - 0.163*(x-298))/286};
\addlegendentry{real cell ($\approx 0.6$ of ideal)}
% mark the crossover (where 1 - 320/x = (237-0.163(x-298))/286), near T = 750 K
\draw[enggray, dashed] (axis cs:750,0) -- (axis cs:750,0.70);
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:750,0.70)
  {crossover};
\end{axis}
\end{tikzpicture}
\caption[Fuel-cell and heat-engine efficiency ceilings versus temperature]{The
ideal efficiency ceilings of two routes from hydrogen to work, drawn against
operating temperature. The fuel cell converts the reaction's Gibbs free energy
directly to electrical work, so its ceiling is $\Delta G/\Delta H$, the blue line,
which starts near $0.83$ at room temperature and falls slowly as the $-T\Delta S$
term eats into $\Delta G$. The heat engine first burns the hydrogen to heat and
then runs a cycle, so its ceiling is the Carnot factor $1 - T_C/T$ against a fixed
cold sink, the red line, which rises from zero only as the hot-side temperature
climbs. At low and moderate temperatures the fuel cell's ceiling lies far above the
Carnot ceiling: the electrochemical route is not bound by Carnot because it never
routes the energy through heat. The two ceilings cross near $750\,\si{\kelvin}$,
above which a hot engine can in principle match the cell. The dashed orange band is
where a real low-temperature cell operates, roughly six-tenths of its own ideal
ceiling after activation, ohmic, and concentration losses, still comfortably above
the Carnot line for the same temperature. Values use the liquid-water reaction
basis, $\Delta H \approx -286\,\text{kJ}/\text{mol}$ and $\Delta G \approx
-237\,\text{kJ}/\text{mol}$ at $298\,\si{\kelvin}$.}
\label{fig:vol04ch04:fuelcell-vs-carnot}
\end{figure}
