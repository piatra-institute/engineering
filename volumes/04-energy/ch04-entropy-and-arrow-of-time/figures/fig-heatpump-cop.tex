% Figure: heating coefficient of performance of an ideal (Carnot) heat pump
% delivering heat to a fixed indoor temperature, plotted against the outdoor
% source temperature. The Carnot heating COP is T_H / (T_H - T_C), which
% diverges as the source approaches the delivery temperature and collapses
% toward one as the source falls. A real air-source unit at a fixed fraction
% of the Carnot value is plotted alongside to show the seasonal penalty.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={outdoor source temperature $T_C$ ($\si{\kelvin}$)},
  ylabel={heating coefficient of performance},
  xmin=250, xmax=290, ymin=0, ymax=16,
  axis lines=left,
  legend pos=north west, legend cell align=left,
  legend style={font=\scriptsize, draw=enggray!40},
  tick label style={font=\scriptsize},
  label style={font=\small},
  smooth,
]
% ideal Carnot heating COP: T_H / (T_H - T_C) with T_H = 293 K
\addplot[engblue, very thick, domain=250:291, samples=120]
  {293/(293-x)};
\addlegendentry{ideal (Carnot) $\mathrm{COP}_H = T_H/(T_H-T_C)$}
% real unit at 45 percent of Carnot
\addplot[engred, very thick, domain=250:291, samples=120]
  {0.45*293/(293-x)};
\addlegendentry{real air-source unit ($0.45\,\mathrm{COP}_\mathrm{Carnot}$)}
% resistance-heating floor at COP = 1
\addplot[enggray, dashed, thick, domain=250:290, samples=2] {1};
\addlegendentry{resistance heater, $\mathrm{COP}=1$}
% mark T_H = 293 delivery temperature
\draw[enggray, dotted] (axis cs:293,0) -- (axis cs:293,16);
\node[enggray, font=\scriptsize, anchor=north east, rotate=90] at (axis cs:290,15)
  {delivery $T_H=293\,\si{\kelvin}$};
\end{axis}
\end{tikzpicture}
\caption[Heating coefficient of performance of a heat pump against source
temperature]{The heating coefficient of performance of an ideal heat pump
delivering warmth at a fixed indoor $T_H=293\,\si{\kelvin}$, drawn against the
outdoor source temperature $T_C$. The Carnot ceiling $\mathrm{COP}_H =
T_H/(T_H-T_C)$ rises without bound as the source warms toward the delivery
temperature, because a vanishing lift needs vanishing work, and falls toward
one as the source cools and the lift widens. A real air-source unit runs at a
fraction of the ceiling, here $0.45$, and its curve sits well below the ideal;
on the coldest days its coefficient of performance approaches the resistance
heater's floor of one, which is why cold climates strain air-source machines
and why a ground source, whose temperature holds near the annual mean, keeps
the lift small all winter. The heat pump always delivers more heat than the
work it draws, because it pumps ambient heat uphill rather than dissipating
work, and the second law caps the multiplier at the Carnot value.}
\label{fig:vol04ch04:heatpump-cop}
\end{figure}
