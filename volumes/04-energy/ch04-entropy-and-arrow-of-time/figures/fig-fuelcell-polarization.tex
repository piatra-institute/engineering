% Figure: the polarization curve of a hydrogen fuel cell. The cell voltage falls
% from the reversible (Nernst) value as current is drawn, through three loss
% regimes: activation loss at low current, ohmic loss across the middle, and
% concentration (mass-transport) loss at high current. The gap between the
% reversible voltage and the operating voltage at a given current is the
% irreversibility, the second-law cost the case study audits.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.2cm,
  xlabel={current density $j$ ($\text{A}/\text{cm}^2$)},
  ylabel={cell voltage (V)},
  xmin=0, xmax=1.6, ymin=0, ymax=1.35,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.97,0.97)}, anchor=north east, draw=enggray!40},
  ymajorgrids=true,
  grid style={enggray!20},
  clip=false,
]
% reversible (thermoneutral-below) Nernst voltage, drawn flat at 1.23 V
\addplot[enggray, thick, dashed, domain=0:1.6, samples=2] {1.23};
\addlegendentry{reversible voltage $E_{\mathrm{rev}} = 1.23\,\text{V}$}
% operating polarization curve: E = 1.23 - b*ln((j+j0)/j0) - R*j - c*ln(jL/(jL-j))
% with b=0.05, j0=0.001, R=0.25, c=0.05, jL=1.5
\addplot[engblue, very thick, domain=0.0:1.45, samples=140]
  {1.23 - 0.05*ln((x+0.001)/0.001) - 0.25*x - 0.05*ln(1.5/(1.5-x))};
\addlegendentry{operating voltage}
% loss-region annotations
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:0.02,0.78)
  {activation};
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:0.7,0.55)
  {ohmic};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:1.42,0.25)
  {concentration};
% mark a typical operating point
\draw[engred, thick] (axis cs:0.7,0) -- (axis cs:0.7,{1.23 - 0.05*ln((0.7+0.001)/0.001) - 0.25*0.7 - 0.05*ln(1.5/(1.5-0.7))});
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.72,0.95)
  {operating point};
\end{axis}
\end{tikzpicture}
\caption[Fuel-cell polarization curve and its loss regimes]{The polarization
curve of a hydrogen fuel cell: cell voltage against current density. The dashed
grey line is the reversible voltage $E_{\mathrm{rev}} = \Delta G/(nF) =
1.23\,\text{V}$, the open-circuit value the cell would hold if it drew no current
and generated no entropy. The blue curve is the voltage the cell actually delivers
once current flows, and the gap between the two is the irreversibility, the loss of
work potential the second law charges. Three regimes set the shape: at low current
the curve falls steeply through the \emph{activation} overpotential, the kinetic
price of driving the electrode reactions; across the middle it falls linearly
through the \emph{ohmic} loss, the resistance of the electrolyte and the contacts;
near the limiting current it plunges through the \emph{concentration} loss, where
reactant cannot reach the electrode fast enough. The red marker is a representative
operating point at $0.7\,\text{A}/\text{cm}^2$; the voltage there divided by
$1.23\,\text{V}$ is the cell's voltage efficiency, and the missing fraction is the
exergy destroyed inside the cell. The curve is the electrochemical mirror of the
turbine's isentropic-efficiency gap in the steam-plant audit: the reversible
benchmark sits at the top, the real device runs below it, and the distance between
is the second-law cost.}
\label{fig:vol04ch04:fuelcell-polarization}
\end{figure}
