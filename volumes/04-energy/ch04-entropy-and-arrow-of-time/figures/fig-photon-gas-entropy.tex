% Figure: energy density and entropy density of blackbody radiation against
% temperature. The photon gas has u = a T^4 (Stefan-Boltzmann) and, from
% integrating du = T ds at fixed volume, s = (4/3) a T^3. Both are plotted on a
% shared temperature axis with a logarithmic vertical scale so the T^4 and T^3
% power laws read off as parallel straight lines of slopes 4 and 3. The lesson:
% radiation carries entropy, and high-temperature radiation carries far more
% energy per unit entropy than low-temperature radiation, which is why sunlight
% is a low-entropy energy source and the Earth's outgoing infrared is a
% high-entropy sink.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogyaxis}[
  width=11cm, height=7cm,
  xlabel={temperature $T$ (units of $1000\,\si{\kelvin}$)},
  ylabel={density (arbitrary log units)},
  xmin=1, xmax=6, ymin=1, ymax=2e3,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  ymajorgrids=true,
  grid style={enggray!22},
  legend style={font=\scriptsize, at={(0.03,0.97)}, anchor=north west, draw=enggray!40},
]
% energy density u = a T^4, take a = 1 in these units, x in units of 1000 K
\addplot[engblue, very thick, domain=1:6, samples=80] {x^4};
\addlegendentry{energy density $u \propto T^4$}
% entropy density s = (4/3) a T^3
\addplot[engred, very thick, dashed, domain=1:6, samples=80] {1.3333*x^3};
\addlegendentry{entropy density $s \propto T^3$}
% mark the ratio u/s = (3/4) T growing linearly: annotate two points
\addplot[mark=*, only marks, englime!60!black, mark size=1.8pt]
  coordinates {(6,{6^4}) (6,{1.3333*6^3})};
\node[font=\scriptsize, anchor=west, englime!45!black] at (axis cs:4.1,900)
  {$u/s = \tfrac34 k_B^{-1}\,T$};
\end{semilogyaxis}
\end{tikzpicture}
\caption[Energy and entropy density of a photon gas]{Energy density and entropy
density of blackbody radiation against temperature, on a logarithmic vertical
axis. The energy density follows the Stefan-Boltzmann law $u = a T^4$ and the
entropy density follows $s = \tfrac43 a T^3$, so on log axes they are straight
lines of slope four and slope three. Their ratio $u/s = \tfrac34 T$ (up to
$k_B$) rises with temperature, which is the working statement that hot radiation
packs more energy per unit entropy than cold radiation. Sunlight arriving from a
$5800\,\si{\kelvin}$ photosphere is a high-quality, low-entropy energy stream;
the infrared the Earth radiates to a $255\,\si{\kelvin}$ sky carries the same
energy back out at far higher entropy. The gap between the two entropies is the
planetary entropy the biosphere and the weather system are driven by, developed
in the vignette of section~\ref{vol04:ch04}.}
\label{fig:vol04ch04:photon-gas-entropy}
\end{figure}
