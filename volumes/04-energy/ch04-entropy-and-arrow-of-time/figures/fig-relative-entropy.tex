% Figure: the relative entropy (Kullback-Leibler divergence) of a biased-coin
% distribution p = (q, 1-q) from a fair reference m = (1/2, 1/2), plotted against
% the bias q. D(p||m) = q log2(q/0.5) + (1-q) log2((1-q)/0.5) = 1 - H(q), where
% H is the binary Shannon entropy in bits. The divergence is zero only at q = 1/2
% (the maximum-entropy distribution matches the reference) and rises to 1 bit at
% the deterministic extremes. The Shannon entropy H(q) is drawn alongside as the
% mirror curve, peaking where the divergence vanishes. The lesson: the
% maximum-entropy distribution is the one closest to uniform given the
% constraints, and relative entropy measures the information gained by moving off
% it.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={bias $q$ (probability of heads)},
  ylabel={bits},
  xmin=0, xmax=1, ymin=0, ymax=1.05,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  ymajorgrids=true,
  grid style={enggray!22},
  legend style={font=\scriptsize, at={(0.5,0.03)}, anchor=south, draw=enggray!40},
]
% Shannon entropy H(q) in bits, clamp near endpoints to avoid log(0)
\addplot[engblue, very thick, domain=0.001:0.999, samples=200]
  {-(x*log2(x) + (1-x)*log2(1-x))};
\addlegendentry{Shannon entropy $H(q)$}
% relative entropy from fair reference: D = 1 - H(q)
\addplot[engred, very thick, dashed, domain=0.001:0.999, samples=200]
  {1 + x*log2(x) + (1-x)*log2(1-x)};
\addlegendentry{relative entropy $D(p\,\|\,\mathrm{uniform})$}
% mark maximum-entropy point
\addplot[mark=*, only marks, englime!60!black, mark size=2pt] coordinates {(0.5,1)};
\node[font=\scriptsize, anchor=south, englime!45!black] at (axis cs:0.5,0.86)
  {max entropy, $D=0$};
\end{axis}
\end{tikzpicture}
\caption[Shannon entropy and relative entropy of a biased coin]{The Shannon
entropy and the relative entropy of a biased coin against its bias $q$, both in
bits. The entropy $H(q) = -q\log_2 q - (1-q)\log_2(1-q)$ peaks at one bit for the
fair coin and falls to zero at the deterministic extremes, where the outcome is
certain and carries no information. The relative entropy $D(p\,\|\,\text{uniform})
= 1 - H(q)$ is its mirror: zero at $q = \tfrac12$, where the distribution equals
the uniform reference, and one bit at the extremes, the information a fair-coin
prior would have to gain to accept a certain outcome. The two curves make the
maximum-entropy principle geometric. Choosing the distribution of largest entropy
consistent with the constraints, here the single constraint that probabilities
sum to one, lands on the uniform distribution, the point where the relative
entropy from uniform is smallest. Every further constraint the modeller imposes
pulls the answer off this peak by exactly the relative entropy the constraint
costs.}
\label{fig:vol04ch04:relative-entropy}
\end{figure}
