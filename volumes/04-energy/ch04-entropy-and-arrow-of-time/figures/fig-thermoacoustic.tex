% Figure: schematic of a standing-wave thermoacoustic refrigerator. A loudspeaker
% drives a standing acoustic wave in a gas-filled resonator; a porous stack sits
% between two heat exchangers. Gas parcels oscillating along the stack are
% compressed (warming) near one end and expanded (cooling) near the other, and the
% stack shuttles heat up the temperature gradient one parcel-step at a time. Drawn
% as labelled blocks, no axes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, font=\small]
% resonator tube outline
\draw[thick, enggray!70, rounded corners=3pt] (0,-1.1) rectangle (11,1.1);
% loudspeaker (driver) at left
\filldraw[engblue!25, draw=engblue] (0,-1.1) -- (0.9,-1.1) -- (0.9,1.1) -- (0,1.1) -- cycle;
\node[font=\scriptsize, engblue!60!black, align=center] at (0.45,0) {drive\\(speaker)};
% hot heat exchanger
\filldraw[engred!25, draw=engred] (4.2,-1.05) rectangle (4.7,1.05);
\node[font=\scriptsize, engred!60!black, rotate=90] at (4.45,0) {hot HX};
% stack (porous)
\filldraw[engamber!22, draw=engamber] (4.9,-1.0) rectangle (6.9,1.0);
\foreach \y in {-0.75,-0.45,-0.15,0.15,0.45,0.75}
  \draw[engamber!70, thin] (4.95,\y) -- (6.85,\y);
\node[font=\scriptsize, engamber!60!black] at (5.9,1.35) {stack};
% cold heat exchanger
\filldraw[engblue!25, draw=engblue] (7.1,-1.05) rectangle (7.6,1.05);
\node[font=\scriptsize, engblue!60!black, rotate=90] at (7.35,0) {cold HX};
% standing-wave sketch along the tube (half wavelength)
\draw[englime!70!black, very thick, domain=0.9:11, samples=120]
  plot (\x, {0.55*sin(deg((\x-0.9)/10.1*3.14159))});
\draw[englime!70!black, very thick, dashed, domain=0.9:11, samples=120]
  plot (\x, {-0.55*sin(deg((\x-0.9)/10.1*3.14159))});
% heat flow arrows: from cold space in, up through stack, out at hot end
\draw[->, engblue, thick] (8.6,0) -- (7.7,0);
\node[font=\scriptsize, engblue!60!black, anchor=west] at (8.6,0) {$Q_C$ in};
\draw[->, engred, thick] (4.1,0) -- (3.2,0);
\node[font=\scriptsize, engred!60!black, anchor=east] at (3.2,0) {$Q_H$ out};
% temperature gradient annotation under the stack
\draw[->, enggray, thin] (4.95,-1.5) -- (6.85,-1.5);
\node[font=\scriptsize, enggray, anchor=north] at (5.9,-1.55) {temperature gradient along stack};
\end{tikzpicture}
\caption[Standing-wave thermoacoustic refrigerator]{A standing-wave
thermoacoustic refrigerator. A loudspeaker drives a half-wavelength acoustic
standing wave in a resonator filled with pressurized gas, sketched as the envelope
in green. A porous stack sits between a hot and a cold heat exchanger. A gas parcel
oscillating along the stack is compressed, and so warmed, as it moves toward the
pressure antinode and expanded, and so cooled, as it moves back; because it
deposits heat to the stack wall where it is hot and picks up heat where it is cold,
each parcel shuttles a small quantity of heat a short distance up the temperature
gradient. The bucket brigade of parcels along the stack carries heat from the cold
exchanger, absorbing $Q_C$ from the refrigerated space, to the hot exchanger,
rejecting $Q_H$ to ambient, driven entirely by the acoustic work of the speaker.
The device has no moving parts beyond the driver, and its second-law accounting is
the vapour-compression cycle's with the compressor's mechanical stroke replaced by
an acoustic one.}
\label{fig:vol04ch04:thermoacoustic}
\end{figure}
