% Figure: voltage standing-wave envelope on a mismatched line.
% |V(z)| = |V+| * |1 + Gamma e^{-2 j beta z}|; the envelope oscillates between
% |V+|(1+|Gamma|) at maxima and |V+|(1-|Gamma|) at minima. VSWR = max/min.
% Drawn for |Gamma| = 1/3 (VSWR = 2). z measured in wavelengths from the load.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xlabel={distance from load $z/\lambda$},
  ylabel={$|V(z)|/|V^{+}|$},
  xmin=0, xmax=1.5, ymin=0, ymax=1.5,
  domain=0:1.5, samples=300,
  axis lines=left,
  ytick={0.667,1.0,1.333},
  yticklabels={$1-|\Gamma|$,$1$,$1+|\Gamma|$},
  xtick={0,0.25,0.5,0.75,1.0,1.25,1.5},
  enlargelimits=false,
]
  % |1 + Gamma e^{-2 j beta z}| with Gamma = 1/3 real (short-like reference phase)
  % magnitude = sqrt( (1 + g cos(phi))^2 + (g sin phi)^2 ), phi = -4 pi z
  \addplot[engblue, very thick]
    {sqrt( (1 + 0.3333*cos(720*x))^2 + (0.3333*sin(720*x))^2 )};
  % envelope max and min
  \addplot[engred, dashed, thick, domain=0:1.5] {1.3333};
  \addplot[engred, dashed, thick, domain=0:1.5] {0.6667};
  \node[engred, font=\scriptsize, anchor=west] at (axis cs:1.5,1.3333) {max};
  \node[engred, font=\scriptsize, anchor=west] at (axis cs:1.5,0.6667) {min};
\end{axis}
\end{tikzpicture}
\caption[Voltage standing wave on a mismatched line]{The magnitude of
the voltage along a lossless line carrying a wave reflected from a
mismatched load, drawn for $|\Gamma| = 1/3$ (VSWR $= 2$). The standing
pattern oscillates between $|V^{+}|(1+|\Gamma|)$ at the voltage maxima
and $|V^{+}|(1-|\Gamma|)$ at the minima, spaced half a wavelength apart.
The voltage standing wave ratio is the ratio of these two, $\text{VSWR}
= (1+|\Gamma|)/(1-|\Gamma|)$, the quantity a bench SWR meter reports.}
\label{fig:vol04ch10:standing-wave}
\end{figure}
