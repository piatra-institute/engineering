% Figure: reflectance of a single quarter-wave anti-reflection coating on
% glass versus wavelength, compared to the bare uncoated surface. The
% coating index is the geometric mean sqrt(n_air * n_glass) and its
% optical thickness is a quarter wave at the design wavelength lambda0.
% Reflectance for a single layer (normal incidence):
%   r = (r1 + r2 e^{-2 i delta}) / (1 + r1 r2 e^{-2 i delta}),
% delta = 2 pi n_f d / lambda = (pi/2)(lambda0/lambda).
% With the geometric-mean index r1 = r2, and at lambda0 the two reflections
% cancel exactly (R = 0). We plot R(lambda) as a percentage.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={wavelength $\lambda/\lambda_0$},
  ylabel={reflectance $R$ (\%)},
  xmin=0.6, xmax=1.6, ymin=0, ymax=5,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  domain=0.6:1.6, samples=300,
]
  % bare glass, n=1.5: R = ((1-1.5)/(1+1.5))^2 = 0.04
  \addplot[enggray, thick, dashed] { 4.0 };
  \addlegendentry{bare glass ($4\%$)}
  % single quarter-wave AR coating, n_f = sqrt(1.5) = 1.2247.
  % interface amplitudes: r1 = (1 - nf)/(1 + nf), r2 = (nf - 1.5)/(nf + 1.5).
  % r1 = (1-1.2247)/(1+1.2247) = -0.1010; r2 = (1.2247-1.5)/(1.2247+1.5) = -0.1010.
  % delta = (pi/2)*(lambda0/lambda). Use degrees for cos: 2 delta in degrees.
  % R = (r1^2 + r2^2 + 2 r1 r2 cos 2delta)/(1 + r1^2 r2^2 + 2 r1 r2 cos 2delta)*100
  \addplot[engblue, very thick] {
    ( 0.0102 + 0.0102 + 2*0.0102*cos( 180*(1/x) ) )
    / ( 1 + 0.000104 + 2*0.0102*cos( 180*(1/x) ) ) * 100
  };
  \addlegendentry{quarter-wave AR coating}
  \draw[engred, dotted, thick] (axis cs:1.0,0) -- (axis cs:1.0,5);
  \node[engred, font=\scriptsize, anchor=south, rotate=90] at (axis cs:1.0,2.5) {design $\lambda_0$};
\end{axis}
\end{tikzpicture}
\caption[Single-layer anti-reflection coating]{Reflectance of a glass
surface with and without a single quarter-wave coating of the
geometric-mean index $n_f=\sqrt{n_{\text{air}}\,n_{\text{glass}}}$. At the
design wavelength the two interface reflections are equal and half a cycle
apart, so they cancel and the reflectance drops to zero; away from the
design point the cancellation is imperfect and the reflectance rises, the
reason a single-layer coating is tuned to the centre of the band and
broadband performance needs a stack of layers. The same construction, in
the impedance picture, is the quarter-wave transformer of Section 10.4.}
\label{fig:vol04ch10:ar-coating}
\end{figure}
