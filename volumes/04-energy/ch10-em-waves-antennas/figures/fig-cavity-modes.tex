% Figure: resonant frequencies of a rectangular cavity, plotting the
% lowest few TE/TM mode frequencies against the cavity's longest
% dimension. Each mode frequency is f = (c/2) sqrt((m/a)^2+(n/b)^2+(p/d)^2)
% for a cavity a x b x d. We hold a and b fixed and sweep d, showing how
% the modes crowd together as the cavity grows, the mode-density growth
% that sets the usable single-mode span of a cavity resonator.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={cavity length $d$ (mm)},
  ylabel={resonant frequency (GHz)},
  xmin=20, xmax=60, ymin=4, ymax=16,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  domain=20:60, samples=200,
]
  % a = 40 mm, b = 20 mm held fixed. c = 300 mm/ns -> c/2 = 150 GHz*mm.
  % f = 150 * sqrt( (m/40)^2 + (n/20)^2 + (p/d)^2 )  with d in mm, f in GHz.
  % TE101: m=1,n=0,p=1
  \addplot[engblue, very thick] { 150*sqrt( (1/40)^2 + (1/x)^2 ) };
  \addlegendentry{$\mathrm{TE}_{101}$}
  % TE102: m=1,n=0,p=2
  \addplot[engred, thick] { 150*sqrt( (1/40)^2 + (2/x)^2 ) };
  \addlegendentry{$\mathrm{TE}_{102}$}
  % TE201: m=2,n=0,p=1
  \addplot[engorange, thick, dashed] { 150*sqrt( (2/40)^2 + (1/x)^2 ) };
  \addlegendentry{$\mathrm{TE}_{201}$}
  % TE011-like using b: m=0,n=1,p=1
  \addplot[enggray, thick, dotted] { 150*sqrt( (1/20)^2 + (1/x)^2 ) };
  \addlegendentry{$\mathrm{TE}_{011}$}
\end{axis}
\end{tikzpicture}
\caption[Resonant frequencies of a rectangular cavity against its
length]{Resonant frequencies of a rectangular cavity of fixed
cross-section ($a = 40\,\mathrm{mm}$, $b = 20\,\mathrm{mm}$) as its
length $d$ is swept. The lowest mode, $\mathrm{TE}_{101}$, falls
steadily as the box lengthens, while higher modes track it and cross
through, so the clean span below the second mode narrows as the cavity
grows. A cavity resonator is built short enough that its wanted mode
sits well below the next one, the three-dimensional cousin of the
single-mode band of a waveguide.}
\label{fig:vol04ch10:cavity-modes}
\end{figure}
