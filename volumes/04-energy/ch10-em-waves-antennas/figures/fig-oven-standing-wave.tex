% Figure: the standing-wave intensity along one axis of a microwave-oven cavity.
% A 2.45 GHz field reflecting between two walls forms a standing wave; the heating
% rate tracks |E|^2 = sin^2(k x) with nulls a half guide-wavelength apart. For
% lambda_0 = 12.24 cm the free-space half wavelength is 6.12 cm, so cold spots sit
% about 6 cm apart, the reason a turntable is needed. We plot sin^2(k x) over a
% 30 cm span with k = 2 pi / 0.1224 m, marking the cold nulls.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=5.2cm,
  xlabel={position across cavity $x$ (cm)},
  ylabel={relative heating $\propto |E|^2$},
  xmin=0, xmax=30,
  ymin=0, ymax=1.15,
  domain=0:30, samples=400,
  axis lines=left,
  enlargelimits=false,
  xtick={0,6.12,12.24,18.36,24.48},
  xticklabel style={font=\scriptsize},
  ytick={0,0.5,1},
]
  % sin^2(k x), x in cm, k = 2 pi / 12.24 cm
  \addplot[engred, very thick] { (sin(deg(2*pi*x/12.24)))^2 };
  % cold-spot markers at the nulls x = 0, 6.12, 12.24, 18.36, 24.48
  \foreach \xn in {0,6.12,12.24,18.36,24.48} {
    \addplot[engblue, only marks, mark=*, mark size=1.6pt, forget plot]
      coordinates {(\xn,0)};
  }
  \node[engblue, font=\scriptsize, anchor=south] at (axis cs:12.24,0.06) {cold spots};
\end{axis}
\end{tikzpicture}
\caption[Standing-wave heating pattern in a microwave-oven cavity]{The
heating rate along one axis of a microwave-oven cavity, proportional to
$|E|^2 = \sin^2(kx)$ for a field reflecting between conducting walls at
$2.45\,\si{\giga\hertz}$ ($\lambda_0 = 12.24\,\si{\centi\meter}$). Nulls
of the field, where almost no heating occurs, fall a half wavelength
($6.12\,\si{\centi\meter}$) apart; food sitting at a null stays cold while
food at an antinode overheats. The turntable carries the load through the
pattern so that the time-averaged exposure smooths out, the working reason
a microwave oven rotates its contents rather than relying on a uniform
field that the standing-wave structure forbids.}
\label{fig:vol04ch10:oven-standing-wave}
\end{figure}
