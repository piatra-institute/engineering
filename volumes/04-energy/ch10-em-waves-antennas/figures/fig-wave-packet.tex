% Figure: a two-tone wave packet, the sum of two sinusoids of nearly equal
% wavenumber, showing the carrier (fast oscillation at the mean wavenumber, moving
% at the phase velocity) inside a slowly moving envelope (the beat, moving at the
% group velocity). Plotted at a single instant against position x.
% E(x) = cos(k1 x) + cos(k2 x) = 2 cos((dk/2) x) cos(k_bar x), with k_bar large,
% dk small. We take k_bar = 9, dk = 1 so the envelope half-period is pi/(dk/2)=2pi.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=5.4cm,
  xlabel={position $x$},
  ylabel={field $E$},
  xmin=0, xmax=12.566,
  ymin=-2.4, ymax=2.4,
  domain=0:12.566, samples=400,
  axis lines=left,
  enlargelimits=false,
  xtick={0,3.1416,6.2832,9.4248,12.566},
  xticklabels={$0$,$\pi$,$2\pi$,$3\pi$,$4\pi$},
  ytick={-2,-1,0,1,2},
  legend pos=north east,
  legend cell align=left,
]
  % the packet: 2 cos(0.5 x) cos(9 x)
  \addplot[engblue, thick] { 2*cos(deg(0.5*x))*cos(deg(9*x)) };
  \addlegendentry{packet}
  % the envelope +/- 2 cos(0.5 x)
  \addplot[engred, very thick, dashed] { 2*cos(deg(0.5*x)) };
  \addlegendentry{envelope}
  \addplot[engred, very thick, dashed, forget plot] { -2*cos(deg(0.5*x)) };
\end{axis}
\end{tikzpicture}
\caption[A two-tone wave packet, carrier and envelope]{The superposition
of two sinusoids of nearly equal wavenumber, $\cos(k_1 x) + \cos(k_2 x) =
2\cos(\tfrac{\Delta k}{2}x)\cos(\bar k x)$, at one instant. The fast
carrier oscillates at the mean wavenumber $\bar k$ and travels at the
phase velocity $v_p = \omega/k$; the slow envelope (dashed) oscillates at
the difference wavenumber $\Delta k/2$ and travels at the group velocity
$v_g = \dd\omega/\dd k$. A signal, which is a modulation imposed on a
carrier, rides the envelope, so energy and information move at $v_g$. In a
non-dispersive medium the two speeds coincide and the packet holds its
shape; where $\dd\omega/\dd k$ varies with $k$ the packet spreads.}
\label{fig:vol04ch10:wave-packet}
\end{figure}
