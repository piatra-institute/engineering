% Figure: E-plane radiation pattern of a half-wave dipole, drawn in a
% Cartesian axis (the pgfplots polar library is not loaded). The normalised
% field pattern is F(theta) = cos((pi/2) cos theta) / sin theta, plotted in
% the x-y plane as (F sin theta, F cos theta) so the dipole axis is vertical.
% theta is the polar angle from the dipole axis; the pattern is the familiar
% figure-of-eight with nulls along the axis (theta = 0, 180 deg).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=8.5cm, height=8.5cm,
  axis equal image,
  xlabel={broadside ($\to$)},
  ylabel={dipole axis ($\uparrow$)},
  xmin=-1.2, xmax=1.2, ymin=-1.2, ymax=1.2,
  axis lines=middle,
  xtick=\empty, ytick=\empty,
]
  % right lobe: theta from 1 deg to 179 deg (avoid the sin->0 endpoints)
  \addplot[engblue, very thick, samples=200, domain=1:179, variable=\t]
    ({ (cos(90*cos(\t))/sin(\t)) * sin(\t) },
     { (cos(90*cos(\t))/sin(\t)) * cos(\t) });
  % left lobe: mirror image (phi = 180 deg), x negated
  \addplot[engblue, very thick, samples=200, domain=1:179, variable=\t]
    ({ -(cos(90*cos(\t))/sin(\t)) * sin(\t) },
     { (cos(90*cos(\t))/sin(\t)) * cos(\t) });
  % dipole element drawn along the vertical axis
  \draw[engred, line width=2pt] (axis cs:0,-0.18) -- (axis cs:0,0.18);
  \node[engred, font=\scriptsize, anchor=west] at (axis cs:0.05,0.32) {dipole};
  % null markers along the axis
  \node[enggray, font=\scriptsize, anchor=south] at (axis cs:0,1.0) {null};
\end{axis}
\end{tikzpicture}
\caption[E-plane radiation pattern of a half-wave dipole]{The E-plane
field pattern of a centre-fed half-wavelength dipole, $F(\theta) =
\cos((\pi/2)\cos\theta)/\sin\theta$, with $\theta$ the angle from the
dipole axis (drawn vertical). Radiation peaks broadside (perpendicular
to the wire) and vanishes along the axis, giving the figure-of-eight
section of the toroidal three-dimensional pattern. The half-power
beamwidth in this plane is $78^{\circ}$ and the peak directivity is
$1.64$ ($2.15\,\si{\decibel\textsubscript{i}}$).}
\label{fig:vol04ch10:dipole-pattern}
\end{figure}
