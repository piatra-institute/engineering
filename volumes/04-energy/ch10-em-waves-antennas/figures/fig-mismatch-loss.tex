% Figure: reflected-power fraction and mismatch (return) loss versus VSWR.
% |Gamma| = (VSWR-1)/(VSWR+1); reflected fraction = |Gamma|^2 (percent);
% the second curve is the fraction of power delivered = 1 - |Gamma|^2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xlabel={VSWR},
  ylabel={power fraction (\%)},
  xmin=1, xmax=5, ymin=0, ymax=100,
  domain=1:5, samples=200,
  axis lines=left,
  legend pos=south east,
  legend cell align=left,
  enlargelimits=false,
  xtick={1,1.5,2,3,4,5},
]
  % reflected fraction = ((x-1)/(x+1))^2 * 100
  \addplot[engred, very thick] { 100*((x-1)/(x+1))^2 };
  \addlegendentry{reflected}
  % delivered fraction = (1 - ((x-1)/(x+1))^2) * 100
  \addplot[engblue, very thick] { 100*(1 - ((x-1)/(x+1))^2) };
  \addlegendentry{delivered}
  % design-target marker at VSWR 1.5
  \draw[enggray, dashed] (axis cs:1.5,0) -- (axis cs:1.5,100);
  \node[enggray, font=\scriptsize, anchor=south west] at (axis cs:1.5,55) {design target $1.5$};
\end{axis}
\end{tikzpicture}
\caption[Reflected and delivered power versus VSWR]{The fraction of
incident power reflected, $|\Gamma|^2$ with $|\Gamma| =
(\text{VSWR}-1)/(\text{VSWR}+1)$, and the fraction delivered to the
load, $1-|\Gamma|^2$, as functions of VSWR. At the common design target
VSWR $1.5$ the reflected fraction is $4\,\%$; at VSWR $2$ it is $11\,\%$;
above VSWR $3$ the reflected power begins to threaten transmitter output
stages and connector dielectrics.}
\label{fig:vol04ch10:mismatch-loss}
\end{figure}
