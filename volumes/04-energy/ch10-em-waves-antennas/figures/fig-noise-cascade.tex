% Figure: the Friis noise cascade. A three-stage receiver chain (LNA,
% mixer, IF amplifier) drawn as blocks, with each stage's noise
% contribution referred to the input plotted as a bar. The point is that
% the first stage dominates because later stages are divided by the gain
% ahead of them: T_e = T1 + T2/G1 + T3/(G1 G2).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  block/.style={draw, thick, minimum width=1.9cm, minimum height=1.0cm,
                align=center, font=\small},
  sig/.style={-{Latex[length=2.2mm]}, thick},
]
  % signal chain of three blocks
  \node[block, fill=engblue!8] (lna) at (0,0) {LNA\\$G_1{=}20$\,dB\\$T_1{=}50$\,K};
  \node[block, fill=engblue!8] (mix) at (3,0) {mixer\\$G_2{=}{-}7$\,dB\\$T_2{=}870$\,K};
  \node[block, fill=engblue!8] (ifa) at (6,0) {IF amp\\$G_3{=}30$\,dB\\$T_3{=}290$\,K};
  \draw[sig] (-2,0) -- (lna);
  \draw[sig] (lna) -- (mix);
  \draw[sig] (mix) -- (ifa);
  \draw[sig] (ifa) -- (8,0);
  \node[font=\scriptsize, anchor=south east] at (-1.0,0.1) {antenna};
  \node[font=\scriptsize, anchor=south west] at (7.1,0.1) {detector};

  % bar chart of input-referred noise contributions (kelvin)
  \begin{scope}[yshift=-3.4cm]
  \begin{axis}[
    width=9.5cm, height=4.2cm,
    ybar,
    bar width=16pt,
    xlabel={},
    ylabel={input-referred $T$ (K)},
    symbolic x coords={stage 1, stage 2, stage 3},
    xtick=data,
    ymin=0, ymax=60,
    nodes near coords,
    every node near coord/.append style={font=\scriptsize},
    axis lines=left,
  ]
    % T1 = 50 ; T2/G1 = 870/100 = 8.7 ; T3/(G1 G2) = 290/(100*0.2) = 14.5
    \addplot[fill=engorange!70, draw=engorange] coordinates {
      (stage 1, 50) (stage 2, 8.7) (stage 3, 14.5) };
  \end{axis}
  \end{scope}
\end{tikzpicture}
\caption[The Friis noise cascade]{A three-stage receiver front end and
each stage's noise temperature referred back to the antenna terminal.
The first stage enters full weight, the second is divided by the first
stage's gain, and the third by the product of the two gains ahead of it,
so $T_e = T_1 + T_2/G_1 + T_3/(G_1 G_2)$. A low-noise amplifier with high
gain at the front makes the later stages, even a lossy mixer, almost
invisible, which is the whole reason the low-noise block sits at the dish
feed rather than at the receiver end of the cable.}
\label{fig:vol04ch10:noise-cascade}
\end{figure}
