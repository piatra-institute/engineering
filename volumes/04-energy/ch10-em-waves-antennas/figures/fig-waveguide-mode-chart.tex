% Figure: cutoff frequencies of the lowest rectangular-waveguide modes for a
% WR-90 guide (a = 22.86 mm, b = 10.16 mm, so a/b = 2.25). Each mode TE_mn / TM_mn
% has cutoff f_c = (c/2) sqrt((m/a)^2 + (n/b)^2). The bars are placed by computed
% cutoff in GHz; the shaded band marks the single-mode operating region between
% the TE10 cutoff and the next-mode (TE20) cutoff.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.2cm,
  xlabel={cutoff frequency $f_c$ (GHz)},
  ylabel={},
  xmin=0, xmax=26,
  ymin=0, ymax=6,
  axis lines=left,
  ytick=\empty,
  enlargelimits=false,
  xtick={0,5,10,13.1,16.4,20,25},
  xticklabel style={font=\scriptsize},
]
  % single-mode band: TE10 (6.557) to TE20 (13.114)
  \addplot[draw=none, fill=englime, fill opacity=0.18] coordinates
    {(6.557,0) (13.114,0) (13.114,6) (6.557,6)} \closedcycle;
  % mode cutoff lines, each a vertical bar with a label
  % TE10 = 6.557
  \draw[engblue, very thick] (axis cs:6.557,0.4) -- (axis cs:6.557,5.6);
  \node[engblue, font=\scriptsize, anchor=south, rotate=90] at (axis cs:6.557,0.5) {$\text{TE}_{10}$};
  % TE20 = 13.114
  \draw[engblue, very thick] (axis cs:13.114,0.4) -- (axis cs:13.114,5.6);
  \node[engblue, font=\scriptsize, anchor=south, rotate=90] at (axis cs:13.114,0.5) {$\text{TE}_{20}$};
  % TE01 = 14.764
  \draw[engred, very thick] (axis cs:14.764,0.4) -- (axis cs:14.764,5.6);
  \node[engred, font=\scriptsize, anchor=south, rotate=90] at (axis cs:14.764,0.5) {$\text{TE}_{01}$};
  % TE11 / TM11 = 16.156
  \draw[engamber, very thick] (axis cs:16.156,0.4) -- (axis cs:16.156,5.6);
  \node[engamber, font=\scriptsize, anchor=south, rotate=90] at (axis cs:16.156,0.5) {$\text{TE}_{11},\,\text{TM}_{11}$};
  % TE30 = 19.671
  \draw[engblue, thick] (axis cs:19.671,0.4) -- (axis cs:19.671,5.6);
  \node[engblue, font=\scriptsize, anchor=south, rotate=90] at (axis cs:19.671,0.5) {$\text{TE}_{30}$};
  % recommended operating band 8.2 to 12.4
  \draw[enggray, dashed, thick] (axis cs:8.2,0) -- (axis cs:8.2,6);
  \draw[enggray, dashed, thick] (axis cs:12.4,0) -- (axis cs:12.4,6);
  \node[enggray, font=\scriptsize, anchor=south] at (axis cs:10.3,5.0) {operating band};
\end{axis}
\end{tikzpicture}
\caption[Mode cutoff chart for a WR-90 rectangular waveguide]{Cutoff
frequencies of the lowest modes of a WR-90 rectangular waveguide
($a = 22.86\,\si{\milli\meter}$, $b = 10.16\,\si{\milli\meter}$),
computed from $f_{c,mn} = (c/2)\sqrt{(m/a)^2 + (n/b)^2}$. The
$\text{TE}_{10}$ mode sets the low edge at $6.56\,\si{\giga\hertz}$ and
the $\text{TE}_{20}$ mode the high edge at $13.1\,\si{\giga\hertz}$; the
shaded band between them is the single-mode region in which only one mode
propagates. The recommended operating band ($8.2$ to $12.4\,\si{\giga
\hertz}$, dashed) sits inside that region with margin at both ends, the
reason a guide is run well above its dominant cutoff and below the next.}
\label{fig:vol04ch10:waveguide-mode-chart}
\end{figure}
