% Figure: simulated VSWR of the microstrip patch versus frequency, modelled
% as a single resonant RLC dip. The patch behaves near resonance like a
% parallel RLC; the reflection magnitude follows from a loaded-Q Lorentzian.
% Curve: |Gamma|(f) = |Q_L (f/f0 - f0/f)| / sqrt(4 + Q_L^2 (f/f0 - f0/f)^2)
% for a critically detuned single resonator referenced to a matched feed,
% then VSWR = (1+|Gamma|)/(1-|Gamma|). With f0 = 2.40 GHz and loaded Q ~ 40
% the VSWR<=2 band is about 60 MHz, the narrow bandwidth typical of a thin
% single-layer patch.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={frequency $f$ (GHz)},
  ylabel={VSWR},
  xmin=2.30, xmax=2.50, ymin=1, ymax=6,
  domain=2.30:2.50, samples=300,
  axis lines=left,
  enlargelimits=false,
  ytick={1,1.5,2,3,4,5,6},
  xtick={2.30,2.35,2.40,2.45,2.50},
  legend pos=north east,
  legend cell align=left,
]
  % loaded Q = 40, f0 = 2.40. Define |Gamma| then VSWR inline.
  % g(f) = 40*(f/2.40 - 2.40/f); |Gamma| = |g|/sqrt(4 + g^2)
  \addplot[engblue, very thick] {
    (abs(40*(x/2.40 - 2.40/x)) / sqrt(4 + (40*(x/2.40 - 2.40/x))^2))
    < 1 ?
    (1 + abs(40*(x/2.40 - 2.40/x))/sqrt(4 + (40*(x/2.40 - 2.40/x))^2))
    /(1 - abs(40*(x/2.40 - 2.40/x))/sqrt(4 + (40*(x/2.40 - 2.40/x))^2))
    : 6 };
  \addlegendentry{patch VSWR}
  % VSWR = 2 band edges
  \draw[enggray, dashed] (axis cs:2.30,2) -- (axis cs:2.50,2);
  \node[enggray, font=\scriptsize, anchor=south east] at (axis cs:2.50,2) {VSWR $=2$};
  % resonant centre
  \draw[engred, dashed] (axis cs:2.40,1) -- (axis cs:2.40,6);
  \node[engred, font=\scriptsize, anchor=south west] at (axis cs:2.40,4.6) {$f_0 = 2.40$ GHz};
\end{axis}
\end{tikzpicture}
\caption[Microstrip patch VSWR versus frequency]{Modelled input VSWR of the
case-study patch, treating the antenna near resonance as a single loaded
resonator of loaded quality factor $Q_L \approx 40$ centred at $f_0 =
2.40\,\si{\giga\hertz}$. The match is near-perfect at the centre and rises
on either side; the band over which VSWR stays below $2$ spans about
$60\,\si{\mega\hertz}$, a $2.5\,\%$ fractional bandwidth that is
characteristic of a thin single-layer patch and the working reason patch
antennas are narrowband unless the substrate is thickened or a stacked or
aperture-coupled geometry is used.}
\label{fig:vol04ch10:patch-vswr}
\end{figure}
