% Figure: normalised array factor of an 8-element uniform linear array spaced
% half a wavelength, in Cartesian form (magnitude in dB versus observation
% angle), for two progressive feed phases: broadside (alpha = 0) and a steered
% beam (alpha chosen to point near 30 degrees). The plot shows the main lobe
% moving with feed phase, the working principle of an electronically steered
% array. Cartesian axes only; pgfplots polar is not relied on.
%
% Array factor magnitude for N elements, spacing d = lambda/2, so
% psi = pi*sin(theta) + alpha. |AF| = |sin(N psi/2)/sin(psi/2)|, normalised to N.
% In dB: 20*log10(|AF|/N), with a small floor added inside the log to keep the
% argument strictly positive at the nulls.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={observation angle $\theta$ (degrees from broadside)},
  ylabel={normalised array factor (dB)},
  xmin=-90, xmax=90, ymin=-30, ymax=2,
  axis lines=left,
  legend pos=south east,
  legend cell align=left,
  enlargelimits=false,
  domain=-90:90, samples=400,
  xtick={-90,-60,-30,0,30,60,90},
  ytick={0,-10,-20,-30},
]
  % Broadside: alpha = 0. Floor 1e-3 inside the log clamps the null depth.
  \addplot[engblue, very thick]
    { 20*log10( (abs(sin(deg(4*(pi*sin(deg(x)) + 0)))
        / sin(deg(0.5*(pi*sin(deg(x)) + 0)))) + 1e-3) / 8 ) };
  \addlegendentry{$\alpha = 0$ (broadside)}
  % Steered: alpha = -pi*sin(30 deg) = -pi/2 points the main lobe to +30 deg.
  \addplot[engred, very thick]
    { 20*log10( (abs(sin(deg(4*(pi*sin(deg(x)) - pi/2)))
        / sin(deg(0.5*(pi*sin(deg(x)) - pi/2)))) + 1e-3) / 8 ) };
  \addlegendentry{$\alpha = -\pi/2$ (steered to $30^{\circ}$)}
\end{axis}
\end{tikzpicture}
\caption[Beam steering of a linear phased array]{Normalised array factor of an
eight-element uniform linear array with half-wavelength spacing, plotted in
decibels against observation angle. With the elements fed in phase
($\alpha = 0$) the main lobe points broadside; applying a progressive feed phase
of $\alpha = -\pi/2$ between neighbours swings the main lobe to $30^{\circ}$,
with no part moving. The steering identity is $\alpha = -(2\pi d/\lambda)
\sin\theta_0$ for a beam pointed at $\theta_0$, here $-\pi\sin 30^{\circ} =
-\pi/2$. The first sidelobes sit about $13\,\si{\decibel}$ below the main lobe,
the signature of uniform amplitude weighting; tapering the element amplitudes
trades main-lobe width for lower sidelobes.}
\label{fig:vol04ch10:beam-steering}
\end{figure}
