% Figure: array factor of a two-element array fed in phase, half-wavelength
% spacing, plotted as normalised magnitude versus angle from broadside.
% For two equal in-phase elements spaced d, AF = 2 cos(psi/2) with
% psi = (2 pi d / lambda) sin theta. At d = lambda/2, psi = pi sin theta,
% so the normalised magnitude is |cos((pi/2) sin theta)|. A second curve
% shows the end-fire case (90-degree feed phase), which steers the peak.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={angle from broadside $\theta$ (degrees)},
  ylabel={normalised array factor},
  xmin=-90, xmax=90, ymin=0, ymax=1.05,
  domain=-90:90, samples=400,
  axis lines=left,
  xtick={-90,-60,-30,0,30,60,90},
  ytick={0,0.25,0.5,0.707,1.0},
  yticklabels={0,0.25,0.5,0.707,1.0},
  enlargelimits=false,
  legend pos=south east,
  legend cell align=left,
]
  % broadside, in-phase: |cos((pi/2) sin theta)|, peak at theta=0
  \addplot[engblue, very thick]
    { abs( cos( 90*sin(x) ) ) };
  \addlegendentry{in phase (broadside)}
  % feed phase delta = -90 deg shifts argument: |cos((pi/2) sin theta - pi/4)|
  \addplot[engred, very thick, dashed]
    { abs( cos( 90*sin(x) - 45 ) ) };
  \addlegendentry{$90^{\circ}$ phase (squinted)}
  % half-power reference line at 0.707
  \draw[enggray, dotted] (axis cs:-90,0.707) -- (axis cs:90,0.707);
\end{axis}
\end{tikzpicture}
\caption[Two-element array factor]{The array factor of a two-element array
with half-wavelength spacing, normalised to unity at its peak. Fed in phase
(solid), the pattern peaks broadside at $\theta = 0$ and is unchanged in the
broadside plane out to the half-power points. A $90^{\circ}$ progressive feed
phase (dashed) squints the peak off broadside, the elementary case of
electronic beam steering. The element pattern of a real dipole multiplies
this array factor to give the full radiation pattern.}
\label{fig:vol04ch10:array-factor-2el}
\end{figure}
