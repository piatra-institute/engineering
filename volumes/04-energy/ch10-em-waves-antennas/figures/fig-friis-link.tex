% Figure: Friis link geometry. A transmit antenna of gain G_t radiates
% power P_t; at range R the power spreads over a sphere; the receive antenna
% of gain G_r and effective area A_eff = G_r lambda^2 / (4 pi) intercepts P_r.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % transmit antenna (left) drawn as a small dipole
  \draw[engblue, line width=2pt] (0,-0.5) -- (0,0.5);
  \node[font=\scriptsize, anchor=north] at (0,-0.6) {Tx};
  \node[font=\scriptsize, anchor=south] at (0,0.6) {$P_t,\ G_t$};
  % receive antenna (right) drawn as a small dish
  \draw[engred, line width=1.5pt] (9,-0.6) arc (-40:40:0.95);
  \draw[engred, ->] (8.6,0) -- (9.1,0);
  \node[font=\scriptsize, anchor=north] at (9,-0.7) {Rx};
  \node[font=\scriptsize, anchor=south] at (9,0.8) {$P_r,\ G_r,\ A_{\text{eff}}$};
  % spreading wavefronts (arcs of growing radius)
  \foreach \r in {1.2,2.4,3.6,4.8,6.0}{
    \draw[enggray, thin] (0,0) +(-25:\r) arc (-25:25:\r);
  }
  % range arrow
  \draw[<->, thick] (0,-1.4) -- (9,-1.4) node[midway, below, font=\small] {range $R$};
  % power-density label
  \node[enggray, font=\scriptsize, anchor=south] at (4.2,1.0)
    {power density $\dfrac{P_t G_t}{4\pi R^{2}}$};
\end{tikzpicture}
\caption[Friis transmission geometry]{The Friis link. A transmit antenna
radiates power $P_t$ with gain $G_t$; at range $R$ the power has spread
over an expanding sphere, giving a power density $P_t G_t/(4\pi R^2)$ at
the receiver. The receive antenna intercepts this density over its
effective area $A_{\text{eff}} = G_r\lambda^2/(4\pi)$, so the received
power is $P_r = P_t G_t G_r \lambda^2/(4\pi R)^2$. The inverse-square
range dependence dominates every free-space link budget.}
\label{fig:vol04ch10:friis-link}
\end{figure}
