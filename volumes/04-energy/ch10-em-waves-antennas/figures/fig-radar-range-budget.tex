% Figure: received echo power versus range for the radar case study, on a
% log range axis, showing the R^-4 fall-off of the radar equation against the
% fixed minimum detectable signal. The crossing sets the maximum range.
% Received power in dBm: P_r(R) = P_r(R_ref) - 40*log10(R/R_ref). We anchor at
% R_ref = 1 km with P_r computed from the case-study constant, and clamp the
% axis to a positive range domain so no log of zero occurs.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xmode=log,
  xlabel={range $R$ (km)},
  ylabel={echo power (dBm)},
  xmin=1, xmax=300, ymin=-160, ymax=-60,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  enlargelimits=false,
  domain=1:300, samples=120,
  grid=major,
  grid style={enggray!25},
]
  % Echo power follows the R^-4 law. Anchor: at R = 1 km the case-study
  % constants give -70.7 dBm; every decade of range costs 40 dB.
  \addplot[engblue, very thick] { -70.7 - 40*log10(x) };
  \addlegendentry{echo power $\propto R^{-4}$}
  % minimum detectable signal line
  \addplot[engred, thick, dashed, domain=1:300] { -127 };
  \addlegendentry{MDS $= -127$ dBm}
  % maximum-range marker where the two cross: R_max approx 113 km
  \draw[enggray, dashed] (axis cs:113,-160) -- (axis cs:113,-127);
  \node[enggray, font=\scriptsize, anchor=south, rotate=90] at
    (axis cs:113,-153) {$R_{\max}\approx 113$ km};
\end{axis}
\end{tikzpicture}
\caption[Radar echo power against range]{Received echo power against range for
the surveillance-radar case study, on a logarithmic range axis. The echo falls
as $R^{-4}$, so it drops $40\,\si{\decibel}$ for every decade of range, a far
steeper slope than the $20\,\si{\decibel}$ per decade of a one-way link. The
echo crosses the minimum detectable signal near $113\,\si{\kilo\meter}$, which
fixes the single-pulse maximum range. Pulse integration lowers the detection
threshold and pushes the crossing outward; the case study carries the
numbers.}
\label{fig:vol04ch10:radar-range-budget}
\end{figure}
