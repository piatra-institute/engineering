% Figure: three rays inside a dense medium striking the boundary to a rarer
% medium at increasing angles. Below the critical angle a ray refracts out;
% at the critical angle the refracted ray grazes the surface; above it the
% ray reflects entirely (total internal reflection). Pure tikz line drawing,
% no computed coordinates beyond fixed literals.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.15, >=latex]
  % dense medium below the boundary, rarer above
  \fill[engblue!8] (-3.4,-2.2) rectangle (3.4,0);
  \draw[thick] (-3.4,0) -- (3.4,0);
  \node[font=\scriptsize, anchor=west] at (3.4,0.28) {rare medium, $n_2$};
  \node[font=\scriptsize, anchor=west] at (3.4,-0.30) {dense medium, $n_1 > n_2$};
  % normal at the common incidence point
  \draw[dashed, enggray] (0,1.4) -- (0,-2.0);
  \node[enggray, font=\scriptsize] at (0.28,1.3) {normal};
  % Ray A: small angle, refracts out and bends away from normal
  \draw[engblue, very thick, ->] (-1.7,-1.9) -- (0,0);
  \draw[engred, very thick, ->] (0,0) -- (1.7,1.35);
  \node[engblue, font=\scriptsize, anchor=east] at (-1.7,-1.85) {A};
  % Ray B: at critical angle, refracted ray grazes the surface
  \draw[engblue, very thick, ->] (-2.5,-1.15) -- (0,0);
  \draw[engred, very thick, ->] (0,0) -- (2.6,0.0);
  \node[engblue, font=\scriptsize, anchor=east] at (-2.5,-1.1) {B};
  \node[engred, font=\scriptsize, anchor=south] at (2.0,0.02) {grazing};
  % Ray C: above critical, fully reflected back into the dense medium
  \draw[engblue, very thick, ->] (-2.9,-0.55) -- (0,0);
  \draw[engamber, very thick, ->] (0,0) -- (2.9,-0.55);
  \node[engblue, font=\scriptsize, anchor=east] at (-2.9,-0.5) {C};
  \node[engamber, font=\scriptsize, anchor=north] at (2.2,-0.5)
    {reflected};
  % critical-angle arc annotation for ray B
  \draw[enggray] (0,-0.75) arc (270:245:0.75);
  \node[font=\scriptsize] at (-0.55,-0.66) {$\theta_c$};
\end{tikzpicture}
\caption[Total internal reflection and the critical angle]{Three rays in
the dense medium meeting the boundary to a rarer medium at growing angles.
Ray A, below the critical angle $\theta_c = \arcsin(n_2/n_1)$, refracts
out and bends away from the normal. Ray B, at $\theta_c$, refracts along
the surface itself. Ray C, past $\theta_c$, cannot satisfy Snell's law
with any real refracted angle, so all of its power reflects back into the
dense medium. This total internal reflection is the confinement mechanism
of an optical fibre and of the right-angle prisms that fold the light path
in binoculars, and its evanescent leakage across a thin gap is the basis
of frustrated-total-reflection couplers.}
\label{fig:vol04ch10:total-internal-reflection}
\end{figure}
