% Figure: reflection and refraction at a planar boundary between two media.
% Incident and reflected rays in medium 1; refracted ray in medium 2.
% Snell's law n1 sin th1 = n2 sin th2; here n2 > n1 so the refracted ray
% bends toward the normal.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.2, >=latex]
  % boundary
  \fill[engblue!8] (-3,0) rectangle (3,-2.2);
  \draw[thick] (-3,0) -- (3,0);
  \node[font=\scriptsize, anchor=west] at (3,0.25) {medium 1, $n_1$};
  \node[font=\scriptsize, anchor=west] at (3,-0.3) {medium 2, $n_2 > n_1$};
  % normal
  \draw[dashed, enggray] (0,1.8) -- (0,-1.8);
  \node[enggray, font=\scriptsize] at (0.25,1.7) {normal};
  % incident ray (from upper left, 35 deg from normal)
  \draw[engblue, very thick, ->] (-1.4,2.0) -- (0,0);
  \node[engblue, font=\scriptsize] at (-1.5,1.6) {incident};
  % reflected ray (upper right, equal angle)
  \draw[engblue, very thick, ->] (0,0) -- (1.4,2.0);
  \node[engblue, font=\scriptsize] at (1.5,1.6) {reflected};
  % refracted ray (lower right, smaller angle from normal, bends toward normal)
  \draw[engred, very thick, ->] (0,0) -- (0.7,-1.9);
  \node[engred, font=\scriptsize] at (1.1,-1.6) {refracted};
  % angle arcs
  \draw[enggray] (0,0.7) arc (90:125:0.7);
  \node[font=\scriptsize] at (-0.42,0.72) {$\theta_1$};
  \draw[enggray] (0,0.7) arc (90:55:0.7);
  \node[font=\scriptsize] at (0.42,0.72) {$\theta_1$};
  \draw[enggray] (0,-0.8) arc (270:290:0.8);
  \node[font=\scriptsize] at (0.4,-0.78) {$\theta_2$};
\end{tikzpicture}
\caption[Reflection and refraction at a planar boundary]{A plane wave
incident from medium 1 ($n_1$) on a planar boundary with medium 2 ($n_2
> n_1$) splits into a reflected wave (angle of reflection equal to the
angle of incidence) and a refracted wave that bends toward the normal,
since $n_1\sin\theta_1 = n_2\sin\theta_2$ requires $\theta_2 < \theta_1$
when $n_2 > n_1$. The Fresnel equations set the amplitude split between
reflected and refracted waves; at normal incidence the reflectivity is
$R = |(n_1-n_2)/(n_1+n_2)|^2$.}
\label{fig:vol04ch10:reflection-boundary}
\end{figure}
