% Figure: the single-slit intensity (sinc^2) plotted against the phase variable
% beta = (pi a / lambda) sin(theta), showing the central maximum, the first
% zeros at beta = +-pi, and the rapidly falling side lobes. This supports the
% rigorous derivation of the diffraction pattern by phasor integration across
% the aperture.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm, height=6.4cm,
  xlabel={$\beta=(\pi a/\lambda)\sin\theta$ (rad)},
  ylabel={$I/I_0$},
  xmin=-10, xmax=10, ymin=0, ymax=1.08,
  axis lines=left,
  xtick={-9.4248,-6.2832,-3.1416,0,3.1416,6.2832,9.4248},
  xticklabels={$-3\pi$,$-2\pi$,$-\pi$,$0$,$\pi$,$2\pi$,$3\pi$},
  ytick={0,0.25,0.5,0.75,1.0},
  tick label style={font=\scriptsize},
  label style={font=\small},
  samples=400, domain=-10:10,
  legend style={font=\scriptsize, at={(0.98,0.97)}, anchor=north east, draw=none},
]
% (sin beta / beta)^2 ; the singularity at beta=0 is removable, value 1.
% Shift the sample grid slightly off zero is unnecessary since pgfplots samples
% avoid exact 0 with even sample count, but guard with a small offset form:
\addplot[engblue, very thick] {(sin(deg(x))/x)^2};
\addlegendentry{$[\sin\beta/\beta]^2$}
% mark the first zeros
\draw[engred, dotted] (axis cs:3.1416,0) -- (axis cs:3.1416,1.0);
\draw[engred, dotted] (axis cs:-3.1416,0) -- (axis cs:-3.1416,1.0);
\node[enggray, font=\scriptsize, anchor=south] at (axis cs:4.712,0.10)
  {side lobe $\approx 4.7\%$};
\end{axis}
\end{tikzpicture}
\caption[Single-slit diffraction intensity]{The Fraunhofer single-slit
intensity $I(\theta)=I_0[\sin\beta/\beta]^2$ with
$\beta=(\pi a/\lambda)\sin\theta$, the result of summing the phasor
contributions of every strip across a slit of width $a$. The central maximum
runs from the first zero at $\beta=-\pi$ to the first zero at $\beta=+\pi$,
twice the width of each side lobe, and the first side lobe carries only about
$4.7\,\%$ of the central peak. Setting $\beta=\pi$ recovers the working
first-minimum condition $\sin\theta=\lambda/a$ used in the wave-optics
section.}
\label{fig:vol04ch11:single-slit-phasor}
\end{figure}
