% Figure: total internal reflection and the evanescent wave. A ray in the dense
% medium strikes the boundary above the critical angle and reflects, while the
% field in the rare medium does not vanish at the interface but decays
% exponentially over a penetration depth of order a wavelength.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, scale=1.0]
% media
\fill[engblue!12] (-4.2,-1.8) rectangle (4.2,0);
\fill[engamber!8] (-4.2,0) rectangle (4.2,2.3);
\draw[thick] (-4.2,0) -- (4.2,0);
\node[engblue, font=\scriptsize, anchor=west] at (-4.15,-1.55) {dense medium $n_1$};
\node[engamber, font=\scriptsize, anchor=west] at (-4.15,2.05) {rare medium $n_2<n_1$};
% incident ray, hitting origin from lower left above critical angle
\draw[engred, ->, very thick] (-3.0,-1.6) -- (0,0);
\node[engred, font=\scriptsize, anchor=east] at (-1.7,-0.75) {incident};
% reflected ray to lower right
\draw[engred, ->, very thick] (0,0) -- (3.0,-1.6);
\node[engred, font=\scriptsize, anchor=west] at (1.7,-0.75) {reflected};
% normal
\draw[enggray, dashed] (0,-1.8) -- (0,2.3);
% angle marks
\draw[enggray] (-0.55,-0.29) arc [start angle=208, end angle=270, radius=0.62];
\node[enggray, font=\tiny] at (-0.30,-0.68) {$\theta_1$};
% evanescent field: exponential decay curve into rare medium along the normal
\draw[engorange, thick, domain=0:2.1, samples=60, variable=\y]
  plot ({1.1*exp(-\y/0.62)}, {\y});
\draw[engorange, thick, domain=0:2.1, samples=60, variable=\y]
  plot ({-1.1*exp(-\y/0.62)}, {\y});
\node[engorange, font=\scriptsize, anchor=west] at (1.15,1.35) {evanescent field $\propto e^{-y/d_p}$};
% penetration depth marker
\draw[<->, engblack] (-1.7,0) -- (-1.7,0.62);
\node[engblack, font=\tiny, anchor=east] at (-1.72,0.31) {$d_p$};
\draw[enggray, dotted] (-1.7,0.62) -- (0,0.62);
\end{tikzpicture}
\caption[Total internal reflection and the evanescent wave]{A ray in the dense
medium striking the boundary at an angle beyond the critical angle reflects
entirely, yet the field in the rare medium does not stop at the interface. It
persists as an evanescent wave whose amplitude decays exponentially with depth
$y$ over a penetration depth $d_p = \lambda/[2\pi\sqrt{n_1^2\sin^2\theta_1 -
n_2^2}]$, of order a wavelength just past the critical angle and shrinking as the
incidence steepens. A second dense medium brought within a few $d_p$ of the
boundary lets the evanescent field couple across the gap and carry power, the
frustrated total internal reflection a beam-splitter cube and a fingerprint
sensor exploit.}
\label{fig:vol04ch11:evanescent-wave}
\end{figure}
