% Figure: Newton's rings. Left: a plano-convex lens of radius R rests on an
% optical flat, trapping a wedge-shaped air gap whose thickness grows as the
% square of the radial coordinate. Right: the resulting concentric fringe
% pattern, dark at the centre and spaced ever closer with radius.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, scale=1.0]
% ---- left panel: cross-section ----
\begin{scope}[shift={(0,0)}]
% flat glass
\fill[engblue!10] (-3.4,-0.9) rectangle (0.4,-0.3);
\draw[thick] (-3.4,-0.3) -- (0.4,-0.3);
\node[font=\scriptsize, anchor=north] at (-1.5,-0.9) {optical flat};
% lens: shallow arc curving up, touching the flat at x=-1.5
% draw a curved lower surface (large radius R)
\draw[thick, engred] (-3.4,0.55) .. controls (-2.2,-0.28) and (-0.8,-0.28) .. (0.4,0.55);
\draw[thick, engred] (-3.4,0.55) -- (0.4,0.55);
\node[engred, font=\scriptsize, anchor=south] at (-1.5,0.55) {plano-convex lens, radius $R$};
% contact point
\filldraw[engblack] (-1.5,-0.3) circle [radius=1.1pt];
\node[font=\scriptsize, anchor=north] at (-1.5,-0.32) {contact};
% air gap label
\draw[<->, enggray] (-2.7,-0.28) -- (-2.7,-0.02);
\node[enggray, font=\scriptsize, anchor=east] at (-2.72,-0.15) {gap $t(r)$};
% incident and reflected arrows
\draw[engorange, ->, thick] (-0.2,1.6) -- (-0.6,0.6);
\draw[engorange, ->, thick] (-0.6,0.6) -- (-1.0,1.6);
\node[engorange, font=\scriptsize, anchor=west] at (-0.2,1.3) {monochromatic};
\end{scope}
% ---- right panel: ring pattern ----
\begin{scope}[shift={(6.0,0.3)}]
% dark centre, then alternating rings; radii ~ sqrt(m)
\fill[engblue!25] (0,0) circle [radius=1.9];
\foreach \m in {1,2,3,4,5} {
  \pgfmathsetmacro{\rin}{1.9*sqrt((\m-0.5)/5.5)}
  \pgfmathsetmacro{\rout}{1.9*sqrt(\m/5.5)}
  \fill[white] (0,0) circle [radius=\rout];
  \fill[engblue!25] (0,0) circle [radius=\rin];
}
\draw[engblack] (0,0) circle [radius=1.9];
\filldraw[engblack] (0,0) circle [radius=1.3pt];
\node[font=\scriptsize, anchor=north] at (0,-2.15) {dark centre, rings crowd outward};
\draw[<-, enggray] (0,0) -- (1.34,1.34);
\node[enggray, font=\scriptsize, anchor=south west] at (1.0,1.0) {$r_m\propto\sqrt{m}$};
\end{scope}
\end{tikzpicture}
\caption[Newton's rings]{A plano-convex lens resting on an optical flat traps a
thin air gap whose thickness grows as $t(r)=r^2/(2R)$ with the radial distance
$r$ from the contact point. Reflections from the two surfaces of the gap
interfere, and the $\lambda/2$ phase step at the lower (air-to-glass) reflection
makes the point of contact dark. Dark rings fall where $2t=m\lambda$, so the
$m$th dark ring sits at $r_m=\sqrt{m\lambda R}$; the rings crowd together with
radius because $r_m$ grows only as $\sqrt{m}$. Measuring two ring radii fixes the
lens radius of curvature $R$ without touching the glass, the classical test of a
polished spherical surface.}
\label{fig:vol04ch11:newton-rings}
\end{figure}
