% Figure: time-of-flight LIDAR pulse timing. A short transmit pulse leaves at
% t=0; the return arrives after the round-trip delay tau = 2R/c, broadened and
% attenuated. The range resolution is set by the pulse width; two targets
% closer than c*tp/2 merge into one return.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm, height=5.8cm,
  xlabel={time (ns)},
  ylabel={detector signal (arb.)},
  xmin=0, xmax=240, ymin=0, ymax=1.25,
  axis lines=left,
  xtick={0,40,80,120,160,200},
  ytick={0,0.5,1.0},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.98,0.97)}, anchor=north east, draw=none},
  enlarge x limits=false,
  samples=200,
]
% transmit pulse: narrow Gaussian centred at t = 6 ns, width ~3 ns
\addplot[engblue, very thick, domain=0:240] {exp(-((x-6)^2)/(2*2.5^2))};
\addlegendentry{transmit pulse}
% return pulse: round trip tau = 2R/c; for R = 22.5 m, tau = 150 ns. Broadened,
% attenuated to 0.45, centred at t = 156 ns.
\addplot[engorange, very thick, domain=0:240] {0.45*exp(-((x-156)^2)/(2*4.0^2))};
\addlegendentry{return pulse}
% round-trip delay annotation
\draw[engred, <->, thick] (axis cs:6,1.12) -- (axis cs:156,1.12);
\node[engred, font=\scriptsize, anchor=south] at (axis cs:81,1.13)
  {$\tau = 2R/c$};
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:158,0.5)
  {attenuated, broadened};
\end{axis}
\end{tikzpicture}
\caption[LIDAR pulse timing]{Time-of-flight ranging measures the round-trip
delay $\tau=2R/c$ between a transmitted pulse and its return. The return is
attenuated by the range equation and broadened by the target depth and the
detector bandwidth. Range follows from $R=c\tau/2$, so a timing resolution of
$\sigma_\tau$ gives a range resolution of $c\,\sigma_\tau/2$; the
$3\,\si{\nano\second}$ transmit pulse drawn here sets a single-shot depth
resolution near $0.45\,\si{\meter}$ before any pulse-centroid or
constant-fraction timing sharpens it.}
\label{fig:vol04ch11:lidar-pulse-timing}
\end{figure}
