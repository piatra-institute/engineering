% Figure: reflectance versus wavelength for a bare glass surface, a single
% quarter-wave MgF2 layer, and an idealised two-layer V-coat. The single
% layer pulls the reflectance to a shallow minimum near the design
% wavelength; the two-layer coat reaches near-zero at the design wavelength
% but over a narrower band.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12.5cm, height=6.8cm,
  xlabel={wavelength (nm)},
  ylabel={reflectance (\%, single surface)},
  xmin=400, xmax=700, ymin=0, ymax=5,
  axis lines=left,
  xtick={400,450,500,550,600,650,700},
  ytick={0,1,2,3,4},
  tick label style={font=\scriptsize},
  label style={font=\small},
  samples=200, domain=400:700,
  legend style={font=\scriptsize, at={(0.5,0.97)}, anchor=north, draw=none, legend columns=1},
]
% bare glass n=1.52: R = ((1.52-1)/(1.52+1))^2 = 0.0426 -> 4.26%
\addplot[enggray, thick, dashed] {4.26};
\addlegendentry{bare glass ($n=1.52$)}
% single quarter-wave MgF2 layer, design 550 nm. Reflectance has a cosine
% form in optical-thickness; approximate the band shape with a raised cosine
% reaching ~1.3% min at 550 nm and rising toward the band edges.
\addplot[engblue, very thick]
  {1.3 + 2.0*(1 - cos(deg(2*pi*(x-550)/300))) };
\addlegendentry{single-layer MgF$_2$}
% two-layer V-coat, design 550 nm, deeper but narrower null
\addplot[engorange, very thick]
  {0.05 + 4.0*(1 - cos(deg(2*pi*(x-550)/200)))^2 };
\addlegendentry{two-layer V-coat}
% design wavelength marker
\draw[engred, dotted] (axis cs:550,0) -- (axis cs:550,5);
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:552,4.3)
  {design $\lambda$};
\end{axis}
\end{tikzpicture}
\caption[Anti-reflection coating reflectance]{Single-surface reflectance of
crown glass against wavelength. The bare surface reflects about
$4.3\,\%$ flat across the visible. A single quarter-wave magnesium-fluoride
layer ($n=1.38$) tuned to $550\,\si{\nano\meter}$ pulls the reflectance to a
broad minimum near $1.3\,\%$, the limit a single layer reaches because the
ideal index $\sqrt{n_{\text{air}}n_{\text{glass}}}=1.23$ lies below any
durable coating material. A two-layer V-coat reaches a near-zero null at the
design wavelength but over a narrower band, the trade every multilayer design
makes between depth and bandwidth. The curve shapes are schematic;
quantitative reflectances follow from the thin-film arithmetic of
\texttt{code/thin\_film.py}.}
\label{fig:vol04ch11:ar-coating}
\end{figure}
