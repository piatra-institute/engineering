% Figure: optical-power waterfall down a fibre link. Power in dBm falls from
% the launch level through coupling loss, two spans of fibre attenuation, a
% mid-span EDFA gain step, splice and connector losses, and lands above the
% receiver sensitivity. The shaded gap between the received power and the
% sensitivity floor is the system margin.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm, height=7.4cm,
  xlabel={distance along link (km)},
  ylabel={optical power (dBm)},
  xmin=0, xmax=132, ymin=-32, ymax=8,
  axis lines=left,
  xtick={0,60,120},
  xticklabels={0,60,120},
  ytick={5,0,-10,-20,-24,-30},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.98,0.97)}, anchor=north east, draw=none},
  enlarge x limits=false,
]
% receiver sensitivity floor
\addplot[engred, thick, dashed, domain=0:132] {-24};
\addlegendentry{receiver sensitivity}
% power profile: launch +3 at 0; couple to fibre -1 (so +2 at 0 entry);
% span 1: 60 km at 0.20 dB/km = -12 dB; reach -10 at 60 km;
% EDFA +14 step at mid-span; up to +4 dBm; span 2: 60 km more = -12;
% reach -8 at 120 km; connectors and splices -1.6; land at -9.6 dBm.
\addplot[engblue, very thick] coordinates {
  (0,3) (0,2) (60,-10) (60,4) (120,-8) (120,-9.6)
};
\addlegendentry{signal power}
% margin annotation at receiver
\draw[engorange, <->, thick] (axis cs:120,-9.6) -- (axis cs:120,-24);
\node[engorange, font=\scriptsize, anchor=west] at (axis cs:121,-17)
  {14 dB margin};
% labels for the events
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:1,3)
  {launch $+3$};
\node[enggray, font=\scriptsize, anchor=north] at (axis cs:30,-4)
  {0.20 dB/km};
\node[englime!60!engblack, font=\scriptsize, anchor=south] at (axis cs:60,4.4)
  {EDFA $+14$};
\node[enggray, font=\scriptsize, anchor=north] at (axis cs:90,-2)
  {0.20 dB/km};
\end{axis}
\end{tikzpicture}
\caption[Fibre-link power waterfall]{Optical power down a two-span
single-mode link at $1.55\,\si{\micro\meter}$. The signal launches at
$+3\,\text{dBm}$, drops $1\,\si{\decibel}$ coupling into the fibre, falls
$12\,\si{\decibel}$ across the first $60\,\si{\kilo\meter}$ span at
$0.20\,\si{\decibel\per\kilo\meter}$, recovers $14\,\si{\decibel}$ at a
mid-span erbium-doped fibre amplifier, falls another $12\,\si{\decibel}$
across the second span, and loses $1.6\,\si{\decibel}$ to connectors and
splices to arrive at $-9.6\,\text{dBm}$. The received power sits about
$14\,\si{\decibel}$ above the $-24\,\text{dBm}$ receiver sensitivity, well
past the $6\,\si{\decibel}$ the design carries against ageing and repair
splices.}
\label{fig:vol04ch11:link-budget}
\end{figure}
