% Figure: Sagnac ring interferometer. A beamsplitter launches two beams that
% counter-propagate around a closed loop (here a triangular ring of three
% mirrors) and recombine at the splitter. Rotation of the whole assembly at
% rate Omega makes the co-rotating beam travel slightly farther, shifting the
% recombination phase in proportion to the enclosed area and the rotation rate.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, scale=1.0]
% ring corners: beamsplitter at bottom-left, two mirrors
\coordinate (BS) at (-2.2,-1.4);
\coordinate (M1) at (2.4,-1.4);
\coordinate (M2) at (0.1,2.0);
% the closed loop
\draw[engblue, very thick] (BS) -- (M1);
\draw[engblue, very thick] (M1) -- (M2);
\draw[engblue, very thick] (M2) -- (BS);
% beamsplitter and mirrors as short bars
\draw[engblack, very thick] ($(BS)+(-0.28,-0.28)$) -- ($(BS)+(0.28,0.28)$);
\node[font=\scriptsize, anchor=north east] at ($(BS)+(-0.1,-0.1)$) {beamsplitter};
\draw[engblack, very thick] ($(M1)+(0.28,-0.28)$) -- ($(M1)+(-0.28,0.28)$);
\node[font=\scriptsize, anchor=west] at ($(M1)+(0.3,0)$) {mirror};
\draw[engblack, very thick] ($(M2)+(-0.3,0.05)$) -- ($(M2)+(0.3,-0.05)$);
\node[font=\scriptsize, anchor=south] at ($(M2)+(0,0.15)$) {mirror};
% input beam into BS
\draw[engred, very thick, ->] (-4.0,-1.4) -- ($(BS)+(-0.35,0)$);
\node[engred, font=\scriptsize, anchor=south] at (-3.4,-1.35) {laser in};
% output to detector
\draw[englime!70!black, very thick, ->] ($(BS)+(0,-0.35)$) -- ($(BS)+(0,-1.3)$);
\node[englime!70!black, font=\scriptsize, anchor=north] at ($(BS)+(0,-1.35)$) {detector};
% counter-propagating direction arrows (clockwise and counter-clockwise)
\node[engorange, font=\scriptsize] at (0.1,-1.65) {$\circlearrowright$ CW};
\node[engorange, font=\scriptsize] at (-1.4,0.55) {$\circlearrowleft$ CCW};
% enclosed-area label and rotation
\node[enggray, font=\scriptsize] at (0.1,0.0) {enclosed area $A$};
\draw[engred, ->, thick] (0.1,-0.55) arc [start angle=-90, end angle=180, radius=0.42];
\node[engred, font=\scriptsize, anchor=west] at (0.55,-0.35) {$\Omega$};
\end{tikzpicture}
\caption[Sagnac ring interferometer]{A Sagnac interferometer splits one beam into
two that circulate the same closed loop in opposite senses and recombine at the
beamsplitter. When the whole assembly rotates at rate $\Omega$ about an axis
normal to the loop, the co-rotating beam meets a slightly longer path and the
counter-rotating beam a shorter one, so the two arrive with a phase difference
$\Delta\phi=8\pi A\,\Omega/(\lambda c)$ set by the enclosed area $A$. The
device measures absolute rotation with no moving parts, the operating principle
of the ring-laser and fibre-optic gyroscopes that hold the heading of aircraft
and spacecraft between satellite fixes.}
\label{fig:vol04ch11:sagnac-gyro}
\end{figure}
