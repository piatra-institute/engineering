% Figure: transmitted intensity through a stack of polarisers obeying Malus's
% law. Curve 1: a single analyser rotated relative to a fixed polariser,
% I/I0 = cos^2(theta). Curve 2: crossed polarisers (90 deg) with one extra sheet
% inserted at angle phi between them, I/I0 = cos^2(phi) cos^2(90-phi), which is
% zero at 0 and 90 deg and peaks at 45 deg. The second curve shows how an
% intermediate sheet leaks light through nominally crossed polarisers.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm, height=6.8cm,
  xlabel={analyser angle (degrees)},
  ylabel={transmitted fraction $I/I_0$},
  xmin=0, xmax=90, ymin=0, ymax=1.05,
  axis lines=left,
  xtick={0,15,30,45,60,75,90},
  ytick={0,0.25,0.5,0.75,1.0},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.5,0.97)}, anchor=north, draw=none},
  samples=181, domain=0:90,
]
% Malus single analyser: cos^2(theta), theta in degrees
\addplot[engblue, very thick] {cos(x)^2};
\addlegendentry{single analyser $\cos^2\theta$}
% inserted middle sheet between crossed polarisers: cos^2(phi)*cos^2(90-phi)
% = cos^2(phi)*sin^2(phi) = (1/4) sin^2(2 phi)
\addplot[engorange, very thick] {0.25*sin(2*x)^2};
\addlegendentry{middle sheet, crossed pair $\tfrac14\sin^2 2\phi$}
% peak marker
\node[engorange, font=\scriptsize, anchor=south] at (axis cs:45,0.27)
  {$0.25$ at $45^{\circ}$};
\end{axis}
\end{tikzpicture}
\caption[Malus's law and a three-sheet stack]{Transmitted fraction through
polarising sheets. The upper curve is Malus's law for a single analyser rotated
by $\theta$ against a fixed polariser, $I/I_0=\cos^2\theta$, which passes all
the polarised light at $0^{\circ}$ and blocks it at $90^{\circ}$. The lower
curve is the counter-intuitive case: two crossed polarisers ($90^{\circ}$
apart) pass no light, yet inserting a third sheet at angle $\phi$ between them
leaks a fraction $\tfrac14\sin^2 2\phi$, which peaks at $0.25$ when the middle
sheet sits at $45^{\circ}$. Light forbidden by the crossed pair reappears
because each sheet projects the field onto a new axis, and the projection is not
reversible. This is the working principle of the liquid-crystal display, whose
pixel is a voltage-controlled middle layer between crossed polarisers.}
\label{fig:vol04ch11:malus-stack}
\end{figure}
