% Figure: second-harmonic generation and phase matching. Left: a fundamental
% beam enters a nonlinear crystal and a doubled-frequency beam grows along it.
% Right: the second-harmonic power against crystal length, oscillating with a
% period of twice the coherence length when the phase is mismatched, and growing
% as the square of the length when the crystal is phase-matched.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Stealth, scale=1.0]
% ---- left panel: crystal schematic ----
\begin{scope}
% crystal block
\fill[engblue!12] (-3.4,-0.7) rectangle (-0.6,0.7);
\draw[thick] (-3.4,-0.7) rectangle (-0.6,0.7);
\node[font=\scriptsize, anchor=south] at (-2.0,0.72) {nonlinear crystal $\chi^{(2)}$};
% fundamental in (red), thick
\draw[engred, very thick, ->] (-4.3,0) -- (-3.4,0);
\node[engred, font=\tiny, anchor=south] at (-3.9,0.02) {$\omega$};
% doubled out (blue/green), growing
\draw[englime!70!black, very thick, ->] (-0.6,0) -- (0.5,0);
\node[englime!70!black, font=\tiny, anchor=south] at (0.0,0.02) {$2\omega$};
% residual fundamental out
\draw[engred, thick, ->] (-0.6,-0.35) -- (0.5,-0.35);
\node[engred, font=\tiny, anchor=north] at (0.0,-0.37) {$\omega$};
\end{scope}
% ---- right panel: SHG power vs length ----
\begin{scope}[shift={(5.4,-0.9)}]
\draw[->] (0,0) -- (0,2.5) node[anchor=south, font=\tiny] {$P_{2\omega}$};
\draw[->] (0,0) -- (4.2,0) node[anchor=east, font=\tiny] {crystal length $L$};
% phase-matched: quadratic growth (clamped for display)
\draw[englime!70!black, very thick, domain=0:3.9, samples=60, variable=\x]
  plot ({\x}, {min(0.14*\x*\x, 2.3)});
\node[englime!70!black, font=\tiny, anchor=west] at (2.35,2.15) {phase-matched $\propto L^2$};
% mismatched: sin^2 oscillation with small amplitude, period = 2 l_c
\draw[engamber, very thick, domain=0:3.9, samples=140, variable=\x]
  plot ({\x}, {0.32*(sin(deg(\x*3.1416/0.85)))^2});
\node[engamber, font=\tiny, anchor=west] at (1.9,0.55) {mismatched};
% coherence-length marker (first peak at l_c = 0.425)
\draw[enggray, dotted] (0.425,0) -- (0.425,0.34);
\draw[<->, engblack] (0,-0.28) -- (0.85,-0.28);
\node[engblack, font=\tiny, anchor=north] at (0.425,-0.28) {$2\ell_c$};
\end{scope}
\end{tikzpicture}
\caption[Second-harmonic generation and phase matching]{A fundamental beam at
frequency $\omega$ driving a nonlinear crystal generates light at $2\omega$. When
the fundamental and the harmonic travel at different phase velocities the two
drift out of step, so the harmonic power built in the first half of a coherence
length $\ell_c = \pi/\Delta k$ is cancelled in the second half and the output
oscillates (amber) with a period $2\ell_c$, never exceeding the tiny amplitude a
single coherence length builds. Matching the phase velocities, by birefringent
angle tuning or by quasi-phase-matching, holds $\Delta k = 0$ so every slice adds
in step and the harmonic power grows as the square of the crystal length (green),
the growth that turns an infrared laser into a visible one.}
\label{fig:vol04ch11:shg-phase-matching}
\end{figure}
