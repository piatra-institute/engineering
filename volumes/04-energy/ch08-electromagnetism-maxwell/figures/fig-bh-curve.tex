% Figure: B-H magnetisation curve of a silicon-steel core with the
% hysteresis loop and the chosen operating point B = 1.5 T below the
% saturation knee near 2 T. Supports the transformer case study.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex, scale=1.0]
\begin{axis}[
  width=10cm, height=7cm,
  xlabel={field intensity $H$ ($\si{\ampere\per\meter}$)},
  ylabel={flux density $B$ ($\si{\tesla}$)},
  xmin=-600, xmax=600,
  ymin=-2.2, ymax=2.2,
  axis lines=middle,
  xtick={-400,-200,200,400},
  ytick={-2,-1,1,2},
  every axis x label/.style={at={(ticklabel* cs:1.0)}, anchor=west},
  every axis y label/.style={at={(ticklabel* cs:1.0)}, anchor=south},
  font=\scriptsize,
  clip=false,
]
% upper branch of the hysteresis loop (tanh-like, shifted by coercivity)
\addplot[engblue, line width=1pt, smooth, domain=-550:550, samples=120]
  {2.0*tanh((x+50)/180)};
% lower branch
\addplot[engblue, line width=1pt, smooth, domain=-550:550, samples=120]
  {2.0*tanh((x-50)/180)};
% saturation asymptote
\addplot[enggray, dashed, domain=-550:550, samples=2] {2.0};
\addplot[enggray, dashed, domain=-550:550, samples=2] {-2.0};
\node[enggray, anchor=south west, font=\scriptsize] at (axis cs:-550,2.0) {$B_{\text{sat}}\approx 2.0\,\si{\tesla}$};
% operating point at B = 1.5 T, H ~ 200 A/m
\addplot[engred, only marks, mark=*, mark size=2pt] coordinates {(200,1.5)};
\node[engred, anchor=west, font=\scriptsize] at (axis cs:220,1.5) {operating point $B=1.5\,\si{\tesla}$};
\draw[engred, dashed] (axis cs:200,0) -- (axis cs:200,1.5);
\draw[engred, dashed] (axis cs:0,1.5) -- (axis cs:200,1.5);
\end{axis}
\end{tikzpicture}
\caption[Silicon-steel B-H curve and operating point]{Magnetisation
loop of a grain-oriented silicon-steel transformer core. The loop area
is the hysteresis loss dissipated per cycle per unit volume; the
coercive field where $B$ crosses zero sets its width. The design places
the peak operating flux density at $B=1.5\,\si{\tesla}$, below the
saturation knee near $2.0\,\si{\tesla}$, so the relative permeability
stays high and the magnetising current stays small. Pushing the
operating point above the knee collapses the permeability, draws a
large and distorted magnetising current, and is the saturation failure
the case study guards against.}
\label{fig:vol04ch08:bh-curve}
\end{figure}
