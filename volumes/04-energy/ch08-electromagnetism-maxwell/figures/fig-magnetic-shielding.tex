% Figure: magnetic shielding factor of a high-permeability spherical shell
% against shell thickness. A mu-metal shell diverts flux around its cavity;
% the shielding factor rises with the product of relative permeability and
% the thickness-to-radius ratio. Thin shells buy little; the return is
% roughly linear in thickness until the shell saturates.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{semilogyaxis}[
  width=11.5cm, height=7cm,
  xlabel={shell thickness $t$ (fraction of radius $t/a$)},
  ylabel={shielding factor $S = B_{\text{out}}/B_{\text{in}}$},
  xmin=0, xmax=0.1,
  ymin=1, ymax=2000,
  xtick={0,0.02,0.04,0.06,0.08,0.10},
  legend pos=north west,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% Thin-shell result for a sphere: S ~ 1 + (2/3) mu_r (t/a). Plot for two
% permeabilities. mu_r = 20000 (mu-metal) and mu_r = 5000 (silicon steel).
\addplot[engblue, line width=1.2pt, domain=0:0.1, samples=80]
  {1 + (2/3)*20000*x};
\addlegendentry{$\mu_r = 20000$ (mu-metal)}
\addplot[engamber, line width=1pt, domain=0:0.1, samples=80]
  {1 + (2/3)*5000*x};
\addlegendentry{$\mu_r = 5000$ (silicon steel)}
% design point: mu-metal at t/a = 0.03 gives S ~ 401
\addplot[enggray, only marks, mark=*, mark size=2pt]
  coordinates {(0.03,401)};
\node[enggray, anchor=west, font=\scriptsize] at (axis cs:0.032,401)
  {$t/a = 0.03$, $S \approx 400$};
\end{semilogyaxis}
\end{tikzpicture}
\caption[Magnetic shielding factor of a high-permeability shell]{Shielding
factor of a spherical high-permeability shell against its thickness, from
the thin-shell result $S \approx 1 + \tfrac23\,\mu_r\,(t/a)$ for a shell of
radius $a$ and wall thickness $t$. The shell works by offering the flux a
low-reluctance path through its wall, so lines that would have crossed the
cavity are diverted around it and the interior field drops by the factor
$S$. The return scales with the product of relative permeability and the
thickness-to-radius ratio, so a modest wall of a very high-permeability
alloy beats a thick wall of ordinary steel. The plot stops short of the
regime where the wall approaches saturation, above which $\mu_r$ collapses
and the shield stops improving with field strength, which is why sensitive
instruments use nested shells rather than one thick one.}
\label{fig:vol04ch08:magnetic-shielding}
\end{figure}
