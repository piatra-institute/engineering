% Figure: energy density stored in an electric field and in a magnetic
% field, each built up quadratically as the field is established. The
% shaded triangle under each linear constitutive line is the stored energy
% density, the geometric origin of the factor of one half.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
% Left panel: electric field. D = eps0 E, energy density = area under line.
\begin{axis}[
  name=left,
  width=7.5cm, height=6cm,
  xlabel={field $E$},
  ylabel={$D = \varepsilon_0 E$},
  xmin=0, xmax=1.1, ymin=0, ymax=1.1,
  xtick=\empty, ytick=\empty,
  axis lines=left,
  font=\scriptsize,
  title={electric: $u_e=\tfrac12\varepsilon_0 E^2$},
  title style={font=\scriptsize},
]
% constitutive line D = eps0 E (slope eps0, drawn as slope 1 in arbitrary units)
\addplot[engblue, line width=1.2pt, domain=0:1] {x};
% shaded triangle = stored energy density
\addplot[engblue!18, fill=engblue!18, draw=none] coordinates {
  (0,0) (1,0) (1,1)
} \closedcycle;
\node[engblue, font=\scriptsize] at (axis cs:0.62,0.28)
  {$\displaystyle\int_0^{E}\! D\,\dd E'$};
\draw[enggray, dashed] (axis cs:1,0) -- (axis cs:1,1);
\end{axis}
% Right panel: magnetic field. B = mu0 H, energy density = area under line.
\begin{axis}[
  name=right,
  at={(left.right of origin)}, anchor=left of origin,
  xshift=1.4cm,
  width=7.5cm, height=6cm,
  xlabel={field $H$},
  ylabel={$B = \mu_0 H$},
  xmin=0, xmax=1.1, ymin=0, ymax=1.1,
  xtick=\empty, ytick=\empty,
  axis lines=left,
  font=\scriptsize,
  title={magnetic: $u_m=\tfrac{1}{2}\mu_0 H^2=\tfrac{B^2}{2\mu_0}$},
  title style={font=\scriptsize},
]
\addplot[engred, line width=1.2pt, domain=0:1] {x};
\addplot[engred!18, fill=engred!18, draw=none] coordinates {
  (0,0) (1,0) (1,1)
} \closedcycle;
\node[engred, font=\scriptsize] at (axis cs:0.62,0.28)
  {$\displaystyle\int_0^{H}\! B\,\dd H'$};
\draw[enggray, dashed] (axis cs:1,0) -- (axis cs:1,1);
\end{axis}
\end{tikzpicture}
\caption[Energy density stored in the electric and magnetic fields]{Energy
density stored in an electric field (left) and a magnetic field (right),
each as the area under its linear constitutive line. Establishing the
field is the work of moving the last increment of charge against the field
already present, or driving the last increment of current against the
back-EMF already present, so the stored energy density is the integral
$\int_0^E D\,\dd E'$ or $\int_0^H B\,\dd H'$. For a linear medium the line
is straight and the integral is the triangle, half the base times the
height: $u_e=\tfrac12\varepsilon_0 E^2$ and
$u_m=\tfrac12\mu_0 H^2 = B^2/(2\mu_0)$. The factor of one half is the
geometry of the triangle, the same factor that appears in
$\tfrac12 CV^2$ and $\tfrac12 LI^2$. A nonlinear medium replaces the
straight line by a curve, and the stored energy is then the area under
that curve rather than a triangle, which is why a hysteretic core does not
return all the energy put into it.}
\label{fig:vol04ch08:field-energy-density}
\end{figure}
