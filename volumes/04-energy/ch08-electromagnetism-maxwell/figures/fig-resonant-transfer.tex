% Figure: fraction of available power delivered across a loosely coupled
% resonant link against the figure of merit k^2 Q1 Q2. A loosely coupled
% pair (small k) still transfers well if the quality factors are high, the
% principle the wireless-charging pad relies on.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{semilogxaxis}[
  width=11.5cm, height=7cm,
  xlabel={figure of merit $k^2 Q_1 Q_2$},
  ylabel={peak transfer efficiency $\eta_{\max}$},
  xmin=0.1, xmax=1000,
  ymin=0, ymax=1,
  legend pos=north west,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% Classic result for two coupled resonators optimally loaded:
% eta_max = U^2 / (1 + sqrt(1+U^2))^2, with U = sqrt(k^2 Q1 Q2).
% Plot against x = U^2 = k^2 Q1 Q2.
\addplot[engblue, line width=1.2pt, domain=0.1:1000, samples=200]
  {x/((1+sqrt(1+x))^2)};
\addlegendentry{$\eta_{\max} = \dfrac{U^2}{(1+\sqrt{1+U^2})^2},\ U^2=k^2Q_1Q_2$}
% mark a practical pad: k=0.3, Q1=Q2=40 => U^2 = 0.09*1600 = 144
\addplot[engred, only marks, mark=*, mark size=2pt]
  coordinates {(144,0.70)};
\node[engred, anchor=north east, font=\scriptsize] at (axis cs:130,0.68)
  {$k=0.3,\ Q=40$};
% mark a poor link: k=0.1, Q=10 => U^2 = 0.01*100 = 1
\addplot[enggray, only marks, mark=square*, mark size=2pt]
  coordinates {(1,0.17)};
\node[enggray, anchor=south west, font=\scriptsize] at (axis cs:1.1,0.18)
  {$k=0.1,\ Q=10$};
\end{semilogxaxis}
\end{tikzpicture}
\caption[Transfer efficiency of a loosely coupled resonant link]{Peak
power-transfer efficiency of two coupled resonators against the figure of
merit $k^2 Q_1 Q_2$, the product of the squared coupling coefficient and
the two quality factors. A pair coupled loosely, with $k$ well below one,
still transfers most of the available power once the figure of merit passes
a few tens, which the coils reach when their quality factors are in the
tens even though the coupling is weak. This is why a wireless-charging pad
works across a gap that couples only a third of the flux: it tunes both
coils to resonance and drives up their quality factors, trading bandwidth
for transfer at a fixed geometry. The marked practical point, coupling of a
third and quality factor of forty, sits near seventy percent efficiency,
while a poorly coupled low-quality link delivers only a fraction of the
available power.}
\label{fig:vol04ch08:resonant-transfer}
\end{figure}
