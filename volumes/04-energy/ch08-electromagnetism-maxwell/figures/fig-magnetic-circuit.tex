% Figure: a two-winding transformer drawn as a magnetic circuit, with
% the reluctance analogy shown alongside. Supports the case study and
% the reluctance-analogy mastery box.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex, scale=1.0]
% ==== left: physical core ====
\begin{scope}[shift={(0,0)}]
% core: a rectangular ring (window in the middle)
\fill[engamber!25] (-1.6,-1.6) rectangle (1.6,1.6);
\fill[white] (-0.9,-0.9) rectangle (0.9,0.9);
\draw[engamber, line width=1pt] (-1.6,-1.6) rectangle (1.6,1.6);
\draw[engamber, line width=1pt] (-0.9,-0.9) rectangle (0.9,0.9);
% primary winding on left limb
\foreach \y in {-0.7,-0.35,0,0.35,0.7}{
  \draw[engblue, line width=1pt] (-1.6,\y) ellipse [x radius=0.18, y radius=0.12];
}
\node[engblue, font=\scriptsize] at (-2.3,1.0) {$N_1$};
% secondary winding on right limb
\foreach \y in {-0.7,-0.35,0,0.35,0.7}{
  \draw[engred, line width=1pt] (1.6,\y) ellipse [x radius=0.18, y radius=0.12];
}
\node[engred, font=\scriptsize] at (2.3,1.0) {$N_2$};
% flux loop arrow inside the core
\draw[enggray, ->, line width=0.8pt] (1.25,-0.5) -- (1.25,0.5);
\node[enggray, font=\scriptsize] at (1.25,0) [right] {$\Phi$};
\node[font=\scriptsize] at (0,-2.1) {physical core};
\end{scope}
% ==== right: reluctance circuit analogy ====
\begin{scope}[shift={(6.2,0)}]
% MMF source (a battery-like symbol)
\draw[engblue, line width=1pt] (-1.6,-1.2) -- (-1.6,1.2);
\draw[engblue, line width=1.4pt] (-1.6,0.3) -- (-1.6,-0.3);
\node[engblue, font=\scriptsize] at (-2.1,0) {$\mathcal{F}=N_1 I_1$};
% loop wires
\draw[enggray, line width=0.8pt] (-1.6,1.2) -- (1.6,1.2);
\draw[enggray, line width=0.8pt] (-1.6,-1.2) -- (1.6,-1.2);
\draw[enggray, line width=0.8pt] (1.6,1.2) -- (1.6,-1.2);
% reluctance as a resistor zig-zag on the top
\draw[engred, line width=1pt] (-0.4,1.2) -- (-0.25,1.45) -- (0.05,0.95) -- (0.35,1.45) -- (0.65,0.95) -- (0.8,1.2);
\node[engred, font=\scriptsize] at (0.2,1.75) {$\mathcal{R}$};
% flux direction
\draw[enggray, ->, line width=0.8pt] (1.6,0.3) -- (1.6,-0.3);
\node[enggray, font=\scriptsize] at (1.6,0) [right] {$\Phi$};
\node[font=\scriptsize] at (0,-2.1) {reluctance circuit};
\end{scope}
% ==== analogy equation between the two ====
\node[font=\scriptsize] at (3.1,-2.7)
  {$\mathcal{F}=\Phi\,\mathcal{R}$ \quad mirrors \quad $V=I\,R$};
\end{tikzpicture}
\caption[Transformer as a magnetic circuit]{The two-winding
transformer drawn as a physical core (left) and as the reluctance
circuit it is analogous to (right). The magnetomotive force
$\mathcal{F}=N_1 I_1$ plays the role of the voltage source, the flux
$\Phi$ plays the role of the current, and the core reluctance
$\mathcal{R}=\ell/(\mu A)$ plays the role of the resistance, so that
$\mathcal{F}=\Phi\,\mathcal{R}$ mirrors $V=IR$. The analogy reduces a
distributed-field problem to a lumped network, and it is the working
tool of every transformer and inductor design in the case study that
follows.}
\label{fig:vol04ch08:magnetic-circuit}
\end{figure}
