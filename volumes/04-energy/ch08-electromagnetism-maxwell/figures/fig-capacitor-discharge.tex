% Figure: pulsed discharge of a capacitor bank into an inductive-
% resistive load. The underdamped current is a damped sinusoid; the
% critically damped case delivers the energy in a single clean pulse.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{axis}[
  width=11.5cm, height=7cm,
  xlabel={time ($\si{\micro\second}$)},
  ylabel={load current (normalised)},
  xmin=0, xmax=12,
  ymin=-0.5, ymax=1.05,
  xtick={0,2,4,6,8,10,12},
  legend pos=north east,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% underdamped: exp(-t/2) * sin(2.2 t), scaled to peak near 1
\addplot[engblue, line width=1pt, domain=0:12, samples=200]
  {1.18*exp(-0.30*x)*sin(deg(0.95*x))};
\addlegendentry{underdamped (ringing)}
% critically damped: (t) exp(-t), normalised; A t exp(-a t), peak at t=1/a
\addplot[engred, line width=1.2pt, domain=0:12, samples=200]
  {2.718*0.55*x*exp(-0.55*x)};
\addlegendentry{critically damped (single pulse)}
% zero line
\addplot[enggray, dashed, domain=0:12, samples=2] {0};
\node[engblue, anchor=south, font=\scriptsize] at (axis cs:5.2,-0.42)
  {polarity reversal stresses the bank};
\end{axis}
\end{tikzpicture}
\caption[Pulsed discharge of a capacitor bank]{Load current during the
discharge of a capacitor bank through a series inductance and
resistance, the $RLC$ transient. When the resistance is small the
circuit is underdamped and the current rings: it overshoots, reverses,
and oscillates while decaying, and the polarity reversal recharges the
bank backwards, a stress that shortens the life of electrolytic and
film capacitors not rated for reversal. When the resistance is raised to
the critical value $R = 2\sqrt{L/C}$ the current delivers the stored
energy in one clean pulse with no reversal, the operating point most
pulsed-power loads are designed around. The trade is speed against
cleanliness: the underdamped case reaches its peak sooner, the
critically damped case protects the bank.}
\label{fig:vol04ch08:capacitor-discharge}
\end{figure}
