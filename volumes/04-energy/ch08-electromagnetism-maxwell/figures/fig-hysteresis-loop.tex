% Figure: the hysteresis B-H loop of a soft magnetic material. The area
% enclosed by the loop is the energy dissipated per unit volume per cycle;
% a soft material traces a narrow loop and a hard (permanent-magnet)
% material a wide one. The loop is drawn as a smooth closed curve with the
% remanence Br and coercivity Hc marked.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex, scale=1.0]
\begin{axis}[
  width=10.5cm, height=7.5cm,
  xlabel={magnetising field $H$ ($\si{\ampere\per\meter}$)},
  ylabel={flux density $B$ ($\si{\tesla}$)},
  xmin=-400, xmax=400,
  ymin=-1.8, ymax=1.8,
  axis lines=middle,
  xtick={-300,-150,0,150,300},
  ytick={-1.5,-0.75,0,0.75,1.5},
  font=\scriptsize,
  every axis x label/.style={at={(axis description cs:0.98,0.42)},anchor=north},
  every axis y label/.style={at={(axis description cs:0.55,0.98)},anchor=east},
]
% Upper (increasing-H) branch: shifted saturating tanh, offset +Hc to the left.
\addplot[engblue, line width=1.2pt, domain=-400:400, samples=120]
  {1.55*tanh((x+60)/120)};
% Lower (decreasing-H) branch: offset -Hc to the right.
\addplot[engred, line width=1.2pt, domain=-400:400, samples=120]
  {1.55*tanh((x-60)/120)};
% Remanence and coercivity markers
\addplot[enggray, only marks, mark=*, mark size=1.6pt]
  coordinates {(0,0.74) (0,-0.74) (-95,0) (95,0)};
\node[engblue, anchor=west, font=\scriptsize] at (axis cs:12,0.9)
  {$+B_r$};
\node[engred, anchor=east, font=\scriptsize] at (axis cs:-12,-0.9)
  {$-B_r$};
\node[enggray, anchor=south, font=\scriptsize] at (axis cs:-120,0.08)
  {$-H_c$};
\node[enggray, anchor=north, font=\scriptsize] at (axis cs:120,-0.08)
  {$+H_c$};
\end{axis}
\end{tikzpicture}
\caption[Hysteresis loop of a soft magnetic material]{The $B$--$H$
hysteresis loop of a soft magnetic core. Driving the field up (upper
branch) and back down (lower branch) traces a closed loop rather than a
single-valued curve, because the domain walls in the material do not
retrace their path: the flux density lags the field. The loop crosses the
$B$ axis at the remanence $\pm B_r$, the flux left when the drive returns
to zero, and crosses the $H$ axis at the coercivity $\pm H_c$, the reverse
field needed to null the flux. The area the loop encloses is the energy
dissipated as heat per unit volume per cycle, so a core driven at
frequency $f$ dissipates that area times $f$ as hysteresis loss, the term
that the two-frequency split separates from the eddy-current loss. A soft
transformer steel traces a thin loop of small area and low coercivity; a
permanent-magnet grade traces a wide loop whose large $B_r$ and $H_c$ are
the properties that let it hold a field, and whose area is deliberately
large because such a magnet is not cycled.}
\label{fig:vol04ch08:hysteresis-loop}
\end{figure}
