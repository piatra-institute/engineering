% Figure: the potential ladder. Sources (rho, J) feed the potentials (phi, A)
% through wave equations in the Lorenz gauge, and the potentials produce the
% fields (E, B) by differentiation. A schematic flow diagram with boxes and
% arrows; no axes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  box/.style={draw, rounded corners, minimum width=2.6cm, minimum height=1.1cm,
    align=center, font=\small},
  src/.style={box, draw=engblue, very thick},
  pot/.style={box, draw=engred, very thick},
  fld/.style={box, draw=engamber, very thick},
  arr/.style={-{Latex}, thick, enggray},
]
% nodes
\node[src] (rho) at (0,1.4) {$\rho,\ \vect{J}$\\sources};
\node[pot] (phi) at (5,1.4) {$\phi,\ \vect{A}$\\potentials};
\node[fld] (EB) at (10,1.4) {$\vect{E},\ \vect{B}$\\fields};
% arrows source -> potential
\draw[arr] (rho) -- node[above, font=\scriptsize, text=engblack]
  {wave equations} node[below, font=\scriptsize, text=engblack]
  {(Lorenz gauge)} (phi);
% arrows potential -> field
\draw[arr] (phi) -- node[above, font=\scriptsize, text=engblack]
  {differentiate} (EB);
% gauge freedom loop on the potential
\draw[arr, engred, dashed] (phi.north) .. controls (5.9,3.0) and (4.1,3.0)
  .. (phi.north);
\node[engred, font=\scriptsize, anchor=south] at (5,3.0)
  {gauge freedom: $\phi\to\phi-\partial_t\chi,\ \vect{A}\to\vect{A}+\grad\chi$};
% formulas below
\node[font=\scriptsize, text=engblack, align=center] at (10,-0.4)
  {$\vect{E}=-\grad\phi-\partial_t\vect{A}$\\$\vect{B}=\curl\vect{A}$};
\node[font=\scriptsize, text=engblack, align=center] at (0,-0.4)
  {measured\\charges, currents};
\end{tikzpicture}
\caption[The potential ladder and gauge freedom]{The potentials sit between
the sources and the fields. The charge and current densities drive the scalar
potential $\phi$ and the vector potential $\vect{A}$ through wave equations
that decouple in the Lorenz gauge; the fields follow by differentiating the
potentials. The potentials are not unique: the gauge transformation
$\phi\to\phi-\partial_t\chi$, $\vect{A}\to\vect{A}+\grad\chi$ for any scalar
$\chi$ leaves the fields $\vect{E}$ and $\vect{B}$ unchanged, the freedom the
gauge condition removes. Only the fields are measured; the potentials are a
calculational convenience whose value depends on the gauge chosen.}
\label{fig:vol04ch08:potentials-gauge}
\end{figure}
