% Figure: shielding effectiveness of a thin copper sheet against a plane
% wave, split into absorption loss (rising with sqrt(f) as the sheet grows
% thick in skin depths) and reflection loss (falling with frequency for a
% plane wave). The total is their sum in decibels.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{semilogxaxis}[
  width=12cm, height=7cm,
  xlabel={frequency $f$ ($\si{\hertz}$)},
  ylabel={shielding effectiveness (\si{\decibel})},
  xmin=1e3, xmax=1e9,
  ymin=0, ymax=260,
  legend pos=north west,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% copper sheet t = 0.5 mm. delta = sqrt(2/(mu0 sigma omega)).
% Absorption A = 8.686 * t/delta = 8.686 * t * sqrt(pi f mu0 sigma)  [dB]
% with t=5e-4, mu0=4pi e-7, sigma=5.96e7:
%  sqrt(pi*mu0*sigma) = sqrt(pi*4pi e-7*5.96e7)=sqrt(2.353e2)=15.34
%  A = 8.686 * 5e-4 * 15.34 * sqrt(f) = 0.06663 * sqrt(f)
\addplot[engred, line width=1.1pt, domain=1e3:1e9, samples=120]
  {0.06663*sqrt(x)};
\addlegendentry{absorption loss $A\propto t\sqrt{f}$}
% Reflection loss for a plane wave on a good conductor:
% R = 168 - 10*log10(f * mu_r / sigma_r) dB (Schelkunoff, sigma_r rel copper=1)
%  = 168 - 10*log10(f)   (approx, mu_r=1, sigma_r=1)
\addplot[engblue, line width=1.1pt, dashed, domain=1e3:1e9, samples=120]
  {168 - 10*log10(x)};
\addlegendentry{reflection loss $R$ (plane wave)}
% Total = A + R
\addplot[englime, line width=1.5pt, domain=1e3:1e9, samples=160]
  {0.06663*sqrt(x) + 168 - 10*log10(x)};
\addlegendentry{total $\mathrm{SE}=A+R$}
\end{semilogxaxis}
\end{tikzpicture}
\caption[Shielding effectiveness of a copper sheet]{Shielding effectiveness
of a thin copper sheet against a plane wave, in decibels, split into its two
mechanisms. Reflection loss is large at low frequency, where the mismatch
between the wave impedance and the conductor's tiny surface impedance turns
most of the wave back, and it falls as the frequency rises. Absorption loss
is small at low frequency, where the sheet is thin in skin depths, and rises
as $\sqrt{f}$ once the sheet is several skin depths thick and the wave that
does enter is attenuated across it. Their sum dips through a shallow minimum
where the falling reflection term meets the rising absorption term, then
climbs steeply as absorption takes over. The dip is where a shield is
weakest, which sets the frequency a designer checks first; below it the shield
works by turning the wave back, above it by soaking it up.}
\label{fig:vol04ch08:shielding-effectiveness}
\end{figure}
