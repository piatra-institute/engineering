% Figure: coaxial-cable attenuation against frequency. Conductor loss
% rises as the square root of frequency through the skin effect;
% dielectric loss rises linearly. Their sum sets the usable bandwidth.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{loglogaxis}[
  width=11.5cm, height=7cm,
  xlabel={frequency $f$ ($\si{\hertz}$)},
  ylabel={attenuation ($\si{\decibel}$ per $100\,\si{\meter}$)},
  xmin=1e6, xmax=1e10,
  ymin=0.3, ymax=200,
  legend pos=north west,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% conductor (skin) loss ~ sqrt(f): 3 dB/100m at 1 GHz reference
\addplot[engblue, line width=1pt, domain=1e6:1e10, samples=60]
  {3.0*sqrt(x/1e9)};
\addlegendentry{conductor loss $\propto \sqrt{f}$}
% dielectric loss ~ f: 0.5 dB/100m at 1 GHz reference
\addplot[engamber, line width=1pt, domain=1e6:1e10, samples=60]
  {0.5*(x/1e9)};
\addlegendentry{dielectric loss $\propto f$}
% total
\addplot[engred, line width=1.2pt, domain=1e6:1e10, samples=80]
  {3.0*sqrt(x/1e9) + 0.5*(x/1e9)};
\addlegendentry{total attenuation}
% crossover where dielectric overtakes conductor: f = 36 GHz off-scale,
% mark instead the 1 GHz design point
\addplot[enggray, only marks, mark=*, mark size=2pt]
  coordinates {(1e9,3.5)};
\node[enggray, anchor=west, font=\scriptsize] at (axis cs:1.1e9,3.5)
  {design point $1\,\si{\giga\hertz}$};
\end{loglogaxis}
\end{tikzpicture}
\caption[Coaxial-cable attenuation against frequency]{Attenuation of a
$50\,\si{\ohm}$ polyethylene coaxial cable against frequency, split into
its two mechanisms. Conductor loss grows as $\sqrt{f}$ because the skin
depth shrinks as $f^{-1/2}$ and the resistance of the current-carrying
surface layer rises with it. Dielectric loss grows linearly with $f$
because the polarisation of the insulator absorbs power in proportion to
the rate the field reverses. Below the gigahertz range the conductor
term dominates and the cable is conductor-loss-limited; the dielectric
term overtakes it only in the tens of gigahertz, where low-loss
dielectrics and air-line construction earn their cost. The curve sets
the usable length of a run: at the design point a $30\,\si{\meter}$
cable loses about $1\,\si{\decibel}$, a tolerable budget for a
laboratory feed.}
\label{fig:vol04ch08:coax-attenuation}
\end{figure}
