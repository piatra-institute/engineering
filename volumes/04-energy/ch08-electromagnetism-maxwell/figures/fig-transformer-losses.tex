% Figure: transformer loss budget versus load fraction. Core loss is
% constant; copper loss rises as the square of load; efficiency peaks
% where the two are equal. Supports the case study.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={load fraction $x = S/S_{\text{rated}}$},
  ylabel={loss ($\si{\watt}$)},
  xmin=0, xmax=1.05,
  ymin=0, ymax=320,
  xtick={0,0.25,0.5,0.75,1.0},
  ytick={0,50,100,150,200,250,300},
  legend pos=north west,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% core loss: constant 90 W
\addplot[engamber, line width=1pt, domain=0:1.0, samples=2] {90};
\addlegendentry{core loss (constant, $90\,\si{\watt}$)}
% copper loss: 250 W at full load, scales as x^2
\addplot[engred, line width=1pt, domain=0:1.0, samples=60] {250*x^2};
\addlegendentry{copper loss ($\propto x^2$, $250\,\si{\watt}$ full load)}
% total loss
\addplot[engblue, line width=1.2pt, domain=0:1.0, samples=60] {90 + 250*x^2};
\addlegendentry{total loss}
% optimum load where copper = core: x = sqrt(90/250) = 0.60
\addplot[enggray, dashed, only marks, mark=*, mark size=2pt]
  coordinates {(0.6,90)};
\draw[enggray, dashed] (axis cs:0.6,0) -- (axis cs:0.6,180);
\node[enggray, anchor=south, font=\scriptsize] at (axis cs:0.6,182)
  {peak efficiency at $x\approx 0.60$};
\end{axis}
\end{tikzpicture}
\caption[Transformer loss budget versus load]{Loss budget of the
case-study transformer against load fraction. Core loss (hysteresis
plus eddy current) is set by the peak flux density and the supply
frequency and does not change with load. Copper loss is the $I^2 R$
dissipation in the windings and rises as the square of the load
current. Total loss is their sum, and efficiency peaks where the two
losses are equal, here near sixty percent of rated load, which is why
distribution transformers are sized so their average daily load sits
near that fraction.}
\label{fig:vol04ch08:transformer-losses}
\end{figure}
