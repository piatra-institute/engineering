% Figure: charge-relaxation time tau = epsilon/sigma against conductivity,
% on log-log axes. A metal neutralises in attoseconds; an insulator holds
% charge for seconds to hours; static-dissipative materials are engineered
% to sit in the millisecond-to-second window between them.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex]
\begin{loglogaxis}[
  width=11.5cm, height=7.2cm,
  xlabel={conductivity $\sigma$ ($\si{\siemens\per\meter}$)},
  ylabel={relaxation time $\tau = \varepsilon/\sigma$ ($\si{\second}$)},
  xmin=1e-14, xmax=1e8,
  ymin=1e-20, ymax=1e4,
  legend pos=north east,
  legend cell align=left,
  font=\scriptsize,
  grid=both,
  grid style={enggray!20},
]
% tau = eps_r * eps0 / sigma. Take a representative eps_r = 4 so
% eps = 4 * 8.85e-12 = 3.54e-11. tau = 3.54e-11 / sigma.
\addplot[engblue, line width=1.2pt, domain=1e-14:1e8, samples=40]
  {3.54e-11/x};
\addlegendentry{$\tau = \varepsilon_r\varepsilon_0/\sigma,\ \varepsilon_r=4$}
% material markers
\addplot[engred, only marks, mark=*, mark size=2pt]
  coordinates {(6e7,5.9e-19) (1e-12,3.54e1)};
\node[engred, anchor=south west, font=\scriptsize] at (axis cs:6e5,5.9e-19)
  {copper};
\node[engred, anchor=north east, font=\scriptsize] at (axis cs:1e-12,3.54e1)
  {glass};
% static-dissipative window
\addplot[engamber, only marks, mark=square*, mark size=2pt]
  coordinates {(1e-4,3.5e-7) (1e-8,3.5e-3)};
\node[engamber, anchor=south west, font=\scriptsize] at (axis cs:2e-8,3.5e-3)
  {static-dissipative};
\end{loglogaxis}
\end{tikzpicture}
\caption[Charge-relaxation time against conductivity]{Charge-relaxation
time $\tau = \varepsilon/\sigma$ against conductivity, on log-log axes, for
a representative relative permittivity of four. Any net charge placed in
the bulk of a homogeneous conductor migrates to its surface with this time
constant, leaving the interior neutral. A metal such as copper sits at the
far right with a relaxation time below an attosecond, so it is neutral the
instant it is charged; an insulator such as glass sits at the far left and
holds charge for tens of seconds. The engineered window between them, marked
as static-dissipative, is where materials for electronics-handling
flooring, wrist straps, and fuel lines are placed: their relaxation time is
long enough not to short a working circuit but short enough to bleed static
charge away before it builds to a sparking voltage. The single straight
line spans twenty-four orders of magnitude in time because conductivity
alone spans that range across engineering materials.}
\label{fig:vol04ch08:charge-relaxation}
\end{figure}
