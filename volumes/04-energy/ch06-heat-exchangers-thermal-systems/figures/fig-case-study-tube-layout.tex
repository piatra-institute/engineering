% Figure: triangular (30-degree) tube layout on the case-study tubesheet, showing
% the pitch, the tube outer diameter, the bundle outer-tube-limit circle, and the
% shell inner diameter, for the shell-and-tube design carried in the case study.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % shell inner circle and outer-tube-limit circle
  \draw[enggray, very thick] (0,0) circle (3.0);
  \draw[engblue, dashed, thick] (0,0) circle (2.65);
  % triangular pitch tube field, pitch = 0.5 cm spacing in figure units
  \def\pitch{0.5}
  \foreach \row in {-5,...,5}{
    \pgfmathsetmacro{\yy}{\row*\pitch*0.866}
    \pgfmathsetmacro{\xoff}{mod(abs(\row),2)==0 ? 0 : \pitch/2}
    \foreach \col in {-5,...,5}{
      \pgfmathsetmacro{\xx}{\col*\pitch + \xoff}
      \pgfmathsetmacro{\rr}{sqrt(\xx*\xx + \yy*\yy)}
      \ifdim \rr cm < 2.6cm
        \fill[engred] (\xx,\yy) circle (0.16);
      \fi
    }
  }
  % pitch annotation between two adjacent tubes near the centre
  \draw[engblack, thick, <->] (0,0) -- ({\pitch},0);
  \node[engblack, font=\scriptsize, anchor=south] at ({\pitch/2},0.05) {$p_t$};
  % shell diameter annotation
  \draw[enggray, thick, <->] (-3.0,-3.4) -- (3.0,-3.4);
  \node[enggray, font=\scriptsize, anchor=north] at (0,-3.45) {shell inner diameter $D_s$};
  % outer-tube-limit annotation
  \node[engblue, font=\scriptsize, anchor=west] at (1.9,2.05) {outer tube limit};
  % single tube outer diameter callout
  \draw[engred, thick, ->] (3.6,1.2) -- (0.18,0.0);
  \node[engred, font=\scriptsize, anchor=west] at (3.6,1.2) {tube $d_o$};
\end{tikzpicture}
\caption[Triangular tube layout on the case-study tubesheet]{Triangular
(30-degree) tube layout for the case-study bundle. Tubes of outer diameter $d_o$
sit on a pitch $p_t$ measured centre to centre; the dashed circle is the outer
tube limit, inside which the bundle fits, and the solid circle is the shell inner
diameter $D_s$. A triangular layout packs more tubes into a given shell than a
square layout and raises the shell-side heat-transfer coefficient through the
crossflow it forces, at the cost of mechanical cleaning access between tubes. The
count drawn here is illustrative; the case-study text fixes the working count from
the required area.}
\label{fig:vol04ch06:case-study-tube-layout}
\end{figure}
