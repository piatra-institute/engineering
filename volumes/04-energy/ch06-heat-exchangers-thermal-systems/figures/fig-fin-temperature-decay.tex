% Figure: temperature decay along a straight fin of length L, the hyperbolic-cosine
% profile that drops from the root (tube-wall) temperature toward the air
% temperature, with the shaded area giving the fin efficiency as the ratio of the
% actual driving difference integrated along the fin to the ideal (root) difference.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={distance along fin $x/L$ (root at left, tip at right)},
  ylabel={excess temperature $\theta/\theta_b$},
  xmin=0, xmax=1, ymin=0, ymax=1.05,
  axis lines=left,
  legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east},
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  samples=80, domain=0:1,
  clip=false,
]
% theta/theta_b = cosh(m L (1-x/L)) / cosh(m L), with m L = 1.6 chosen for the figure
\addplot[engred, very thick] {cosh(1.6*(1-x))/cosh(1.6)};
\addlegendentry{$\theta/\theta_b = \cosh[mL(1-x/L)]/\cosh(mL)$}
% ideal isothermal fin at theta_b
\addplot[enggray, dashed, thick] {1};
\addlegendentry{ideal fin (whole fin at root)}
% shade the area under the actual profile to suggest the efficiency integral
\addplot[engred!18, fill, draw=none, domain=0:1] {cosh(1.6*(1-x))/cosh(1.6)} \closedcycle;
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:0.06,0.30) {$\eta_f = \dfrac{\text{shaded}}{\text{box}} = \dfrac{\tanh(mL)}{mL}$};
\end{axis}
\end{tikzpicture}
\caption[Temperature decay along a straight fin]{Excess temperature along a
straight fin of uniform cross-section, normalised to the root value. Heat must
conduct out along the fin before it reaches the air, so the local temperature
difference driving convection falls from the root toward the tip, following a
hyperbolic cosine. The fin efficiency is the ratio of the heat the real fin
sheds to the heat an ideal fin held everywhere at the root temperature would
shed, which is the shaded area divided by the dashed box, and equals
$\tanh(mL)/mL$ for an adiabatic-tip straight fin. A taller or thinner fin, or one
of poorer conductor, raises $mL$ and lowers the efficiency, which is the trade
extended surface always carries.}
\label{fig:vol04ch06:fin-temperature-decay}
\end{figure}
