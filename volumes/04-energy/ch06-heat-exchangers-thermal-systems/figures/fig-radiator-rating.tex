% Figure: heat rejection of an automotive radiator against road speed (which sets the
% ram-air mass flow over the core), with the fan-only point at zero road speed marked.
% The curve rises and rolls off as the air-side film stops being the sole bottleneck.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={road speed $v$ (km/h), setting ram-air flow over the core},
  ylabel={heat rejected $\dot{Q}$ (kW)},
  xmin=0, xmax=120, ymin=0, ymax=80,
  axis lines=left,
  legend pos=south east,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  samples=80, domain=0:120,
]
% Effectiveness model: air mass flow grows with road speed, NTU = UA/Cmin rises with
% the air-side film (h ~ flow^0.6), epsilon rises and rolls off. Rendered as a smooth
% saturating curve scaled to a ~70 kW peak duty.
\addplot[engblue, very thick] {70*(1 - exp(-(0.18 + 0.020*x)))};
\addlegendentry{$\dot{Q}$ with ram air plus fan}
% Fan-only operating point at v = 0 (idling in traffic): the fan alone moves the air.
\addplot[engred, only marks, mark=*, mark size=2pt] coordinates {(0, 11.5)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:3,16) {fan only (idle in traffic)};
% Highway design point
\addplot[engamber, only marks, mark=square*, mark size=2pt] coordinates {(100, 60.3)};
\node[engamber, font=\scriptsize, anchor=north east] at (axis cs:98,57) {highway cruise};
\end{axis}
\end{tikzpicture}
\caption[Automotive radiator heat rejection against road speed]{Heat rejected by an
automotive radiator against road speed, which sets the ram-air mass flow drawn
through the finned core. The air side is the weak film, so the duty climbs steeply
with road speed as the faster air lifts the air-side coefficient and the number of
transfer units, then rolls off as the air side stops being the sole bottleneck and
the coolant-side and core conductance begin to share the resistance. The worst case
is not the highway but the idle in slow traffic at the red point, where road speed
gives no ram air and the engine-driven or electric fan alone must move enough air to
reject the heat the engine still produces; the radiator and fan are sized for that
point, not for the cruise at the amber square where ram air does most of the work.
The shape is the same saturating effectiveness the chapter's air cooler showed, now
read against a cold-side flow the driver sets with the accelerator rather than one a
plant operator sets with a fan.}
\label{fig:vol04ch06:radiator-rating}
\end{figure}
