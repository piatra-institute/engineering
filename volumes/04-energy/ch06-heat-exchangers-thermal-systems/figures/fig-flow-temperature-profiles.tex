% Figure: temperature profiles along the exchanger length for counter-flow and
% parallel-flow arrangements, showing how counter-flow holds a more uniform
% driving difference and can let the cold outlet exceed the hot outlet.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
% ==== Panel 1: counter-flow ====
\begin{scope}
  \begin{axis}[
    name=cf,
    width=7.5cm, height=6cm,
    title={counter-flow},
    title style={font=\small},
    xlabel={position along exchanger $x/L$}, ylabel={temperature ($^{\circ}$C)},
    xmin=0, xmax=1, ymin=10, ymax=130,
    axis lines=left,
    tick label style={font=\scriptsize},
    label style={font=\small},
    legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east},
    legend cell align=left,
  ]
  % hot enters at x=0 (120) leaves at x=1 (60); cold enters at x=1 (20) leaves x=0 (75)
  \addplot[engred, very thick] coordinates {(0,120) (0.25,104) (0.5,90) (0.75,73) (1,60)};
  \addlegendentry{hot}
  \addplot[engblue, very thick] coordinates {(0,75) (0.25,63) (0.5,51) (0.75,36) (1,20)};
  \addlegendentry{cold}
  \end{axis}
\end{scope}
% ==== Panel 2: parallel-flow ====
\begin{scope}[shift={(8.5cm,0)}]
  \begin{axis}[
    name=pf,

    title style={font=\small},

    xmin=0, xmax=1, ymin=10, ymax=130,
    axis lines=left,
    tick label style={font=\scriptsize},
    label style={font=\small},
    legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east},
    legend cell align=left,

  % both enter at x=0; converge toward a common temperature
  \addplot[engred, very thick] coordinates {(0,120) (0.25,98) (0.5,83) (0.75,73) (1,67)};
  \addlegendentry{hot}
  \addplot[engblue, very thick] coordinates {(0,20) (0.25,40) (0.5,52) (0.75,59) (1,63)};
  \addlegendentry{cold}
  \end{axis}
\end{scope}
\end{tikzpicture}
\caption[Temperature profiles for counter-flow and parallel-flow]{Temperature
profiles along the exchanger for the two limiting single-pass arrangements,
drawn for the same inlet temperatures and the same duty. In counter-flow the
streams move in opposite directions, the driving temperature difference stays
relatively uniform along the length, and the cold outlet can rise above the hot
outlet. In parallel-flow both streams enter at the same end, the difference is
largest at the inlet and collapses toward a common temperature, and the cold
outlet can never reach the hot outlet. Counter-flow yields the larger log-mean
temperature difference for the same terminal temperatures and is the
thermodynamically preferred configuration.}
\label{fig:vol04ch06:flow-temperature-profiles}
\end{figure}
