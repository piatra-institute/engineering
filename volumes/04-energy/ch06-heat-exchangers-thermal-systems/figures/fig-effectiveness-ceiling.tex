% Figure: the large-NTU ceiling of effectiveness plotted against the capacity-rate
% ratio, showing that counter-flow always reaches unity at infinite area while
% parallel-flow tops out at 1/(1+Cr), the gap being the duty the routing forfeits.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={capacity-rate ratio $C_r = C_{\min}/C_{\max}$},
  ylabel={effectiveness ceiling as NTU $\to\infty$},
  xmin=0, xmax=1, ymin=0.4, ymax=1.05,
  axis lines=left,
  legend pos=south west,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  samples=80,
  domain=0:1,
  clip=false,
]
% counter-flow ceiling is 1 for all Cr
\addplot[engblue, very thick] {1};
\addlegendentry{counter-flow ($\varepsilon_{\max}=1$)}
% parallel-flow ceiling is 1/(1+Cr)
\addplot[engred, very thick] {1/(1+x)};
\addlegendentry{parallel-flow ($\varepsilon_{\max}=1/(1+C_r)$)}
% shade the forfeited region between the two ceilings
\addplot[engamber!18, fill, draw=none, domain=0:1] {1/(1+x)} \closedcycle;
% mark the balanced case Cr = 1
\addplot[only marks, mark=*, engred] coordinates {(1,0.5)};
\node[engred, font=\scriptsize, anchor=north east] at (axis cs:0.98,0.53) {balanced: $0.5$};
\node[engamber!70!black, font=\scriptsize, anchor=south] at (axis cs:0.55,0.80) {duty forfeited by parallel routing};
\end{axis}
\end{tikzpicture}
\caption[Effectiveness ceilings of counter-flow and parallel-flow]{The most
effectiveness each arrangement can reach if given infinite area, plotted against
the capacity-rate ratio. Counter-flow climbs to full unity for every ratio: buy
enough area and one stream is brought all the way to the other's inlet
temperature. Parallel-flow cannot, because its two outlets must approach a common
temperature that lies between the two inlets, so its ceiling is $1/(1+C_r)$,
falling to exactly one half in the balanced case where the two capacity rates
match. The shaded gap between the curves is the duty the parallel routing forfeits
no matter how much metal is spent, which is the quantitative statement of why
counter-flow is the thermodynamically preferred arrangement and why an engineer
reaches for parallel-flow only when a mechanical or fouling constraint forces it.}
\label{fig:vol04ch06:effectiveness-ceiling}
\end{figure}
