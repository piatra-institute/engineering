% Figure: the Wilson plot. Measured total resistance 1/(U A) is plotted against the
% inverse tube-side velocity group u^{-0.8}; the data fall on a straight line whose
% slope is the tube-side film group and whose intercept is the sum of the
% velocity-independent resistances (wall, fouling, and the shell-side film).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.5cm,
  xlabel={tube-side velocity group $u^{-0.8}$ ($(\text{m/s})^{-0.8}$)},
  ylabel={measured resistance $1/(U_o)$ ($10^{-3}\,\text{m}^2\text{K}/\text{W}$)},
  xmin=0, xmax=1.2, ymin=0, ymax=2.2,
  axis lines=left,
  legend pos=north west,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% synthetic straight-line data: intercept 0.55, slope 1.30, with small scatter
\addplot[only marks, mark=*, engblue, mark size=1.6pt] coordinates {
  (0.20,0.79) (0.35,1.02) (0.50,1.22) (0.66,1.38) (0.83,1.66) (1.00,1.83)
};
\addlegendentry{measured points}
% best-fit line y = 0.55 + 1.30 x
\addplot[engred, very thick, domain=0:1.15] {0.55 + 1.30*x};
\addlegendentry{fit: intercept $=$ non-tube resistances}
% mark the intercept
\addplot[enggray, dashed, thick] coordinates {(0,0.55) (1.15,0.55)};
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:0.02,0.42) {intercept $R_{\text{wall}}+R_{f}+1/h_o$};
\node[engred, font=\scriptsize, anchor=west, rotate=27] at (axis cs:0.55,1.28) {slope $\propto 1/h_i$};
\end{axis}
\end{tikzpicture}
\caption[Wilson plot separating the tube-side film from the fixed resistances]{The
Wilson plot separates a measured overall resistance into its tube-side film and its
velocity-independent remainder. Running the exchanger at several tube-side velocities
and plotting the measured total resistance $1/U_o$ against the group $u^{-0.8}$ that
the tube-side film follows in turbulent flow yields a straight line: the slope is
the tube-side film group and the intercept is the sum of the wall, the fouling, and
the shell-side film, none of which move with tube-side velocity. A clean unit gives
one line; a fouled unit gives a parallel line lifted by the fouling resistance, so
the vertical shift of the intercept over time is a direct field measurement of the
fouling that has grown, without ever opening the exchanger.}
\label{fig:vol04ch06:wilson-plot}
\end{figure}
