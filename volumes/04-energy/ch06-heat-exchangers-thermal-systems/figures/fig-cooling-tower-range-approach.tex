% Figure: cooling-tower operating diagram. Water enters hot at the top and cools
% as it falls; the difference between hot inlet and cold outlet is the range, and
% the gap between the cold outlet and the air wet-bulb temperature is the
% approach. The wet-bulb temperature is the thermodynamic floor the tower cannot
% pass.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={fractional height down the fill (water inlet at top, $z=0$)},
  ylabel={temperature ($^{\circ}$C)},
  xmin=0, xmax=1, ymin=18, ymax=42,
  axis lines=left,
  legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east},
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% water cools from 38 (hot return, top z=0) to 29 (cold to basin, z=1)
\addplot[engblue, very thick] coordinates {(0,38) (0.25,35.2) (0.5,32.6) (0.75,30.6) (1,29)};
\addlegendentry{water temperature}
% wet-bulb temperature floor (constant 24 C) -- the thermodynamic limit
\addplot[enggray, thick, dashed] coordinates {(0,24) (1,24)};
\addlegendentry{air wet-bulb $24\,^{\circ}$C (floor)}
% range: hot inlet minus cold outlet, marked on the left
\draw[engred, thick, <->] (axis cs:0.04,38) -- (axis cs:0.04,29);
\node[engred, font=\scriptsize, anchor=west] at (axis cs:0.06,33.5) {range $=9$ K};
% approach: cold outlet minus wet-bulb, marked on the right
\draw[engamber, thick, <->] (axis cs:0.96,29) -- (axis cs:0.96,24);
\node[engamber, font=\scriptsize, anchor=east] at (axis cs:0.94,26.5) {approach $=5$ K};
\end{axis}
\end{tikzpicture}
\caption[Cooling-tower range and approach]{Water temperature falling down an
induced-draft cooling tower as it is cooled by direct contact with rising air.
The range is the temperature the water loses across the tower; the approach is
the gap left between the cooled water and the air wet-bulb temperature. The
wet-bulb temperature is the floor: no amount of fill or air can cool the water
below it, and a tight approach (here $5$ K) costs disproportionate size because
the driving difference shrinks toward zero near the floor. A tower is selected
on range and approach against the design wet-bulb, not on an outlet temperature
alone.}
\label{fig:vol04ch06:cooling-tower-range-approach}
\end{figure}
