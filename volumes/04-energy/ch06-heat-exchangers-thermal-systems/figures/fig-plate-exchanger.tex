% Figure: exploded view of a plate-and-frame exchanger, showing the corrugated
% plates clamped in a frame and the alternating hot/cold channels formed between
% them, with the counter-current port arrangement.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=latex, scale=1.0]
% draw a stack of corrugated plates as parallel slanted shading lines
\foreach \i in {0,1,2,3,4,5} {
  \pgfmathsetmacro\x{\i*1.0}
  % plate outline
  \draw[thick] (\x,0) rectangle (\x+0.4,4);
  % corrugation hatching
  \foreach \yy in {0.3,0.8,1.3,1.8,2.3,2.8,3.3,3.8} {
    \draw[enggray, thin] (\x,\yy) -- (\x+0.4,\yy-0.2);
  }
  % port holes top and bottom
  \draw[fill=white] (\x+0.1,3.6) circle (0.07);
  \draw[fill=white] (\x+0.1,0.4) circle (0.07);
  \draw[fill=white] (\x+0.3,3.6) circle (0.07);
  \draw[fill=white] (\x+0.3,0.4) circle (0.07);
}
% frame end plates (heavier)
\draw[very thick] (-0.3,-0.2) rectangle (0.0,4.2);
\draw[very thick] (5.4,-0.2) rectangle (5.7,4.2);
% tie-bolt
\draw[enggray, thick] (-0.3,2) -- (5.7,2);
% alternating channel flow arrows (hot down, cold up between successive plates)
\foreach \i in {0,2,4} {
  \pgfmathsetmacro\xa{\i*1.0+0.4}
  \draw[engred, ->, thick] (\xa+0.3,3.5) -- (\xa+0.3,0.6);
}
\foreach \i in {1,3} {
  \pgfmathsetmacro\xb{\i*1.0+0.4}
  \draw[engblue, ->, thick] (\xb+0.3,0.6) -- (\xb+0.3,3.5);
}
% labels
\node[engred, font=\scriptsize, anchor=west] at (6.0,3.4) {hot channel (down)};
\node[engblue, font=\scriptsize, anchor=west] at (6.0,2.6) {cold channel (up)};
\node[font=\scriptsize, anchor=west] at (6.0,1.6) {corrugated plates};
\node[font=\scriptsize, anchor=west] at (6.0,0.8) {corner ports};
\node[font=\scriptsize, anchor=north] at (2.7,-0.4) {clamped plate pack};
\end{tikzpicture}
\caption[Plate-and-frame exchanger construction]{Construction of a
plate-and-frame exchanger. Corrugated metal plates are clamped between two heavy
frame plates by tie-bolts, and the gaskets around each plate route the two fluids
into alternating channels so that every hot channel is bounded by two cold
channels and the streams run counter-current. The corrugations force early
turbulence and roughly triple the heat-transfer coefficient relative to a smooth
channel at the same flow, which is why plate units reach close temperature
approaches in a small volume. The gaskets cap the temperature near $180\,^{\circ}$C
and the pressure near $25\,\text{bar}$; the same bolted assembly that limits the
envelope is what lets the pack be opened for inspection and mechanical cleaning.}
\label{fig:vol04ch06:plate-exchanger}
\end{figure}
