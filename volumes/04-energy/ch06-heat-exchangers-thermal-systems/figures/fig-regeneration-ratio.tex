% Figure: regeneration ratio (effectiveness) of the regeneration section of a
% balanced plate pasteurizer as a function of NTU, which is proportional to plate
% count. The balanced case C_r = 1 gives R = NTU/(1+NTU); the curve approaches
% unity slowly, so the last few points of regeneration cost disproportionate area.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={number of transfer units $\mathrm{NTU} = UA/C_{\min}$ (proportional to plate count)},
  ylabel={regeneration ratio $R$},
  xmin=0, xmax=20, ymin=0, ymax=1,
  axis lines=left,
  legend style={font=\scriptsize, at={(0.98,0.04)}, anchor=south east},
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
  domain=0.01:20, samples=120,
]
% balanced counter-current regeneration: R = NTU/(1+NTU)
\addplot[engblue, very thick] {x/(1+x)};
\addlegendentry{$R = \mathrm{NTU}/(1+\mathrm{NTU})$ (balanced)}
% mark the 90% regeneration design point at NTU = 9
\draw[enggray, dashed] (axis cs:0,0.9) -- (axis cs:9,0.9) -- (axis cs:9,0);
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:9.2,0.7) {$R = 0.90$ at $\mathrm{NTU}=9$};
% mark the 95% point at NTU = 19 to show the diminishing return
\draw[engred, dashed] (axis cs:0,0.95) -- (axis cs:19,0.95) -- (axis cs:19,0);
\node[engred, font=\scriptsize, anchor=south east] at (axis cs:18.8,0.55) {$R = 0.95$ at $\mathrm{NTU}=19$};
\end{axis}
\end{tikzpicture}
\caption[Regeneration ratio versus plate count]{Regeneration ratio of a balanced
($C_r = 1$) plate-exchanger regeneration section against NTU, which grows in
proportion to the number of plates. The ratio rises steeply at first and then
flattens, so raising regeneration from $0.90$ to $0.95$ doubles the plate count
of the regeneration section. The dairy economics therefore settle on a
regeneration ratio in the high eighties to low nineties: high enough to slash
the heating and cooling utility, low enough to keep the plate stack and its
capital cost in hand.}
\label{fig:vol04ch06:regeneration-ratio}
\end{figure}
