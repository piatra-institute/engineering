% Figure: the three-zone temperature profile of a de-superheating condenser with
% subcooling. The vapour enters superheated, gives up sensible heat in the
% desuperheating zone, condenses at constant saturation temperature across the wide
% middle zone, then subcools as liquid; the coolant warms monotonically throughout.
% The kink in U (not drawn) at each zone boundary is why such units are zoned.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.8cm,
  xlabel={cumulative duty $\dot{Q}/\dot{Q}_{\text{total}}$},
  ylabel={temperature (\si{\degreeCelsius})},
  xmin=0, xmax=1, ymin=20, ymax=140,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Hot (process) side: desuperheat 130->80 over first 0.12 of duty, condense at 80
% over 0.12..0.85, subcool 80->55 over 0.85..1.0. Duty axis runs from hot outlet
% end (left) to hot inlet end (right) so the coolant rises left to right.
\addplot[engred, very thick] coordinates {
  (0.00,55) (0.15,80) (0.88,80) (1.00,130)
};
\addlegendentry{process (condensing) side}
% Cold (coolant) side warming monotonically 30 -> 60
\addplot[engblue, very thick] coordinates {(0.00,30) (1.00,60)};
\addlegendentry{cooling water}
% zone boundaries
\addplot[enggray, dashed] coordinates {(0.15,20) (0.15,140)};
\addplot[enggray, dashed] coordinates {(0.88,20) (0.88,140)};
\node[font=\scriptsize, anchor=south] at (axis cs:0.075,120) {subcool};
\node[font=\scriptsize, anchor=south] at (axis cs:0.50,120) {condense};
\node[font=\scriptsize, anchor=south] at (axis cs:0.94,120) {desuperheat};
\end{axis}
\end{tikzpicture}
\caption[Three-zone profile of a de-superheating condenser with subcooling]{The
temperature profile of a condenser that receives superheated vapour and delivers
subcooled liquid, read against cumulative duty from the cold end on the left. The
process side falls steeply while it gives up sensible superheat, holds flat across
the wide condensing zone where the latent heat leaves at constant saturation
temperature, then falls again as the condensate subcools. The coolant warms along a
single straight line because its sensible heat capacity is constant. The overall
coefficient is different in each zone, high in the condensing middle and lower in
the two single-phase ends, so the exchanger cannot be sized on one log-mean and one
coefficient; it is split into three zones, each sized on its own driving difference
and its own coefficient, and the areas are summed. The flat condensing plateau
carries most of the duty across the narrowest temperature difference, which is why
the condensing zone dominates the area.}
\label{fig:vol04ch06:condenser-zones}
\end{figure}
