% Figure: rating curve for the air-cooled cooler, duty and gas outlet temperature
% against the air-side mass flow set by the fan. As more air is moved the duty
% rises and the gas outlet falls, both with diminishing return as the air-side
% film coefficient flattens and the exchanger approaches its effectiveness ceiling.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={air mass flow $\dot{m}_{\text{air}}$ (kg\,s$^{-1}$)},
  ylabel={duty $\dot{Q}$ (MW)},
  xmin=20, xmax=120, ymin=1.0, ymax=3.2,
  axis y line=left,
  axis x line=bottom,
  legend style={font=\scriptsize, at={(0.02,0.98)}, anchor=north west},
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% duty rising with diminishing return (concave in air flow)
\addplot[engblue, very thick] coordinates {
  (25,1.55) (40,2.05) (55,2.38) (70,2.58) (85,2.72) (100,2.82) (115,2.89)};
\addlegendentry{duty $\dot{Q}$}
% design point marker at 70 kg/s, 2.58 MW
\addplot[only marks, mark=*, engred] coordinates {(70,2.58)};
\node[engred, font=\scriptsize, anchor=west] at (axis cs:72,2.46) {design point};
\end{axis}
\begin{axis}[
  width=12cm, height=7.5cm,

  ylabel near ticks,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize, at={(0.98,0.98)}, anchor=north east},
  legend cell align=left,
  clip=false,
]
% gas outlet falling with diminishing return
\addplot[engamber, very thick, dashed] coordinates {
  (25,98) (40,85) (55,76) (70,70) (85,66) (100,63) (115,61)};
\addlegendentry{gas outlet}
\end{axis}
\end{tikzpicture}
\caption[Rating curve for the air-cooled cooler]{Rating curve for the
case-study air-cooled cooler as the fan moves more air. The duty rises and the
gas outlet temperature falls, both with diminishing return: doubling the air flow
past the design point buys only a few more degrees of cooling because the
air-side film coefficient grows only as roughly the air velocity to the
$0.6$ power and the exchanger approaches its effectiveness ceiling. The fan power,
which grows as the cube of the air flow, is what makes operating far up the curve
uneconomic, so the design point sits where the marginal degree of cooling no
longer earns its fan power.}
\label{fig:vol04ch06:aircooler-rating}
\end{figure}
