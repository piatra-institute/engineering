% Figure: decline of the overall heat-transfer coefficient U over service time as
% fouling builds, with the asymptotic-fouling model, and the matching rise in
% pressure drop. Shows the cleaning sawtooth.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={service time (months)},
  ylabel={overall coefficient $U$ (W\,m$^{-2}$\,K$^{-1}$)},
  xmin=0, xmax=24, ymin=400, ymax=1300,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
]
% asymptotic fouling: U(t) = U_clean / (1 + U_clean*Rf_inf*(1 - exp(-t/tau)))
% U_clean = 1200, Rf_inf chosen so U falls toward ~650, tau ~ 6 months. First cycle.
\addplot[engblue, very thick, samples=60, domain=0:12]
  {1200/(1 + 1200*0.0007*(1 - exp(-x/6)))};
\addlegendentry{$U(t)$, asymptotic fouling}
% cleaning event at t=12: jump back near clean, then decline again
\draw[enggray, dashed, thick] (axis cs:12,650) -- (axis cs:12,1180);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:12.1,1100) {clean};
\addplot[engblue, very thick, samples=60, domain=12:24]
  {1200/(1 + 1200*0.0007*(1 - exp(-(x-12)/6)))};
% design-floor line
\addplot[engred, dashed, thick, domain=0:24] {700};
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:0.2,705) {design floor (fouled $U$)};
\end{axis}
\end{tikzpicture}
\caption[Decline of the overall coefficient under fouling]{Decline of the
overall heat-transfer coefficient as a fouling layer accumulates, following the
asymptotic-fouling model in which the deposit thickness approaches a steady value
as removal balances deposition. The exchanger is sized so that the fouled
coefficient at the end of a cleaning cycle still meets the duty (red design
floor); the gap between the clean value and the floor is the over-design margin
the fouling allowance pays for. Cleaning restores the coefficient toward the
clean value and the cycle repeats, giving the characteristic sawtooth of
operating data. Tracking the slope of this curve is the primary diagnostic for
scheduling the next cleaning.}
\label{fig:vol04ch06:fouling-decline}
\end{figure}
