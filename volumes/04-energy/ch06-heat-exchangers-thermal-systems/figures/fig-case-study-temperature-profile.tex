% Figure: temperature profile along the case-study exchanger for the kerosene-to-
% cooling-water duty, with the two terminal differences marked, drawn for the
% counter-flow arrangement the case study sizes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={position along exchanger $x/L$ (hot inlet at left)},
  ylabel={temperature ($^{\circ}$C)},
  xmin=0, xmax=1, ymin=20, ymax=170,
  axis lines=left,
  legend style={font=\scriptsize, at={(0.5,0.02)}, anchor=south},
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% hot kerosene cools 160 -> 60 from left to right
\addplot[engred, very thick] coordinates {(0,160) (0.25,135) (0.5,109) (0.75,84) (1,60)};
\addlegendentry{hot kerosene}
% cold water heats from right inlet (30) to left outlet (52) == counter-flow
\addplot[engblue, very thick] coordinates {(0,52) (0.25,46.5) (0.5,41) (0.75,35.5) (1,30)};
\addlegendentry{cooling water}
% terminal difference at hot-inlet end (left): 160 - 52 = 108
\draw[enggray, thick, <->] (axis cs:0.03,160) -- (axis cs:0.03,52);
\node[enggray, font=\scriptsize, anchor=west] at (axis cs:0.05,106) {$\Delta T_1 = 108$ K};
% terminal difference at hot-outlet end (right): 60 - 30 = 30
\draw[enggray, thick, <->] (axis cs:0.97,60) -- (axis cs:0.97,30);
\node[enggray, font=\scriptsize, anchor=east] at (axis cs:0.95,45) {$\Delta T_2 = 30$ K};
\end{axis}
\end{tikzpicture}
\caption[Temperature profile for the case-study duty]{Temperature profile along
the case-study counter-flow exchanger as kerosene cools from $160$ to
$60\,^{\circ}$C and cooling water heats from $30$ to $52\,^{\circ}$C. The two
terminal differences, $\Delta T_1 = 108$\,K at the hot-inlet end and
$\Delta T_2 = 30$\,K at the hot-outlet end, set the log-mean temperature
difference used to size the area. The curved profiles reflect the changing
local driving difference; the log-mean condenses them into one effective
difference for the design equation.}
\label{fig:vol04ch06:case-study-temperature-profile}
\end{figure}
