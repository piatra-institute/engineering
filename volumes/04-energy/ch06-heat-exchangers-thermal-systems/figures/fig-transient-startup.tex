% Figure: the exchanger's transient warm-up after flow starts, the outlet approaching
% its steady value as a first-order exponential set by the metal thermal time constant
% tau = m c / (UA); the curve reaches 63 percent of the step in one tau and settles
% within a few percent after three, which is the lag a control loop must respect.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={time since flow start (multiples of $\tau$)},
  ylabel={outlet approach to steady value (\%)},
  xmin=0, xmax=5, ymin=0, ymax=105,
  axis lines=left,
  legend pos=south east,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  samples=80,
  domain=0:5,
  clip=false,
]
% first-order approach 100(1 - e^{-t/tau})
\addplot[engblue, very thick] {100*(1 - exp(-x))};
\addlegendentry{$1 - e^{-t/\tau}$ metal warm-up}
% asymptote at 100 percent
\addplot[enggray, dashed, domain=0:5] {100};
% one-tau marker at 63 percent
\addplot[only marks, mark=*, engred] coordinates {(1,63.2)};
\node[engred, font=\scriptsize, anchor=north west] at (axis cs:1.05,60) {$63\%$ at $t=\tau$};
% three-tau marker at 95 percent
\addplot[only marks, mark=square*, engamber] coordinates {(3,95.0)};
\node[engamber, font=\scriptsize, anchor=north west] at (axis cs:3.05,90) {$95\%$ at $t=3\tau$};
% vertical guide at one tau
\addplot[enggray, dotted] coordinates {(1,0) (1,63.2)};
\end{axis}
\end{tikzpicture}
\caption[Transient warm-up of an exchanger outlet]{The outlet temperature of a heat
exchanger approaching its steady value after flow starts, plotted against time in
multiples of the metal thermal time constant $\tau = m_{\text{metal}}
c_{\text{metal}}/(UA)$. The metal, and with it the outlet, warms as a first-order
exponential because the exchanger cannot deliver heat faster than its own mass
absorbs it, reaching $63\%$ of the step in one time constant and settling within a
few percent after three. A massive shell unit carries a large metal mass and lags
for minutes, while a thin-plate exchanger holds little metal and settles in seconds,
which is why plate units are favoured where a hold temperature must not droop when
the feed rate steps. A control loop tuned faster than this lag hunts, because the
setpoint cannot be tracked faster than the metal warms, so the time constant read
off this curve sets the fastest useful loop tuning.}
\label{fig:vol04ch06:transient-startup}
\end{figure}
