% Figure: forced-draft air-cooled (fin-fan) condenser arrangement, showing the
% finned-tube bundle in the horizontal plane, the air drawn up through it by the
% fan below, and the process gas making two passes across the bundle. Drawn for
% the third case study.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % bundle box (finned-tube section), viewed from the side
  \draw[enggray, very thick] (-3.4,1.4) rectangle (3.4,2.4);
  \node[enggray, font=\scriptsize] at (0,2.85) {finned-tube bundle (process gas inside)};
  % finned tubes drawn as a row of small circles with fin hatching
  \foreach \x in {-3.0,-2.4,-1.8,-1.2,-0.6,0,0.6,1.2,1.8,2.4,3.0}{
    \fill[engred] (\x,1.9) circle (0.12);
    \draw[engred, thin] (\x,1.72) -- (\x,2.08);
  }
  % process gas headers at the two ends
  \draw[engblue, very thick] (-3.9,1.3) rectangle (-3.4,2.5);
  \draw[engblue, very thick] (3.4,1.3) rectangle (3.9,2.5);
  \node[engblue, font=\scriptsize, anchor=east] at (-3.95,2.6) {gas in};
  \node[engblue, font=\scriptsize, anchor=west] at (3.95,1.2) {gas out};
  \draw[engblue, thick, ->] (-4.6,2.4) -- (-3.95,2.4);
  \draw[engblue, thick, ->] (3.95,1.5) -- (4.6,1.5);
  % air drawn up by the fan below
  \foreach \x in {-2.4,-1.2,0,1.2,2.4}{
    \draw[englime, thick, ->] (\x,-1.4) -- (\x,1.3);
  }
  \node[englime, font=\scriptsize, anchor=north] at (0,-1.5) {ambient air drawn up through the bundle};
  % fan ring below
  \draw[engblack, thick] (-1.6,-1.0) -- (1.6,-1.0);
  \draw[engblack, thick] (-0.2,-1.0) -- (-0.9,-0.55);
  \draw[engblack, thick] (0.2,-1.0) -- (0.9,-1.45);
  \node[engblack, font=\scriptsize, anchor=north] at (0,-1.05) {fan};
\end{tikzpicture}
\caption[Forced-draft air-cooled condenser arrangement]{Forced-draft air-cooled
(fin-fan) cooler for the third case study. Process gas flows inside the finned
tubes; a fan below the bundle draws ambient air up across the finned surface. The
fins multiply the air-side area by an order of magnitude to compensate for the
low air-side film coefficient, and the fan power is the running cost the design
must keep small. The bundle is shown with two gas passes; the air makes a single
unmixed pass, the cross-flow arrangement the rating uses.}
\label{fig:vol04ch06:aircooler-arrangement}
\end{figure}
