% Figure: flowsheet of an HTST milk pasteurizer built from three plate-exchanger
% sections on one frame, the regenerator recovering heat from the hot pasteurized
% stream into the cold raw stream, with a heater section to reach holding
% temperature and a cooler section to chill the product.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  font=\scriptsize,
  sect/.style={draw, thick, minimum width=2.2cm, minimum height=1.9cm, align=center},
  raw/.style={engblue, very thick, -{Latex}},
  past/.style={engred, very thick, -{Latex}},
  util/.style={enggray, thick, -{Latex}},
]
% three plate sections on the frame
\node[sect] (regen) at (0,0) {regeneration\\section};
\node[sect] (heat) at (4,0) {heating\\section};
\node[sect] (cool) at (8,0) {cooling\\section};
% holding tube above the heater
\node[draw, thick, minimum width=2.8cm, minimum height=0.55cm, align=center] (hold) at (4,2.6) {holding tube\\$72\,^{\circ}$C, $15$ s};

% cold raw milk in at 4 C into bottom of regenerator
\draw[raw] (-3.0,-0.5) node[anchor=east]{raw $4\,^{\circ}$C} -- (-1.1,-0.5);
% raw preheated, leaves top of regenerator up and across to heater
\draw[raw] (0,0.95) -- (0,1.6) -- (4,1.6) -- (4,0.95);
\node[engblue, anchor=south, font=\tiny] at (2,1.6) {preheated $\sim64\,^{\circ}$C};
% heater to holding tube
\draw[raw] (4,0.95) -- (4,2.33);
% holding tube returns hot pasteurized milk to top-left of regenerator
\draw[past] (2.6,2.6) -- (-1.6,2.6) -- (-1.6,0.5) -- (-1.1,0.5);
\node[engred, anchor=south, font=\tiny] at (0.5,2.6) {pasteurized $72\,^{\circ}$C};
% pasteurized milk gives up heat in regenerator, exits low toward cooler
\draw[past] (1.1,-0.3) -- (2.9,-0.3);
\node[engred, anchor=north, font=\tiny] at (2,-0.3) {$\sim12\,^{\circ}$C};
\draw[past] (5.1,-0.3) -- (6.9,-0.3);

\draw[past] (9.1,-0.5) -- (10.6,-0.5) node[anchor=west]{product $4\,^{\circ}$C};

% utilities
\draw[util] (4,-0.95) -- (4,-1.7) node[anchor=north]{hot water};

\end{tikzpicture}
\caption[Plate pasteurizer with regeneration]{An HTST milk pasteurizer built as
three plate-exchanger sections on one frame. Cold raw milk enters the
regeneration section and is preheated by the outgoing hot pasteurized stream,
then a small heating section lifts it the last few kelvin to the holding
temperature, where it is held in the holding tube for the legally required time.
The hot pasteurized milk returns through the regeneration section, giving its
heat to the incoming raw milk, before a final cooling section chills the product.
The regeneration section carries most of the duty; the heater and cooler supply
only the difference the regenerator cannot recover, which is what makes the unit
cheap to run.}
\label{fig:vol04ch06:pasteurizer-flowsheet}
\end{figure}
