% Figure: the concentration-polarisation boundary layer at a rejecting
% membrane. Rejected solute builds toward the wall in an exponential film,
% raising the wall concentration above the bulk by the factor exp(Jw/k). Two
% curves show a well-swept film (high k) and a poorly swept one (low k).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={distance from membrane wall $y/\delta$},
  ylabel={solute concentration $c/c_b$},
  xmin=0, xmax=1.2, ymin=0.8, ymax=3.2,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=0:1,
]
% Poorly swept film, Jw/k = 1.0: c/cb = exp((1-y/delta)*1.0), wall at exp(1)=2.72
\addplot[engred, very thick] {exp((1-x)*1.0)};
\addlegendentry{poorly swept, $J_w/k = 1.0$}
% Well swept film, Jw/k = 0.3: wall at exp(0.3)=1.35
\addplot[engblue, very thick, dashed] {exp((1-x)*0.3)};
\addlegendentry{well swept, $J_w/k = 0.3$}
% bulk line beyond the film
\addplot[engblack, thin] coordinates {(1.0,1.0) (1.2,1.0)};
\node[anchor=west, font=\footnotesize] at (axis cs:1.02,1.15) {bulk};
% membrane wall marker
\draw[enggray, very thick] (axis cs:0,0.8) -- (axis cs:0,3.2);
\node[anchor=south, rotate=90, font=\footnotesize, enggray] at (axis cs:-0.03,2.0) {membrane};
\end{axis}
\end{tikzpicture}
\caption[The concentration-polarisation film at a rejecting
membrane]{The concentration-polarisation boundary layer at a rejecting
membrane, with the rejected solute swept toward the wall by the permeate
flow and diffusing back toward the bulk. The wall concentration rises above
the bulk by the exponential factor $\exp(J_w/k)$, so a poorly swept film,
where the permeation velocity $J_w$ is large against the mass-transfer
coefficient $k$, piles solute far higher at the wall than a well-swept one.
The raised wall concentration lifts the local osmotic pressure exactly where
the membrane is working hardest, stealing driving force, which is why a unit
sized on the bulk osmotic pressure underdelivers flux and why the cure is
always to raise $k$ by faster cross-flow rather than to raise the applied
pressure.}
\label{fig:vol04ch07:polarisation-profile}
\end{figure}
