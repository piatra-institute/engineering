% Figure: evaporation from a liquid surface into a moving air stream.
% Saturation at the surface, drier bulk; wind sweeps vapour away.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % liquid pool
  \fill[engblue!20] (0,0) rectangle (8,-1.0);
  \draw[thick] (0,0) -- (8,0);
  \node[font=\scriptsize, engblue] at (4,-0.55) {liquid pool};
  % wavy surface hint
  \draw[engblue, thick] (0,0) sin (1,0.1) cos (2,0) sin (3,0.1) cos (4,0)
    sin (5,0.1) cos (6,0) sin (7,0.1) cos (8,0);
  % evaporation arrows (vapour leaving surface)
  \foreach \x in {1,2.5,4,5.5,7}{
    \draw[->, engred, thick] (\x,0.15) -- (\x,1.1);
  }
  \node[font=\scriptsize, engred] at (1.0,1.4) {vapour flux $N_A$};
  % wind arrows
  \foreach \y in {1.7,2.3,2.9}{
    \draw[->, engamber, thick] (-0.6,\y) -- (1.0,\y);
  }
  \node[font=\scriptsize, engamber] at (0.2,3.3) {wind $u$};
  % concentration labels
  \node[font=\scriptsize] at (4,0.6) {$p_{A,s}=p_{\text{sat}}(T_s)$ at surface};
  \node[font=\scriptsize] at (5.6,2.9) {$p_{A,\infty}=\phi\,p_{\text{sat}}(T_\infty)$ in bulk};
\end{tikzpicture}
\caption[Evaporation from a liquid surface into moving air]{Evaporation
from a free liquid surface. The air at the surface is saturated at the
surface temperature, $p_{A,s}=p_{\text{sat}}(T_s)$; the bulk air carries
the vapour at $p_{A,\infty}=\phi\,p_{\text{sat}}(T_\infty)$ for relative
humidity $\phi$. The wind thins the boundary layer and raises the
mass-transfer coefficient, so the flux rises with wind speed. In the
constant-rate drying period the surface holds at the wet-bulb
temperature, where evaporative cooling balances the convective heat
supply.}
\label{fig:vol04ch07:evaporation}
\end{figure}
