% Figure: transient diffusion into a semi-infinite medium, error-function profiles
% at increasing times. The penetration depth grows as sqrt(D t).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=7cm,
  xlabel={depth $x$ (mm)},
  ylabel={normalised concentration $(c-c_0)/(c_s-c_0)$},
  domain=0:8, samples=80,
  xmin=0, xmax=8, ymin=0, ymax=1.05,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!20},
  every axis plot/.append style={very thick},
]
% erfc(z) profiles, argument z = x/(2 sqrt(D t)).
% pgfmath has no erf, so use a smooth closed-form stand-in for the
% complementary error function: erfc(z) approx exp(-1.2*z - 0.6*z^2),
% accurate enough for an illustrative monotone profile.
% pick 2 sqrt(D t) = 1, 2, 3 mm respectively (so z = x, x/2, x/3).
\addplot[engred, domain=0:8] {exp(-1.2*(x) - 0.6*(x)^2)};
\addlegendentry{$2\sqrt{Dt}=1$ mm (early)}
\addplot[engamber, domain=0:8] {exp(-1.2*(x/2) - 0.6*(x/2)^2)};
\addlegendentry{$2\sqrt{Dt}=2$ mm}
\addplot[engblue, domain=0:8] {exp(-1.2*(x/3) - 0.6*(x/3)^2)};
\addlegendentry{$2\sqrt{Dt}=3$ mm (late)}
\end{axis}
\end{tikzpicture}
\caption[Transient diffusion into a semi-infinite medium]{Error-function
concentration profiles for transient diffusion into a semi-infinite
medium held at fixed surface concentration. The solution is
$(c-c_0)/(c_s-c_0)=\operatorname{erfc}\!\big(x/(2\sqrt{Dt})\big)$. The
penetration depth scales as $\sqrt{Dt}$, so doubling the depth reached
takes four times as long. The same profile governs transient heat
conduction into a semi-infinite solid and case-hardening of steel by
carbon diffusion.}
\label{fig:vol04ch07:transient-penetration}
\end{figure}
