% Figure: concentration boundary layer developing over a flat plate.
% The concentration BL thickness grows along the plate; flux is set by the wall gradient.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % plate
  \fill[enggray!25] (0,0) rectangle (8,-0.3);
  \draw[thick] (0,0) -- (8,0);
  \node[font=\scriptsize] at (4,-0.6) {wetted plate surface, $c_{A,s}$};
  % bulk flow arrows
  \foreach \y in {2.2,2.7,3.2}{
    \draw[->, engblue] (-0.6,\y) -- (0.6,\y);
  }
  \node[font=\scriptsize, engblue] at (-0.2,3.6) {$u_\infty,\ c_{A,\infty}$};
  % concentration boundary layer edge (grows as sqrt x)
  \draw[engred, very thick, samples=60, domain=0.05:8, variable=\xb]
    plot ({\xb}, {0.85*sqrt(\xb)});
  \node[font=\scriptsize, engred] at (7.0,2.7) {$\delta_c(x)$};
  % concentration profile insets at two stations
  \foreach \xs in {2.5,5.5}{
    \draw[enggray] (\xs,0) -- (\xs,{0.85*sqrt(\xs)});
    \draw[engamber, thick, samples=30, domain=0:1, variable=\t]
      plot ({\xs + 0.7*(1-(1-\t)*(1-\t))}, {0.85*sqrt(\xs)*\t});
  }
  \node[font=\scriptsize, engamber] at (3.6,1.1) {$c_A(y)$};
\end{tikzpicture}
\caption[Concentration boundary layer over a flat plate]{The
concentration boundary layer over a soluble or evaporating flat plate.
Species leave the surface at concentration $c_{A,s}$ and mix into the
free stream at $c_{A,\infty}$; the boundary-layer thickness $\delta_c$
grows along the plate. The wall concentration gradient sets the local
flux, which the Sherwood-number correlation
$\mathrm{Sh}_x=0.332\,\mathrm{Re}_x^{1/2}\mathrm{Sc}^{1/3}$ packages into
a mass-transfer coefficient. For liquids the Schmidt number is large, so
$\delta_c$ is much thinner than the velocity boundary layer.}
\label{fig:vol04ch07:boundary-layer}
\end{figure}
