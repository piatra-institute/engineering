% Figure: the distillation trade between reflux ratio and stage count. Stages
% diverge as reflux approaches its minimum and fall toward N_min as reflux
% grows without bound. The economic operating band sits at 1.2 to 1.5 times
% minimum reflux, where the curve has turned its knee.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={reflux ratio $R$},
  ylabel={number of equilibrium stages $N$},
  xmin=1.0, xmax=6.0, ymin=0, ymax=40,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=200, domain=1.55:6.0,
]
% Illustrative stage-reflux curve: diverges at R_min = 1.44, falls to N_min = 6.4.
% N = N_min + k/(R - R_min), a hyperbola in R with the right asymptotes.
\addplot[engblue, very thick] {6.4 + 3.2/(x - 1.44)};
\addlegendentry{stages $N(R)$}
% N_min asymptote
\addplot[engamber, dashed, thick] coordinates {(1.0,6.4) (6.0,6.4)};
\addlegendentry{$N_{\min} = 6.4$ (total reflux)}
% R_min asymptote (vertical)
\addplot[engred, dashed, thick] coordinates {(1.44,0) (1.44,40)};
\addlegendentry{$R_{\min} = 1.44$}
% economic operating band 1.2 to 1.5 R_min = 1.73 to 2.16
\addplot[englime!60!black, ultra thick] coordinates {(1.73,0) (1.73,40)};
\addplot[englime!60!black, ultra thick] coordinates {(2.16,0) (2.16,40)};
\node[anchor=south, font=\footnotesize, englime!50!black] at (axis cs:1.95,30) {economic band};
\end{axis}
\end{tikzpicture}
\caption[The reflux-against-stages trade in distillation]{The trade a
distillation designer turns between reflux ratio and equilibrium-stage count.
As the reflux falls toward its minimum the stages diverge, because the
operating line closes on the equilibrium curve and each stage buys a
vanishing step; as the reflux grows without bound the stage count falls
toward the total-reflux minimum set by the Fenske equation. Neither limit is
built: infinite stages cost infinite shell, infinite reflux costs infinite
reboiler and condenser duty. The economic operating band sits at $1.2$ to
$1.5$ times the minimum reflux, at the knee of the curve, where a small
increase in reflux still buys a useful drop in stages before the return
flattens. The curve is the reason both the stage count and the reflux ratio
are design variables, and why a column is sized on the knee rather than at
either asymptote.}
\label{fig:vol04ch07:reflux-stages}
\end{figure}
