% Figure: diffusion through a membrane/film with partition jumps at the faces.
% Solubility partition makes the concentration discontinuous at each interface.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % membrane block
  \fill[engblue!8] (3,0) rectangle (6,4);
  \draw[thick] (3,0) rectangle (6,4);
  \node[font=\scriptsize] at (4.5,-0.4) {membrane};
  % upstream and downstream fluid regions
  \node[font=\scriptsize, align=center] at (1.6,3.6) {feed\\ fluid};
  \node[font=\scriptsize, align=center] at (7.4,3.6) {permeate\\ fluid};
  % axis hint
  \draw[->, thick] (0.4,0) -- (8.4,0) node[right, font=\small] {$x$};
  \draw[->, thick] (0.4,0) -- (0.4,4.3) node[above, font=\small] {$c$};
  % feed bulk concentration (flat then partition jump up into membrane)
  \draw[engred, very thick] (0.8,3.3) -- (3,3.3);
  \draw[engred, dashed] (3,3.3) -- (3,3.8);
  % membrane-side high concentration at left face
  \draw[engblue, very thick] (3,3.8) -- (6,1.4);
  % partition jump down at right face to permeate bulk
  \draw[engred, dashed] (6,1.4) -- (6,0.9);
  \draw[engred, very thick] (6,0.9) -- (8.1,0.9);
  % labels for partition jumps
  \node[font=\scriptsize, engblue, right] at (3.05,3.95) {$c_{m,1}=Sc_1$};
  \node[font=\scriptsize, engblue, left] at (5.95,1.25) {$c_{m,2}=Sc_2$};
  \node[font=\scriptsize, engred, left] at (2.95,3.3) {$c_1$};
  \node[font=\scriptsize, engred, right] at (6.05,0.9) {$c_2$};
\end{tikzpicture}
\caption[Diffusion through a membrane with partition jumps]{Steady
permeation through a dense membrane. The dissolved-species concentration
inside the membrane is set by the partition (solubility) coefficient $S$
at each face, so the concentration is discontinuous at the two interfaces
even though the chemical potential is continuous. The permeability
$P=DS$ collapses the diffusivity and the solubility into one design
number; the flux is $N_A=P(c_1-c_2)/L$ for a feed-to-permeate
concentration drop across a membrane of thickness $L$.}
\label{fig:vol04ch07:membrane-flux}
\end{figure}
