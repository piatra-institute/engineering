% Figure: reverse-osmosis driving force and specific energy versus recovery.
% As recovery rises the brine concentrates, its osmotic pressure climbs, the
% net driving pressure shrinks, and the specific energy per cubic metre of
% permeate passes through a minimum then climbs as the feed pump must lift an
% ever larger reject flow. Two curves on a shared recovery axis.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.2cm,
  xlabel={recovery $r$ (permeate fraction of feed)},
  ylabel={brine osmotic pressure $\pi_b$ (MPa)},
  xmin=0, xmax=0.7, ymin=2, ymax=10,
  axis y line*=left,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% brine osmotic pressure pi_b = pi_f /(1-r), pi_f = 2.7 MPa (perfect rejection)
\addplot[engred, very thick, domain=0:0.65, samples=80] {2.7/(1-x)};
\addlegendentry{brine osmotic pressure $\pi_b=\pi_f/(1-r)$}
% applied feed pressure held at 7 MPa for reference
\addplot[engblue, thick, dashed, domain=0:0.7] {7};
\addlegendentry{feed pressure $\Delta p = 7$ MPa}
% design recovery marker
\addplot[only marks, mark=*, mark size=1.6pt, black] coordinates {(0.45,4.909)};
\node[font=\scriptsize, anchor=west] at (axis cs:0.46,4.4) {design $r=0.45$};
\end{axis}
\begin{axis}[
  width=12cm, height=7.2cm,
  xmin=0, xmax=0.7, ymin=2, ymax=6,
  axis y line*=right,
  ylabel={specific energy (kWh/m$^3$)},
  axis x line=none,
  legend pos=north east,
  legend cell align=left,
]
% schematic specific energy: U-shaped, minimum near r = 0.45 to 0.5
\addplot[engamber, very thick, domain=0.1:0.65, samples=80]
  {2.6 + 9*(x-0.47)^2 + 0.02/x};
\addlegendentry{specific energy (with recovery device)}
\end{axis}
\end{tikzpicture}
\caption[Reverse-osmosis driving force and specific energy versus
recovery]{The two competing pressures of a single-pass seawater
reverse-osmosis stage. As the recovery $r$ rises, the brine concentrates
by the factor $1/(1-r)$ and its osmotic pressure (left axis, red) climbs
toward the applied feed pressure (blue dashed). The net driving pressure
$\Delta p - \pi_b$ shrinks, so the last increment of permeate is the most
expensive. The specific energy per cubic metre of permeate (right axis,
amber) is U-shaped: at low recovery the feed pump lifts a large reject
flow for little product, while at high recovery the osmotic
back-pressure dominates. The minimum near $r\approx 0.45$ to $0.50$ is
where seawater units run, with an energy-recovery device returning the
pressure of the reject brine to the incoming feed.}
\label{fig:vol04ch07:ro-recovery}
\end{figure}
