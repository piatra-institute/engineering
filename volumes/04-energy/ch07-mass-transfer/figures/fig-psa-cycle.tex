% Figure: the four-step Skarstrom pressure-swing-adsorption cycle on a
% pressure-versus-time axis for a two-bed unit. Bed A and bed B run the same
% cycle a half-period out of phase, so that one bed adsorbs at high pressure
% while the other regenerates at low pressure. The four steps are
% pressurisation, high-pressure adsorption (feed/production), blowdown, and
% low-pressure purge.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.4cm,
  xlabel={time within one cycle},
  ylabel={bed pressure (bar)},
  xmin=0, xmax=8, ymin=0, ymax=7,
  xtick={0,2,4,6,8},
  xticklabels={0,$t_1$,$t_2$,$t_3$,$t_4$},
  ytick={1,6},
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% bed A: pressurise 0-2, adsorb 2-4 at 6 bar, blowdown 4-6, purge 6-8 at 1 bar
\addplot[engblue, very thick] coordinates {
  (0,1) (2,6) (4,6) (6,1) (8,1)
};
\addlegendentry{bed A}
% bed B: half period out of phase
\addplot[engred, very thick, dashed] coordinates {
  (0,6) (2,1) (4,1) (6,6) (8,6)
};
\addlegendentry{bed B}
% step labels for bed A
\node[font=\scriptsize, anchor=south, engblue] at (axis cs:1.0,2.6) {pressurise};
\node[font=\scriptsize, anchor=south, engblue] at (axis cs:3.0,6.05) {adsorb};
\node[font=\scriptsize, anchor=north, engblue] at (axis cs:5.0,3.4) {blowdown};
\node[font=\scriptsize, anchor=south, engblue] at (axis cs:7.0,1.05) {purge};
\node[font=\scriptsize, anchor=west, enggray] at (axis cs:0.1,6.55) {$p_H = 6$ bar};
\node[font=\scriptsize, anchor=west, enggray] at (axis cs:0.1,0.5) {$p_L = 1$ bar};
\end{axis}
\end{tikzpicture}
\caption[The four-step Skarstrom pressure-swing-adsorption cycle]{The
four-step Skarstrom cycle for a two-bed pressure-swing-adsorption unit.
Each bed cycles through pressurisation to the high pressure $p_H$,
high-pressure adsorption while feed passes and product is withdrawn,
blowdown to the low pressure $p_L$, and low-pressure purge with a slip of
product gas that sweeps the desorbed contaminant out. The two beds run a
half-period out of phase, so one bed always produces while the other
regenerates, and the feed sees a continuous high-pressure sink. The
working capacity that does the separation is the difference between the
loading at $p_H$ and the loading at $p_L$, the swing the next figure
reads off the isotherm.}
\label{fig:vol04ch07:psa-cycle}
\end{figure}
