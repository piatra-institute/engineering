% Figure: linear steady concentration profile across a slab, Fick's first law.
% Constant flux drives a straight-line concentration drop.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % axes
  \draw[->, thick] (0,0) -- (6.3,0) node[right, font=\small] {$x$};
  \draw[->, thick] (0,0) -- (0,4.2) node[above, font=\small] {$c_A$};
  % slab boundaries
  \draw[dashed, enggray] (1,0) -- (1,3.7);
  \draw[dashed, enggray] (5,0) -- (5,1.0);
  \node[font=\scriptsize] at (1,-0.35) {$x=0$};
  \node[font=\scriptsize] at (5,-0.35) {$x=L$};
  % concentration values
  \node[font=\scriptsize, left] at (0,3.5) {$c_{A,1}$};
  \node[font=\scriptsize, left] at (0,1.0) {$c_{A,2}$};
  \draw[dashed, enggray] (0,3.5) -- (1,3.5);
  \draw[dashed, enggray] (0,1.0) -- (5,1.0);
  % linear profile
  \draw[engblue, very thick] (1,3.5) -- (5,1.0);
  % flux arrow
  \draw[->, engred, thick] (2.6,2.7) -- (3.6,2.07)
    node[midway, above, font=\scriptsize, sloped] {$J_A=-D_{AB}\,dc_A/dx$};
  % constant gradient label
  \node[font=\scriptsize, engblue] at (3.0,1.45) {constant gradient};
\end{tikzpicture}
\caption[Steady linear concentration profile across a slab]{Steady
diffusion across a slab of thickness $L$ with fixed surface concentrations
$c_{A,1}$ and $c_{A,2}$. With constant diffusivity and no generation, the
steady profile is linear and the molar flux $J_A=-D_{AB}\,dc_A/dx$ is
uniform across the slab. The same straight-line solution governs steady
heat conduction through a plane wall, with temperature replacing
concentration and $\alpha$ replacing $D_{AB}$.}
\label{fig:vol04ch07:fick-profile}
\end{figure}
