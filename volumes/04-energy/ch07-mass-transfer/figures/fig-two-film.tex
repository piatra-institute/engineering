% Figure: two-film theory at a gas-liquid interface.
% Two stagnant films carry the resistance; equilibrium holds at the interface.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, >=latex]
  % interface
  \draw[very thick] (4,-0.2) -- (4,4.2);
  \node[font=\scriptsize] at (4,4.5) {interface};
  % film boundaries
  \draw[dashed, enggray] (2.4,-0.2) -- (2.4,4.2);
  \draw[dashed, enggray] (5.6,-0.2) -- (5.6,4.2);
  \node[font=\scriptsize] at (1.0,3.9) {gas bulk};
  \node[font=\scriptsize] at (7.0,3.9) {liquid bulk};
  \node[font=\scriptsize, align=center] at (3.2,-0.6) {gas\\ film};
  \node[font=\scriptsize, align=center] at (4.8,-0.6) {liquid\\ film};
  % gas-side composition (y), high in bulk, drops across gas film to interface y_i
  \draw[engblue, very thick] (0.4,3.4) -- (2.4,3.4);
  \draw[engblue, very thick] (2.4,3.4) -- (4,2.5);
  \node[font=\scriptsize, engblue, left] at (2.3,3.4) {$y_{A,b}$};
  \node[font=\scriptsize, engblue, above right] at (4,2.5) {$y_{A,i}$};
  % liquid-side composition (x), rises across liquid film from interface x_i to bulk
  \draw[engred, very thick] (4,1.6) -- (5.6,1.0);
  \draw[engred, very thick] (5.6,1.0) -- (7.6,1.0);
  \node[font=\scriptsize, engred, below right] at (4,1.6) {$x_{A,i}$};
  \node[font=\scriptsize, engred, right] at (5.6,1.0) {$x_{A,b}$};
  % equilibrium note at interface
  \node[font=\scriptsize, align=center] at (4,0.2) {$y_{A,i}=m\,x_{A,i}$};
\end{tikzpicture}
\caption[Two-film theory at a gas-liquid interface]{Two-film theory.
Two thin stagnant films, one on each side of the interface, carry all of
the mass-transfer resistance; the well-mixed bulks are uniform. The
phases reach equilibrium at the interface, $y_{A,i}=m\,x_{A,i}$. The
overall coefficient adds the film resistances in series,
$1/K_y=1/k_y+m/k_x$, and the phase with the larger resistance sets the
separation rate.}
\label{fig:vol04ch07:two-film}
\end{figure}
