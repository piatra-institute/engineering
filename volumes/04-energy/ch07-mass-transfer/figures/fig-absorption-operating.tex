% Figure: gas absorption operating line above equilibrium line, in mole-ratio
% coordinates; the vertical gap between them is the local driving force whose
% reciprocal is integrated to get the number of transfer units.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=7.5cm,
  xlabel={liquid solute ratio $X = x/(1-x)$},
  ylabel={gas solute ratio $Y = y/(1-y)$},
  xmin=0, xmax=0.012, ymin=0, ymax=0.022,
  scaled x ticks=false, scaled y ticks=false,
  xtick={0,0.003,0.006,0.009,0.012},
  ytick={0,0.005,0.010,0.015,0.020},
  x tick label style={/pgf/number format/fixed, /pgf/number format/precision=3, font=\scriptsize},
  y tick label style={/pgf/number format/fixed, /pgf/number format/precision=3, font=\scriptsize},
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% equilibrium line Y* = m X with m = 0.86 (SO2 into dilute caustic, working slope)
\addplot[engred, very thick, domain=0:0.012] {0.86*x};
\addlegendentry{equilibrium $Y^{*}=mX$}
% operating line from bottom (X0=0, Y1=0.0008) to top (XN=0.0108, YN+1=0.0204)
\addplot[engblue, thick] coordinates {(0,0.0008) (0.0108,0.0204)};
\addlegendentry{operating line}
% endpoints
\addplot[only marks, mark=*, mark size=1.7pt, black] coordinates {(0,0.0008) (0.0108,0.0204)};
\node[font=\scriptsize, anchor=north west] at (axis cs:0.0002,0.0012) {bottom $(X_0,Y_1)$};
\node[font=\scriptsize, anchor=south east] at (axis cs:0.0106,0.0204) {top $(X_N,Y_{N+1})$};
% a representative driving-force gap
\draw[enggray, thick, <->] (axis cs:0.0060,0.00516) -- (axis cs:0.0060,0.01140);
\node[font=\scriptsize, anchor=west] at (axis cs:0.0062,0.0083) {$Y-Y^{*}$};
\end{axis}
\end{tikzpicture}
\caption[Absorption operating and equilibrium lines]{Operating and
equilibrium lines for a dilute counter-current absorber in mole-ratio
coordinates. The operating line is the column material balance connecting
the lean-solvent, clean-gas bottom corner to the loaded-solvent,
dirty-gas top corner; the equilibrium line is the solubility relation
$Y^{*}=mX$. The vertical gap $Y-Y^{*}$ at any height is the local
driving force. Its reciprocal, integrated over the change in gas
composition, is the number of transfer units. The closer the operating
line approaches the equilibrium line, the smaller the driving force and
the taller the column.}
\label{fig:vol04ch07:absorption-operating}
\end{figure}
