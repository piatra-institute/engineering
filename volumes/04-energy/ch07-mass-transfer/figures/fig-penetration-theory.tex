% Figure: the penetration-theory concentration profiles in a semi-infinite
% medium at three successive times. The complementary error function spreads
% the front as sqrt(D t); the depth at which the concentration falls to a
% fixed fraction advances in proportion to the square root of time. Profiles
% normalised so that c/c_s = erfc( x / (2 sqrt(D t)) ).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6.6cm,
  xlabel={depth $x$ (mm)},
  ylabel={normalised concentration $c/c_s$},
  xmin=0, xmax=12, ymin=0, ymax=1.05,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=0:12,
]
% Take D = 1e-9 m^2/s. Then 2 sqrt(D t) in mm:
%   t1 = 1e3 s -> 2 sqrt(1e-6) = 2e-3 m = 2.0 mm
%   t2 = 4e3 s -> 2 sqrt(4e-6) = 4e-3 m = 4.0 mm
%   t3 = 1.6e4 s -> 2 sqrt(1.6e-5) = 8e-3 m = 8.0 mm
% pgfmath has no built-in erf, so use the Abramowitz-Stegun 7.1.25 rational
% approximation for erfc(z) on z >= 0 (here z = x/(2 sqrt(D t)) >= 0):
%   t = 1/(1 + 0.47047 z),
%   erfc(z) = (0.3480242 t - 0.0958798 t^2 + 0.7478556 t^3) exp(-z^2).
\addplot[engblue, very thick] {
  (0.3480242*(1/(1+0.47047*x/2.0))
   - 0.0958798*(1/(1+0.47047*x/2.0))^2
   + 0.7478556*(1/(1+0.47047*x/2.0))^3)*exp(-(x/2.0)^2)};
\addlegendentry{$t_1$ (front depth $\sim 2$ mm)}
\addplot[engred, very thick] {
  (0.3480242*(1/(1+0.47047*x/4.0))
   - 0.0958798*(1/(1+0.47047*x/4.0))^2
   + 0.7478556*(1/(1+0.47047*x/4.0))^3)*exp(-(x/4.0)^2)};
\addlegendentry{$t_2 = 4t_1$ (front $\sim 4$ mm)}
\addplot[engamber, very thick] {
  (0.3480242*(1/(1+0.47047*x/8.0))
   - 0.0958798*(1/(1+0.47047*x/8.0))^2
   + 0.7478556*(1/(1+0.47047*x/8.0))^3)*exp(-(x/8.0)^2)};
\addlegendentry{$t_3 = 16t_1$ (front $\sim 8$ mm)}
% the half-concentration locus advances as sqrt(t): erfc(0.477)=0.5
% x_half = 0.954 * sqrt(D t) -> at the three times: 0.954, 1.908, 3.816 mm
\addplot[enggray, dashed, thick] coordinates {(0,0.5) (12,0.5)};
\node[font=\scriptsize, enggray, anchor=south west] at (axis cs:8.4,0.5) {half-saturation};
\end{axis}
\end{tikzpicture}
\caption[Penetration-theory profiles spreading as the square root of
time]{The complementary-error-function profiles of transient diffusion
into a semi-infinite medium at three successive times, with the diffusion
coefficient held fixed and the times in the ratio $1:4:16$. Each profile
is $c/c_s = \operatorname{erfc}\!\big(x/(2\sqrt{Dt})\big)$, so the
characteristic front depth scales as $\sqrt{Dt}$ and the depth at any
fixed concentration fraction advances in proportion to the square root of
time. Quadrupling the time only doubles the penetration depth, the
signature of a diffusion-limited process and the reason that thin films,
not bulk volumes, set the rate of every two-film contactor.}
\label{fig:vol04ch07:penetration-theory}
\end{figure}
