% Figure: McCabe-Thiele diagram, equilibrium curve with operating lines and stepped stages.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10cm, height=10cm,
  xlabel={liquid mole fraction $x$},
  ylabel={vapour mole fraction $y$},
  xmin=0, xmax=1, ymin=0, ymax=1,
  axis equal image,
  legend pos=south east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% y = x diagonal
\addplot[enggray, thick, domain=0:1] {x};
\addlegendentry{$y=x$}
% equilibrium curve: relative volatility alpha = 2.5
\addplot[engred, very thick, domain=0:1, samples=80] {2.5*x/(1+1.5*x)};
\addlegendentry{equilibrium, $\alpha=2.5$}
% rectifying operating line from (xD,xD) with slope R/(R+1); xD=0.9, R=2 -> slope 0.667, intercept 0.3
\addplot[engblue, thick, domain=0.45:0.9] {0.667*x + 0.3};
\addlegendentry{rectifying line}
% stripping operating line from (xW,xW)=(0.05,0.05) up to feed pinch near (0.45, 0.6)
\addplot[engamber, thick, domain=0.05:0.45] {1.375*x - 0.019};
\addlegendentry{stripping line}
% feed point marker
\addplot[only marks, mark=*, mark size=1.6pt, black] coordinates {(0.45,0.6)};
\node[font=\scriptsize, anchor=west] at (axis cs:0.46,0.55) {feed};
% a few stepped stages (illustrative staircase between eq curve and operating lines)
\addplot[black, thick] coordinates {(0.9,0.9) (0.72,0.9) (0.72,0.78) (0.55,0.78) (0.55,0.667) (0.40,0.667) (0.40,0.53) (0.24,0.53) (0.24,0.31) (0.10,0.31) (0.10,0.12) (0.05,0.12)};
\end{axis}
\end{tikzpicture}
\caption[McCabe-Thiele construction for binary distillation]{The
McCabe-Thiele construction for a binary distillation. The equilibrium
curve, plotted here for a constant relative volatility $\alpha=2.5$, lies
above the $y=x$ diagonal. The rectifying and stripping operating lines
are material balances around the two column sections; they meet at the
feed. Stepping a staircase between the equilibrium curve and the
operating lines counts the equilibrium stages between the distillate and
bottoms compositions. The staircase shown is illustrative; an exact
construction lands the final step on the bottoms specification.}
\label{fig:vol04ch07:mccabe-thiele}
\end{figure}
