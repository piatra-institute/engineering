% Figure: three adsorption-isotherm shapes on a common axis of loading q
% against fluid-phase concentration c. The favourable (Langmuir, concave-down)
% isotherm rises steeply then saturates and sharpens a breakthrough front; the
% linear (Henry) isotherm is the reference; the unfavourable (concave-up)
% isotherm spreads the front and wastes bed. The separation constant R_s of the
% dimensionless-isotherm reading is what sorts the three: R_s < 1 favourable,
% R_s = 1 linear, R_s > 1 unfavourable.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.8cm,
  xlabel={fluid-phase concentration $c/c_0$},
  ylabel={solid loading $q/q_m$},
  xmin=0, xmax=1, ymin=0, ymax=1.05,
  legend pos=south east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=0:1,
]
% Favourable Langmuir: q/qm = (K c)/(1 + K c) normalised to reach ~1 at c=c0.
% With K c0 = 20, q/qm at c0 = 20/21 = 0.952; rescale to hit 1 at c0.
\addplot[engblue, very thick] {(20*x/(1+20*x))/(20/21)};
\addlegendentry{favourable (Langmuir)}
% Linear Henry reference
\addplot[enggray, thick, dashed] {x};
\addlegendentry{linear (Henry)}
% Unfavourable, concave-up: a power law q/qm = c^2 (rises slowly then steepens)
\addplot[engred, very thick] {x*x};
\addlegendentry{unfavourable}
\end{axis}
\end{tikzpicture}
\caption[Three adsorption-isotherm shapes and the front they make]{Three
adsorption isotherms on a common axis of solid loading against fluid-phase
concentration. The favourable Langmuir isotherm rises steeply at low
concentration and saturates, so the fast-moving concentrated part of a
loading front catches the slow-moving dilute part and the wave self-sharpens
into a near-constant-pattern shape that uses the bed almost to its equilibrium
capacity. The linear Henry isotherm is the reference case where the front
neither sharpens nor spreads. The unfavourable concave-up isotherm reverses
the wave speeds, so the front spreads as it travels, the mass-transfer zone
broadens, and much of the bed sits unused at breakthrough. The isotherm shape,
read as the separation constant of the section, decides how much bed the
design wastes.}
\label{fig:vol04ch07:isotherm-shapes}
\end{figure}
