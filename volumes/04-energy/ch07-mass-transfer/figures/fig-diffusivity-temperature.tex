% Figure: gas binary diffusivity rising with temperature and falling with
% pressure, the Chapman-Enskog scaling D ~ T^{3/2}/p. Two curves show the
% same system at two pressures, with the higher pressure sitting lower and
% the T^{3/2} rise shared by both.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={temperature $T$ (K)},
  ylabel={binary diffusivity $D_{AB}$ ($10^{-5}\,\mathrm{m}^2/\mathrm{s}$)},
  xmin=250, xmax=600, ymin=0, ymax=8,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=250:600,
]
% D at 1 atm scaled to 2.0e-5 m2/s at 298 K, rising as T^{3/2}.
\addplot[engblue, very thick] {2.0*(x/298)^(1.5)};
\addlegendentry{$p = 1\,\mathrm{atm}$}
% D at 2 atm: half the value at every T (inverse-pressure scaling).
\addplot[engred, very thick, dashed] {1.0*(x/298)^(1.5)};
\addlegendentry{$p = 2\,\mathrm{atm}$}
% marker at the tabulated room-temperature value
\addplot[only marks, mark=*, engblack, mark size=2pt] coordinates {(298,2.0)};
\node[anchor=west, font=\footnotesize] at (axis cs:305,2.35) {$2.0$ at $298\,\mathrm{K}$};
\end{axis}
\end{tikzpicture}
\caption[Gas diffusivity rising as $T^{3/2}$ and falling with
pressure]{The binary diffusivity of a gas pair rising with temperature as
$T^{3/2}$ and falling inversely with pressure, the Chapman-Enskog scaling of
the section's estimation mastery box. Doubling the pressure halves the
diffusivity at every temperature, because the mean free path shortens; the
two curves share the same $T^{3/2}$ rise, because raising the temperature
speeds the molecules regardless of pressure. The marked point fixes the
curve to the tabulated oxygen-in-nitrogen value at room temperature, and the
spread of a factor of several across the plotted range is why a diffusivity
carried from a table must be corrected to the operating temperature and
pressure before it is used in a flux.}
\label{fig:vol04ch07:diffusivity-temperature}
\end{figure}
