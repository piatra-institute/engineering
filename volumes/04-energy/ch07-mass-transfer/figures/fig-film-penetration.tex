% Figure: the three interfacial-coefficient models on a common log-log axis
% of mass-transfer coefficient against diffusivity. The film model predicts
% k_c proportional to D^1; the penetration and surface-renewal models both
% predict k_c proportional to D^(1/2). The slope on a log-log plot is the
% diagnostic that tells a steady film (slope 1) from a renewing surface
% (slope 1/2).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=11cm, height=7cm,
  xlabel={diffusivity $D$ (m$^2$/s)},
  ylabel={coefficient $k_c$ (m/s)},
  xmin=1e-10, xmax=1e-8, ymin=1e-5, ymax=1e-3,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% Film model: k_c = D / delta, with delta = 1e-5 m, slope 1 on log-log.
\addplot[engred, very thick, domain=1e-10:1e-8, samples=40]
  {x/1e-5};
\addlegendentry{film, $k_c=D/\delta\ (\propto D^{1})$}
% Penetration model: k_c = 2 sqrt(D/(pi t_c)), t_c = 0.05 s, slope 1/2.
\addplot[engblue, very thick, domain=1e-10:1e-8, samples=40]
  {2*sqrt(x/(3.14159*0.05))};
\addlegendentry{penetration, $k_c=2\sqrt{D/\pi t_c}\ (\propto D^{1/2})$}
% Surface-renewal model: k_c = sqrt(D s), s = 25 1/s, slope 1/2.
\addplot[engamber, very thick, dashed, domain=1e-10:1e-8, samples=40]
  {sqrt(x*25)};
\addlegendentry{surface renewal, $k_c=\sqrt{Ds}\ (\propto D^{1/2})$}
\node[font=\scriptsize, engred, anchor=south east] at (axis cs:8e-9,7e-4)
  {slope $1$};
\node[font=\scriptsize, engblue, anchor=north west] at (axis cs:1.3e-10,9e-5)
  {slope $\tfrac{1}{2}$};
\end{loglogaxis}
\end{tikzpicture}
\caption[Film, penetration, and surface-renewal coefficients against
diffusivity]{The three interfacial models for the mass-transfer
coefficient, plotted as $k_c$ against the diffusivity $D$ on log-log axes.
The film model of a steady stagnant layer gives $k_c=D/\delta$, a slope of
one. The penetration model of Higbie and the surface-renewal model of
Danckwerts both rest on transient diffusion into renewing fluid elements
and give $k_c\propto\sqrt{D}$, a slope of one-half, differing only in
whether a single contact time $t_c$ or a renewal rate $s$ sets the scale.
The slope is the working diagnostic: sweep a contactor across solutes of
different diffusivity, measure $k_c$, and read the slope. A slope near one
says the resistance is a steady film; a slope near one-half says the
surface is renewing and a contact time, not a film thickness, controls the
rate.}
\label{fig:vol04ch07:film-penetration}
\end{figure}
