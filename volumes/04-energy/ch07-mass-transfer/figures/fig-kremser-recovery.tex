% Figure: Kremser recovery against stage count for three absorption factors.
% For A>1 the recovery climbs toward unity with stages; for A=1 it climbs as
% N/(N+1); for A<1 it saturates below unity no matter how many stages are added.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={number of equilibrium stages $N$},
  ylabel={fractional recovery of solute},
  xmin=0, xmax=10, ymin=0, ymax=1.05,
  legend pos=south east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=200, domain=0.05:10,
]
% A = 1.5: recovery = (A^{N+1} - A)/(A^{N+1} - 1)
\addplot[engblue, very thick] {(1.5^(x+1) - 1.5)/(1.5^(x+1) - 1)};
\addlegendentry{$A = 1.5$}
% A = 1: recovery = N/(N+1)
\addplot[engamber, very thick, dashed] {x/(x+1)};
\addlegendentry{$A = 1.0$}
% A = 0.7: saturates at A = 0.7 as N grows
\addplot[engred, very thick, dotted] {(0.7^(x+1) - 0.7)/(0.7^(x+1) - 1)};
\addlegendentry{$A = 0.7$}
% asymptote line for A<1 ceiling at 0.7
\addplot[engred, thin] coordinates {(0,0.7) (10,0.7)};
\node[anchor=west, font=\footnotesize, engred] at (axis cs:6.2,0.63) {ceiling $A=0.7$};
\end{axis}
\end{tikzpicture}
\caption[Kremser recovery against stage count for three absorption
factors]{The Kremser recovery of a solute against the number of equilibrium
stages, drawn for three absorption factors. Above unity, at $A = 1.5$, the
recovery climbs toward one and stages buy ever more recovery; at $A = 1$ the
recovery climbs only as $N/(N+1)$, so each added stage helps less than the
last; below unity, at $A = 0.7$, the recovery saturates against a ceiling
set by $A$ itself and no stage count can reach complete recovery. The chart
is the visual statement of why an absorber must be run with $A$ comfortably
above one, and why an under-solvented column ($A < 1$) leaves a floor of
unrecovered solute that only more solvent, not more height, can lift.}
\label{fig:vol04ch07:kremser-recovery}
\end{figure}
