% Figure: schematic generalised pressure-drop correlation (GPDC). The capacity
% parameter falls as the flow parameter rises; constant-pressure-drop contours
% run roughly parallel below the flooding envelope. A design point sits on a
% chosen pressure-drop contour, at a fraction of the flooding capacity.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogxaxis}[
  width=11cm, height=7cm,
  xlabel={flow parameter $F_{LV}=(L/G)\sqrt{\rho_G/\rho_L}$},
  ylabel={capacity parameter $C_S F^{0.5}\,\mathrm{Sc}^{0.5}$},
  xmin=0.01, xmax=10, ymin=0.02, ymax=3,
  log basis x=10,
  legend pos=south west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% flooding envelope (top curve): capacity decreases with flow parameter
\addplot[engred, very thick, domain=0.01:10, samples=80] {1.6*exp(-0.45*ln(1+8*x))};
\addlegendentry{flooding}
% a constant pressure-drop contour, lower and roughly parallel
\addplot[engblue, thick, domain=0.01:10, samples=80] {1.0*exp(-0.45*ln(1+8*x))};
\addlegendentry{$\Delta p = 400\,\text{Pa/m}$}
% a lower pressure-drop contour
\addplot[engamber, thick, domain=0.01:10, samples=80] {0.62*exp(-0.45*ln(1+8*x))};
\addlegendentry{$\Delta p = 200\,\text{Pa/m}$}
% design point on the 200 Pa/m contour at FLV = 0.3
\addplot[only marks, mark=*, mark size=1.8pt, black] coordinates {(0.3,0.394)};
\node[font=\scriptsize, anchor=west] at (axis cs:0.34,0.45) {design point};
\end{semilogxaxis}
\end{tikzpicture}
\caption[Generalised pressure-drop correlation for packed columns]{The
generalised pressure-drop correlation in schematic form. The capacity
parameter on the vertical axis sets the allowable gas mass velocity; the
flow parameter on the horizontal axis is the liquid-to-gas loading ratio
weighted by the density ratio. The upper curve is the flooding envelope,
the gas velocity at which the column will not pass the liquid. Below it
run contours of constant irrigated pressure drop. A design selects a
pressure-drop contour, typically near $200$ to $400\,\text{Pa/m}$ of
packing, which places the operating point at roughly $60$ to $70\,\%$ of
the flooding capacity. The capacity parameter then fixes the gas velocity
and, with the volumetric gas rate, the column cross-section and diameter.}
\label{fig:vol04ch07:flooding}
\end{figure}
