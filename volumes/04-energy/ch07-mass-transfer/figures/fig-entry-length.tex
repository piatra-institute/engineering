% Figure: the local Sherwood number decaying down the entry length of a tube
% toward its fully developed asymptote. Near the inlet the concentration
% boundary layer is thin and the local coefficient is large; as the layer
% fills the tube the local Sherwood number falls toward the constant-wall-
% concentration asymptote Sh = 3.66. The dimensionless axis is the inverse
% Graetz number x* = (z/d)/(Re Sc).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={dimensionless axial position $x^{*} = (z/d)/(\mathrm{Re}\,\mathrm{Sc})$},
  ylabel={local Sherwood number $\mathrm{Sh}_z$},
  xmin=0.002, xmax=0.3, ymin=2, ymax=14,
  xmode=log,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=0.002:0.3,
]
% Entry-length correlation for constant wall concentration, laminar tube:
% Sh_z ~ 3.66 + a leading-edge term ~ 1.077 (x*)^(-1/3) - 0.7 (Leveque-like).
% Clamp with a simple engineering fit that starts high and decays to 3.66.
\addplot[engblue, very thick] {3.66 + 1.2*(x^(-0.333)) - 1.2*(0.3^(-0.333))};
\addlegendentry{local $\mathrm{Sh}_z$ (developing)}
% fully developed asymptote
\addplot[engred, dashed, thick] coordinates {(0.002,3.66) (0.3,3.66)};
\addlegendentry{fully developed, $\mathrm{Sh}=3.66$}
\end{axis}
\end{tikzpicture}
\caption[The local Sherwood number down the entry length of a
tube]{The local mass-transfer coefficient, as a Sherwood number, decaying
along the entry length of a tube toward its fully developed value. Near the
inlet the concentration boundary layer is thin, the wall gradient is steep,
and the local coefficient is large; as the layer grows to fill the tube the
local Sherwood number falls toward the constant-wall-concentration asymptote
of $3.66$. The horizontal axis is the inverse Graetz number, which measures
how far the flow has travelled in units of the distance needed to develop the
profile. A short contactor run in the entry region carries a coefficient
several times the fully developed value, which is why wetted-wall and
hollow-fibre devices are often kept short on purpose, and why using the
fully developed asymptote to size a short device underpredicts its transfer.}
\label{fig:vol04ch07:entry-length}
\end{figure}
