% Figure: catalyst-pellet effectiveness factor versus Thiele modulus for a
% first-order reaction in a sphere. Plateau near unity at small modulus
% (reaction-limited), and the asymptotic 3/phi falloff at large modulus
% (diffusion-limited). Log-log axes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{loglogaxis}[
  width=11cm, height=7cm,
  xlabel={Thiele modulus $\phi$},
  ylabel={effectiveness factor $\eta$},
  xmin=0.1, xmax=100, ymin=0.01, ymax=1.5,
  legend pos=south west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
]
% spherical-pellet effectiveness factor, eta = (3/phi^2)(phi*coth(phi) - 1),
% with coth written through exponentials so the math engine evaluates it.
\addplot[engblue, very thick, domain=0.1:100, samples=120]
  {(3/(x^2))*(x*(exp(x)+exp(-x))/(exp(x)-exp(-x)) - 1)};
\addlegendentry{sphere, $\eta=(3/\phi^2)(\phi\coth\phi-1)$}
% large-modulus asymptote eta approaches 3/phi
\addplot[engred, dashed, thick, domain=1:100, samples=60] {3/x};
\addlegendentry{large-$\phi$ asymptote $\eta=3/\phi$}
% reaction-limited plateau marker
\node[font=\scriptsize, anchor=south west] at (axis cs:0.12,0.55)
  {reaction-limited};
\node[font=\scriptsize, anchor=north east] at (axis cs:90,0.06)
  {diffusion-limited};
\end{loglogaxis}
\end{tikzpicture}
\caption[Catalyst-pellet effectiveness factor versus Thiele
modulus]{The effectiveness factor $\eta$ of a porous catalyst pellet for
a first-order reaction, plotted against the Thiele modulus
$\phi=L\sqrt{k/D_e}$ that compares the reaction rate to the
intraparticle diffusion rate. At small $\phi$ the reactant penetrates the
whole pellet and $\eta\to 1$: the pellet runs at its intrinsic kinetics.
At large $\phi$ the reactant is consumed in a thin outer shell, the core
sits starved, and $\eta\to 3/\phi$: the observed rate is set by diffusion
into the pellet, not by the chemistry. The crossover near $\phi\approx 1$
is where pellet size, pore structure, and operating temperature are
traded against the catalyst utilisation. The curve shown is for a sphere;
slabs and cylinders give the same plateau and the same inverse falloff
with a different constant.}
\label{fig:vol04ch07:effectiveness}
\end{figure}
