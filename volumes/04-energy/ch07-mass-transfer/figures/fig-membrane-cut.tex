% Figure: permeate purity versus stage cut for a binary gas-separation membrane
% at fixed selectivity. High purity demands a small stage cut; taking more of the
% feed as permeate dilutes it with the slower gas.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={stage cut $\theta$ (permeate fraction of feed)},
  ylabel={permeate mole fraction of fast gas $y_P$},
  xmin=0, xmax=1, ymin=0, ymax=1,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  every axis plot/.append style={very thick},
]
% feed composition reference
\draw[enggray, dashed] (axis cs:0,0.21) -- (axis cs:1,0.21);
\node[font=\scriptsize, anchor=west, enggray] at (axis cs:0.02,0.27) {feed $x_F=0.21$};
% selectivity alpha = 6 curve (declines from enriched permeate toward feed at full cut)
\addplot[engblue, domain=0.02:1, samples=80] {0.62*exp(-1.1*x) + 0.21*(1-exp(-1.1*x))};
\addlegendentry{$\alpha = 6$}
% selectivity alpha = 3 curve, lower enrichment
\addplot[engamber, domain=0.02:1, samples=80] {0.45*exp(-1.0*x) + 0.21*(1-exp(-1.0*x))};
\addlegendentry{$\alpha = 3$}
\end{axis}
\end{tikzpicture}
\caption[Permeate purity versus stage cut for gas separation]{Permeate
enrichment as a function of stage cut for a single-stage binary
gas-separation membrane, shown for two selectivities. The fast gas is
enriched most when only a thin slice of the feed is taken as permeate
(small stage cut), because the membrane then sees nearly the feed
composition as its driving condition. Taking a larger fraction of the
feed as permeate dilutes it: at full cut the permeate must equal the feed.
Higher selectivity lifts the whole curve. The design trade-off is between
permeate purity and recovery, and high purity at high recovery requires
either a multi-stage cascade or a recycle.}
\label{fig:vol04ch07:membrane-cut}
\end{figure}
