% Figure: the Stefan correction factor, the ratio of the stagnant-film flux
% to the bare equimolar-counterdiffusion flux, plotted against the mole
% fraction of the diffusing species at the high-concentration face (with the
% low face held near zero). The factor is 1/x_B,lm and rises above unity as
% the diffusing species becomes a large fraction of the gas, the quantitative
% statement of when the two diffusion modes diverge.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={diffusing-species mole fraction at the high face $x_{A,1}$},
  ylabel={Stefan factor $N_A^{\mathrm{Stefan}}/N_A^{\mathrm{equi}}$},
  xmin=0, xmax=0.8, ymin=1, ymax=3,
  legend pos=north west,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  samples=120, domain=0.001:0.8,
]
% Low face held at x_A2 = 0.001 so x_B2 ~ 0.999. The factor is
% 1/x_B,lm with x_B,lm = (x_B2 - x_B1)/ln(x_B2/x_B1), x_B1 = 1 - x_A1.
% Substituting x_A1 = x and x_B1 = 1-x, x_B2 = 0.999:
%   factor = ln(0.999/(1-x)) / (0.999 - (1-x)) = ln(0.999/(1-x))/(x - 0.001).
\addplot[engblue, very thick] {ln(0.999/(1-x))/(x-0.001)};
\addlegendentry{$1/x_{B,\mathrm{lm}}$, low face near zero}
% reference line at unity (the equimolar / dilute limit)
\addplot[enggray, dashed, thick] coordinates {(0,1) (0.8,1)};
\node[font=\scriptsize, enggray, anchor=south west] at (axis cs:0.02,1.02)
  {bare Fick / equimolar limit};
% mark the worked-example operating point near x_A1 = 0.40
\addplot[engred, only marks, mark=*, mark size=1.6pt]
  coordinates {(0.40,1.51)};
\node[font=\scriptsize, engred, anchor=west] at (axis cs:0.42,1.55)
  {worked example};
\end{axis}
\end{tikzpicture}
\caption[The Stefan correction factor against diffusing-species mole
fraction]{The ratio of the Stefan stagnant-film flux to the bare
equimolar-counterdiffusion flux for the same end compositions, plotted
against the mole fraction of the diffusing species at the high face with
the low face held near zero. The two diffusion modes coincide in the
dilute limit, where the factor is one and the choice between them does not
matter. As the diffusing species becomes a large fraction of the gas the
factor climbs, because the diffusing species must push aside the stagnant
carrier it moves into and that induced convection adds to the transport.
Sizing a concentrated stagnant-film flux with the bare Fick value, or a
distillation tray with the Stefan factor, is the error this curve guards
against.}
\label{fig:vol04ch07:stefan-factor}
\end{figure}
