% diagram. The solubility curve is the equilibrium floor; the supersolubility
% curve is the spontaneous-nucleation ceiling. Controlled cooling holds the
% path inside the band; uncontrolled cooling crosses the ceiling and floods
% the solution with fines.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={temperature $T$ ($^{\circ}\mathrm{C}$)},
  ylabel={concentration $c$ (g solute per 100 g solvent)},
  xmin=10, xmax=70, ymin=20, ymax=80,
  legend pos=north west,
  legend cell align=left,

  samples=120, domain=10:70,

% Solubility curve (equilibrium): rises with temperature.
\addplot[engblue, very thick] {25 + 0.7*x};
\addlegendentry{solubility $c^{*}(T)$}
% Supersolubility curve (nucleation ceiling): parallel, higher.
\addplot[engred, very thick, dashed] {35 + 0.7*x};
\addlegendentry{nucleation ceiling}
% Controlled cooling path: seed hot, ride down inside the band.
\addplot[englime!55!black, ultra thick] coordinates {(60,60) (45,58) (30,52) (18,44)};
\addlegendentry{controlled cooling}
% Uncontrolled crash cooling: vertical drop crossing the ceiling.
\addplot[engamber, ultra thick, dotted] coordinates {(60,60) (25,60)};
\addlegendentry{crash cooling}
\node[anchor=west, font=\footnotesize, engamber] at (axis cs:26,63) {crosses ceiling};
\node[anchor=west, font=\footnotesize, enggray] at (axis cs:40,40) {metastable band};
\end{axis}
\end{tikzpicture}
\caption[The metastable zone and the crystallisation cooling
policy]{The crystallisation metastable zone on a solubility-temperature
diagram. The lower curve is the solubility, the equilibrium floor below which
crystals dissolve; the upper curve is the supersolubility, the ceiling above
which the solution nucleates spontaneously and floods with fines. The band
between the two is where crystals grow on existing seeds without a burst of
new nuclei. A controlled cooling policy seeds the hot solution and rides the
concentration down inside the band, letting growth onto the seeds consume the
supersaturation as fast as the cooling generates it. A crash cool drops the
temperature so fast that the concentration crosses the ceiling before growth
can catch up, and the solution nucleates a cloud of small crystals that will
not filter. The picture is why the operating policy, not the endpoint, sets
the crystal-size distribution.}
\label{fig:vol04ch07:metastable-zone}
\end{figure}
