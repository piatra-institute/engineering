% Figure: Bode magnitude and phase of a first-order RC high-pass filter,
% H(jw) = jwRC/(1 + jwRC). Corner at f_c. Magnitude rises at +20 dB/decade
% below the corner and flattens to 0 dB above it; phase runs from +90 deg
% at low frequency to 0 deg at high frequency, passing +45 deg at the corner.
% Frequency axis normalised to f/f_c on a log scale, decades -2..+2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
% --- magnitude panel ---
\begin{axis}[
  name=mag,
  width=12cm, height=5.2cm,
  xmode=log,
  xlabel={}, ylabel={magnitude (dB)},
  xmin=0.01, xmax=100, ymin=-42, ymax=6,
  xtick={0.01,0.1,1,10,100},
  xticklabels={},
  ytick={0,-10,-20,-30,-40},
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% exact magnitude: 20 log10( x / sqrt(1 + x^2) ), x = f/f_c
\addplot[engblue, very thick, domain=0.01:100, samples=200]
  {20*log10(x/sqrt(1 + x^2))};
% low-frequency asymptote +20 dB/decade: 20 log10(x)
\addplot[engred, dashed, domain=0.01:1, samples=2] {20*log10(x)};
% high-frequency asymptote 0 dB
\addplot[engred, dashed, domain=1:100, samples=2] {0};
% corner marker at f/f_c = 1, exact value -3.01 dB
\filldraw[engorange] (axis cs:1,-3.01) circle [radius=1.8pt];
\node[engorange, font=\scriptsize, anchor=north west] at (axis cs:1.1,-3.4) {$-3\,\mathrm{dB}$ at $f_c$};
\end{axis}
% --- phase panel ---
\begin{axis}[
  at={(mag.below south west)}, anchor=north west, yshift=-6mm,
  width=12cm, height=4.6cm,
  xmode=log,
  xlabel={normalised frequency $f/f_c$}, ylabel={phase (deg)},
  xmin=0.01, xmax=100, ymin=-5, ymax=95,
  xtick={0.01,0.1,1,10,100},
  ytick={0,45,90},
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% exact phase: arctan(1/x) in degrees, x = f/f_c
\addplot[engblue, very thick, domain=0.01:100, samples=200]
  {atan(1/x)};
% +45 deg marker at corner
\filldraw[engorange] (axis cs:1,45) circle [radius=1.8pt];
\node[engorange, font=\scriptsize, anchor=south west] at (axis cs:1.1,46) {$+45^{\circ}$ at $f_c$};
\end{axis}
\end{tikzpicture}
\caption[First-order high-pass Bode plot]{Magnitude and phase of a first-order
RC high-pass filter. The magnitude rises at twenty decibels per decade below
the corner and flattens toward the pass band above it, the two straight-line
asymptotes meeting at the corner where the exact response is three decibels
down. The phase leads by ninety degrees far below the corner, where the
capacitor dominates and the output is the derivative of the input, and falls to
zero far above it, where the filter passes the signal through, crossing
forty-five degrees at the corner. The high-pass is the mirror of the low-pass
of Figure~\ref{fig:vol04ch09:bode-lowpass}: a zero at the origin lifts the low
frequencies away and a pole at the corner levels the response above it, which is
why an AC-coupling capacitor blocks the DC term and passes the signal.}
\label{fig:vol04ch09:highpass-bode}
\end{figure}
