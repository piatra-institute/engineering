% Figure: magnitude (flat, unity) and phase (running from 0 to -180 degrees)
% of a first-order all-pass network. The point is that the magnitude is
% frequency-independent while the phase sweeps a full 180 degrees through the
% break frequency, the signature that distinguishes an all-pass delay stage
% from a filter. pgfplots semilog with two y axes.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogxaxis}[
  width=11.5cm, height=5.6cm,
  xlabel={normalised frequency $\omega/\omega_0$},
  ylabel={phase (deg)},
  xmin=0.01, xmax=100, ymin=-190, ymax=10,
  axis y line=left, axis x line=bottom,
  tick label style={font=\tiny},
  label style={font=\scriptsize},
  ytick={0,-45,-90,-135,-180},
  clip=false,
]
% first-order all-pass phase: H = (1 - j w/w0)/(1 + j w/w0),
% phase = -2 arctan(w/w0), in degrees
\addplot[engblue, very thick, domain=0.01:100, samples=300]
  {-2*atan(x)};
\node[engblue, font=\tiny, anchor=north east] at (axis cs:90,-150) {phase};
% break frequency marker at w/w0 = 1, phase = -90
\addplot[enggray, dotted] coordinates {(1,-190) (1,10)};
\node[enggray, font=\tiny, anchor=south west] at (axis cs:1.05,-185)
  {$\omega=\omega_0$};
\filldraw[engblue] (axis cs:1,-90) circle [radius=1.6pt];
\node[engblue, font=\tiny, anchor=south west] at (axis cs:1.1,-88)
  {$-90^\circ$};
\end{semilogxaxis}
% overlay a flat unity magnitude line on a second axis
\begin{semilogxaxis}[
  width=11.5cm, height=5.6cm,
  xmin=0.01, xmax=100, ymin=-30, ymax=3,
  axis y line=right, axis x line=none,
  ylabel={magnitude (dB)},
  tick label style={font=\tiny},
  label style={font=\scriptsize},
  ytick={0,-10,-20},
  clip=false,
]
\addplot[engred, thick, dashed, domain=0.01:100, samples=2] {0};
\node[engred, font=\tiny, anchor=south east] at (axis cs:90,0.3)
  {magnitude (0 dB, flat)};
\end{semilogxaxis}
\end{tikzpicture}
\caption[First-order all-pass network]{The first-order all-pass network passes
every frequency at unity magnitude (red, flat at zero decibels) while its
phase (blue) sweeps from zero at low frequency through $-90^\circ$ at the break
frequency to $-180^\circ$ at high frequency. A network that changes phase
without changing amplitude is the tool for delay equalisation and for the
phase-shift oscillator, where three such stages accumulate the $180^\circ$ an
inverting amplifier needs to close the loop in phase. The flat magnitude is
what separates an all-pass stage from a filter: it moves the signal in time
without shaping its spectrum.}
\label{fig:vol04ch09:allpass-phase}
\end{figure}
