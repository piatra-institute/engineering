% Figure: the Joule-Thomson inversion curve on pressure-temperature axes. Inside
% the dome the coefficient is positive and throttling cools the gas; outside it
% is negative and throttling warms it. The maximum inversion temperature caps the
% dome. Schematic, drawn from hand-placed points shaped like a real inversion loop.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=0.80\textwidth, height=0.52\textwidth,
    xlabel={pressure $p$ (arbitrary units)},
    ylabel={temperature $T$ (arbitrary units)},
    xmin=0, xmax=10, ymin=0, ymax=10,
    axis lines=left,
    tick label style={font=\scriptsize},
    label style={font=\small},
    xtick=\empty, ytick=\empty,
    clip=false,
]
  % inversion curve: a closed loop opening on the temperature axis, drawn as an
  % upper branch (high inversion temperature falling with pressure) and a lower
  % branch that turns over at the maximum-pressure nose
  \addplot[engblue, very thick, smooth] coordinates
    {(0,9.2) (1.5,8.9) (3,8.3) (4.5,7.3) (6,5.9) (7,4.6) (7.5,3.4)};
  \addplot[engblue, very thick, smooth] coordinates
    {(7.5,3.4) (7.2,2.3) (6,1.4) (4.5,0.95) (3,0.7) (1.5,0.55) (0,0.5)};
  % cooling region label (inside the loop)
  \node[engblue, font=\scriptsize, align=center] at (axis cs:3.2,4.6)
    {$\mu_{JT}>0$\\cooling on\\throttling};
  % warming region label (outside, upper right)
  \node[engred, font=\scriptsize, align=center] at (axis cs:8.6,7.2)
    {$\mu_{JT}<0$\\warming on\\throttling};
  % maximum inversion temperature on the T axis
  \draw[enggray, dashed] (axis cs:0,9.2) -- (axis cs:0.6,9.2);
  \node[enggray, font=\scriptsize, anchor=west] at (axis cs:0.65,9.2)
    {max.\ inversion temperature};
  % a low-temperature throttle path that cools (arrow pointing to lower pressure,
  % staying inside the dome)
  \draw[englime!70!engblack, line width=1.2pt, -{Stealth[length=3mm]}]
    (axis cs:6,5.9) -- (axis cs:2,5.4);
  \node[englime!70!engblack, font=\scriptsize, anchor=north] at (axis cs:4,5.1)
    {throttle (Linde)};
  % a high-temperature throttle path that warms (arrow above the dome)
  \draw[engred, line width=1.2pt, -{Stealth[length=3mm]}]
    (axis cs:7,8.2) -- (axis cs:3,8.5);
\end{axis}
\end{tikzpicture}
\caption[The Joule-Thomson inversion curve]{The Joule-Thomson inversion curve on
pressure-temperature axes. Inside the loop the coefficient $\mu_{JT} =
(\partial T/\partial p)_h$ is positive, so a pressure drop at constant enthalpy
cools the gas, the regime the Linde air-liquefaction process operates in. Outside
the loop the coefficient is negative and the same throttle warms the gas. The
curve meets the temperature axis at the maximum inversion temperature, above
which no throttling from any pressure cools the gas; hydrogen and helium have
maximum inversion temperatures below room temperature, which is why they must be
pre-cooled before a throttle will liquefy them. The ideal-gas result, no
temperature change on throttling, is the boundary case that the whole curve is
drawn around.}
\label{fig:vol04ch01:jt-inversion}
\end{figure}
