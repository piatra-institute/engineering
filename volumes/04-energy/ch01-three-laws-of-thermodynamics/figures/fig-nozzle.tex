% Figure: control volume around a converging nozzle. A high-enthalpy, low-speed
% stream enters at 1; a low-enthalpy, high-speed stream leaves at 2. No shaft
% work, adiabatic, so the enthalpy drop converts to kinetic energy.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\small, scale=1.0]
  % nozzle wall (converging duct), upper and lower profiles
  \draw[engblack, thick] (0,1.1) -- (3.2,1.1) -- (5.0,0.45);
  \draw[engblack, thick] (0,-1.1) -- (3.2,-1.1) -- (5.0,-0.45);
  % control-volume boundary (dashed) around the duct interior
  \draw[engred, very thick, dashed] (0.1,1.0) -- (3.2,1.0) -- (4.9,0.42)
    -- (4.9,-0.42) -- (3.2,-1.0) -- (0.1,-1.0) -- cycle;
  % inlet arrow (slow, fat)
  \draw[engblue, line width=3pt, -{Stealth[length=4mm]}] (-1.3,0) -- (-0.2,0);
  \node[engblue, font=\scriptsize, anchor=east] at (-1.35,0) {$V_1$ small};
  % outlet arrow (fast, thin, long)
  \draw[englime!70!engblack, line width=1.5pt, -{Stealth[length=5mm]}]
    (5.1,0) -- (7.2,0);
  \node[englime!70!engblack, font=\scriptsize, anchor=west] at (7.25,0) {$V_2$ large};
  % station labels
  \node[font=\scriptsize, anchor=south] at (0.3,1.15) {state 1: $h_1$};
  \node[font=\scriptsize, anchor=south] at (4.6,0.55) {state 2: $h_2$};
  % CV annotation
  \node[engred, font=\scriptsize] at (2.0,0) {control volume};
  \node[enggray, font=\scriptsize, anchor=north] at (2.5,-1.25)
    {adiabatic, no shaft work: $h_1 + \tfrac12 V_1^2 = h_2 + \tfrac12 V_2^2$};
\end{tikzpicture}
\caption[Control volume around an adiabatic nozzle]{A control volume drawn
around an adiabatic nozzle with no shaft work. A stream enters slowly at high
enthalpy and leaves quickly at lower enthalpy; the steady-flow first law forces
the enthalpy drop to reappear as kinetic energy. The nozzle is the one piece of
process equipment for which the kinetic-energy term dominates the energy
balance, the opposite of the turbine and heat exchanger where it is discarded.
The same balance run backward describes a diffuser, which trades speed for
pressure recovery at the inlet of a jet engine.}
\label{fig:vol04ch01:nozzle}
\end{figure}
