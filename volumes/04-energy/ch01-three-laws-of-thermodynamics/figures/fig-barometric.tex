% Figure: the barometric (isothermal-atmosphere) pressure profile. Pressure
% falls exponentially with height, p/p0 = exp(-z/H), with scale height
% H = RT/(Mg) about 7.3 km for air at 250 K. A dashed marker sits at one scale
% height, where the pressure has fallen to 1/e of its surface value.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=8cm,
  xlabel={pressure ratio $p/p_0$}, ylabel={height $z$ (km)},
  xmin=0, xmax=1.05, ymin=0, ymax=16,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% barometric curve: z as function of p/p0, z = -H ln(p/p0), H=7.306 km
% plot p/p0 on x, z on y by sweeping p/p0
\addplot[engblue, very thick, smooth, domain=0.08:1.0, samples=80]
  ({x}, {-7.306*ln(x)});
% one-scale-height marker: p/p0 = 1/e = 0.368, z = 7.306 km
\draw[enggray, dashed] (axis cs:0.368,0) -- (axis cs:0.368,7.306);
\draw[enggray, dashed] (axis cs:0,7.306) -- (axis cs:0.368,7.306);
\filldraw[engred] (axis cs:0.368,7.306) circle [radius=1.8pt];
\node[engred, font=\scriptsize, anchor=south west] at (axis cs:0.368,7.306)
  {$z=H$, $p/p_0=1/e$};
\end{axis}
\end{tikzpicture}
\caption[The isothermal-atmosphere barometric profile]{Pressure against height
for an isothermal atmosphere, $p/p_0=e^{-z/H}$ with scale height $H=RT/(Mg)$.
Every rise of one scale height, about $7.3\,\text{km}$ for air near
$250\,\si{\kelvin}$, cuts the pressure by a factor $e$, marked by the dashed
lines where the pressure reaches $1/e$ of its surface value. The exponential is
the hydrostatic balance solved at constant temperature; the real atmosphere,
cooling with height along the lapse rate of the previous figure, falls off a
little faster low down and the two treatments bracket the truth. The scale
height also fixes why aircraft cruise near ten kilometres, where the pressure
has dropped to a quarter of sea level and the drag with it, and why a cabin must
be pressurised to keep its occupants near the surface value.}
\label{fig:vol04ch01:barometric}
\end{figure}
