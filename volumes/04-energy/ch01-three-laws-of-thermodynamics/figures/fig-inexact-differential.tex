% Figure: two paths between the same two states on a p-V plane, with the area
% under each shaded, to show that the work integral (an inexact differential)
% depends on the path while the change in any state function does not.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=0.74\textwidth, height=0.52\textwidth,
    xlabel={volume $V$ (arb.\ units)},
    ylabel={pressure $p$ (arb.\ units)},
    xmin=0.6, xmax=4.4, ymin=0, ymax=4.6,
    axis lines=left,
    legend pos=north east, legend cell align=left,
    legend style={font=\scriptsize, draw=enggray},
    tick label style={font=\scriptsize},
    label style={font=\small},
    clip=false,
]
  % state A at (1,4), state B at (4,1)
  % Path I: high-pressure leg first (isobar at p=4 then drop), large area
  \addplot[engblue, very thick] coordinates {(1,4) (4,4) (4,1)};
  \addlegendentry{path I (high road)}
  % Path II: low-pressure leg first (drop then isobar at p=1), small area
  \addplot[engred, very thick] coordinates {(1,4) (1,1) (4,1)};
  \addlegendentry{path II (low road)}
  % endpoint markers
  \addplot[only marks, mark=*, engblack, mark size=2.2pt] coordinates {(1,4) (4,1)};
  \node[engblack, font=\scriptsize, anchor=south west] at (axis cs:1,4) {state $A$};
  \node[engblack, font=\scriptsize, anchor=north west] at (axis cs:4,1) {state $B$};
\end{axis}
\end{tikzpicture}
\caption[Two paths between the same two states]{Two quasi-static paths between
the same end states $A$ and $B$ on pressure-volume axes. Every state function
returns the same change for both paths: $\Delta U$, $\Delta H$, and $\Delta S$
read off the endpoints alone. The boundary work, the area swept under the
curve, differs: path I sweeps the larger area and delivers more work, path II
the smaller. The work is an inexact differential, written $\delta W$ rather
than $\mathrm{d}W$, because no single function $W(p,V)$ has the area as its
value. The same statement holds for the heat $\delta Q$. The gap between the
two areas is the net work the cycle $A \to B$ by path I and $B \to A$ by path
II would deliver, the area enclosed by the loop.}
\label{fig:vol04ch01:inexact-differential}
\end{figure}
