% Figure: the air-standard Diesel cycle on a p-V diagram. An adiabatic
% compression 1->2, a constant-pressure heat addition 2->3 (the injection and
% burn), an adiabatic expansion 3->4 (power stroke), and a constant-volume
% rejection 4->1 (exhaust). Compression ratio r = 18, cutoff ratio rc = 2.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={volume $V$ (arb.)}, ylabel={pressure $p$ (arb.)},
  xmin=0, xmax=19, ymin=0, ymax=64,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% compression 1->2 : adiabat p V^1.4 = C1, from V=18 (p=1) to V=1 (p=57.2)
% C1 = 1 * 18^1.4 = 57.20
\addplot[engblue, very thick, smooth, domain=1:18, samples=80] {57.20/(x^1.4)};
% heat addition 2->3 : isobar at p=57.2, V from 1 to 2
\addplot[engred, very thick] coordinates {(1,57.20) (2,57.20)};
% expansion 3->4 : adiabat p V^1.4 = C2, from V=2 (p=57.2) to V=18
% C2 = 57.20 * 2^1.4 = 151.0
\addplot[engorange, very thick, smooth, domain=2:18, samples=80] {151.0/(x^1.4)};
% rejection 4->1 : isochore at V=18, p from 2.64 to 1.0
\addplot[enggray, very thick] coordinates {(18,2.64) (18,1.0)};
% state markers
\filldraw[engblack] (axis cs:18,1) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=north west] at (axis cs:18,1) {1};
\filldraw[engblack] (axis cs:1,57.20) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:1,57.20) {2};
\filldraw[engblack] (axis cs:2,57.20) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:2,57.20) {3};
\filldraw[engblack] (axis cs:18,2.64) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:18,2.64) {4};
\end{axis}
\end{tikzpicture}
\caption[The air-standard Diesel cycle]{The air-standard Diesel cycle, the
idealisation of a compression-ignition engine. Air alone is compressed
adiabatically along $1\to2$ to a high enough temperature to ignite injected
fuel, heat is added at constant pressure along $2\to3$ while the injector meters
fuel into the expanding charge, the gas expands adiabatically along $3\to4$, and
heat is rejected at constant volume along $4\to1$. Because only air is
compressed, the compression ratio $r=V_1/V_2$ is not capped by fuel
pre-ignition and reaches values past twenty, well above the gasoline engine of
the previous figure. The efficiency depends on both $r$ and the cutoff ratio
$r_c=V_3/V_2$ through $\eta=1-r^{-(\gamma-1)}(r_c^{\gamma}-1)/[\gamma(r_c-1)]$,
and for a fixed $r$ it falls as $r_c$ rises, so a Diesel engine is most
efficient at light load where the fuel is cut off early.}
\label{fig:vol04ch01:diesel-pv}
\end{figure}
