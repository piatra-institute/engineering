% Figure: temperature versus altitude, comparing the dry adiabatic lapse rate
% (9.8 K/km) with a typical measured environmental lapse rate (6.5 K/km).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=8cm,
  xlabel={temperature $T$ ($^{\circ}\mathrm{C}$)},
  ylabel={altitude $z$ (km)},
  xmin=-40, xmax=25, ymin=0, ymax=8,
  axis lines=left,
  legend pos=north east,
  legend cell align=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
  clip=false,
]
% dry adiabatic: T = 15 - 9.8*z, z 0..6
\addplot[engred, very thick, domain=0:6, samples=2] {15 - 9.8*x};
\addlegendentry{dry adiabat $-9.8\,\text{K/km}$}
% environmental: T = 15 - 6.5*z, z 0..8
\addplot[engblue, very thick, domain=0:8, samples=2] {15 - 6.5*x};
\addlegendentry{environmental $-6.5\,\text{K/km}$}
% surface marker
\filldraw[engblack] (axis cs:15,0) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=north east] at (axis cs:15,0) {surface};
\end{axis}
\end{tikzpicture}
\caption[Dry adiabatic and environmental lapse rates]{A parcel of dry air lifted
without heat exchange cools along the dry adiabatic lapse rate of about
$9.8\,\text{K}$ per kilometre, fixed by the first law as $g/c_p$. The surrounding
atmosphere usually cools more slowly, near $6.5\,\text{K}$ per kilometre, so a
lifted parcel finds itself colder and denser than its surroundings and sinks back:
the atmosphere is statically stable. Where the environmental gradient steepens past
the adiabat the parcel stays warmer and keeps rising, the unstable condition that
drives convection and thunderstorms. The two slopes are a first-law result and an
observation set side by side.}
\label{fig:vol04ch01:lapse-rate}
\end{figure}
