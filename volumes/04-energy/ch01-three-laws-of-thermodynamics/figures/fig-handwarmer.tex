% Figure: the temperature history of a sodium-acetate hand warmer through one
% use, schematic. Supersaturated liquid sits at ambient, snaps to crystallise
% on triggering, plateaus near the melt temperature while latent heat releases,
% then cools back to ambient. Drawn with pgfplots from hand-placed points.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=0.78\textwidth, height=0.50\textwidth,
    xlabel={time (min)},
    ylabel={temperature ($^{\circ}\mathrm{C}$)},
    xmin=0, xmax=120, ymin=10, ymax=65,
    axis lines=left,

    label style={font=\small},
    clip=false,

  % supersaturated liquid resting at ambient before the trigger
  \addplot[engblue, very thick] coordinates {(0,21) (10,21)};
  % trigger: rapid temperature jump as crystallisation front sweeps through
  \addplot[engred, very thick] coordinates {(10,21) (12,54)};
  % plateau near the melt temperature while latent heat is released
  \addplot[engred, very thick] coordinates {(12,54) (45,52)};
  % long cooling tail back toward ambient (Newtonian decay, drawn smooth)
  \addplot[engorange, very thick, smooth] coordinates
    {(45,52) (60,44) (75,37) (90,31) (105,26) (120,23)};

  \node[engblack, font=\scriptsize, anchor=south west] at (axis cs:12,54) {trigger};

\caption[Temperature history of a sodium-acetate hand warmer]{Temperature
history of a sodium-acetate hand warmer through one use. The pad holds a
supersaturated solution resting at ambient, a metastable state carrying stored
crystallisation enthalpy with no temperature signature. Flexing a metal disc
seeds a crystallisation front that sweeps the pad in seconds, and the latent
heat of fusion released as the solution freezes lifts the pad to the melt
temperature near $54\,^{\circ}\mathrm{C}$, where it plateaus while the
remaining liquid solidifies. Once fully crystallised the pad cools toward
ambient along a Newtonian decay. The stored energy is recovered by boiling the
pad to redissolve the crystals, returning it to the metastable liquid; the
charge-store-release structure is the same as the compressed-air plant, with
phase change standing in for compression.}
\label{fig:vol04ch01:handwarmer}
\end{figure}
