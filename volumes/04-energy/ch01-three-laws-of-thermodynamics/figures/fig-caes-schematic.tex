% Figure: schematic of a compressed-air energy storage (CAES) plant.
% Charge side: motor drives a compressor train, intercoolers and an aftercooler
% reject heat, air is stored in a cavern. Discharge side: air is reheated and
% expanded through a turbine that drives a generator. The diabatic plant burns
% fuel on discharge; the adiabatic plant stores the compression heat instead.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\small, scale=1.0,
    box/.style={draw=engblack, thick, minimum width=1.5cm, minimum height=0.9cm, align=center, font=\scriptsize},
    flow/.style={-{Latex[length=2mm]}, thick, engblack}]
  % charge path (top row)
  \node[box, fill=englime!20] (motor) at (0,2.6) {motor};
  \node[box, fill=engblue!15] (comp) at (2.4,2.6) {compressor\\train};
  \node[box, fill=engred!15] (cool) at (5.0,2.6) {inter- and\\aftercoolers};
  % store (middle)
  \node[box, fill=engamber!25, minimum width=2.2cm, minimum height=1.3cm] (store) at (7.8,1.3) {air store\\(cavern)};
  % discharge path (bottom row)
  \node[box, fill=engred!15] (reheat) at (5.0,0.0) {reheat};
  \node[box, fill=engblue!15] (turb) at (2.4,0.0) {expander\\(turbine)};
  \node[box, fill=englime!20] (gen) at (0,0.0) {generator};
  % charge flows
  \draw[flow] (motor) -- (comp);
  \draw[flow] (comp) -- (cool);
  \draw[flow] (cool) -- (store.north west);

  \draw[flow] (store.south west) -- (reheat);
  \draw[flow] (reheat) -- (turb);

\end{tikzpicture}
\caption[Compressed-air energy storage plant]{A compressed-air energy
storage plant. During charging (top path) off-peak electricity drives a
motor and a compressor train; intercoolers and an aftercooler reject the
heat of compression, and the cooled high-pressure air fills an underground
store. During discharge (bottom path) the stored air is reheated and
expanded through a turbine that drives a generator at on-peak demand. The
diabatic plant supplies the reheat from burning fuel, throwing the
compression heat away during charging. The adiabatic plant captures the
compression heat in a thermal store (dashed loop) and returns it on
discharge, removing the need for fuel and raising the round-trip
efficiency. The first law fixes the size of every stream; the placement of
the reheat heat is what separates the two plant types.}
\label{fig:vol04ch01:caes-schematic}
\end{figure}
