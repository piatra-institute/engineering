% Figure: schematic of a Ranque-Hilsch vortex tube. Compressed air enters
% tangentially at a swirl chamber, a hot stream leaves the far end past a
% throttle cone, a cold stream leaves the near end through a central orifice.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[>=Latex, font=\small]
% main tube body
\draw[thick, fill=engblue!5] (0,0) rectangle (8,1.6);
% swirl chamber (wider block near the inlet, at x=1.5)
\draw[thick, fill=engamber!12] (1.2,-0.15) rectangle (1.9,1.75);
% tangential inlet arrow
\draw[->, very thick, engblue] (1.55,2.7) -- (1.55,1.8);
\node[engblue, anchor=south] at (1.55,2.75) {compressed air in};
% cold orifice at left end
\draw[thick] (0,0.55) -- (0,1.05);

\end{tikzpicture}
\caption[The Ranque-Hilsch vortex tube]{The vortex tube splits one stream of
compressed air into a hot stream and a cold stream with no moving parts. Air
enters tangentially and spins down the tube; the outer layers leave hot past a
throttle cone at the far end while the inner core reverses and leaves cold through
a central orifice at the near end. The first law is satisfied exactly, the
enthalpy carried out by the two streams summing to the enthalpy carried in, and
the second law is satisfied because the total entropy of the two exit streams
exceeds that of the inlet: the device redistributes energy without violating a law,
paying for the temperature split with the pressure energy of the compressed feed.}
\label{fig:vol04ch01:vortex-tube}
\end{figure}
