% Figure: the air-standard Otto cycle on a p-V diagram. Two adiabats
% (compression 1->2 and expansion 3->4) joined by two isochores
% (heat addition 2->3 and heat rejection 4->1).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=8cm,
  xlabel={volume $V$ (arb.)}, ylabel={pressure $p$ (arb.)},
  xmin=0.6, xmax=9, ymin=0, ymax=42,
  axis lines=left,
  tick label style={font=\scriptsize},
  label style={font=\small},
  clip=false,
]
% compression stroke 1->2 : adiabat p V^1.4 = C1 from V=8 (p=1) to V=1
% C1 = 1 * 8^1.4 = 18.38
\addplot[engblue, very thick, smooth, domain=1:8, samples=60] {18.38/(x^1.4)};
% heat addition 2->3 : isochore at V=1, p from 18.38 to 36.76 (double)
\addplot[engred, very thick] coordinates {(1,18.38) (1,36.76)};
% expansion 3->4 : adiabat p V^1.4 = C2 from V=1 (p=36.76) to V=8
% C2 = 36.76 * 1 = 36.76
\addplot[engorange, very thick, smooth, domain=1:8, samples=60] {36.76/(x^1.4)};
% heat rejection 4->1 : isochore at V=8, p from 2.0 to 1.0
\addplot[enggray, very thick] coordinates {(8,2.0) (8,1.0)};
% state markers
\filldraw[engblack] (axis cs:8,1) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=north west] at (axis cs:8,1) {1};
\filldraw[engblack] (axis cs:1,18.38) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south east] at (axis cs:1,18.38) {2};
\filldraw[engblack] (axis cs:1,36.76) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south east] at (axis cs:1,36.76) {3};
\filldraw[engblack] (axis cs:8,2.0) circle [radius=1.6pt];
\node[engblack, font=\scriptsize, anchor=south west] at (axis cs:8,2.0) {4};
\end{axis}
\end{tikzpicture}
\caption[The air-standard Otto cycle]{The air-standard Otto cycle, the
idealisation of a spark-ignition engine. The working gas is compressed
adiabatically along $1\to2$, heat is added at constant volume along $2\to3$
(the idealised combustion), the gas expands adiabatically along $3\to4$ (the
power stroke), and heat is rejected at constant volume along $4\to1$ (the
exhaust). The enclosed area is the net work per cycle. The efficiency depends
only on the compression ratio $r=V_1/V_2$ through $\eta=1-r^{-(\gamma-1)}$, so a
higher compression ratio raises efficiency until the fuel pre-ignites, the knock
limit that caps the ratio of a real gasoline engine near ten.}
\label{fig:vol04ch01:otto-pv}
\end{figure}
