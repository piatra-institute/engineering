% Figure: energy-flow (Sankey-style) diagram for a coal-fired steam power plant.
% Fuel chemical energy in (100 units) splits into net electrical work (39),
% condenser reject (50), and stack + radiative + auxiliary losses (11).
% Band heights are drawn to scale: 1 unit = 0.05 vertical units.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\small, scale=1.0]
  % fuel input band: 100 units, full height 5.0, from x=0 to x=2.2
  \fill[engamber!40] (0,0) rectangle (2.2,5.0);
  \node[font=\scriptsize, anchor=east, align=right] at (-0.1,2.5)
    {fuel energy in\\$100\%$\\($\dot{Q}_{\text{fuel}}$)};
  \node[engamber, font=\scriptsize, rotate=90] at (1.1,2.5) {coal LHV};

  % boiler: stack loss peels off the top (11 units), height 0.55
  \fill[enggray!35] (2.2,4.45) -- (4.0,4.45) -- (4.0,5.0) -- (2.2,5.0) -- cycle;
  \node[enggray, font=\scriptsize, anchor=west] at (4.05,4.72)
    {stack, radiation, auxiliaries $11\%$};

  % heat into the cycle (steam to turbine): 89 units, height 4.45
  \fill[engred!30] (2.2,0) rectangle (4.0,4.45);
  \node[font=\scriptsize, align=center] at (3.1,2.0) {heat to\\working\\fluid $89\%$};

  % turbine: net work peels off the top (39 units), height 1.95
  \fill[englime!55] (4.0,2.5) -- (6.6,2.5) -- (6.6,4.45) -- (4.0,4.45) -- cycle;
  \node[englime!70!engblack, font=\scriptsize, anchor=west, align=left] at (6.65,3.5)

  % condenser reject: 50 units, height 2.5
  \fill[engblue!35] (4.0,0) rectangle (6.6,2.5);
  \node[engblue, font=\scriptsize, anchor=west, align=left] at (6.65,1.2)
    {condenser reject\\$50\%$ to cooling water\\($Q_c$)};

  % station markers
  \draw[engblack, very thick, dashed] (2.2,-0.3) -- (2.2,5.3);
  \node[enggray, font=\scriptsize, anchor=north] at (2.2,-0.3) {boiler};
  \draw[engblack, very thick, dashed] (4.0,-0.3) -- (4.0,5.3);
  \node[enggray, font=\scriptsize, anchor=north] at (4.0,-0.3) {turbine};
\end{tikzpicture}
\caption[Energy flow through a coal-fired steam power plant]{Energy flow
through a coal-fired steam power plant, drawn to scale for the case study of
this section. The chemical energy released by burning the coal (the fuel
lower heating value) enters at the left. The boiler loses a part of it up
the stack and to radiation and auxiliaries; the rest heats the working
fluid. The turbine converts a fraction of that heat to net electrical work,
and the condenser rejects the largest single stream to the cooling water.
The widths obey the first law: every band that leaves equals the band that
arrives. The net work is $39\%$ of the fuel energy, close to the best
single-cycle steam plants in service. The condenser reject, half the fuel
energy, is the price the second law charges for running a heat engine
between a flame and a river.}
\label{fig:vol04ch01:plant-sankey}
\end{figure}
