% Order of limits matters: a_{m,n} = m/(m+n). Rows fix m and run n; columns
% fix n and run m. Row limits (n to infinity) are all 0; the limit of those
% is 0. Column limits (m to infinity) are all 1; the limit of those is 1.
% The grid shows the two marginal passes disagreeing, so the double limit
% does not exist. Cell shading tracks the value from 0 (light) to 1 (dark).

\def\cell#1#2#3{%
  \pgfmathsetmacro\sh{#3*70}%
  \fill[engblue!\sh] (#1,#2) rectangle ({#1+1},{#2+1});%
  \draw[enggray] (#1,#2) rectangle ({#1+1},{#2+1});%
  \node[font=\footnotesize] at ({#1+0.5},{#2+0.5}) {#3};%
}

% grid: columns n=1,2,3,4 (x=0..4), rows m=1,2,3,4 from top (y=3..0)
% values a_{m,n}=m/(m+n)
% row m=4 (top, y=3): 4/5,4/6,4/7,4/8 = .80,.67,.57,.50
\cell{0}{3}{0.80}\cell{1}{3}{0.67}\cell{2}{3}{0.57}\cell{3}{3}{0.50}
% row m=3 (y=2): 3/4,3/5,3/6,3/7 = .75,.60,.50,.43
\cell{0}{2}{0.75}\cell{1}{2}{0.60}\cell{2}{2}{0.50}\cell{3}{2}{0.43}
% row m=2 (y=1): 2/3,2/4,2/5,2/6 = .67,.50,.40,.33
\cell{0}{1}{0.67}\cell{1}{1}{0.50}\cell{2}{1}{0.40}\cell{3}{1}{0.33}

\caption[Order of limits matters for a double sequence]{The double sequence
$a_{m,n}=m/(m+n)$, tabulated for small $m$ and $n$ with cell shading from
light (value near $0$) to dark (value near $1$). Fixing a row and running
the column index $n\to\infty$ sends every row to $0$, so the iterated limit
$\lim_{m}\lim_{n}a_{m,n}=0$. Fixing a column and running the row index
$m\to\infty$ sends every column to $1$, so the reversed iterated limit
$\lim_{n}\lim_{m}a_{m,n}=1$. The two orders disagree, so the double limit
does not exist, and no single value can be interchanged past both indices.
Uniform convergence of one index in the other is exactly the missing
condition that would license the swap, the same guarantee the chapter's
term-by-term integration argument requires.}
