% Diagnostic ratio e_{n+1}/e_n against step for Newton's iteration on
% a simple root and a double root. Simple root: ratio collapses toward 0
% (superlinear / quadratic). Double root: ratio holds at 1 - 1/m = 0.5.
% This is the fingerprint the engineer reads from the residual sequence.
% x = step, mapped x_cm = (step-2)*1.5. y = ratio in [0,1], mapped y_cm = ratio*6.

\draw[->, thick] (-0.2, 0) -- (7.2, 0) node[right] {iteration step $n$};
\draw[->, thick] (0, -0.2) -- (0, 6.6) node[above] {ratio $e_{n+1}/e_{n}$};

\foreach \s in {2,3,4,5,6} {
  \pgfmathsetmacro\xc{(\s-2)*1.5}
  \draw (\xc, -0.08) -- (\xc, 0.08) node[below, yshift=-0.12cm] {\s};
}
\foreach \r/\lbl in {0/$0$, 0.25/$0.25$, 0.5/$0.5$, 0.75/$0.75$, 1/$1$} {
  \pgfmathsetmacro\yc{\r*6}
  \draw (-0.08, \yc) -- (0.08, \yc) node[left, xshift=-0.1cm] {\lbl};
}

% reference line at 0.5 (the 1 - 1/m rate for m = 2)
\draw[dashed, enggray] (0, 3) -- (6, 3) node[right, enggray] {\footnotesize $1-\tfrac{1}{m}=\tfrac12$};

% Double root: ratio constant 0.5 across all steps
\draw[very thick, engred, mark=square*, mark size=1.3pt] plot coordinates {
  (0,3) (1.5,3) (3,3) (4.5,3) (6,3) };
\node[engred, anchor=south west] at (3.1, 3.05) {double root ($m=2$)};

% Simple root: ratios at steps 2..6: 0.165, 0.038, 0.0015, ~0, ~0
% map y_cm = ratio*6: 0.99, 0.23, 0.009, 0, 0
\draw[very thick, engblue, mark=*, mark size=1.3pt] plot coordinates {
  (0,0.99) (1.5,0.23) (3,0.01) (4.5,0.0) (6,0.0) };
\node[engblue, anchor=west] at (1.6, 0.6) {simple root};

\end{tikzpicture}
\caption[Reading Newton's convergence rate from the error ratio]{The
diagnostic ratio $e_{n+1}/e_{n}$ of successive Newton errors, plotted
against iteration step for a simple root and a double root of the same
form. At a simple root the ratio collapses toward zero: each error is a
vanishing fraction of the last, the signature of quadratic convergence.
At a root of multiplicity $m$ the ratio settles at the constant
$1-1/m$, which is $\tfrac12$ for the double root shown, so the residual
shrinks by a fixed factor per step rather than squaring. A residual
sequence whose successive ratios level off near $0.5$ rather than
collapsing is the fingerprint of a double root, and an engineer who
expected quadratic convergence reads a correctly working solver as
broken. The fix is modified Newton with the multiplicity folded in, or
Steffensen's derivative-free acceleration.}

\end{figure}
