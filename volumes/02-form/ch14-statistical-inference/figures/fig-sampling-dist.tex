% Figure: sampling distribution of the sample mean for three sample
% sizes. Underlying population has mu = 10, sigma = 3. The sampling
% distribution of xbar is N(10, sigma^2/n), so the standard error is
% 3/sqrt(n): 1.342 (n=5), 0.671 (n=20), 0.387 (n=60). The three
% Gaussian densities are drawn from those standard deviations.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=12cm, height=6.5cm,
    xmin=6, xmax=14,
    ymin=0, ymax=1.10,
    axis lines=left,
    xlabel={Value of the sample mean $\bar{x}$},
    ylabel={Sampling density},
    every axis plot/.append style={very thick, mark=none},
    legend style={font=\small, draw=none, fill=none, at={(0.02,0.98)}, anchor=north west},
    samples=200,
    domain=6:14,
    xtick={6,8,10,12,14},
]
% n = 5, se = 1.342
\addplot[engblue] {1/(1.342*sqrt(2*pi))*exp(-(x-10)^2/(2*1.342^2))};
\addlegendentry{$n = 5$,\ \ se $= 1.34$}
% n = 20, se = 0.671
\addplot[engred] {1/(0.671*sqrt(2*pi))*exp(-(x-10)^2/(2*0.671^2))};
\addlegendentry{$n = 20$,\ se $= 0.67$}
% n = 60, se = 0.387
\addplot[englime!70!black, dashed] {1/(0.387*sqrt(2*pi))*exp(-(x-10)^2/(2*0.387^2))};
\addlegendentry{$n = 60$,\ se $= 0.39$}
% true mean
\addplot[engblack, dotted] coordinates {(10,0) (10,1.10)};
\end{axis}
\end{tikzpicture}
\caption[Sampling distribution of the sample mean]{The sampling
distribution of the sample mean $\bar{X}$ for a population with
$\mu = 10$ and $\sigma = 3$, at three sample sizes. The standard
error $\sigma/\sqrt{n}$ shrinks as $1/\sqrt{n}$: quadrupling the
sample halves the width. The shape approaches the Gaussian by the
central limit theorem regardless of the population's shape, which is
why the $z$ and $t$ constructions apply to non-Gaussian data at
large $n$. The dotted vertical line is the fixed true mean; the
estimator, not the parameter, is what varies across resamples.}
\label{fig:vol02:ch14:sampling-dist}
\end{figure}
