% Figure: Weibull probability plot for the censored bearing-life
% case study. The x-axis is ln(t); the y-axis is the standard
% Weibull plotting position ln(-ln(1-F)). Complete (failed) units
% appear as plotted points; the fitted line has slope beta-hat and
% intercept -beta-hat*ln(eta-hat). Coordinates are precomputed from
% the median-rank plotting positions of the eight observed failures
% in the worked case study.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=11cm, height=7cm,
    xmin=4.6, xmax=6.6,
    ymin=-3.2, ymax=1.4,
    axis lines=left,
    xlabel={$\ln t$ (life in hours, log scale)},
    ylabel={$\ln\!\left(-\ln(1 - \hat{F})\right)$},
    every axis plot/.append style={very thick},
    legend style={font=\small, draw=none, fill=none, at={(0.02, 0.98)}, anchor=north west},
    grid=both,
    grid style={dashed, draw=enggray!20},
]
% Plotting positions for the eight observed failures (median-rank
% F = (i-0.3)/(n+0.4), n = 10). x = ln(t_failure),
% y = ln(-ln(1 - F_hat)).
\addplot[only marks, engblue, mark=*, mark size=2pt] coordinates {
  (4.79, -2.66) (5.19, -1.72) (5.42, -1.20) (5.60, -0.82)
  (5.75, -0.51) (5.88, -0.23) (6.01, 0.03) (6.18, 0.30)
};
\addlegendentry{observed failures}
% Fitted Weibull line: slope = beta-hat = 2.33, passing through the
% data; line y = 2.33*(ln t - ln eta-hat), ln eta-hat = 6.00.
\addplot[engred, domain=4.6:6.6, samples=2]{2.33*(x - 6.00)};
\addlegendentry{fitted line, $\hat{\beta} = 2.33$}
% Marker of the fitted characteristic life eta-hat (F = 0.632), at
% y = 0, x = ln eta-hat = 6.00.
\addplot[engblack, dotted] coordinates {(4.6, 0) (6.00, 0)};
\addplot[engblack, dotted] coordinates {(6.00, -3.2) (6.00, 0)};
\node[anchor=south west, font=\footnotesize] at (axis cs:6.00, -3.0) {$\ln\hat{\eta}$};
\end{axis}
\end{tikzpicture}
\caption[Weibull probability plot for the censored bearing-life
data]{Weibull probability plot for the eight observed bearing
failures, with the two right-censored units accounted for in the
total sample size of the median-rank positions. A straight-line fit on
these transformed axes is the graphical evidence that the Weibull
model is defensible: the slope estimates the shape $\hat{\beta} =
2.33$ and the abscissa at $\ln(-\ln(1 - \hat{F})) = 0$ estimates the
characteristic life $\ln\hat{\eta}$, the life by which a fraction
$1 - e^{-1} = 0.632$ of the population has failed. The plot is read
before the maximum-likelihood fit is trusted; a systematic curve
here is the signal that the single-Weibull model has missed a second
failure mode.}
\label{fig:vol02:ch14:weibull-plot}
\end{figure}
