% Figure: Beta prior and two sequential posteriors for a defect rate.
% Prior Beta(2,98) concentrated near 0.02; posterior after batch one
% Beta(5,145); posterior after both batches Beta(9,189). The densities
% are drawn from the Beta density formula evaluated pointwise; each is
% normalised by its own peak so the shape comparison is legible on one
% axis. The rightward march of the mode is the visual signal a
% quality engineer watches.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=12cm, height=7cm,
    xmin=0, xmax=0.12,
    ymin=0, ymax=1.15,
    axis lines=left,
    xlabel={Defect probability $p$},
    ylabel={Posterior density (peak-normalised)},
    every axis plot/.append style={very thick, mark=none},
    legend style={font=\small, draw=none, fill=none, at={(0.98,0.95)}, anchor=north east},
    xtick={0,0.02,0.04,0.06,0.08,0.10},
    ytick=\empty,
    domain=0.001:0.119, samples=120,
]
% Beta(2,98): kernel p^1 (1-p)^97; peak-normalise at mode 0.010204.
% Evaluated through ln to avoid large-exponent overflow:
% ratio = exp( 1*ln(x/mode) + 97*ln((1-x)/(1-mode)) ).
\addplot[enggray!70]{ exp( 1*ln(x/0.010204) + 97*ln((1-x)/0.989796) ) };
\addlegendentry{prior $\mathrm{Beta}(2,98)$}
% Beta(5,145): kernel p^4 (1-p)^144; mode = 4/148 = 0.027027.
\addplot[engblue]{ exp( 4*ln(x/0.027027) + 144*ln((1-x)/0.972973) ) };
\addlegendentry{after batch 1: $\mathrm{Beta}(5,145)$}
% Beta(9,189): kernel p^8 (1-p)^188; mode = 8/196 = 0.040816.
\addplot[engred]{ exp( 8*ln(x/0.040816) + 188*ln((1-x)/0.959184) ) };
\addlegendentry{after batch 2: $\mathrm{Beta}(9,189)$}
% Action threshold at p = 0.05.
\addplot[englime!70!black, dashed] coordinates {(0.05,0) (0.05,1.15)};
\node[anchor=south, font=\footnotesize, englime!60!black, rotate=90] at (axis cs:0.05,0.30) {action threshold $0.05$};
\end{axis}
\end{tikzpicture}
\caption[Sequential Beta posterior for a defect rate]{A conjugate
Beta prior on a supplier defect rate and its two sequential
posteriors, each peak-normalised so the shapes compare on one axis.
The prior $\mathrm{Beta}(2,98)$ sits near two percent; each batch of
defect counts shifts and sharpens the posterior, the mode marching
from $0.010$ toward $0.041$ as evidence accumulates. The sequential
update is coherent: processing the two batches one at a time gives the
same $\mathrm{Beta}(9,189)$ as pooling them, which is the property that
makes the model the standard engine for online quality-control
monitoring. The engineer watches the posterior mass crossing the
action threshold, not the point estimate alone.}
\label{fig:vol02:ch14:beta-posterior-update}
\end{figure}
