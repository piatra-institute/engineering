% Figure: response-surface contour for the central-composite design
% worked example. Two coded factors x1 (temperature) and x2 (time)
% over [-1.6, 1.6]; the fitted second-order surface has a stationary
% point (a maximum) at (0.50, -0.79) in coded units. Contours are
% drawn as nested ellipses; the canonical axes are tilted because the
% interaction coefficient is nonzero. Coordinates are precomputed
% ellipse traces of the fitted quadratic at three response levels.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=10cm, height=8cm,
    xmin=-1.7, xmax=1.7,
    ymin=-1.9, ymax=1.3,
    axis lines=middle,
    axis line style={-},
    xlabel={Coded temperature $x_1$},
    ylabel={Coded time $x_2$},
    xlabel style={at={(axis description cs:0.97,0.59)}, anchor=north},
    ylabel style={at={(axis description cs:0.51,0.97)}, anchor=east},
    every axis plot/.append style={very thick, mark=none},
    grid=both,
    grid style={dashed, draw=enggray!15},
]
% Innermost contour (response = 91.5), a small ellipse around the
% stationary point (0.50, -0.79).
\addplot[engred, smooth] coordinates {
  (0.78,-0.79) (0.70,-0.50) (0.50,-0.40) (0.30,-0.50) (0.22,-0.79)
  (0.30,-1.08) (0.50,-1.18) (0.70,-1.08) (0.78,-0.79)
};
\addlegendentry{$\hat{y} = 91.5$}
% Middle contour (response = 89).
\addplot[engblue, smooth] coordinates {
  (1.10,-0.79) (0.98,-0.25) (0.50,-0.02) (0.02,-0.25) (-0.10,-0.79)
  (0.02,-1.33) (0.50,-1.56) (0.98,-1.33) (1.10,-0.79)
};
\addlegendentry{$\hat{y} = 89$}
% Outer contour (response = 84).
\addplot[englime!70!black, smooth, dashed] coordinates {
  (1.55,-0.79) (1.35,-0.05) (0.50,0.35) (-0.35,-0.05) (-0.55,-0.79)
  (-0.35,-1.53) (0.50,-1.80) (1.35,-1.53) (1.55,-0.79)
};
\addlegendentry{$\hat{y} = 84$}

\addplot[only marks, engblack, mark=*, mark size=2.5pt] coordinates {(0.50, -0.79)};
\node[anchor=south west, font=\footnotesize] at (axis cs:0.55, -0.74) {stationary point};
\end{axis}
\end{tikzpicture}
\caption[Response-surface contours from a central-composite
design]{Fitted second-order response surface in the two coded
factors of the central-composite-design example, drawn as contours
of constant predicted yield. The nested closed contours identify the
stationary point at coded $(0.50, -0.79)$ as a maximum; an open or
saddle-shaped contour set would have signalled a ridge or a saddle,
each calling for a different optimisation move. The tilt of the
contour axes away from the coordinate axes is the visible signature of
the temperature-time interaction coefficient. The engineer reads the
location of the stationary point and the steepness of the contours
together: a flat region near the optimum means the process tolerates
drift, and a tight contour spacing means it does not.}
\label{fig:vol02:ch14:response-surface}
\end{figure}
