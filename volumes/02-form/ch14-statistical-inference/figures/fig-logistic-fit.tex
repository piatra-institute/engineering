% Figure: a fitted logistic regression curve through binary data.
% The open markers at y=0 and y=1 are the raw pass/fail observations
% jittered onto the two outcome levels; the curve is the fitted
% probability sigma(beta0 + beta1 x). Coordinates trace the worked
% IRLS example (beta0 = 4.0, beta1 = -0.5) so the figure and the
% numbers in the text agree.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=11cm, height=6cm,
    xmin=0, xmax=16,
    ymin=-0.08, ymax=1.08,
    axis lines=left,
    xlabel={Travel speed $x$ (mm/s)},
    ylabel={$\Pr[\text{weld passes}]$},
    xtick={0,2,4,6,8,10,12,14,16},
    ytick={0,0.25,0.5,0.731,1.0},
    yticklabels={0,0.25,0.5,0.731,1.0},
    grid=both,
    grid style={dashed, draw=enggray!20},
    every axis plot/.append style={very thick, mark=none},
]
% fitted sigmoid sigma(4 - 0.5 x)
\addplot[engblue, smooth, domain=0:16, samples=80]
  {1/(1 + exp(-(4 - 0.5*x)))};
% raw observations: passes near low speed, fails near high speed
\addplot[only marks, mark=o, mark size=2pt, engblack] coordinates {
  (2,1) (3,1) (4,1) (5,1) (6,1) (6,1) (7,1) (5,0) (8,1)
};
\addplot[only marks, mark=o, mark size=2pt, engred] coordinates {
  (9,0) (10,0) (11,0) (12,0) (13,0) (14,0) (8,0) (10,0)
};
% median-effective level at x = 8
\addplot[engorange, dotted] coordinates {(8,-0.08) (8,0.5)};
\addplot[engorange, dotted] coordinates {(0,0.5) (8,0.5)};
\node[engorange, anchor=west, font=\small] at (axis cs:8.2,0.12) {$x_{50} = 8$};
\end{axis}
\end{tikzpicture}
\caption[Fitted logistic regression curve]{The fitted logistic curve
$\sigma(\hat{\beta}_0 + \hat{\beta}_1 x)$ for the weld-pass model,
with $\hat{\beta}_0 = 4.0$ and $\hat{\beta}_1 = -0.5$ from the
iteratively reweighted least-squares fit. Open circles are the raw
binary observations at the two outcome levels; passes cluster at low
travel speed and failures at high speed. The dotted lines mark the
median-effective level $x_{50} = -\hat{\beta}_0/\hat{\beta}_1 = 8$,
the speed at which the fitted pass probability is $0.5$ and the curve
is steepest. The coefficient is constant on the log-odds scale but
the probability change per unit speed varies along the curve, which
is the geometry the odds-ratio interpretation respects and the
probability-difference interpretation does not.}
\label{fig:vol02:ch14:logistic-fit}
\end{figure}
