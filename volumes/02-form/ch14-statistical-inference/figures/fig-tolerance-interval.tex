% Figure: the three interval half-widths for a Gaussian mean, in units
% of the sample standard deviation s, plotted against sample size n. The
% confidence-interval half-width shrinks as 1/sqrt(n) toward zero; the
% prediction-interval half-width shrinks toward the fixed z-multiple of
% s; the 95/95 two-sided tolerance-interval half-width shrinks toward
% the same z-multiple but stays above the prediction curve at every n,
% because it must cover a stated proportion of the population with
% stated confidence, not a single future draw. Half-widths are the
% standard closed forms: CI = t(0.975,n-1)*s/sqrt(n); PI =
% t(0.975,n-1)*s*sqrt(1+1/n); TI = k2(n,0.95,0.95)*s with the
% Howe approximation for the two-sided tolerance factor, precomputed and
% tabulated below so the figure carries no runtime dependency.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=12cm, height=7cm,
    xmin=3, xmax=50,
    ymin=0, ymax=4.2,
    axis lines=left,
    xlabel={Sample size $n$},
    ylabel={Interval half-width (units of $s$)},
    every axis plot/.append style={very thick, mark=none},
    legend style={font=\small, draw=none, fill=none, at={(0.98,0.95)}, anchor=north east},
    xtick={5,10,15,20,25,30,35,40,45,50},
]
% Confidence-interval half-width, tabulated at t(0.975,n-1)/sqrt(n).
\addplot[engblue] coordinates {
  (3,1.807) (5,1.111) (10,0.715) (15,0.554) (20,0.468)
  (25,0.413) (30,0.373) (40,0.320) (50,0.284)
};
\addlegendentry{confidence (mean)}
% Prediction-interval half-width, t(0.975,n-1)*sqrt(1+1/n).
\addplot[engred] coordinates {
  (3,3.612) (5,2.720) (10,2.371) (15,2.276) (20,2.229)
  (25,2.201) (30,2.183) (40,2.160) (50,2.147)
};
\addlegendentry{prediction (next draw)}
% 95/95 two-sided tolerance factor k2 (Howe approximation), tabulated.
\addplot[englime!70!black] coordinates {
  (3,9.916) (5,5.079) (10,3.379) (15,2.965) (20,2.760)
  (25,2.631) (30,2.549) (40,2.445) (50,2.379)
};
\addlegendentry{tolerance ($95/95$ two-sided)}
% Asymptotic z-multiple 1.96 that PI and TI both approach.
\draw[enggray, dashed] (axis cs:3,1.96) -- (axis cs:50,1.96);
\node[enggray, anchor=south east, font=\footnotesize] at (axis cs:50,1.96) {$z_{0.975}=1.96$};
\end{axis}
\end{tikzpicture}
\caption[Confidence, prediction, and tolerance interval half-widths]{Half-widths
of the three intervals for a Gaussian population, in units of the sample
standard deviation $s$, against sample size $n$. The confidence interval
for the mean shrinks to zero as $1/\sqrt{n}$, because more data pin the
mean ever tighter. The prediction interval for one future draw shrinks
toward the fixed multiple $z_{0.975}=1.96$, because a single new
observation carries irreducible spread the sample size cannot remove.
The two-sided $95/95$ tolerance interval, which must cover $95\,\%$ of
the population with $95\,\%$ confidence, sits above the prediction curve
at every finite $n$ and approaches the same limit only as the estimation
uncertainty vanishes. The three answer three different questions, and an
engineer who quotes a confidence interval where a specification demands
population coverage has quoted the narrowest of the three by mistake.}
\label{fig:vol02ch14:tolerance-interval}
\end{figure}
