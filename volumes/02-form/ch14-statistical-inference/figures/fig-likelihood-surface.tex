% Figure: contours of the Gaussian log-likelihood over (mu, sigma) for
% a fixed sample, with the MLE at the peak. The contours are drawn as
% nested ellipse-like level sets of a precomputed concave surface; the
% ridge of the surface along mu = xbar shows that the mu-MLE does not
% depend on sigma, while the sigma-MLE sits at a finite height. The
% marked point is the joint MLE (xbar, sigmahat). Coordinates are a
% schematic quadratic-in-mu, penalised-in-log-sigma illustration.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
    width=11cm, height=8cm,
    xmin=8.4, xmax=11.6,
    ymin=0.6, ymax=2.6,
    axis lines=left,
    xlabel={Mean $\mu$},
    ylabel={Standard deviation $\sigma$},
    xtick={9,10,11},
    ytick={1.0,1.5,2.0,2.5},
]
% Precomputed contour set: nested closed curves around the MLE at
% (10, 1.4). We approximate level sets of a concave surface with
% scaled ellipses whose vertical axis is stretched to reflect the
% skew of the sigma-direction of the log-likelihood.
% Outer contour.
\addplot[engblue, thick] coordinates {
 (8.7,1.4) (9.0,2.05) (10.0,2.30) (11.0,2.05) (11.3,1.4)
 (11.0,0.95) (10.0,0.82) (9.0,0.95) (8.7,1.4)};
% Middle contour.
\addplot[engblue, thick] coordinates {
 (9.2,1.4) (9.4,1.85) (10.0,2.02) (10.6,1.85) (10.8,1.4)
 (10.6,1.08) (10.0,1.00) (9.4,1.08) (9.2,1.4)};
% Inner contour.
\addplot[engblue, thick] coordinates {
 (9.6,1.4) (9.72,1.66) (10.0,1.74) (10.28,1.66) (10.4,1.4)
 (10.28,1.20) (10.0,1.14) (9.72,1.20) (9.6,1.4)};
% Ridge line mu = xbar = 10 (the mu-MLE is 10 regardless of sigma).
\addplot[enggray!60, dashed] coordinates {(10.0,0.6) (10.0,2.6)};
% Joint MLE marker.
\addplot[only marks, engred, mark=*, mark size=2.5pt] coordinates {(10.0,1.4)};
\node[anchor=south west, font=\footnotesize, engred] at (axis cs:10.05,1.44) {$(\hat{\mu},\hat{\sigma})$};

\caption[Contours of the Gaussian log-likelihood]{Level sets of the
two-parameter Gaussian log-likelihood over the mean $\mu$ and the
standard deviation $\sigma$ for a fixed sample. The vertical ridge at
$\mu = \bar{x}$ shows that the maximum-likelihood mean is the sample
mean whatever the variance is, the algebraic fact behind the block
diagonal Hessian: the two estimates are asymptotically independent.
The joint maximum sits at $(\hat{\mu},\hat{\sigma})$, and the contours
stretch further along the $\sigma$ axis than along the $\mu$ axis,
which is the geometric signature of the variance estimate carrying a
skewed sampling distribution at small $n$.}
\label{fig:vol02:ch14:likelihood-surface}
\end{figure}
