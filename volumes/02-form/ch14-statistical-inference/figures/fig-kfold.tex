% Figure: schematic of k=5 cross-validation. Five rows of five
% horizontal cells; in each row a different cell is the held-out
% fold (red), the others are training (blue).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  every node/.style={font=\small},
  train/.style={draw=engblack, fill=engblue!18,
    minimum width=1.6cm, minimum height=0.7cm},
  test/.style={draw=engblack, fill=engred!30,
    minimum width=1.6cm, minimum height=0.7cm},
]

\foreach \row in {1, ..., 5}{
  \pgfmathsetmacro{\y}{-0.85*\row}
  \node[anchor=east] at (-0.25, \y) {Fold \row};
  \foreach \col in {1, ..., 5}{
    \pgfmathsetmacro{\x}{1.65*\col}
    \ifnum\row=\col
      \node[test] at (\x, \y) {test};
    \else
      \node[train] at (\x, \y) {train};
    \fi
  }
}

% Header
\foreach \col in {1, ..., 5}{
  \pgfmathsetmacro{\x}{1.65*\col}
  \node[anchor=south, font=\small] at (\x, 0) {block \col};
}

% Right-side bracket with caption text
\node[anchor=west, align=left, font=\small]
  at (10.0, -2.5)
  {Five fits in total.\\
   Held-out error per row:\\
   $L_j = \frac{1}{|\mathcal{F}_j|}\sum_{i \in \mathcal{F}_j} L(y_i, \hat{f}^{(-j)}(x_i))$.\\[4pt]
   Cross-validated estimate:\\
   $\widehat{\mathrm{CV}}_{5} = \frac{1}{5}\sum_{j=1}^{5} L_j$.};

\draw[engblack, decorate, decoration={brace, amplitude=6pt}]
  (8.75, -0.6) -- (8.75, -4.6);
\end{tikzpicture}
\caption[Five-fold cross-validation as a schedule of fits]{The
schedule of fits in $k=5$ cross-validation. Each row is one fit:
the cell marked ``test'' is the held-out fold; the other four
provide the training data. The held-out errors $L_1, \ldots, L_5$
are averaged to give $\widehat{\mathrm{CV}}_5$, an estimate of the
generalisation error \cite{text:v2ch14-stone-cv-1974}. Every
operation that adapts to the data (standardisation, hyperparameter
search, variable selection) must sit inside the per-row training
block. A pre-processing step applied to the whole dataset before
the split contaminates every row and produces an optimistic
$\widehat{\mathrm{CV}}_5$.}
\label{fig:vol02:ch14:kfold}
\end{figure}
