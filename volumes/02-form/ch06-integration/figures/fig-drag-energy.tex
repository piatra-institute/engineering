% Figure: aerodynamic drag-energy case study. Top: a measured drive-cycle
% speed profile v(t) over a 180 s window (a stylised urban-to-highway
% segment). The drag power is P(t) = (1/2) rho C_d A v(t)^3, so the
% cumulative drag energy is the integral of a cubic-in-v integrand. We
% plot v(t) and the cubic weighting v^3 (scaled) to make the point that
% the energy integral is dominated by the high-speed portion.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=5.8cm,
  xlabel={$t$ (s)}, ylabel={speed $v$ (m/s)},
  xmin=0, xmax=180, ymin=0, ymax=36,
  axis lines=left, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=north west, legend style={font=\footnotesize},
  axis y line*=left,
]
% measured-style speed profile: idle, urban, accelerate to highway, cruise
\addplot[engblue, very thick] coordinates {
  (0,0)(10,5)(25,8)(40,6)(55,11)(70,9)(85,14)(100,22)(115,30)(130,33)
  (145,32)(160,33)(175,31)(180,28)};
\addlegendentry{$v(t)$ measured}
\end{axis}
\begin{axis}[
  width=12cm, height=5.8cm,
  xmin=0, xmax=180, ymin=0, ymax=1.05,
  axis lines=none, font=\small,
  legend pos=north west, legend style={font=\footnotesize, yshift=-1.3em},
]
% cumulative drag energy fraction (normalised), a monotone S-curve that
% rises fastest where v is largest because the integrand scales as v^3
\addplot[engred, very thick, dashed] coordinates {
  (0,0)(10,0.002)(25,0.010)(40,0.018)(55,0.035)(70,0.055)(85,0.085)
  (100,0.16)(115,0.33)(130,0.55)(145,0.74)(160,0.90)(175,0.99)(180,1.0)};
\addlegendentry{cumulative drag energy (normalised)}
\end{axis}
\end{tikzpicture}
\caption[Drag energy over a measured drive cycle]{A measured speed
profile $v(t)$ over a 180-second drive cycle (blue, left axis) and the
normalised cumulative aerodynamic drag energy
$\int_{0}^{t}\frac{1}{2}\rho\,C_{d}A\,v(\tau)^{3}\,\mathrm{d}\tau$ (red
dashed, right axis). Because the drag power scales as the cube of speed,
the energy integral is dominated by the high-speed cruise segment after
$t\approx 100\,$s, even though that segment occupies less than half the
window. The cubic weighting is the structural reason a measurement
error in the high-speed portion of $v(t)$ propagates three times as
strongly into the energy as the same fractional error at low speed,
which the case study's error budget makes quantitative.}
\label{fig:vol02:ch06:drag-energy}
\end{figure}
