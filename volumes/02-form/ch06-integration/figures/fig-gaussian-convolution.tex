% Figure: convolution of two Gaussians is a Gaussian whose variance is
% the sum of the variances (widths add in quadrature). Shows two narrow
% Gaussians (sigma1, sigma2) and their convolution (sigma = sqrt(s1^2 +
% s2^2)), all normalised to unit area.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xlabel={$x$}, ylabel={density (unit area)},
  xmin=-5, xmax=5, ymin=0, ymax=0.62,
  domain=-5:5, samples=200,
  axis lines=middle, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=north east, legend style={font=\footnotesize},
]
% sigma1 = 0.8
\addplot[engblue, very thick]
  {1/(0.8*sqrt(2*pi)) * exp(-x^2/(2*0.8^2))};
\addlegendentry{$f$: $\sigma_{1} = 0.8$}
% sigma2 = 1.2
\addplot[engamber, very thick]
  {1/(1.2*sqrt(2*pi)) * exp(-x^2/(2*1.2^2))};
\addlegendentry{$g$: $\sigma_{2} = 1.2$}
% convolution: sigma = sqrt(0.8^2 + 1.2^2) = sqrt(2.08) ~ 1.442
\addplot[engred, very thick, dashed]
  {1/(1.4422*sqrt(2*pi)) * exp(-x^2/(2*1.4422^2))};
\addlegendentry{$f * g$: $\sigma = \sqrt{\sigma_{1}^{2} + \sigma_{2}^{2}}
  \approx 1.44$}
\end{axis}
\end{tikzpicture}
\caption[Convolution of two Gaussians]{The convolution of two
Gaussians is a Gaussian whose variance is the sum of the variances, so
the widths combine in quadrature: $\sigma = \sqrt{\sigma_{1}^{2} +
\sigma_{2}^{2}}$. Convolving a Gaussian of width $\sigma_{1} = 0.8$
(blue) with one of width $\sigma_{2} = 1.2$ (amber) gives a Gaussian
of width $\sigma \approx 1.44$ (red dashed), wider and lower than
either input but of the same unit area. The result is the integral
form of the add-in-quadrature rule for independent Gaussian errors
(Volume I Chapter 4) and the reason cascaded Gaussian blurs in imaging
never sharpen. Repeated convolution drives any finite-variance density
toward the Gaussian, which is the integral content of the central
limit theorem (Volume II Chapter 13).}
\label{fig:vol02:ch06:gaussian-convolution}
\end{figure}
