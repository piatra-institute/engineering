% Figure: a Type I region (bounded by two curves in y over an x-range)
% and the inner integration strip. We draw the region between y = x
% (upper) and y = x^2 (lower) on [0,1], with a vertical strip showing
% the inner dy-integration with limits y = x^2 to y = x. The fill is
% built by overlaying a fill under the upper curve with a white fill
% under the lower curve, the closedcycle technique used elsewhere in
% this chapter (no fillbetween library required).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=9.5cm, height=7cm,
  xlabel={$x$}, ylabel={$y$},
  xmin=-0.05, xmax=1.15, ymin=-0.05, ymax=1.1,
  axis lines=middle, font=\small,
  xtick={0,0.5,1}, ytick={0,0.5,1},
]
% region fill: under y = x, minus under y = x^2
\addplot[draw=none, fill=engblue!16, domain=0:1, samples=100] {x} \closedcycle;
\addplot[draw=none, fill=white, domain=0:1, samples=100] {x^2} \closedcycle;
% the bounding curves
\addplot[engblue, very thick, domain=0:1, samples=100] {x};
\addplot[engred, very thick, domain=0:1, samples=100] {x^2};
% a representative vertical strip at x = 0.6 from y = x^2 = 0.36 to y = x = 0.6
\draw[engamber, very thick, ->] (axis cs:0.6,0.36) -- (axis cs:0.6,0.6);
\node[engamber, font=\footnotesize, anchor=west] at (axis cs:0.62,0.48)
  {$\displaystyle\int_{x^{2}}^{x}\mathrm{d}y$};
\node[engblue, font=\footnotesize, anchor=south east] at (axis cs:1.0,1.0) {$y=x$};
\node[engred, font=\footnotesize, anchor=north west] at (axis cs:1.0,1.0) {$y=x^{2}$};
\end{axis}
\end{tikzpicture}
\caption[A Type I region and its inner integration]{A Type I (vertically
simple) region $D = \{(x,y): 0 \le x \le 1,\ x^{2} \le y \le x\}$,
bounded below by $y = x^{2}$ (red) and above by $y = x$ (blue). The
double integral $\iint_{D} f\,\mathrm{d}A$ is evaluated as an iterated
integral whose inner integration runs in $y$ from the lower curve to the
upper curve at fixed $x$ (amber strip), and whose outer integration runs
in $x$ across the projection $[0,1]$. The variable inner limits, set by
the two bounding curves, are the only new ingredient beyond the
rectangle case: $\iint_{D} f\,\mathrm{d}A = \int_{0}^{1}\!\int_{x^{2}}^{x}
f(x,y)\,\mathrm{d}y\,\mathrm{d}x$. A horizontal strip would give the Type
II description with the order of integration swapped.}
\label{fig:vol02:ch06:nonrectangular-region}
\end{figure}
