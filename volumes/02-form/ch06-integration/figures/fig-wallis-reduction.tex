% Figure: the Wallis integrals W_n = int_0^{pi/2} sin^n x dx, computed by
% the reduction formula W_n = ((n-1)/n) W_{n-2}. The values decrease with
% n; even and odd n follow two interleaved sequences that both fall like
% sqrt(pi/(2n)). Shown as points against n.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.0cm,
  xlabel={power $n$},
  ylabel={$W_{n} = \int_{0}^{\pi/2}\sin^{n}x\,\dd x$},
  xmin=0, xmax=13, ymin=0, ymax=1.7,
  axis lines=left, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=north east, legend style={font=\footnotesize},
  xtick={0,2,4,6,8,10,12},
]
% even n: W0=pi/2=1.5708, W2=pi/4=0.7854, W4=3pi/16=0.5890,
% W6=15pi/96=0.4909, W8=105pi/768=0.4295, W10=945pi/9216=0.3866,
% W12=0.3547
\addplot[engblue, thick, mark=*, mark size=1.8pt, only marks]
  coordinates {(0,1.5708) (2,0.7854) (4,0.5890) (6,0.4909)
               (8,0.4295) (10,0.3866) (12,0.3547)};
\addlegendentry{even $n$}
% odd n: W1=1, W3=2/3=0.6667, W5=8/15=0.5333, W7=16/35=0.4571,
% W9=128/315=0.4063, W11=256/693=0.3694
\addplot[engred, thick, mark=square*, mark size=1.6pt, only marks]
  coordinates {(1,1.0) (3,0.6667) (5,0.5333) (7,0.4571)
               (9,0.4063) (11,0.3694)};
\addlegendentry{odd $n$}
% asymptote sqrt(pi/(2n))
\addplot[engamber, very thick, dashed, domain=1:12, samples=80]
  {sqrt(pi/(2*x))};
\addlegendentry{$\sqrt{\pi/2n}$ asymptote}
\end{axis}
\end{tikzpicture}
\caption[Wallis integrals from the reduction formula]{The Wallis
integrals $W_{n} = \int_{0}^{\pi/2}\sin^{n}x\,\dd x$, generated by the
reduction formula $W_{n} = \tfrac{n-1}{n}W_{n-2}$ that integration by
parts produces, with seeds $W_{0} = \pi/2$ and $W_{1} = 1$. The even-$n$
values (blue circles) carry a factor of $\pi$ and the odd-$n$ values
(red squares) are rational, yet both sequences decrease and merge onto
the same envelope $\sqrt{\pi/2n}$ (amber dashed) for large $n$. Taking
the ratio $W_{2n}/W_{2n+1} \to 1$ is Wallis's product for $\pi$. The
recursion is the reason a single integration by parts, tabulated once,
delivers every member of the family without re-deriving the parts step
for each power.}
\label{fig:vol02ch06:wallis-reduction}
\end{figure}
