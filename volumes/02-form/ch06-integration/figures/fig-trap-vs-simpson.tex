% Figure: trapezoid panel vs Simpson parabola panel on the same data
% (the function f(x) = exp(-x^2) on [0, 1]), illustrating the
% interpolant under each rule.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}

% Left panel: trapezoid
\begin{axis}[
  name=trap,
  width=7cm, height=5.5cm,
  xlabel={$x$},
  ylabel={$f(x)$},
  xmin=0, xmax=1,
  ymin=0, ymax=1.1,
  domain=0:1,
  samples=150,
  axis lines=left,
  grid=both,
  major grid style={engblack!15},
  font=\small,
  title={Trapezoidal panel ($n = 4$)},
]
% Shaded trapezoids
\foreach \i in {0,1,2,3}{
  \pgfmathsetmacro{\xl}{\i*0.25}
  \pgfmathsetmacro{\xr}{(\i+1)*0.25}
  \pgfmathsetmacro{\yl}{exp(-\xl*\xl)}
  \pgfmathsetmacro{\yr}{exp(-\xr*\xr)}
  \addplot[draw=engblue, fill=engblue!12, fill opacity=0.5]
    coordinates {(\xl,0) (\xr,0) (\xr,\yr) (\xl,\yl)} \closedcycle;
}
\addplot[engblack, very thick] {exp(-x*x)};
\end{axis}

% Right panel: Simpson
\begin{axis}[
  at={(trap.east)}, anchor=west, xshift=12pt,
  width=7cm, height=5.5cm,
  xlabel={$x$},
  ylabel={},
  xmin=0, xmax=1,
  ymin=0, ymax=1.1,
  domain=0:1,
  samples=150,
  axis lines=left,
  grid=both,
  major grid style={engblack!15},
  font=\small,
  title={Simpson panel ($n = 4$)},
]
% Simpson: each pair of subintervals carries a parabola through three
% endpoint values; we plot the parabola fits as smooth curves over
% two subintervals each.
\foreach \i in {0,2}{
  \pgfmathsetmacro{\xl}{\i*0.25}
  \pgfmathsetmacro{\xm}{(\i+1)*0.25}
  \pgfmathsetmacro{\xr}{(\i+2)*0.25}
  \pgfmathsetmacro{\yl}{exp(-\xl*\xl)}
  \pgfmathsetmacro{\ym}{exp(-\xm*\xm)}
  \pgfmathsetmacro{\yr}{exp(-\xr*\xr)}
  % Coefficients of the Lagrange parabola through (xl,yl),(xm,ym),(xr,yr).
  % Plot the shaded region under the parabola; approximate by a fine
  % piecewise sampling.
  \addplot[draw=engred, fill=engred!12, fill opacity=0.5,
    domain=\xl:\xr, samples=40]
    ({x},
     {\yl*((x-\xm)*(x-\xr))/((\xl-\xm)*(\xl-\xr)) +
      \ym*((x-\xl)*(x-\xr))/((\xm-\xl)*(\xm-\xr)) +
      \yr*((x-\xl)*(x-\xm))/((\xr-\xl)*(\xr-\xm))})
    \closedcycle;
}
\addplot[engblack, very thick] {exp(-x*x)};
\end{axis}

\end{tikzpicture}
\caption[Trapezoidal and Simpson interpolants]{Trapezoidal panels
(blue, left) and Simpson panels (red, right) compared on the same
integrand $f(x) = e^{-x^{2}}$ over $[0, 1]$ with $n = 4$
subintervals. The trapezoidal rule joins each pair of adjacent
endpoint values with a straight line and sums the trapezoids;
Simpson's rule fits a parabola through three endpoint values on
each pair of subintervals. Both rules require the same five
function evaluations on a $4$-subinterval grid; the parabolic fit
hugs the curve more closely than the chord and recovers an order
$h^{4}$ error against the order $h^{2}$ error of the trapezoidal
rule \cite{text:v2ch06-burden-faires-numerical}.}
\label{fig:vol02:ch06:trap-vs-simpson}
\end{figure}
