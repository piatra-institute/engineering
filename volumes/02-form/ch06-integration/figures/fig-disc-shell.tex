% Figure: volume of revolution by disc method and by shell method on the
% same generating region (y = sqrt(x) on [0, 1] revolved about the x-axis
% for discs, and about the y-axis for shells).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}

% Left panel: disc method (revolve about x-axis)
\begin{axis}[
  name=disc,
  width=6.4cm, height=5.2cm,
  xlabel={$x$}, ylabel={$y$},
  xmin=-0.1, xmax=1.15, ymin=-1.15, ymax=1.15,
  domain=0:1, samples=120,
  axis lines=middle, font=\footnotesize,
  title={Disc method (about $x$-axis)},
]
% the generating curve y = sqrt(x) and its reflection
\addplot[engblack, very thick] {sqrt(x)};
\addplot[engblack, very thick, dashed] {-sqrt(x)};
% a representative disc at x = 0.6 shown as a thin vertical strip
\pgfmathsetmacro{\xc}{0.6}
\pgfmathsetmacro{\rc}{sqrt(\xc)}
\addplot[draw=engblue, fill=engblue!18]
  coordinates {(\xc-0.03,-\rc) (\xc+0.03,-\rc) (\xc+0.03,\rc) (\xc-0.03,\rc)} \closedcycle;
\node[font=\scriptsize, engblue] at (axis cs:0.6,0.95) {radius $y$};
\end{axis}

% Right panel: shell method (revolve about y-axis)
\begin{axis}[
  at={(disc.east)}, anchor=west, xshift=14pt,
  width=6.4cm, height=5.2cm,
  xlabel={$x$}, ylabel={$y$},
  xmin=-1.15, xmax=1.15, ymin=-0.1, ymax=1.15,
  domain=0:1, samples=120,
  axis lines=middle, font=\footnotesize,
  title={Shell method (about $y$-axis)},
]
\addplot[engblack, very thick] {sqrt(x)};
\addplot[engblack, very thick] ({-x},{sqrt(x)});
% a representative cylindrical shell at x = 0.55
\pgfmathsetmacro{\xs}{0.55}
\pgfmathsetmacro{\hs}{sqrt(\xs)}
\addplot[draw=engred, fill=engred!18]
  coordinates {(\xs-0.025,0) (\xs+0.025,0) (\xs+0.025,\hs) (\xs-0.025,\hs)} \closedcycle;
\addplot[draw=engred, fill=engred!18]
  coordinates {(-\xs-0.025,0) (-\xs+0.025,0) (-\xs+0.025,\hs) (-\xs-0.025,\hs)} \closedcycle;
\node[font=\scriptsize, engred] at (axis cs:0.55,1.0) {height $y$};
\end{axis}

\end{tikzpicture}
\caption[Disc and shell methods of revolution]{Two ways to revolve the
region under $y = \sqrt{x}$ on $[0, 1]$ into a solid. Left: revolving
about the $x$-axis builds the solid from thin discs of radius $y =
\sqrt{x}$ and thickness $\dd x$, each contributing $\pi y^{2}\dd x$,
giving $V = \pi\int_{0}^{1} x \dd x = \pi/2$. Right: revolving about
the $y$-axis builds the same kind of solid from thin cylindrical
shells of radius $x$, height $y = \sqrt{x}$, and thickness $\dd x$,
each contributing $2\pi x \cdot \sqrt{x}\,\dd x$, giving $V =
2\pi\int_{0}^{1} x^{3/2}\dd x = 4\pi/5$. The choice of method follows
the geometry: discs when slicing perpendicular to the axis is clean,
shells when slicing parallel to it is.}
\label{fig:vol02:ch06:disc-shell}
\end{figure}
