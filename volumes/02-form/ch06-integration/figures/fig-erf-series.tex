% Figure: the error function and its Maclaurin partial sums. The
% integrand e^{-t^2} has no elementary antiderivative, but its Taylor
% series integrates term by term to a convergent series for erf(x).
% Partial sums with 2, 3, and 5 terms are shown against the exact erf,
% illustrating that a few terms suffice near the origin and more are
% needed as x grows.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.2cm,
  xlabel={$x$}, ylabel={$\mathrm{erf}(x)$ and partial sums},
  xmin=0, xmax=2.2, ymin=0, ymax=1.35,
  domain=0:2.2, samples=160,
  axis lines=left, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=south east, legend style={font=\footnotesize},
]
% "exact" erf via a high-order Maclaurin sum (accurate over [0, 2.2])
\addplot[engblack, very thick, samples=200]
  {2/sqrt(pi) * (x - x^3/3 + x^5/10 - x^7/42 + x^9/216 - x^11/1320
   + x^13/9360 - x^15/75600 + x^17/685440 - x^19/6894720)};
\addlegendentry{$\mathrm{erf}(x)$ (exact)}
% 2-term partial sum: (2/sqrt(pi)) (x - x^3/3)
\addplot[engblue, very thick, dashed, domain=0:1.2]
  {2/sqrt(pi) * (x - x^3/3)};
\addlegendentry{$2$ terms}
% 3-term: + x^5/10
\addplot[engamber, very thick, dotted, domain=0:1.5]
  {2/sqrt(pi) * (x - x^3/3 + x^5/10)};
\addlegendentry{$3$ terms}
% 5-term: - x^7/42 + x^9/216
\addplot[engred, very thick, domain=0:1.9]
  {2/sqrt(pi) * (x - x^3/3 + x^5/10 - x^7/42 + x^9/216)};
\addlegendentry{$5$ terms}
\end{axis}
\end{tikzpicture}
\caption[Error function by term-by-term integration]{The error
function $\mathrm{erf}(x) = \tfrac{2}{\sqrt{\pi}}\int_{0}^{x}
e^{-t^{2}}\dd t$ has no elementary antiderivative, yet integrating the
Maclaurin series of $e^{-t^{2}}$ term by term gives a convergent power
series. The two-term partial sum (blue dashed) already tracks the exact
curve (black) out to about $x = 0.7$; three terms (amber dotted) reach
$x \approx 1$; five terms (red) hold to about $x \approx 1.4$ before the
alternating series needs still more terms to control the growing powers.
The lesson is that a non-elementary integral is not an intractable one:
term-by-term integration of a convergent series is a working
antiderivative wherever the series converges, and the number of terms is
set by the accuracy the application demands over the range of interest.}
\label{fig:vol02ch06:erf-series}
\end{figure}
