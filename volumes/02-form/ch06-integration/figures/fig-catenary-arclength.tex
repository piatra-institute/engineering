% Figure: a hanging cable takes the catenary shape y = a cosh(x/a); its
% arc length between two supports is an integral whose integrand carries a
% hyperbolic cosine and closes exactly in hyperbolic-sine form.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.0cm,
  xlabel={horizontal position $x$ (m)},
  ylabel={height $y$ (m)},
  xmin=-2.2, xmax=2.2, ymin=0, ymax=3.2,
  domain=-2:2, samples=200,
  axis lines=left, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=north east, legend style={font=\footnotesize},
]
% catenary y = a cosh(x/a) with a = 1
\addplot[engblue, very thick] {cosh(x)};
\addlegendentry{catenary $y=a\cosh(x/a)$}
% parabola through same three points for contrast: passes (0,1),(2,cosh2)
% cosh(2) approx 3.7622; parabola y = 1 + c x^2 with c = (3.7622-1)/4 = 0.6906
\addplot[engred, thick, dashed, domain=-2:2] {1 + 0.6906*x^2};
\addlegendentry{parabola (same endpoints)}
% supports
\addplot[engamber, only marks, mark=*, mark size=1.8pt]
  coordinates {(-2,3.7622) (2,3.7622)};
% clamp the y-axis view: annotate the sag point
\addplot[engblue, only marks, mark=*, mark size=1.5pt]
  coordinates {(0,1)};
\node[font=\scriptsize, engblue, anchor=north]
  at (axis cs:0,0.97) {lowest point};
\end{axis}
\end{tikzpicture}
\caption[Catenary shape and its arc-length integral]{A flexible cable
hanging under its own weight takes the catenary shape $y = a\cosh(x/a)$,
not the parabola (red dashed) a first guess suggests, though the two are
close for a shallow sag. The arc length between supports at $\pm b$ is
$s = \int_{-b}^{b}\sqrt{1 + \sinh^2(x/a)}\,\dd x = \int_{-b}^{b}
\cosh(x/a)\,\dd x = 2a\sinh(b/a)$, the rare arc-length integral that
closes in elementary form because the identity $1 + \sinh^2 =
\cosh^2$ removes the radical exactly. The cable's length exceeds the
straight-line span, and the integral gives the excess a rigger must
allow when cutting a span to length.}
\label{fig:vol02ch06:catenary-arclength}
\end{figure}
