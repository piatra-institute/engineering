% Figure: Fourier-coefficient pipeline. A square wave reconstructed from
% its first few odd harmonics, computed by the coefficient integral
% b_n = (2/T) int s(t) sin(n omega t) dt. Shows the target square wave
% and partial sums N = 1, 3, 9 harmonics, with Gibbs overshoot visible.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=6cm,
  xlabel={$t/T$}, ylabel={amplitude},
  xmin=0, xmax=1, ymin=-1.45, ymax=1.45,
  domain=0:1, samples=400,
  axis lines=middle, grid=both,
  major grid style={engblack!12}, font=\small,
  legend pos=north east, legend style={font=\footnotesize},
]
% target square wave: +1 on (0,0.5), -1 on (0.5,1)
\addplot[engblack, thick, dashed, samples=2, domain=0.001:0.499] {1};
\addplot[engblack, thick, dashed, samples=2, domain=0.501:0.999] {-1};
% N = 1 (fundamental): (4/pi) sin(2 pi t)
\addplot[engred, thick] {(4/pi)*sin(deg(2*pi*x))};
\addlegendentry{$N=1$}
% N = 3 harmonics (n = 1,3,5)
\addplot[engamber, thick]
  {(4/pi)*(sin(deg(2*pi*x)) + sin(deg(6*pi*x))/3 + sin(deg(10*pi*x))/5)};
\addlegendentry{$N=3$ terms}
% N = 9 odd harmonics (n = 1,3,...,17)
\addplot[engblue, very thick]
  {(4/pi)*(sin(deg(2*pi*x)) + sin(deg(6*pi*x))/3 + sin(deg(10*pi*x))/5
  + sin(deg(14*pi*x))/7 + sin(deg(18*pi*x))/9 + sin(deg(22*pi*x))/11
  + sin(deg(26*pi*x))/13 + sin(deg(30*pi*x))/15 + sin(deg(34*pi*x))/17)};
\addlegendentry{$N=9$ terms}
\end{axis}
\end{tikzpicture}
\caption[Fourier-coefficient reconstruction of a square wave]{A unit
square wave (black dashed) reconstructed from its Fourier sine
coefficients $b_{n} = 4/(n\pi)$ for odd $n$, computed by the projection
integral $b_{n} = (2/T)\int_{0}^{T} s(t)\sin(n\omega t)\,\mathrm{d}t$.
The fundamental alone ($N=1$, red) captures the gross shape; three terms
(amber) sharpen the edges; nine terms (blue) track the flat tops while
the persistent overshoot near each jump is the Gibbs phenomenon, which
does not vanish as terms are added but narrows. The orthogonality of the
trigonometric family makes each coefficient an independent integral that
picks out one harmonic and annihilates the rest; the coefficient
pipeline is the working content of the Fourier series of Volume II
Chapter 12.}
\label{fig:vol02:ch06:fourier-coefficients}
\end{figure}
