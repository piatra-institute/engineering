% Figure: area between two curves y = x and y = x^2 on [0, 1], the
% region whose area is the integral of the difference of the two
% bounding functions.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=8.5cm, height=6cm,
  xlabel={$x$}, ylabel={$y$},
  xmin=-0.05, xmax=1.1, ymin=-0.05, ymax=1.1,
  domain=0:1, samples=120,
  axis lines=left, grid=both,
  major grid style={engblack!12},
  font=\small,
  legend pos=north west, legend style={font=\footnotesize},
]
% shaded region between y = x (upper) and y = x^2 (lower)
\addplot[draw=none, fill=engblue!15, domain=0:1] {x} \closedcycle;
\addplot[draw=none, fill=white, domain=0:1] {x*x} \closedcycle;
\addplot[engblue, very thick] {x}; \addlegendentry{$y = x$}
\addplot[engred, very thick] {x*x}; \addlegendentry{$y = x^{2}$}
% a representative vertical strip at x = 0.55
\pgfmathsetmacro{\xs}{0.55}
\addplot[engblack, thick] coordinates {(\xs,\xs*\xs) (\xs,\xs)};
\node[font=\scriptsize, anchor=west] at (axis cs:0.57,0.40)
  {$\bigl(x - x^{2}\bigr)\,\dd x$};
\end{axis}
\end{tikzpicture}
\caption[Area between two curves]{The area between $y = x$ and $y =
x^{2}$ on $[0, 1]$. A vertical strip of width $\dd x$ at position $x$
has height equal to the gap between the upper and lower bounding
curves, $x - x^{2}$, and contributes $(x - x^{2})\dd x$ to the area.
Summing the strips gives $A = \int_{0}^{1}(x - x^{2})\dd x = \tfrac{1}{2}
- \tfrac{1}{3} = \tfrac{1}{6}$. The integrand is the difference of the
two functions, ordered upper-minus-lower so the contribution of each
strip is positive; reversing the order would give the negative of the
area and signals that the two curves crossed.}
\label{fig:vol02:ch06:area-between}
\end{figure}
