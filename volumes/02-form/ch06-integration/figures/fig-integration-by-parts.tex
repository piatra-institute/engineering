% Figure: integration by parts as the area between two curves in the
% (u, v) plane. The rectangle [0, u(b)] x [0, v(b)] decomposes into
% the integral of v du (under the curve) and the integral of u dv
% (to the left of the curve).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\small]

% Axes
\draw[->, thick] (0,0) -- (7.2,0) node[below right] {$u$};
\draw[->, thick] (0,0) -- (0,5.2) node[left] {$v$};

% The curve v(u) = 4 - 4 exp(-u/2) on u in [0, 3], a strictly
% increasing concave curve from (0,0) to (3, 4(1-e^{-1.5})) ~ (3, 3.11).

% Shade region under the curve (the int v du part)
\fill[engred!15, opacity=0.7]
  (0,0)
  \foreach \u in {0,0.1,...,3.0}{
    -- ({\u}, {4 - 4*exp(-\u/2)})
  }
  -- (3,0) -- cycle;

% Shade region to the left of the curve (the int u dv part)
\fill[engblue!15, opacity=0.7]
  (0,0)
  \foreach \u in {0,0.1,...,3.0}{
    -- ({\u}, {4 - 4*exp(-\u/2)})
  }
  -- (0, 3.11) -- cycle;

% The curve itself, drawn on top
\draw[very thick, engblack]
  (0,0)
  \foreach \u in {0,0.1,...,3.0}{
    -- ({\u}, {4 - 4*exp(-\u/2)})
  };

% Outer dashed rectangle
\draw[engblack, dashed] (3, 0) -- (3, 3.11) -- (0, 3.11);

% Corner annotations
\node[anchor=west] at (3.05, 3.30) {$(u(b), v(b))$};
\node[anchor=north] at (3, 0) {$u(b)$};
\node[anchor=east] at (0, 3.11) {$v(b)$};

% Region labels
\node[engred, anchor=center] at (1.9, 0.7)
  {$\displaystyle\int_{0}^{u(b)} v \dd u$};
\node[engblue, anchor=center] at (0.85, 2.4)
  {$\displaystyle\int_{0}^{v(b)} u \dd v$};

\end{tikzpicture}
\caption[Integration by parts as area decomposition]{Integration by
parts read as area decomposition. The curve $v = v(u)$ between the
origin and the point $(u(b), v(b))$ divides the rectangle
$[0, u(b)] \times [0, v(b)]$ into two regions. The region under
the curve has area $\int_{0}^{u(b)} v \dd u$; the region to the
left of the curve has area $\int_{0}^{v(b)} u \dd v$. The two
regions together fill the rectangle of area $u(b)\,v(b)$, which
gives the geometric content of $\int u \dd v = u v - \int v \dd u$
with the boundary contribution from the lower-left corner taken to
vanish. The technique trades one integral for another and is
useful when the second integral is closer to the catalogue of
elementary antiderivatives than the first.}
\label{fig:vol02:ch06:by-parts-geometry}
\end{figure}
