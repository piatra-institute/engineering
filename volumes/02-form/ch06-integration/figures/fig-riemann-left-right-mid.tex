% Figure: left, right, and midpoint Riemann sums for the same integrand
% f(x) = 1 + x^2/2 on [0, 2] with n = 4, showing the systematic over-
% and under-estimate of left/right sums on a monotone integrand and the
% near-cancellation of the midpoint rule.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}

\begin{axis}[
  name=left,
  width=5.2cm, height=4.6cm,
  xlabel={$x$}, ylabel={$f(x)$},
  xmin=0, xmax=2, ymin=0, ymax=3.5,
  domain=0:2, samples=120,
  axis lines=left, font=\footnotesize,
  title={Left endpoints},
]
\foreach \i in {0,1,2,3}{
  \pgfmathsetmacro{\xl}{\i*0.5}
  \pgfmathsetmacro{\xr}{(\i+1)*0.5}
  \pgfmathsetmacro{\yh}{1+\xl*\xl/2}
  \addplot[draw=engblue, fill=engblue!12]
    coordinates {(\xl,0) (\xr,0) (\xr,\yh) (\xl,\yh)} \closedcycle;
}
\addplot[engblack, very thick] {1+x*x/2};
\end{axis}

\begin{axis}[
  at={(left.east)}, anchor=west, xshift=10pt,
  name=mid,
  width=5.2cm, height=4.6cm,
  xlabel={$x$}, ylabel={},
  xmin=0, xmax=2, ymin=0, ymax=3.5,
  domain=0:2, samples=120,
  axis lines=left, font=\footnotesize,
  title={Midpoints},
]
\foreach \i in {0,1,2,3}{
  \pgfmathsetmacro{\xl}{\i*0.5}
  \pgfmathsetmacro{\xr}{(\i+1)*0.5}
  \pgfmathsetmacro{\xm}{(\xl+\xr)/2}
  \pgfmathsetmacro{\yh}{1+\xm*\xm/2}
  \addplot[draw=engamber, fill=engamber!12]
    coordinates {(\xl,0) (\xr,0) (\xr,\yh) (\xl,\yh)} \closedcycle;
}
\addplot[engblack, very thick] {1+x*x/2};
\end{axis}

\begin{axis}[
  at={(mid.east)}, anchor=west, xshift=10pt,
  width=5.2cm, height=4.6cm,
  xlabel={$x$}, ylabel={},
  xmin=0, xmax=2, ymin=0, ymax=3.5,
  domain=0:2, samples=120,
  axis lines=left, font=\footnotesize,
  title={Right endpoints},
]
\foreach \i in {0,1,2,3}{
  \pgfmathsetmacro{\xl}{\i*0.5}
  \pgfmathsetmacro{\xr}{(\i+1)*0.5}
  \pgfmathsetmacro{\yh}{1+\xr*\xr/2}
  \addplot[draw=engred, fill=engred!12]
    coordinates {(\xl,0) (\xr,0) (\xr,\yh) (\xl,\yh)} \closedcycle;
}
\addplot[engblack, very thick] {1+x*x/2};
\end{axis}

\end{tikzpicture}
\caption[Left, midpoint, and right Riemann sums]{Left-endpoint
(blue), midpoint (green), and right-endpoint (red) Riemann sums for
$f(x) = 1 + x^{2}/2$ on $[0, 2]$ with $n = 4$ subintervals. On an
increasing integrand the left sum systematically under-estimates and
the right sum systematically over-estimates; the two errors are equal
and opposite to leading order, which is why their average (the
trapezoidal rule) is one order more accurate than either. The
midpoint rule's over- and under-shoots on each panel nearly cancel,
giving it the same $h^{2}$ order as the trapezoidal rule but with
roughly half the error constant.}
\label{fig:vol02:ch06:riemann-lrm}
\end{figure}
