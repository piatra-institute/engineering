% Water-filling picture: four channels with floor heights 1/g_i (inverse
% channel gain, the "ground level"), a common water level nu, and the
% allocated power p_i = nu - 1/g_i above each floor.
% Floors (bar bottoms): heights h_i. Water level nu.
% h = {0.6, 1.9, 1.1, 2.6}, nu = 2.3.
\def\waterlevel{2.3}

% Channel bar positions and floor heights.
\foreach \i/\h in {1/0.6, 2/1.9, 3/1.1, 4/2.6} {
  \pgfmathsetmacro{\xa}{1.4*\i - 0.55}
  \pgfmathsetmacro{\xb}{1.4*\i + 0.55}
  % floor block (the 1/g_i "ground"), gray.
  \fill[enggray!45] (\xa, 0) rectangle (\xb, \h);
  \draw[enggray!80, thin] (\xa, 0) rectangle (\xb, \h);
  % allocated power (water) above the floor up to the level, blue,
  % only where floor is below the level.
  \pgfmathsetmacro{\top}{max(\h,\waterlevel)}
  \ifdim \h pt < \waterlevel pt
    \fill[engblue!35] (\xa, \h) rectangle (\xb, \waterlevel);
    \draw[engblue!80, thin] (\xa, \h) rectangle (\xb, \waterlevel);
  \fi
  \node[anchor=north, font=\scriptsize] at ({1.4*\i}, -0.15) {ch \i};
  \node[enggray, anchor=north, font=\scriptsize] at ({1.4*\i}, \h)
    {$\tfrac{1}{g_{\i}}$};
}

% Water level line across all channels.
\draw[engred, very thick, dashed] (0.6, \waterlevel) -- (6.4, \waterlevel);
\node[engred, anchor=west, font=\scriptsize] at (6.5, \waterlevel)
  {level $\nu$};

% Axis.

\node[anchor=north, font=\scriptsize] at (3.5, -0.75)

\end{tikzpicture}
\caption[Water-filling power allocation]{The water-filling solution of
the power-allocation KKT problem. Each channel $i$ is a container whose
floor sits at height $1/g_{i}$, the inverse of its gain: a strong
channel (large $g_{i}$) has a low floor and a weak channel a high one.
Pouring a fixed budget of power in until the surface reaches a common
level $\nu$ fills the low-floored channels most and leaves the very
weakest channel (its floor above the water line, channel 4 here)
allocated nothing at all. The allocated power is the water depth
$p_{i}^{*} = \max(\nu - 1/g_{i},\, 0)$, and the common level $\nu$ is
the single Lagrange multiplier on the total-power constraint, fixed by
requiring the depths to sum to the budget $P$. The picture is the
literal reading of the KKT conditions: the multiplier is the water
level, complementary slackness is the fact that a channel is either wet
(active, $p_{i} > 0$) or bone dry (its nonnegativity bound active,
$p_{i} = 0$), and the shadow price of one more watt of budget is the
marginal rate at which raising $\nu$ improves the objective.}
