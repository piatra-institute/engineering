% Two intersecting curves whose crossing is the root of a 2x2 nonlinear
% system, with the vector-Newton iterates spiralling in toward it.
\draw[->, thick] (-0.3,0) -- (5.3,0) node[right] {$x$};
\draw[->, thick] (0,-0.3) -- (0,4.3) node[above] {$y$};

% Curve 1: x^2 + y^2 = 16 (circle arc) drawn schematically
\draw[engblue, thick] plot[smooth, domain=0.2:3.98, samples=40]
  ({\x}, {sqrt(max(0,16-\x*\x))});
\node[engblue, anchor=west, font=\scriptsize] at (0.9,3.85)
  {$x^{2}+y^{2}=16$};

% Curve 2: y = x^2/3 (parabola) intersecting the arc
\draw[engamber, thick] plot[smooth, domain=0:3.4, samples=40]
  ({\x}, {\x*\x/3});
\node[engamber!70!black, anchor=north west, font=\scriptsize] at (3.1,3.4)
  {$y = x^{2}/3$};

% Intersection (root) near (2.7, 2.43)
\fill[engred] (2.70,2.43) circle (2.4pt);
\node[engred, anchor=west, font=\scriptsize] at (2.85,2.30) {root $\vect{x}^{*}$};

% Newton iterates converging to the root
\fill[enggray] (1.20,0.60) circle (1.6pt);

\end{tikzpicture}
\caption[Vector Newton iteration for a $2\times2$ nonlinear system]{The
solution of a nonlinear system is the intersection of the curves each
equation defines. Vector Newton linearises both curves at the current iterate,
solves the resulting $2\times2$ linear system for the step, and lands nearer
the crossing. From a coarse start the iterates approach the root and the error
squares each step once the estimate enters the quadratic-convergence
neighbourhood, the same rate the scalar method achieves, now governed by the
Jacobian in place of the scalar derivative. Source: authors' schematic; iterate
positions illustrative.}
