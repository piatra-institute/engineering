% Two 2-D positioning geometries: receiver at origin, range circles from
% satellites. Well-spread satellites give a small intersection region
% (low GDOP); clustered satellites give a long thin region (high GDOP).

% ----- Left: good geometry -----
\begin{scope}[shift={(0,0)}]
  \fill[engblue] (0,0) circle (2.2pt);
  \node[engblue, anchor=north, font=\scriptsize] at (0,-0.15) {receiver};
  % satellite directions well spread
  \foreach \a in {60,160,290} {
    \draw[enggray, thick] (0,0) -- (\a:2.6);
    \fill[engamber] (\a:2.6) circle (2pt);
  }
  % small uncertainty region (near-circular) around receiver
  \draw[engred, thick, fill=engred!12] (0,0) circle (0.35);
  \node[align=center, font=\scriptsize, anchor=north] at (0,-1.6)
    {well-spread lines of sight\\small, near-round error region\\low GDOP};
\end{scope}

% ----- Right: bad geometry -----
\begin{scope}[shift={(6.5,0)}]
  \fill[engblue] (0,0) circle (2.2pt);
  \node[engblue, anchor=north, font=\scriptsize] at (0,-0.15) {receiver};
  % satellite directions clustered
  \foreach \a in {70,90,110} {
    \draw[enggray, thick] (0,0) -- (\a:2.6);
    \fill[engamber] (\a:2.6) circle (2pt);
  }
  % elongated uncertainty region along the poorly-constrained direction
  \draw[engred, thick, fill=engred!12, rotate=0]
    (0,0) ellipse (0.28 and 1.15);
  \node[align=center, font=\scriptsize, anchor=north] at (0,-1.6)
    {clustered lines of sight\\long, thin error region\\high GDOP};
\end{scope}

\caption[Geometric dilution of precision in positioning]{The same
pseudorange measurement noise produces very different position
uncertainty depending on the satellite geometry, the effect the
positioning literature calls geometric dilution of precision (GDOP). The
receiver solves a nonlinear least-squares problem whose design matrix has
rows along the lines of sight to the satellites. When the lines of sight
are well spread (left) the matrix is well conditioned and the position
error region is small and near-round. When the visible satellites cluster
in one part of the sky (right) two rows of the design matrix are nearly
parallel, the matrix is ill conditioned, and the position error stretches
into a long thin band along the poorly-constrained direction. GDOP is the
condition number of section 16.2 wearing its field name: a property of the
problem geometry that no solver can beat. Source: authors' schematic.}
\label{fig:vol02ch16:gps-gdop}
\end{figure}
