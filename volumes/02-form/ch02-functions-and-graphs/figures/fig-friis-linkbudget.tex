% Figure: a radio link budget on log-log axes.
% Received power (dBm) versus range (m). Free-space loss gives a power
% law P_r ~ r^{-2}, a straight line of slope -2 on log-log axes (here
% plotted as dBm vs log-range, so slope -20 dB per decade). The receiver
% sensitivity floor and the resulting maximum range read off directly.
\begin{figure}[htbp]
\centering
\resizebox{0.82\linewidth}{!}{%
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.4cm,
  xmode=log, log basis x=10,
  xlabel={range $r$ (m)},
  ylabel={received power $P_r$ (dBm)},
  xmin=10, xmax=100000,
  ymin=-120, ymax=-30,
  ytick={-120, -100, -80, -60, -40},
  axis lines=left,
  tick align=outside,
  grid=both,
  major grid style={draw=engblack!12},
  minor grid style={draw=engblack!6},
  every axis plot/.append style={very thick},
  label style={font=\footnotesize},
  tick label style={font=\footnotesize},
  legend style={font=\footnotesize, at={(0.98,0.98)}, anchor=north east,
    draw=engblack!30},
  clip=false,
]
  % Free-space path: P_r(dBm) = C - 20*log10(r). Choose C so P_r = -40 dBm
  % at r = 10 m: -40 = C - 20*1 -> C = -20.
  \addplot[engblue, domain=10:100000, samples=60] {-20 - 20*log10(x)};
  \addlegendentry{$P_r \propto r^{-2}$ (free space)}
  % receiver sensitivity floor
  \addplot[engred, dashed, domain=10:100000, samples=2] {-100};
  \addlegendentry{receiver sensitivity ($-100$ dBm)}
  % maximum range: -100 = -20 - 20 log10(r) -> log10 r = 4 -> r = 10000 m
  \addplot[engamber, only marks, mark=*, mark size=1.8pt]
    coordinates {(10000, -100)};
  \draw[engamber, thick, dotted]
    (axis cs:10000, -120) -- (axis cs:10000, -100);
  \node[font=\footnotesize, text=engamber, anchor=south west]
    at (axis cs:10500, -99) {max range $\approx 10\,\si{\kilo\meter}$};
  \node[font=\footnotesize, text=engblue, anchor=south west]
    at (axis cs:40, -55) {slope $-20\,\si{\deci\bel}$/decade};
\end{axis}
\end{tikzpicture}%
}%
\caption[A radio link budget on log axes]{A free-space radio link
budget. The received power follows the inverse-square law
$P_r \propto r^{-2}$, which on a plot of power in decibels against
log range is a straight line of slope $-20\,\si{\deci\bel}$ per decade,
the power-law exponent $-2$ read through the $10\log_{10}$ power
convention. The receiver sensitivity is a horizontal floor: the
smallest power the receiver can still demodulate. The two lines cross
at the maximum usable range, and reading that crossing off the plot
answers the link-design question directly. Every term in the budget is
an additive decibel contribution, so antenna gains and cable losses
slide the sloped line up or down without changing its slope, which is
why link budgets are worked entirely in decibels.}
\label{fig:vol02:ch02:friis-linkbudget}
\end{figure}
