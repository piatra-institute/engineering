% Figure: two Lissajous figures, the parametric curves traced when one
% sinusoid drives the horizontal axis and another the vertical. A 1:1
% ratio with a phase offset gives an ellipse; a 3:2 ratio gives a closed
% multi-lobe curve. The shape reads off the frequency ratio and phase.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
% Panel (a): 1:1 ratio, phase pi/4 -> tilted ellipse.
\begin{axis}[
  name=plota,
  width=6.4cm, height=6.4cm,
  title={(a) ratio $1{:}1$, phase $\pi/4$},
  title style={font=\footnotesize},
  xlabel={$x = \sin(\omega t)$},
  ylabel={$y = \sin(\omega t + \pi/4)$},
  xmin=-1.25, xmax=1.25, ymin=-1.25, ymax=1.25,
  xtick={-1,0,1}, ytick={-1,0,1},
  axis lines=middle,
  every axis plot/.append style={very thick},
  label style={font=\scriptsize},
  tick label style={font=\scriptsize},
  clip=false,
]
  \addplot[engblue, domain=0:360, samples=240]
    ({sin(x)}, {sin(x + 45)});
\end{axis}
% Panel (b): 3:2 ratio -> three-by-two lobe closed curve.
\begin{axis}[
  name=plotb,
  at={($(plota.east)+(1.6cm,0)$)}, anchor=west,
  width=6.4cm, height=6.4cm,
  title={(b) ratio $3{:}2$, phase $0$},
  title style={font=\footnotesize},
  xlabel={$x = \sin(3\omega t)$},
  ylabel={$y = \sin(2\omega t)$},
  xmin=-1.25, xmax=1.25, ymin=-1.25, ymax=1.25,
  xtick={-1,0,1}, ytick={-1,0,1},
  axis lines=middle,
  every axis plot/.append style={very thick},
  label style={font=\scriptsize},
  tick label style={font=\scriptsize},
  clip=false,
]
  \addplot[engred, domain=0:360, samples=360]
    ({sin(3*x)}, {sin(2*x)});
\end{axis}
\end{tikzpicture}
\caption[Lissajous figures read frequency ratio and phase]{Two sinusoids
of the same frequency, one on each axis, trace a closed ellipse whose
tilt and openness read the phase between them; panel (a) shows the
$\pi/4$ case, an ellipse tilted along the diagonal. When the two
frequencies stand in a small integer ratio the parametric curve closes
into a multi-lobe pattern whose lobe counts read the ratio directly:
panel (b) shows a $3{:}2$ Lissajous figure, with three lobes across and
two up. Counting the tangencies on each edge recovers the frequency
ratio, which is how an oscilloscope in $x$-$y$ mode was long used to
compare an unknown frequency against a reference before digital counters
made the measurement direct.}
\label{fig:vol02:ch02:lissajous}
\end{figure}
