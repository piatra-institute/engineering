% Figure: the beat pattern from two sinusoids of nearly equal frequency.
% Sum of cos(2*pi*10 t) and cos(2*pi*11 t) shows a fast carrier at the
% mean frequency inside a slow envelope at half the difference frequency.
\begin{figure}[htbp]
\centering
\resizebox{0.9\linewidth}{!}{%
\begin{tikzpicture}[scale=1.0,
  ax/.style={->, draw=engblack, thick},
  tick/.style={draw=engblack!60, thin},
  carrier/.style={draw=engblue, thick},
  env/.style={draw=engred, very thick, dashed},
  ann/.style={font=\footnotesize, align=center},
]
  % axes: horizontal 0..6 s maps to 0..10.8 cm (1.8 cm/s); vertical +-2.5.
  \draw[ax] (0, 0) -- (11.0, 0) node[below right, ann] {$t$ (s)};
  \draw[ax] (0, -2.5) -- (0, 2.6) node[left, ann] {$y$ (arb.)};
  \foreach \x in {1, 2, 3, 4, 5, 6} {
    \draw[tick] (\x*1.8, -0.06) -- (\x*1.8, 0.06);
    \node[ann, below, yshift=-2pt] at (\x*1.8, 0) {\x};
  }
  \foreach \y in {-2, 2} {
    \draw[tick] (-0.06, \y) -- (0.06, \y);
    \node[ann, left, xshift=-2pt] at (0, \y) {\y};
  }
  % f1 = 4.5 Hz, f2 = 5.5 Hz: mean 5 Hz, difference 1 Hz.
  % Envelope at half-difference 0.5 Hz = 180 deg/s; carrier at 5 Hz = 1800 deg/s.
  % Domain kept to 6 s so the largest angle argument (1800*6 = 10800 deg)
  % stays inside pgf's fixed-point range.
  \draw[env, domain=0:6, samples=180, smooth]
    plot (\x*1.8, {2*cos(180*\x)});
  \draw[env, domain=0:6, samples=180, smooth]
    plot (\x*1.8, {-2*cos(180*\x)});
  % Carrier * envelope: 2 cos(180 t) cos(1800 t). No smooth: dense line
  % segments avoid the large Bezier control points that overflow pgf math.
  \draw[carrier, domain=0:6, samples=700]
    plot (\x*1.8, {2*cos(180*\x)*cos(1800*\x)});
  \node[ann, text=engred, anchor=south west] at (0.15, 2.15)
    {slow envelope\\at $\tfrac{1}{2}(f_2 - f_1)$};
  \node[ann, text=engblue, anchor=north] at (5.4, -2.15)
    {fast carrier at mean frequency $\tfrac{1}{2}(f_1 + f_2)$};
  % beat period marker between two envelope minima (t=0.5 and t=1.5)
  \draw[<->, draw=engamber, thick] (0.5*1.8, -2.35) -- (1.5*1.8, -2.35);
  \node[ann, text=engamber, below] at (1.0*1.8, -2.35) {beat period $T_b = 1/(f_2 - f_1)$};
\end{tikzpicture}%
}%
\caption[The beat pattern of two nearby frequencies]{Two sinusoids of
nearly equal frequency, here $f_1 = 4.5\,\si{\hertz}$ and $f_2 =
5.5\,\si{\hertz}$ of equal amplitude, sum to a single fast oscillation
at the mean frequency $\tfrac{1}{2}(f_1 + f_2)$ multiplied by a slow
envelope at half the difference frequency. The product identity
$\cos(2\pi f_1 t) + \cos(2\pi f_2 t) = 2\cos\!\bigl(\pi(f_2 -
f_1)t\bigr)\cos\!\bigl(\pi(f_1 + f_2)t\bigr)$ turns a sum of two tones
into a product of two, and the ear hears the envelope as a loudness
that rises and falls at the beat frequency $f_2 - f_1$. A piano tuner
counts beats per second to read the frequency mismatch between a note
and a reference directly, without measuring either frequency.}
\label{fig:vol02:ch02:beat-envelope}
\end{figure}
