% Figure: the photographic exposure value (EV) as a base-2 logarithmic
% scale. Panel-free single axis: shutter time on a log2 axis against
% exposure value, showing that each stop is a factor of two in light and
% one unit on the EV scale, so a multiplicative light ratio becomes an
% additive stop count.
\begin{figure}[htbp]
\centering
\resizebox{0.82\linewidth}{!}{%
\begin{tikzpicture}
\begin{semilogyaxis}[
  width=12cm, height=7.2cm,
  log basis y=2,
  xlabel={exposure value $\mathrm{EV}$ (stops from reference)},
  ylabel={shutter time $t$ (\si{\second}), $\log_2$ axis},
  xmin=-1, xmax=9,
  ymin=0.002, ymax=1.4,
  xtick={0,1,2,3,4,5,6,7,8},
  ytick={1, 0.5, 0.25, 0.125, 0.0625, 0.03125, 0.015625, 0.0078125, 0.00390625},
  yticklabels={$1$, $1/2$, $1/4$, $1/8$, $1/16$, $1/32$, $1/64$, $1/128$, $1/256$},
  axis lines=left,
  tick align=outside,
  every axis plot/.append style={very thick},
  label style={font=\footnotesize},
  tick label style={font=\footnotesize},
  grid=both,
  major grid style={engblack!12},
  minor grid style={engblack!5},
  clip=false,
]
  % Each stop halves the shutter time: t = t0 * 2^{-EV}, t0 = 1 s at EV = 0.
  \addplot[engblue, domain=0:8, samples=9] {2^(-x)};
  \addplot[engred, only marks, mark=*, mark size=1.5pt, domain=0:8, samples=9]
    {2^(-x)};
  % Annotate a single stop as a factor of two and one EV unit.
  \draw[<->, engamber, thick] (axis cs:3, 0.125) -- (axis cs:4, 0.125);
  \node[font=\footnotesize, text=engamber, anchor=south west]
    at (axis cs:3.02, 0.128) {one stop};
  \draw[<->, engamber, thick] (axis cs:4, 0.125) -- (axis cs:4, 0.0625);
  \node[font=\footnotesize, text=engamber, anchor=west]
    at (axis cs:4.05, 0.088) {$\times 1/2$ light};
  \node[font=\footnotesize, text=engblack!70, anchor=west]
    at (axis cs:4.6, 0.4) {equal spacing $=$ equal ratio};
\end{semilogyaxis}
\end{tikzpicture}%
}%
\caption[Exposure value as a base-2 log scale]{The photographic
exposure value counts stops of light on a base-2 logarithmic scale, so
each unit step in $\mathrm{EV}$ is a factor of two in the light reaching
the sensor and each halving of shutter time is one stop. Plotted with
shutter time on a $\log_2$ axis, the reference sequence
$1, 1/2, 1/4, \dots$ falls on a straight line whose equal vertical
spacing marks equal ratios, the signature of a logarithmic axis. A
photographer who wants two stops less light shortens the exposure by a
factor of four, computed by adding two on the stop scale rather than
multiplying by four on the light scale, which is the same
addition-for-multiplication trade the decibel makes for power ratios.}
\label{fig:vol02:ch02:exposure-stops}
\end{figure}
