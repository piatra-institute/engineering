\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Subdifferential of f(x) = |x| at the origin: the set of all lines
% through (0,0) that stay on or below the graph, slopes in [-1, 1].
% Plot scale: x by 1.6, y by 1.6. Domain x in [-2.4, 2.4].

% Axes
\draw[->, thick] (-4.2, 0) -- (4.4, 0) node[right] {$x$};
\draw[->, thick] (0, -0.4) -- (0, 4.4) node[above] {$y$};

% f(x) = |x|: two rays, plotx = 1.6*x, ploty = 1.6*|x|
\draw[very thick, engblue] (-1.6*2.4, 1.6*2.4) -- (0, 0)
  -- (1.6*2.4, 1.6*2.4);
\node[engblue, anchor=south east, font=\scriptsize] at (-1.6*1.6, 1.6*1.7)
  {$f(x) = |x|$};

% Supporting lines through origin with slopes in [-1, 1].
% Slope s: ploty = (s)*plotx*(1) but must scale: y = s*x -> in plot units
% ploty = 1.6*(s * (plotx/1.6)) = s*plotx. So a line of slope s passes as
% ploty = s*plotx and stays below |x| for |s| <= 1. Draw several.
\foreach \s in {-1, -0.6, -0.2, 0.2, 0.6, 1} {
  \draw[thin, engamber] (-1.6*2.2, {\s*(-1.6*2.2)}) -- (1.6*2.2, {\s*(1.6*2.2)});
}
% the two boundary supporting lines emphasised
\draw[thick, engred, dashed] (-1.6*2.4, {-1*(-1.6*2.4)}) -- (1.6*2.4, {-1*(1.6*2.4)});
\draw[thick, engred, dashed] (-1.6*2.4, {1*(-1.6*2.4)}) -- (1.6*2.4, {1*(1.6*2.4)});

% kink point
\fill[engblack] (0, 0) circle (2pt);
\node[engblack, anchor=north west, font=\scriptsize] at (0.15, -0.15)
  {kink at $x = 0$};

% annotate the slope interval
\node[engred, anchor=west, font=\scriptsize] at (0.6, 3.9)
  {supporting slopes: $s \in [-1, 1]$};
\node[engamber, anchor=west, font=\scriptsize] at (0.6, 3.4)
  {$\partial f(0) = [-1, 1]$};

% mark that s = 0 (horizontal) is in the set -> minimum
\draw[thick, englime] (-1.6*2.2, 0) -- (1.6*2.2, 0);
\node[engblue, anchor=north east, font=\scriptsize] at (-0.2, -0.15)
  {$0 \in \partial f(0)$};

\end{tikzpicture}
\caption[The subdifferential of $|x|$ at its kink]{At the corner of
$f(x) = |x|$ the ordinary derivative does not exist, because the left
slope is $-1$ and the right slope is $+1$. The subdifferential replaces
the single slope by the whole set of slopes whose line through the
corner stays on or below the graph. Every line of slope $s$ with
$|s| \le 1$ (amber) supports the graph from below; the two extreme
supports (red dashed) have slopes $\pm 1$, the one-sided derivatives.
The set is the closed interval $\partial f(0) = [-1, 1]$. The
horizontal support of slope zero (green) lies in the set, so
$0 \in \partial f(0)$, which is the optimality condition that locates
the minimum at the corner the ordinary condition $f'(x) = 0$ could not
reach.}
\label{fig:vol02:ch05:subgradient}
\end{figure}
