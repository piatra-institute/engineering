\draw[->, thick] (0, -0.3) -- (0, 4.6) node[above] {};

% N(t) = 1/(1+exp(-1.2 t)), t = plotx/0.7 - 5
\draw[very thick, engblue] plot[domain=0:7, samples=120]

  {$N(t)$};

\draw[thin, enggray, dashed] (0, {2.6 + 1.6}) -- (7.4, {2.6 + 1.6});
\node[enggray, anchor=east, font=\scriptsize] at (-0.05, {2.6 + 1.6}) {$K$};
\draw[thin, enggray, dotted] (0, {2.6 + 0.8}) -- (7.4, {2.6 + 0.8});
\node[enggray, anchor=east, font=\scriptsize] at (-0.05, {2.6 + 0.8}) {$K/2$};

\caption[Logistic growth and its rate, with the inflection at half
capacity]{The logistic population $N(t)$ (blue, upper) rises from near
zero and saturates at the carrying capacity $K$. Its rate of change
$\dv{N}{t} = rN(1 - N/K)$ (orange, lower) is small at both ends and
peaks in the middle. The peak of the rate sits exactly at the
inflection point of $N(t)$, where the population equals half the
carrying capacity, $N = K/2$, and the maximum absolute growth rate is
$rK/4$. The second derivative of $N$ vanishes there and changes sign,
which is why the population curve switches from accelerating to
decelerating growth at the same instant the rate curve turns over. A
fermentation harvested for maximum throughput is drawn down at
$N = K/2$.}
