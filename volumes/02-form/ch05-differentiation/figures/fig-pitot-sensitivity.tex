\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% Airspeed from dynamic pressure: v = sqrt(2 q / rho), so v grows as the
% square root of q, and the sensitivity dv/dq = 1/sqrt(2 rho q) diverges
% as q -> 0. Normalise rho and constants so v = sqrt(q) and
% dv/dq = 1/(2 sqrt(q)). Domain q in [0.06, 4]; at q = 0.06
% dv/dq = 1/(2*0.245) approx 2.04 (finite, clamped by the lower limit).

% Axes
\draw[->, thick] (-0.3, 0) -- (5.0, 0)
  node[right, font=\scriptsize] {dynamic pressure $q$};
\draw[->, thick] (0, -0.3) -- (0, 4.0) node[above] {value};

% Airspeed v = sqrt(q), plotx = q, plotic scale x by 1.15, y by 1.6
% Use v(q) = sqrt(q): plot (1.15*q, 1.6*sqrt(q))
\draw[very thick, engblue] plot[domain=0.02:4, samples=120]
  ({1.15*\x}, {1.6*sqrt(\x)});
\node[engblue, anchor=south, font=\scriptsize] at (3.6, 2.7)
  {airspeed $v = \sqrt{2q/\rho}$};

% Sensitivity dv/dq = 1/(2 sqrt(q)); clamp near q = 0.06 to avoid blow-up
% plot (1.15*q, 1.6 * 1/(2 sqrt(q))) for q in [0.06, 4]
\draw[thick, engred, dashed] plot[domain=0.06:4, samples=120]
  ({1.15*\x}, {1.6*(1/(2*sqrt(\x)))});
\node[engred, anchor=west, font=\scriptsize] at (2.0, 1.0)
  {sensitivity $\dv{v}{q} = \dfrac{1}{2\sqrt{2\rho q}}$};

% Danger band at low q where sensitivity diverges
\fill[engred!12] (0, 0) rectangle (0.85, 3.7);
\node[engred, anchor=south, font=\tiny, rotate=90] at (0.42, 1.6)
  {low speed: sensitivity diverges};

% mark a cruise operating point q = 2.25 -> v = 1.5
\fill[engblack] ({1.15*2.25}, {1.6*1.5}) circle (1.7pt);
\draw[thin, enggray, dotted] ({1.15*2.25}, 0) -- ({1.15*2.25}, {1.6*1.5});
\node[anchor=north, font=\scriptsize] at ({1.15*2.25}, -0.05) {cruise};

\end{tikzpicture}
\caption[Airspeed and its sensitivity to dynamic pressure]{Airspeed
inferred from a pitot-static dynamic pressure follows
$v = \sqrt{2q/\rho}$ (blue), and its sensitivity to the pressure reading,
$\dv{v}{q} = 1/(2\sqrt{2\rho q})$ (red), grows without bound as the
dynamic pressure falls toward zero. At cruise the curve is flat and a
small pressure error maps to a small speed error; near stall the same
pressure error maps to a large speed error, which is why a pitot airspeed
reading is least trustworthy exactly at the low speeds where a stall
margin matters most. The shaded band marks the low-pressure regime the
derivative flags as hazardous.}
\label{fig:vol02ch05:pitot-sensitivity}
\end{figure}
