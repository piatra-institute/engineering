\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% A hanging cable follows y(x) = a cosh(x/a); its slope is
% y'(x) = sinh(x/a), the tangent of the angle the cable makes with the
% horizontal. Take a = 1 for the picture. The chain drops to its
% lowest point at x = 0 where the slope is zero and rises symmetrically.
% Domain x in [-1.6, 1.6]; cosh(1.6) approx 2.577, sinh(1.6) approx
% 2.376, both finite and small so pgf math is safe.

% Axes
\draw[->, thick] (-2.0, 0) -- (2.2, 0) node[right] {$x/a$};
\draw[->, thick] (0, -0.4) -- (0, 3.4) node[above] {height and slope};

% The cable y = cosh(x), plotted directly (a = 1)
\draw[very thick, engblue] plot[domain=-1.6:1.6, samples=80]
  (\x, {cosh(\x)});
\node[engblue, anchor=south east, font=\scriptsize] at (-0.7, 2.55)
  {cable $y = a\cosh(x/a)$};

% The slope y' = sinh(x), an odd function through the origin
\draw[thick, engred, dashed] plot[domain=-1.6:1.6, samples=80]
  (\x, {sinh(\x)});
\node[engred, anchor=north west, font=\scriptsize] at (0.9, 1.05)
  {slope $y' = \sinh(x/a)$};

% Lowest point of the cable: slope zero
\fill[engblack] (0, 1) circle (1.8pt);
\node[anchor=west, font=\scriptsize] at (0.12, 1.18) {lowest point};
\draw[thin, enggray, dotted] (-1.6, 1) -- (0, 1);

% Mark a support point at x = 1.6 and its tangent slope
\fill[engblue] (1.6, {cosh(1.6)}) circle (1.6pt);
\draw[thin, enggray, dotted] (1.6, 0) -- (1.6, {cosh(1.6)});
\node[anchor=north, font=\scriptsize] at (1.6, -0.05) {support};

% Slope value at the support read off the red curve
\fill[engred] (1.6, {sinh(1.6)}) circle (1.4pt);
\draw[<-, enggray, very thin] (1.62, {sinh(1.6)}) -- (2.0, {sinh(1.6)});
\node[enggray, anchor=west, font=\tiny] at (1.95, {sinh(1.6)+0.28})
  {tension angle};

\end{tikzpicture}
\caption[The hanging cable and its slope]{A cable hanging under its own
weight takes the shape $y = a\cosh(x/a)$ (blue), and its slope
$y' = \sinh(x/a)$ (red) is the odd function that measures the angle the
cable makes with the horizontal at each point. The slope is zero at the
lowest point, where the cable runs flat, and grows toward the supports,
where the cable is steepest and the tension has its largest vertical
component. The horizontal tension in the cable is constant along its
length; the slope read here converts that constant horizontal pull into
the varying total tension the anchors must hold.}
\label{fig:vol02ch05:catenary-slope}
\end{figure}
