% the central secant against the true tangent. The central secant lies
% closer to the tangent because its bias cancels to second order.
% Plot scale: x by 2.2 (x measured from origin), y by 1.0.

% Axes

\draw[very thick, engblue] plot[domain=0.2:3.1, samples=90]
  ({2.2*\x}, {0.5*\x*\x + 1});

% Working point x0 = 1.5: f = 2.125; plot (3.3, 2.125)
\fill[engblack] (3.3, 2.125) circle (1.8pt);
\node[engblack, anchor=north west, font=\scriptsize] at (3.35, 2.02)
  {$(x_{0}, f(x_{0}))$};

% Points at x0-h and x0+h with h=1: x=0.5 f=1.125; x=2.5 f=4.125
\fill[enggray] (1.1, 1.125) circle (1.4pt);   % x0 - h
\fill[enggray] (5.5, 4.125) circle (1.4pt);   % x0 + h

% True tangent at x0=1.5: slope f'=1.5. In plot units slope = 1.5/2.2.
% line through (3.3, 2.125): y = 2.125 + (1.5/2.2)*(X - 3.3)
\draw[thick, engamber, dashdotted]
  (0.6, {2.125 + (1.5/2.2)*(0.6 - 3.3)})
  -- (6.6, {2.125 + (1.5/2.2)*(6.6 - 3.3)});
\node[engamber, anchor=south, font=\scriptsize] at (6.2, 4.55)
  {tangent, slope $f'(x_{0})$};

% Forward secant through (3.3,2.125) and (5.5,4.125): plot slope
% = (4.125-2.125)/(5.5-3.3) = 2.0/2.2 = 0.909. True derivative estimate
% is (f(x0+h)-f(x0))/h = 2.0, steeper than tangent 1.5.
\draw[thick, engred, dashed]
  (2.5, {2.125 + (2.0/2.2)*(2.5 - 3.3)})
  -- (6.2, {2.125 + (2.0/2.2)*(6.2 - 3.3)});
\node[engred, anchor=north west, font=\scriptsize] at (5.6, 4.05)
  {forward secant};

\caption[Forward and central differences as secant slopes]{The two
one-step difference formulas read as secant slopes on
$f(x) = \tfrac{1}{2}x^{2} + 1$ at the working point $x_{0}$ with step
$h$. The forward secant (red dashed) joins $x_{0}$ to $x_{0}+h$ and
overshoots the true tangent slope (amber) by $\tfrac{1}{2}f''(x_{0})h$,
the first-order truncation bias. The central secant (green) joins
$x_{0}-h$ to $x_{0}+h$; its even-order bias cancels, so for a quadratic
it recovers the tangent slope exactly and in general it errs only at
order $h^{2}$. The picture is the geometric root of the accuracy
ordering derived in the truncation-error analysis: the symmetric
stencil is one order more accurate than the one-sided one.}
\label{fig:vol02:ch05:forward-central-error}
