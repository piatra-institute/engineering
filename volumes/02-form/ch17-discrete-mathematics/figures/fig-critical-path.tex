% Activity-on-node project DAG. Node label is the task letter; the
% number below each node is its duration. The critical path (longest
% path from Start to End) is drawn in red.
\node[crit] (S) at (0, 0)      {$S$};
\node[crit] (A) at (2.2, 1.4)  {$A$};
\node[task] (B) at (2.2, -1.4) {$B$};
\node[crit] (C) at (4.6, 1.4)  {$C$};
\node[task] (D) at (4.6, -1.4) {$D$};
\node[crit] (E) at (7.0, 0)    {$E$};
\node[crit] (T) at (9.2, 0)    {$T$};

% Durations under the nodes.
\node[font=\scriptsize, anchor=north] at (S.south) {$0$};
\node[font=\scriptsize, anchor=south] at (A.north) {$3$};

\node[font=\scriptsize, anchor=north] at (D.south) {$2$};
\node[font=\scriptsize, anchor=south] at (E.north) {$2$};

\draw[cedge] (S) -- (A);
\draw[edge]  (S) -- (B);

\draw[edge]  (A) -- (E);

\caption{An activity-on-node project graph. Each node is a task; the
number below it is the task duration. An edge $u \to v$ means task $u$
must finish before task $v$ starts. The earliest finish time of a task
is the longest path from Start $S$ to that task, computed by a single
sweep in topological order. The red path $S \to A \to C \to E \to T$
has total duration $3 + 4 + 2 = 9$, the project makespan, and every
task on it has zero slack: a delay to any red task delays the whole
project. The gray tasks $B$ and $D$ carry slack and can slip without
moving the finish date.}
