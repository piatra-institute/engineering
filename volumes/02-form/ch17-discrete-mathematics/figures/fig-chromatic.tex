% Deletion-contraction on one edge of a 4-cycle plus chord (the "house"
% is avoided; we use a triangle-with-pendant to keep the arithmetic short).
% Left: the graph G. Middle: G minus edge e. Right: G contract e.

% ---- G ----
\begin{scope}[xshift=0cm]
\node[vertex] (a) at (0, 1.4)   {$a$};
\node[vertex] (b) at (-0.9, 0)  {$b$};
\node[vertex] (c) at (0.9, 0)   {$c$};
\draw[edge] (a) -- (b);
\draw[cedge] (a) -- (c);
\draw[edge] (b) -- (c);
\node[lbl] at (0, -0.7) {$G$};
\node[font=\scriptsize, engred, anchor=west] at (0.55, 0.75) {$e$};
\end{scope}

% ---- G - e ----
\begin{scope}[xshift=4cm]
\node[vertex] (a2) at (0, 1.4)   {$a$};
\node[vertex] (b2) at (-0.9, 0)  {$b$};
\node[vertex] (c2) at (0.9, 0)   {$c$};
\draw[edge] (a2) -- (b2);
\draw[edge] (b2) -- (c2);
\node[lbl] at (0, -0.7) {$G - e$};
\end{scope}

% ---- G / e ----
\begin{scope}[xshift=8cm]
\node[vertex] (ac) at (0, 1.4)   {$ac$};
\node[vertex] (b3) at (0, 0)     {$b$};
\draw[edge] (ac) -- (b3);
\node[lbl] at (0, -0.7) {$G / e$};
\end{scope}
\end{tikzpicture}
\caption{Deletion-contraction on one edge $e = \set{a, c}$ of a
triangle. The chromatic polynomial obeys $P(G, k) = P(G - e, k) -
P(G / e, k)$. The middle graph $G - e$ is a path on three vertices with
$P = k(k-1)^{2}$; the right graph $G / e$ merges $a$ and $c$ into a
single vertex and is an edge on two vertices with $P = k(k-1)$.
Subtracting gives $P(G, k) = k(k-1)^{2} - k(k-1) = k(k-1)(k-2)$, the
triangle's chromatic polynomial, which is zero for $k < 3$ and confirms
$\chi = 3$.}
\label{fig:vol02:ch17:chromatic}
\end{figure}
