\begin{figure}[ht]
\centering
\begin{tikzpicture}[
  vertex/.style = {circle, draw=engblack, thick, fill=englime!25,
                   inner sep=0pt, minimum size=8mm, font=\small},
  term/.style   = {circle, draw=engblue, very thick, fill=engblue!18,
                   inner sep=0pt, minimum size=8mm, font=\small},
  edge/.style   = {thick, draw=enggray, ->, >=stealth},
  cut/.style    = {thick, draw=engred, ->, >=stealth},
  flbl/.style   = {font=\scriptsize, fill=white, inner sep=1pt},
]
% Six-vertex flow network used in the max-flow worked example.
\node[term]   (s)  at (0, 1.5)     {$s$};
\node[vertex] (v1) at (2.6, 3)     {$1$};
\node[vertex] (v2) at (2.6, 0)     {$2$};
\node[vertex] (v3) at (5.2, 3)     {$3$};
\node[vertex] (v4) at (5.2, 0)     {$4$};
\node[term]   (t)  at (7.8, 1.5)   {$t$};

% Capacities; the min-cut edges (S = {s,1,2}) drawn engred.
\draw[edge] (s)  -- node[flbl] {$10$} (v1);
\draw[edge] (s)  -- node[flbl] {$10$} (v2);
\draw[edge] (v1) -- node[flbl] {$4$}  (v2);
\draw[cut]  (v1) -- node[flbl] {$8$}  (v3);
\draw[cut]  (v2) -- node[flbl] {$9$}  (v4);
\draw[edge] (v2) -- node[flbl, pos=0.35] {$6$} (v3);
\draw[edge] (v3) -- node[flbl] {$10$} (t);
\draw[edge] (v4) -- node[flbl] {$10$} (t);
\draw[edge] (v4) -- node[flbl, pos=0.5] {$6$} (v3);

% The minimum cut, drawn as a dashed boundary.
\draw[engred, dashed, very thick]
  ($(v1)!0.5!(v3) + (0, 1.0)$) -- ($(v2)!0.5!(v4) + (0, -1.0)$);
\node[engred, font=\footnotesize, anchor=south]
  at ($(v1)!0.5!(v3) + (0, 1.0)$) {cut $S = \set{s, 1, 2}$};
\node[engred, font=\footnotesize, anchor=north]
  at (3.9, -1.6) {capacity $= 8 + 9 = 17$};
\end{tikzpicture}
\caption{A flow network with source $s$, sink $t$, and edge
capacities. The maximum $s$--$t$ flow has value $17$, certified by
the minimum cut $S = \set{s, 1, 2}$ (red boundary), whose two
forward edges $1 \to 3$ and $2 \to 4$ carry capacity $8 + 9 = 17$.
The max-flow min-cut theorem makes the saturated cut a proof of
optimality: no flow can exceed the capacity of any cut, and this cut
is achieved.}
\label{fig:vol02:ch17:max-flow}
\end{figure}
