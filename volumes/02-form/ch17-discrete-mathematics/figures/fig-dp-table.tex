\begin{tikzpicture}[

                    font=\small, anchor=center},
  hdr/.style     = {font=\small\bfseries, anchor=center},
  opt/.style     = {draw=engblue, very thick, fill=engblue!15,
                    minimum width=8mm, minimum height=7mm,

]
% 0/1 knapsack DP table: items (wt, val) = (2,3),(3,4),(4,5),(5,6); W = 5.
% rows i = 0..4 (items considered), cols w = 0..5 (capacity).
\def\rowstep{-0.72}
\def\colstep{0.85}

% column headers (capacity 0..5)
\foreach \w in {0,...,5} {
  \node[hdr] at ({\colstep*\w + \colstep}, 0.72) {$\w$};
}
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, 0.72) {$w\!\to$};
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, {\rowstep*0}) {$i\!=\!0$};
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, {\rowstep*1}) {$i\!=\!1$};
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, {\rowstep*2}) {$i\!=\!2$};
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, {\rowstep*3}) {$i\!=\!3$};
\node[hdr, anchor=east] at ({\colstep*0 - 0.1}, {\rowstep*4}) {$i\!=\!4$};

% row 0: all zeros (no items)
\foreach \w in {0,...,5} {
  \node[cell] at ({\colstep*\w + \colstep}, {\rowstep*0}) {$0$};
}
% row 1: item (2,3)
\foreach \w/\val in {0/0, 1/0, 2/3, 3/3, 4/3, 5/3} {
  \node[cell] at ({\colstep*\w + \colstep}, {\rowstep*1}) {$\val$};
}
% row 2: items (2,3),(3,4)
\foreach \w/\val in {0/0, 1/0, 2/3, 3/4, 4/4, 5/7} {
  \node[cell] at ({\colstep*\w + \colstep}, {\rowstep*2}) {$\val$};
}
% row 3: + item (4,5)
\foreach \w/\val in {0/0, 1/0, 2/3, 3/4, 4/5, 5/7} {
  \node[cell] at ({\colstep*\w + \colstep}, {\rowstep*3}) {$\val$};
}
% row 4: + item (5,6); optimum at (4,5) = 7
\foreach \w/\val in {0/0, 1/0, 2/3, 3/4, 4/5} {
  \node[cell] at ({\colstep*\w + \colstep}, {\rowstep*4}) {$\val$};
}
\node[opt] at ({\colstep*5 + \colstep}, {\rowstep*4}) {$7$};

\node[font=\footnotesize, anchor=north, engblue]
  at ({\colstep*3 + \colstep}, {\rowstep*4 - 0.55})
  {optimum $V[4,5] = 7$ (items $1$ and $2$)};
\end{tikzpicture}
\caption{The dynamic-programming table for the $0/1$ knapsack
instance with items $(w, v) = (2,3), (3,4), (4,5), (5,6)$ and
capacity $W = 5$. Entry $V[i, w]$ is the best value achievable using
the first $i$ items within capacity $w$. Each entry is filled from
two earlier entries by the recurrence $V[i,w] = \max(V[i-1,w],\,
v_i + V[i-1, w - w_i])$. The optimum $V[4,5] = 7$ takes items $1$
and $2$ (weights $2 + 3 = 5$, value $3 + 4 = 7$); the heavier
items never improve on it.}
