\begin{tikzpicture}[
  vertex/.style = {circle, draw=engblack, thick, fill=englime!25,
                   inner sep=0pt, minimum size=8mm, font=\small},
  edge/.style   = {thick, draw=enggray, ->, >=stealth},
  hl/.style     = {thick, draw=engblue, ->, >=stealth},
  flbl/.style   = {font=\scriptsize, fill=white, inner sep=1pt},
]
% Four-vertex directed weighted graph for the Floyd-Warshall trace.
\node[vertex] (v1) at (0, 2)   {$1$};
\node[vertex] (v2) at (3, 2)   {$2$};
\node[vertex] (v3) at (3, -1)  {$3$};
\node[vertex] (v4) at (0, -1)  {$4$};

% The route 1 -> 2 -> 3 -> 4 (cost 5) is highlighted in blue;
% it beats the direct edge 1 -> 4 (cost 7), drawn gray.
\draw[hl]   (v1) -- node[flbl] {$3$} (v2);
\draw[hl]   (v2) -- node[flbl] {$1$} (v3);
\draw[hl]   (v3) -- node[flbl] {$1$} (v4);
\draw[edge] (v1) -- node[flbl, pos=0.45] {$7$} (v4);
\draw[edge] (v3) -- node[flbl, pos=0.55] {$2$} (v1);
\draw[edge] (v4) -- node[flbl, pos=0.55] {$2$} (v2);
\end{tikzpicture}
\caption{The directed weighted graph of the Floyd-Warshall worked
trace. The direct edge $1 \to 4$ has weight $7$, but the two-hop
route $1 \to 2 \to 3 \to 4$ (blue) costs only $3 + 1 + 1 = 5$. The
algorithm discovers this saving in round $k = 3$, when vertex $3$
first becomes an allowed intermediate; the final distance matrix
records $D_{14} = 5$. The diagonal stays at $0$ throughout, the
running certificate that no negative cycle exists.}
\label{fig:vol02:ch17:floyd-warshall}
