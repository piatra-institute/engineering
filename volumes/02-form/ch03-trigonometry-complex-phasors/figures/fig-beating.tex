% Figure: beating phenomenon: two sinusoids of nearly equal frequency
% and their sum showing the slow envelope.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  % Upper panel: two sinusoids cos(2 pi 10 t) and cos(2 pi 11 t)
  % Domain t in [0, 1], so each sample at delta t = 0.005
  \begin{scope}
    \draw[->, engblack, thin] (0, 0) -- (12.5, 0) node[right, font=\small] {$t$ (s)};
    \draw[->, engblack, thin] (0, -1.4) -- (0, 1.5);
    \node[font=\small, anchor=east] at (-0.1, 1.0) {$+1$};
    \node[font=\small, anchor=east] at (-0.1, -1.0) {$-1$};
    \draw[engblack, dashed, thin] (0, 1.0) -- (12.0, 1.0);
    \draw[engblack, dashed, thin] (0, -1.0) -- (12.0, -1.0);
    % cos(2 pi 1.0 t) at scale 12 (so t=1 maps to x=12), 10 cycles
    \draw[engblue, thin, smooth, samples=240, domain=0:12.0] plot (\x, {cos(360*1.0*\x/12.0 * 10)});
    % cos(2 pi 1.1 t), 11 cycles
    \draw[engred, thin, dashed, smooth, samples=240, domain=0:12.0] plot (\x, {cos(360*1.0*\x/12.0 * 11)});
    \node[engblue, font=\footnotesize, anchor=west] at (12.0, 1.0) {$\cos(2\pi \cdot 10 t)$};
    \node[engred, font=\footnotesize, anchor=west] at (12.0, 0.55) {$\cos(2\pi \cdot 11 t)$};
  \end{scope}

  % Lower panel: sum, shown two units below, with the envelope 2 cos(pi t)
  \begin{scope}[yshift=-4cm]
    \draw[->, engblack, thin] (0, 0) -- (12.5, 0) node[right, font=\small] {$t$ (s)};
    \draw[->, engblack, thin] (0, -2.4) -- (0, 2.5);
    \node[font=\small, anchor=east] at (-0.1, 2.0) {$+2$};
    \node[font=\small, anchor=east] at (-0.1, -2.0) {$-2$};
    \draw[engblack, dashed, thin] (0, 2.0) -- (12.0, 2.0);
    \draw[engblack, dashed, thin] (0, -2.0) -- (12.0, -2.0);
    % sum cos(2 pi 10 t) + cos(2 pi 11 t)
    % equals 2 cos(2 pi 0.5 t) cos(2 pi 10.5 t)
    \draw[engblack, thin, smooth, samples=480, domain=0:12.0] plot (\x, {cos(360*1.0*\x/12.0 * 10) + cos(360*1.0*\x/12.0 * 11)});
    % envelope +/- 2 cos(2 pi 0.5 t)
    \draw[engred, thick, smooth, samples=200, domain=0:12.0] plot (\x, {2*abs(cos(180*1.0*\x/12.0))});
    \draw[engred, thick, smooth, samples=200, domain=0:12.0] plot (\x, {-2*abs(cos(180*1.0*\x/12.0))});
    \node[engred, font=\footnotesize, anchor=west] at (12.0, 1.0) {envelope $|2\cos(2\pi\cdot 0.5 t)|$};
    \node[font=\footnotesize, anchor=west] at (12.0, 0.0) {sum};
  \end{scope}
\end{tikzpicture}
\caption[Beating between two nearby sinusoids]{Two sinusoids of
amplitude one at $10\,\si{\hertz}$ (solid, blue) and
$11\,\si{\hertz}$ (dashed, red) and their sum (lower panel). The
sum-to-product identity from section 3.1 gives
$\cos(2\pi\cdot 10 t)+\cos(2\pi\cdot 11 t) = 2\cos(2\pi\cdot 0.5
t)\cos(2\pi\cdot 10.5 t)$: a fast carrier at the average
frequency $10.5\,\si{\hertz}$ modulated by a slow envelope
(red, lower panel) at half the difference frequency,
$0.5\,\si{\hertz}$. The musician tuning two strings hears the
slow envelope as a periodic loud-soft beat; the frequency-comparison
instrument reads the envelope period to recover the frequency
difference to whatever precision the envelope can be timed.}
\label{fig:vol02:ch03:beating}
\end{figure}
