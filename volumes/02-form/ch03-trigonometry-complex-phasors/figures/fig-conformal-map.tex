% Figure: a conformal map preview. The analytic map w = z^2 carries the
% orthogonal Cartesian grid of the z-plane to a family of mutually
% orthogonal parabolas in the w-plane. Angles between curves are
% preserved (right angles stay right angles), which is the defining
% property of a conformal map.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex]
  % z-plane: a Cartesian grid in the first quadrant
  \begin{scope}
    \draw[->, enggray, thin] (-0.2,0) -- (2.6,0) node[right, font=\small]{$\Re z$};
    \draw[->, enggray, thin] (0,-0.2) -- (0,2.6) node[above, font=\small]{$\Im z$};
    \foreach \c in {0.5,1.0,1.5,2.0} {
      \draw[engblue, thick] (\c,0) -- (\c,2.2);  % vertical lines x = c
      \draw[engred, thick] (0,\c) -- (2.2,\c);   % horizontal lines y = c
    }
    \node[font=\small, anchor=north] at (1.3,-0.5) {(a) $z$-plane grid};
  \end{scope}
  % arrow
  \node[font=\large] at (3.5,1.2) {$\xrightarrow{\,w=z^2\,}$};
  % w-plane: images are parabolas
  \begin{scope}[xshift=5.2cm, scale=0.62]
    \draw[->, enggray, thin] (-0.3,0) -- (5.0,0) node[right, font=\small]{$\Re w$};
    \draw[->, enggray, thin] (0,-0.3) -- (0,5.0) node[above, font=\small]{$\Im w$};
    % image of x = c (vertical lines): u = c^2 - y^2, v = 2 c y  (parabola opening left)
    \foreach \c in {0.5,1.0,1.5,2.0} {
      \draw[engblue, thick, domain=0:2.2, samples=60, smooth]
        plot ({\c*\c - \x*\x}, {2*\c*\x});
    }
    % image of y = c (horizontal lines): u = x^2 - c^2, v = 2 x c (parabola opening right)
    \foreach \c in {0.5,1.0,1.5,2.0} {
      \draw[engred, thick, domain=0:2.2, samples=60, smooth]
        plot ({\x*\x - \c*\c}, {2*\x*\c});
    }
    \node[font=\small, anchor=north] at (2.4,-0.8) {(b) $w$-plane images};
  \end{scope}
\end{tikzpicture}
\caption[A conformal map preview]{The analytic map $w = z^2$ as a
conformal map. Panel (a): the orthogonal Cartesian grid in the first
quadrant of the $z$-plane, vertical lines $\Re z = c$ in blue and
horizontal lines $\Im z = c$ in red. Panel (b): the images under
$w = z^2$ are two families of confocal parabolas. The right angles at
which the grid lines crossed in the $z$-plane survive as right angles
between the image parabolas in the $w$-plane. Angle preservation at
every point where the derivative is nonzero is the defining property of
a conformal map and the geometric content of complex
differentiability.}
\label{fig:vol02:ch03:conformal-map}
\end{figure}
