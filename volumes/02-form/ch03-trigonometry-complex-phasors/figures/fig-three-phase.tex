% Figure: balanced three-phase phasor star (left) and the corresponding
% time-domain waveforms (right). The three phasors are spaced 120 degrees
% and sum to zero; the three sinusoids sum to zero at every instant.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[line cap=round, >=latex]
  % Left panel: phasor star.
  \begin{scope}[scale=1.3]
    \draw[->, engblack, thin] (-1.4, 0) -- (1.5, 0) node[right, font=\small] {$\Re$};
    \draw[->, engblack, thin] (0, -1.4) -- (0, 1.5) node[above, font=\small] {$\Im$};
    \draw[engblack!30, thin, dashed] (0, 0) circle (1);
    % V_a at 0 deg, V_b at -120 deg, V_c at -240 = +120 deg.
    \draw[engblue, very thick, ->] (0, 0) -- (0:1);
    \node[engblue, font=\footnotesize, anchor=west] at (0:1.05) {$\tilde V_a$};
    \draw[engred, very thick, ->] (0, 0) -- (-120:1);
    \node[engred, font=\footnotesize, anchor=north] at (-120:1.1) {$\tilde V_b$};
    \draw[engamber, very thick, ->] (0, 0) -- (120:1);
    \node[engamber, font=\footnotesize, anchor=south] at (120:1.1) {$\tilde V_c$};
    \node[font=\small, anchor=north] at (0, -1.6) {(a) phasor star};
  \end{scope}
  % Right panel: time-domain waveforms.
  \begin{scope}[xshift=6.2cm, scale=1.0]
    \draw[->, engblack, thin] (0, 0) -- (5.0, 0) node[right, font=\small] {$\omega t$};
    \draw[->, engblack, thin] (0, -1.3) -- (0, 1.4) node[above, font=\small] {$v$};
    \draw[engblue, thick, domain=0:4.6, samples=80, smooth]
      plot (\x, {cos(deg(\x))});
    \draw[engred, thick, domain=0:4.6, samples=80, smooth]
      plot (\x, {cos(deg(\x) - 120)});
    \draw[engamber, thick, domain=0:4.6, samples=80, smooth]
      plot (\x, {cos(deg(\x) + 120)});
    \node[engblue, font=\footnotesize, anchor=south] at (0.2, 1.0) {$v_a$};
    \node[engred, font=\footnotesize, anchor=north] at (2.5, -0.6) {$v_b$};
    \node[engamber, font=\footnotesize, anchor=south] at (1.6, 0.6) {$v_c$};
    \node[font=\small, anchor=north] at (2.3, -1.5) {(b) the three waveforms};
  \end{scope}
\end{tikzpicture}
\caption[Balanced three-phase system]{A balanced three-phase voltage
set. Panel (a): the three phasors $\tilde V_a$, $\tilde V_b$,
$\tilde V_c$ are equal in magnitude and spaced by $120^{\circ}$; they
are the cube roots of $V_m^3$ and sum to zero. Panel (b): the three
time-domain sinusoids $v_a(t) = V_m\cos\omega t$, $v_b(t) =
V_m\cos(\omega t - 2\pi/3)$, $v_c(t) = V_m\cos(\omega t + 2\pi/3)$ sum
to zero at every instant. The zero sum is the engineering reason a
balanced three-phase system needs no neutral return conductor and the
basis of the rotating-field machine.}
\label{fig:vol02:ch03:three-phase}
\end{figure}
