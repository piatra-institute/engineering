% Figure: the impedance triangle of a series RLC circuit. Resistance R on
% the real axis, net reactance X = wL - 1/(wC) on the imaginary axis, and
% impedance Z = R + iX as the hypotenuse with phase angle phi_Z. Two cases:
% net inductive (X > 0) and net capacitive (X < 0).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex]
  % Inductive case (left).
  \begin{scope}[xshift=-3.6cm]
    \draw[->, engblack, thin] (-0.4, 0) -- (3.4, 0) node[right, font=\small] {$\Re$};
    \draw[->, engblack, thin] (0, -2.0) -- (0, 2.4) node[above, font=\small] {$\Im$};
    \draw[engblue, very thick, ->] (0, 0) -- (2.4, 1.8);
    \filldraw[engblue] (2.4, 1.8) circle (0.05);
    \node[engblue, font=\footnotesize, anchor=south west] at (2.4, 1.8) {$Z$};
    \draw[engblue, thick] (0, 0) -- (2.4, 0) node[midway, below, font=\footnotesize] {$R$};
    \draw[engblue, thick] (2.4, 0) -- (2.4, 1.8) node[midway, right, font=\footnotesize] {$X > 0$};
    \draw[engblue] (0.7, 0) arc (0:36.87:0.7);
    \node[engblue, font=\footnotesize] at (0.95, 0.22) {$\varphi_Z$};
    \node[font=\small, anchor=north] at (1.5, -1.6) {(a) net inductive};
  \end{scope}
  % Capacitive case (right).
  \begin{scope}[xshift=3.0cm]
    \draw[->, engblack, thin] (-0.4, 0) -- (3.4, 0) node[right, font=\small] {$\Re$};
    \draw[->, engblack, thin] (0, -2.4) -- (0, 2.0) node[above, font=\small] {$\Im$};
    \draw[engred, very thick, ->] (0, 0) -- (2.4, -1.8);
    \filldraw[engred] (2.4, -1.8) circle (0.05);
    \node[engred, font=\footnotesize, anchor=north west] at (2.4, -1.8) {$Z$};
    \draw[engred, thick] (0, 0) -- (2.4, 0) node[midway, above, font=\footnotesize] {$R$};
    \draw[engred, thick] (2.4, 0) -- (2.4, -1.8) node[midway, right, font=\footnotesize] {$X < 0$};
    \draw[engred] (0.7, 0) arc (0:-36.87:0.7);
    \node[engred, font=\footnotesize] at (0.95, -0.25) {$\varphi_Z$};
    \node[font=\small, anchor=north] at (1.5, -2.2) {(b) net capacitive};
  \end{scope}
\end{tikzpicture}
\caption[The impedance triangle]{The impedance triangle of a series RLC
circuit. The resistance $R$ lies on the real axis, the net reactance
$X = \omega L - 1/(\omega C)$ on the imaginary axis, and the impedance
$Z = R + iX = |Z| e^{i\varphi_Z}$ as the hypotenuse. Panel (a): above
resonance the inductive reactance dominates, $X > 0$, $\varphi_Z > 0$,
and the current lags the voltage. Panel (b): below resonance the
capacitive reactance dominates, $X < 0$, $\varphi_Z < 0$, and the
current leads the voltage. At resonance $X = 0$, the impedance reduces
to $R$, and voltage and current are in phase.}
\label{fig:vol02:ch03:impedance-triangle}
\end{figure}
