% Figure: the two-branch AC network of the worked case study. A source Vs
% drives a series R1-L branch (mesh 1) sharing a node with a parallel
% R2 || C branch to the return. The phasor diagram on the right shows the
% source voltage, the two branch currents, and their sum (the supply
% current).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex,
    every node/.style={font=\footnotesize}]
  % Circuit schematic (left)
  \begin{scope}
    % outer rectangle nodes
    \coordinate (sl) at (0, 0);     % source low
    \coordinate (sh) at (0, 2.6);   % source high
    \coordinate (a)  at (2.4, 2.6); % top-left node
    \coordinate (b)  at (4.4, 2.6); % mid node (after R1 L)
    \coordinate (br) at (4.4, 0);   % return node under b
    % source
    \draw[engblue, thick] (sl) -- (sh);
    \draw[engblue, thick] (sl) ++(-0.18,0.9) -- ++(0.36,0);
    \draw[engblue, thick] (sl) ++(-0.30,1.7) -- ++(0.60,0);
    \node[engblue, anchor=east] at (-0.35, 1.3) {$\tilde V_s$};
    % top wire with R1 and L
    \draw[engblack, thick] (sh) -- (a);
    \draw[engblack, thick] (a) -- ++(0.15,0) -- ++(0,0.18) -- ++(0.3,-0.36)
      -- ++(0.3,0.36) -- ++(0.3,-0.36) -- ++(0.15,0.18) -- (b);
    \node[anchor=south] at (3.4, 2.85) {$R_1,\ L$};
    % R2 || C branch from b down to br
    \draw[engred, thick] (b) -- ++(0,-0.4);
    \draw[engred, thick] (4.0,1.0) rectangle (4.4,1.8);
    \draw[engred, thick] (4.4,1.4) -- (4.4,1.8);
    \draw[engred, thick] (4.4,1.0) -- (4.4,0.4);
    \node[engred, anchor=east] at (3.95, 1.4) {$R_2$};
    % capacitor branch parallel
    \draw[engred, thick] (b) -- ++(0.8,0) -- ++(0,-0.7);
    \draw[engred, thick] (5.0,1.7) -- (5.4,1.7);
    \draw[engred, thick] (5.0,1.5) -- (5.4,1.5);
    \draw[engred, thick] (5.2,1.9) -- (5.2,1.7);
    \draw[engred, thick] (5.2,1.5) -- (5.2,0.4) -- (4.4,0.4);
    \node[engred, anchor=west] at (5.45, 1.6) {$C$};
    % return wire
    \draw[engblack, thick] (sl) -- (br);
    \draw[engblack, thick] (b) ++(0,-0.4) -- (br);
    % node dots
    \filldraw[engblack] (b) circle (0.05);
    \node[anchor=south west] at (b) {node $\mathrm{N}$};
  \end{scope}
  % Phasor diagram (right)
  \begin{scope}[xshift=8.2cm, yshift=1.3cm, scale=0.95]
    \draw[->, enggray, thin] (-1.7,0) -- (1.9,0) node[right]{$\Re$};
    \draw[->, enggray, thin] (0,-1.7) -- (0,1.4) node[above]{$\Im$};
    % Vs along real axis (reference), scaled.
    \draw[engblue, very thick, ->] (0,0) -- (1.6,0)
      node[anchor=south east]{$\tilde V_s$};
    % branch current I1 (lagging, fourth quadrant).
    \draw[engred, thick, ->] (0,0) -- (0.9,-1.1)
      node[anchor=north west]{$\tilde I_1$};
    % branch current I2 (leading-ish, first quadrant small).
    \draw[engamber, thick, ->] (0,0) -- (0.5,0.55)
      node[anchor=south]{$\tilde I_2$};
    % supply current I = I1 + I2 (sum).
    \draw[engblack, very thick, ->] (0,0) -- (1.4,-0.55)
      node[anchor=north]{$\tilde I$};
    % dashed parallelogram
    \draw[enggray, dashed] (0.9,-1.1) -- (1.4,-0.55);
    \draw[enggray, dashed] (0.5,0.55) -- (1.4,-0.55);
  \end{scope}
\end{tikzpicture}
\caption[A two-branch AC network and its phasor diagram]{The two-branch
AC network of the worked case study. A source $\tilde V_s$ drives a
series $R_1$-$L$ branch to node $\mathrm{N}$; at node $\mathrm{N}$ a
parallel combination of $R_2$ and $C$ returns the current to the source.
Right: the phasor diagram with $\tilde V_s$ as the real-axis reference,
the series-branch current $\tilde I_1$ lagging (the inductor dominates
its branch), the shunt-branch current $\tilde I_2$ leading (the capacitor
dominates its branch), and the supply current $\tilde I = \tilde I_1 +
\tilde I_2$ as their vector sum. The dashed lines complete the
addition parallelogram.}
\label{fig:vol02:ch03:ac-network}
\end{figure}
