% Figure: the array factor of a uniform linear array of N isotropic
% elements, plotted as magnitude versus the electrical angle psi. Shows
% the main lobe at psi = 0, the grating-lobe repeat at psi = 2*pi, and
% the N-1 nulls between, for N = 6. The array factor is a phasor sum of N
% unit phasors with progressive phase psi, summing as a geometric series.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[line cap=round, >=latex, font=\footnotesize]
  \draw[->, engblack, thin] (-0.3,0) -- (6.6,0) node[right] {$\psi$ (rad)};
  \draw[->, engblack, thin] (0,-0.2) -- (0,3.2) node[above] {$|AF|$};
  % Dirichlet kernel |sin(N psi/2)/sin(psi/2)| for N=6, scaled.
  % Plot magnitude; avoid the removable singularity at psi=0 by starting
  % the sampled curve slightly off zero and drawing the peak by hand.
  \draw[engblue, thick] plot[domain=0.06:6.22, samples=400, smooth]
    ({\x}, { 0.5*abs( sin(3*\x r) / sin(0.5*\x r) ) });
  % main lobe peak (value N = 6, scaled by 0.5 = 3.0) at psi = 0 and 2*pi
  \filldraw[engblue] (0,3.0) circle (0.04);
  \filldraw[engblue] (6.283,3.0) circle (0.04);
  \draw[engblue, thick] (0,3.0) .. controls (0.25,2.6) .. (0.5,1.4);
  \node[engblue, anchor=south] at (0,3.05) {main lobe};
  \node[engblue, anchor=south] at (6.0,3.05) {grating lobe};
  \draw[engblack!40, dashed] (6.283,0) -- (6.283,3.0);
  \node[engblack, anchor=north] at (6.283,-0.05) {$2\pi$};
  \node[engblack, anchor=north] at (3.14,-0.05) {$\pi$};
\end{tikzpicture}
\caption[Array factor of a uniform linear array]{The array factor of a
uniform linear array of $N = 6$ equally-spaced isotropic elements is the
magnitude of a phasor sum of $N$ unit phasors with progressive
inter-element phase $\psi = k d \cos\vartheta + \beta$, a geometric
series that sums to $|AF(\psi)| = |\sin(N\psi/2) / \sin(\psi/2)|$, the
Dirichlet kernel. The main lobe at $\psi = 0$ reaches the coherent sum
$N$; between it and the grating lobe at $\psi = 2\pi$ sit $N - 1 = 5$
nulls where the $N$ phasors close into a regular polygon and cancel
exactly. Steering the beam means choosing the fixed phase increment
$\beta$ so the main lobe points in the wanted direction. The pattern is
the same roots-of-unity cancellation that zeroes the balanced
three-phase current, read as a function of angle.}
\label{fig:vol02:ch03:array-factor}
\end{figure}
