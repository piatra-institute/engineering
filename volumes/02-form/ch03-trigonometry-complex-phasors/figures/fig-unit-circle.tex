% Figure: the unit circle with canonical angles and (cos, sin) coordinates.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=2.6, line cap=round, >=latex]
  % Axes
  \draw[->, engblack, thin] (-1.35, 0) -- (1.4, 0) node[right, font=\small] {$\Re,\ x = \cos\theta$};
  \draw[->, engblack, thin] (0, -1.35) -- (0, 1.4) node[above, font=\small] {$\Im,\ y = \sin\theta$};
  % Unit circle
  \draw[engblue, thick] (0, 0) circle (1);
  % Canonical angles: 0, 30, 45, 60, 90, 120, 135, 150, 180, 210, 225, 240, 270, 300, 315, 330
  \foreach \ang/\lab/\xa/\ya in {
    0/0/1.05/0.05,
    30/{\pi/6}/0.95/0.55,
    45/{\pi/4}/0.78/0.78,
    60/{\pi/3}/0.55/0.95,
    90/{\pi/2}/0.05/1.05,
    120/{2\pi/3}/-0.55/0.95,
    135/{3\pi/4}/-0.78/0.78,
    150/{5\pi/6}/-0.95/0.55,
    180/{\pi}/-1.05/0.05,
    270/{3\pi/2}/0.05/-1.05}
  {
    \filldraw[engblue] ({cos(\ang)}, {sin(\ang)}) circle (0.022);
  }
  % Selected labels (cos, sin) for the four key first-quadrant angles
  \node[font=\tiny, anchor=west] at (1.05, 0.05) {$(1,0)$};
  \node[font=\tiny, anchor=south] at (0.07, 1.05) {$(0,1)$};
  \node[font=\tiny, anchor=east] at (-1.05, 0.05) {$(-1,0)$};
  \node[font=\tiny, anchor=north] at (0.07, -1.05) {$(0,-1)$};
  \node[font=\tiny, anchor=south west, engblue] at (0.78, 0.78) {$\!\!(\tfrac{\sqrt2}{2},\tfrac{\sqrt2}{2})$};
  \node[font=\tiny, anchor=south west, engblue] at (0.55, 0.95) {$\!\!(\tfrac12,\tfrac{\sqrt3}{2})$};
  \node[font=\tiny, anchor=west, engblue] at (0.95, 0.55) {$\!\!(\tfrac{\sqrt3}{2},\tfrac12)$};
  % A representative angle theta, with radius and projections
  \draw[engred, thick] (0, 0) -- ({cos(55)}, {sin(55)});
  \draw[engred, thin, dashed] ({cos(55)}, {sin(55)}) -- ({cos(55)}, 0);
  \draw[engred, thin, dashed] ({cos(55)}, {sin(55)}) -- (0, {sin(55)});
  \draw[engred, thick] (0.28, 0) arc (0:55:0.28);
  \node[engred, font=\small] at (0.18, 0.10) {$\theta$};
  \node[engred, font=\footnotesize, anchor=south] at ({cos(55)/2}, -0.18) {$\cos\theta$};
  \node[engred, font=\footnotesize, anchor=east, rotate=90] at (-0.10, {sin(55)/2}) {$\sin\theta$};
  \filldraw[engred] ({cos(55)}, {sin(55)}) circle (0.03);
  \node[engred, font=\footnotesize, anchor=south west] at ({cos(55)}, {sin(55)}) {$(\cos\theta,\sin\theta)$};
\end{tikzpicture}
\caption[Unit circle with canonical angles]{The unit circle in the
Cartesian plane carries the working definition of cosine and sine
for all real $\theta$. The point at arc length $\theta$
counterclockwise from $(1,0)$ has Cartesian coordinates
$(\cos\theta,\sin\theta)$. Canonical angles at multiples of
$\pi/6$ and $\pi/4$ are marked; the first-quadrant labels reproduce
the values tabulated in section 3.1. The same diagram, with the
axes relabelled $\Re$ and $\Im$, displays the complex exponential
$e^{i\theta}$.}
\label{fig:vol02:ch03:unit-circle}
\end{figure}
