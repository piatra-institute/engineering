% Figure: the Smith chart as the image of the impedance right-half-plane
% under the Mobius map Gamma = (z - 1)/(z + 1) (z = normalised
% impedance). Draws the unit reflection-coefficient circle, a few
% constant-resistance circles and constant-reactance arcs, and marks the
% match point at the centre (Gamma = 0).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[line cap=round, >=latex, font=\footnotesize, scale=2.4]
  % boundary: |Gamma| = 1
  \draw[engblack, thick] (0,0) circle (1);
  \draw[engblack!30, thin] (-1.05,0) -- (1.1,0) node[right] {$\Re\Gamma$};
  \draw[engblack!30, thin] (0,-1.1) -- (0,1.15) node[above] {$\Im\Gamma$};
  % constant-resistance circles: centre (r/(1+r),0), radius 1/(1+r)
  \foreach \r in {0.5, 1, 2} {
    \pgfmathsetmacro{\cx}{\r/(1+\r)}
    \pgfmathsetmacro{\rad}{1/(1+\r)}
    \draw[engblue!70, thin] (\cx,0) circle (\rad);
  }
  % constant-reactance arcs (x = +1 and x = -1): centre (1, 1/x), radius
  % 1/|x|; clip to inside the unit circle.
  \begin{scope}
    \clip (0,0) circle (1);
    \draw[engred!70, thin] (1,1) circle (1);
    \draw[engred!70, thin] (1,-1) circle (1);
  \end{scope}
  % open circuit (z -> inf) at Gamma = +1; short (z = 0) at Gamma = -1
  \filldraw[engblack] (1,0) circle (0.02);
  \node[engblack, anchor=west] at (1.02,0.12) {open};
  \filldraw[engblack] (-1,0) circle (0.02);
  \node[engblack, anchor=east] at (-1.02,0.12) {short};
  % match point at centre
  \filldraw[engred] (0,0) circle (0.025);
  \node[engred, anchor=north] at (0,-0.06) {match, $\Gamma=0$};
  % a sample load point: z = 0.5 - 0.6 i  ->  Gamma computed offline
  % Gamma = (z-1)/(z+1) for z = 0.5 - 0.6i is approx -0.30 - 0.39 i
  \filldraw[engblue] (-0.30,-0.39) circle (0.025);
  \node[engblue, anchor=north west] at (-0.28,-0.40) {load};
\end{tikzpicture}
\caption[The Smith chart as a Mobius image]{The Smith chart is the image
of the normalised-impedance right-half-plane under the Mobius map
$\Gamma = (z - 1)/(z + 1)$, where $z = Z/Z_0$ is the impedance
normalised to the reference $Z_0$ and $\Gamma$ is the reflection
coefficient. The map carries the constant-resistance lines of the
$z$-plane to the blue circles and the constant-reactance lines to the red
arcs, all packed inside the unit circle $|\Gamma| = 1$ because a passive
load reflects no more than it receives. The centre $\Gamma = 0$ is the
matched condition $Z = Z_0$. A matching network moves the load point to
the centre by adding series and shunt reactances, each a glide along a
constant-resistance circle or a constant-conductance circle. Circles and
lines map to circles and lines, the defining property of the Mobius
transformation, which is why the chart is drawn entirely with circular
arcs.}
\label{fig:vol02:ch03:smith-chart}
\end{figure}
