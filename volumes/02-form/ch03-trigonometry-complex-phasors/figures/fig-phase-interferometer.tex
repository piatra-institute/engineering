% Figure: a two-element phase interferometer for direction finding. A distant
% source at angle theta from broadside sends a plane wave onto two antennas a
% baseline d apart; the extra path d sin theta introduces a measurable phase
% difference. The ambiguity: baselines longer than half a wavelength wrap.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex, font=\footnotesize]
  % two antennas on a horizontal baseline
  \coordinate (A1) at (0,0);
  \coordinate (A2) at (4.0,0);
  \draw[engblack, thick] (A1) -- (A2) node[midway, below=6pt] {baseline $d$};
  \filldraw[engblue] (A1) circle (0.07) node[below left] {ant.\ 1};
  \filldraw[engblue] (A2) circle (0.07) node[below right] {ant.\ 2};
  % incoming wavefronts (plane wave from upper right, angle theta from vertical)
  \foreach \s in {0,1,2} {
    \draw[enggray, thin] ($(1.0+1.1*\s,3.0)$) -- ($(2.6+1.1*\s,1.2)$);
  }
  \node[enggray] at (2.0,3.2) {wavefronts};
  % arrival direction arrow
  \draw[engred, thick, ->] (5.1,3.0) -- (3.9,1.65);
  \node[engred, anchor=west] at (5.1,3.0) {from source};
  % broadside normal (vertical dashed) at antenna 2
  \draw[engamber, dashed] (A2) -- ($(A2)+(0,2.6)$);
  \node[engamber, anchor=south] at ($(A2)+(0,2.6)$) {broadside};
  % angle theta between broadside and arrival direction
  \draw[engamber] ($(A2)+(0,1.4)$) arc (90:132:1.4);
  \node[engamber] at ($(A2)+(-0.55,1.75)$) {$\theta$};
  % extra path segment d sin theta at antenna 1 side
  \draw[engred, thick] (A1) -- ($(A1)+(0.74,0.99)$);
  \node[engred, anchor=south east] at ($(A1)+(0.74,0.99)$) {$d\sin\theta$};
\end{tikzpicture}
\caption[A two-element phase interferometer]{A two-element phase
interferometer for finding the direction of a distant source. A plane wave
arriving at angle $\theta$ from broadside travels an extra distance
$d\sin\theta$ to the nearer antenna, which appears at the receiver as a phase
difference $\Delta\varphi = (2\pi/\lambda)\,d\sin\theta$ between the two
channels. Measuring $\Delta\varphi$ and inverting gives the arrival angle. The
measurement inherits the wrap discipline of the failure section: a baseline
longer than half a wavelength lets $\Delta\varphi$ exceed $\pi$ for some
$\theta$, so the measured wrapped phase maps several true angles onto one
reading. A short unambiguous baseline paired with a long precise baseline
resolves the ambiguity, the same coarse-plus-fine strategy that unwraps a
laser interferometer.}
\label{fig:vol02ch03:phase-interferometer}
\end{figure}
