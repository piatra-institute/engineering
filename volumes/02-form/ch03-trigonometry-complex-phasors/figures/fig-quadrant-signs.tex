% Figure: the four quadrants of the complex plane with the signs of the
% real and imaginary parts, and the angle returned by atan2 versus the
% angle returned by the naive arctan(b/a). Illustrates the section 3.7
% quadrant-error failure mode.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.3, line cap=round, >=latex]
  \draw[->, engblack, thin] (-2.2, 0) -- (2.4, 0) node[right, font=\small] {$\Re$};
  \draw[->, engblack, thin] (0, -2.2) -- (0, 2.4) node[above, font=\small] {$\Im$};
  % quadrant labels with signs
  \node[engblack, font=\footnotesize, align=center] at (1.3, 1.3)
    {QI\\$(+,+)$};
  \node[engblack, font=\footnotesize, align=center] at (-1.3, 1.3)
    {QII\\$(-,+)$};
  \node[engblack, font=\footnotesize, align=center] at (-1.3, -1.3)
    {QIII\\$(-,-)$};
  \node[engblack, font=\footnotesize, align=center] at (1.3, -1.3)
    {QIV\\$(+,-)$};
  % point z = -1 + 1 i in QII (atan2 gives 3pi/4; arctan(1/-1) gives -pi/4)
  \draw[engblue, very thick, ->] (0, 0) -- (-1, 1);
  \filldraw[engblue] (-1, 1) circle (0.05);
  \node[engblue, font=\footnotesize, anchor=south east] at (-1, 1) {$z = -1 + i$};
  % correct angle arc (3pi/4 from +Re axis, going CCW)
  \draw[engblue, thick] (0.6, 0) arc (0:135:0.6);
  \node[engblue, font=\footnotesize] at (0.35, 0.85) {$\tfrac{3\pi}{4}$};
  % wrong angle: arctan(1/-1) = -pi/4 maps to the point 1 - i in QIV
  \draw[engred, very thick, dashed, ->] (0, 0) -- (1, -1);
  \filldraw[engred] (1, -1) circle (0.05);
  \node[engred, font=\footnotesize, anchor=north west] at (1, -1)
    {naive $\arctan$: $-\tfrac{\pi}{4}$};
\end{tikzpicture}
\caption[Quadrant signs and the atan2 convention]{The four quadrants of
the complex plane with the signs of the real and imaginary parts. The
point $z = -1 + i$ lies in the second quadrant at argument $3\pi/4$
(blue), correctly recovered by $\operatorname{atan2}(\Im z, \Re z)$. The
naive expression $\arctan(\Im z / \Re z) = \arctan(1/{-1}) =
\arctan(-1) = -\pi/4$ (red, dashed) collapses the second and third
quadrants onto the fourth and first, producing an error of exactly
$\pi$ radians. Every phase computation uses the four-quadrant
$\operatorname{atan2}$; the naive $\arctan$ is silently wrong whenever
$\Re z < 0$.}
\label{fig:vol02:ch03:quadrant-signs}
\end{figure}
