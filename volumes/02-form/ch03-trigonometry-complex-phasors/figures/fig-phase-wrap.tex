% the wrapped phase the instrument reports.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
  \draw[->, engblack, thin] (0, 0) -- (11.5, 0) node[right, font=\small] {$t$};
  \draw[->, engblack, thin] (0, -3.6) -- (0, 5.5);
  % horizontal guides at +/- pi (1.0 unit per pi)
  \draw[engblack, dashed, thin] (0, 3.1416) -- (11.0, 3.1416);
  \node[font=\footnotesize, anchor=east] at (-0.1, 3.1416) {$+\pi$};
  \draw[engblack, dashed, thin] (0, -3.1416) -- (11.0, -3.1416);
  \node[font=\footnotesize, anchor=east] at (-0.1, -3.1416) {$-\pi$};
  % Unwrapped: phi(t) = 0.6 t for t in [0, 11] (slope 0.6)
  \draw[engblue, very thick, smooth, samples=2, domain=0:11.0] plot (\x, {0.6*\x});
  \node[engblue, font=\footnotesize, anchor=south] at (8.5, 5.3) {unwrapped $\varphi(t)$};
  % Wrapped: the same phase modulo 2 pi, restricted to (-pi, pi]
  % Compute the wrap segments by hand. phi advances by 0.6 t.
  % Wraps occur where phi crosses pi (3.1416), 3 pi (9.4248), ...
  % Segment 1: t in [0, pi/0.6 = 5.236], phi in [0, pi], wrapped = phi.
  \draw[engred, very thick, smooth, samples=2, domain=0:5.236] plot (\x, {0.6*\x});
  % Jump down at t = 5.236 to -pi
  \draw[engred, dotted, thin] (5.236, 3.1416) -- (5.236, -3.1416);
  % Segment 2: t in [5.236, 3 pi/0.6 = 15.708] but cut at 11 since 0.6*11=6.6, wrapped = 0.6 t - 2 pi
  % at t=11, wrapped = 6.6 - 6.283 = 0.317 (just above 0). Range: at t=5.236 wrapped = -pi.
  \draw[engred, very thick, smooth, samples=2, domain=5.236:11.0] plot (\x, {0.6*\x - 6.2832});
  \node[engred, font=\footnotesize, anchor=south west] at (5.4, -2.8) {wrapped $\varphi_{\mathrm{w}}(t)$ on $(-\pi,\pi]$};
  \node[engred, font=\footnotesize, anchor=west, align=left] at (5.3, -1.2) {jump of $2\pi$\\at every wrap};
\end{tikzpicture}
\caption[Phase wrap-around in measurement]{The underlying physical
phase $\varphi(t)$ (blue, solid) advances linearly in time. An
instrument that reports the wrapped phase $\varphi_{\mathrm{w}}(t)$
on the principal range $(-\pi, \pi]$ (red, solid) jumps by $2\pi$
at every crossing of the wrap boundary. A calculation that depends
on the unwrapped phase, for example a phase-velocity integration
or a phase-shift range measurement, must add $2\pi k$ to recover
$\varphi(t)$ from $\varphi_{\mathrm{w}}(t)$, where $k$ is the
running cycle count. A wrap that is missed because the
sample-to-sample phase change exceeds $\pi$ between samples is
indistinguishable from a wrap that did not occur; the unwrap fails
silently, and the recovered $\varphi(t)$ is off by an integer
multiple of $2\pi$.}
\label{fig:vol02:ch03:phase-wrap}
\end{figure}
