% Figure: the frequency-response locus of a first-order low-pass filter,
% H(iw) = 1/(1 + i w/w_c), traced in the complex plane as w runs from 0 to
% infinity. The locus is the lower half of the circle of radius 1/2 centred
% at (1/2, 0): H starts at 1 (w=0), passes through (1/2, -1/2) at w = w_c,
% and spirals into the origin as w -> infinity.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=3.2, line cap=round, >=latex]
  % axes
  \draw[->, engblack, thin] (-0.2, 0) -- (1.25, 0) node[right, font=\small] {$\Re H$};
  \draw[->, engblack, thin] (0, -0.75) -- (0, 0.35) node[above, font=\small] {$\Im H$};
  % the semicircle: centre (0.5,0), radius 0.5, lower half (angles 0 to -180)
  \draw[engblue, very thick] (1,0) arc (0:-180:0.5);
  % marker at w = 0: H = 1
  \filldraw[engred] (1, 0) circle (0.018);
  \node[engred, font=\footnotesize, anchor=south west] at (1, 0) {$\omega = 0$};
  % marker at w = w_c: H = 0.5 - 0.5 i  (angle -90 on the small circle)
  \filldraw[engred] (0.5, -0.5) circle (0.018);
  \node[engred, font=\footnotesize, anchor=north] at (0.5, -0.56) {$\omega = \omega_c$};
  % marker at w -> infinity: origin
  \filldraw[engred] (0, 0) circle (0.018);
  \node[engred, font=\footnotesize, anchor=south east] at (0, 0.02) {$\omega \to \infty$};
  % radius vector at w_c showing magnitude 1/sqrt2 and phase -45
  \draw[engamber, thick, ->] (0,0) -- (0.5, -0.5);
  \node[engamber, font=\footnotesize, anchor=north west] at (0.27, -0.24) {$|H| = \tfrac{1}{\sqrt 2}$};
  % phase arc at origin from real axis down to the radius vector (-45 deg)
  \draw[engamber] (0.22, 0) arc (0:-45:0.22);
  \node[engamber, font=\footnotesize] at (0.33, -0.075) {$-45^{\circ}$};
  % centre dot
  \filldraw[enggray] (0.5, 0) circle (0.012);
  \node[enggray, font=\scriptsize, anchor=south] at (0.5, 0.02) {$\tfrac{1}{2}$};
\end{tikzpicture}
\caption[Low-pass frequency-response locus]{The frequency-response locus of a
first-order low-pass filter $H(i\omega) = 1/(1 + i\omega/\omega_c)$, traced in
the complex plane as the frequency runs from zero to infinity. The locus is the
lower half of the circle of radius $\tfrac{1}{2}$ centred at $(\tfrac{1}{2},
0)$: at $\omega = 0$ the response sits at $H = 1$ (unit gain, zero phase); at
the corner frequency $\omega = \omega_c$ it passes through $\tfrac{1}{2} -
\tfrac{1}{2}i$, where the gain has fallen to $1/\sqrt 2$ (the half-power point,
$-3\,$dB) and the phase reaches $-45^{\circ}$; as $\omega \to \infty$ the
response spirals into the origin, all gain lost and the phase approaching
$-90^{\circ}$. The single complex number $H(i\omega)$ carries both the gain
(its distance from the origin) and the phase shift (its argument) at every
frequency, and the whole steady-state behaviour of the filter is the shape of
this one curve.}
\label{fig:vol02ch03:lowpass-locus}
\end{figure}
