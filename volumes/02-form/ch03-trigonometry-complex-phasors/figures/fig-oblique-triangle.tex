% Figure: an oblique (non-right) triangle labelled for the law of sines and
% the law of cosines, in a surveying-resection setting. Two known stations A
% and B a baseline apart, an unknown point C sighted from both; the measured
% angles at A and B and the baseline c fix the triangle.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex, font=\footnotesize]
  % vertices
  \coordinate (A) at (0,0);
  \coordinate (B) at (5.2,0);
  \coordinate (C) at (3.7,2.7);
  % sides
  \draw[engblue, very thick] (A) -- (B) node[midway, below] {$c$ (baseline)};
  \draw[engblue, thick] (B) -- (C) node[midway, right] {$a$};
  \draw[engblue, thick] (C) -- (A) node[midway, above left] {$b$};
  % vertex dots and labels
  \filldraw[engblack] (A) circle (0.05) node[below left] {$A$};
  \filldraw[engblack] (B) circle (0.05) node[below right] {$B$};
  \filldraw[engred] (C) circle (0.05) node[above] {$C$ (unknown)};
  % interior angle arcs
  \draw[engamber] (A) ++(0:0.75) arc (0:36.1:0.75);
  \node[engamber] at ($(A)+(18:1.05)$) {$\alpha$};
  \draw[engamber] (B) ++(180:0.75) arc (180:119:0.75);
  \node[engamber] at ($(B)+(150:1.05)$) {$\beta$};
  \draw[engamber] (C) ++(-121:0.7) arc (-121:-165:0.7);
  \node[engamber] at ($(C)+(-143:1.0)$) {$\gamma$};
\end{tikzpicture}
\caption[An oblique triangle for the law of sines]{An oblique triangle, the
setting for the law of sines and the law of cosines. In the surveying
resection, the baseline $c = \overline{AB}$ between two known stations is
measured directly, and the interior angles $\alpha$ at $A$ and $\beta$ at $B$
are read from a sighting to the unknown point $C$. The third angle is
$\gamma = \pi - \alpha - \beta$, and the law of sines
$a/\sin\alpha = b/\sin\beta = c/\sin\gamma$ returns the two unknown ranges
$a$ and $b$. The law of cosines $c^{2} = a^{2} + b^{2} - 2ab\cos\gamma$ is
the companion identity when two sides and the included angle are known
instead. Neither identity requires a right angle; the right-triangle ratios
are the special case $\gamma = \pi/2$.}
\label{fig:vol02ch03:oblique-triangle}
\end{figure}
