% Figure: the magnitude of the impedance of a parallel RLC tank versus
% frequency, peaking at resonance, with the two half-power points that fix the
% bandwidth marked. Normalised so the peak is one and omega0 sits at x = 1.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=6.2cm,
  xlabel={$\omega/\omega_{0}$}, ylabel={$|Z|/R$},
  xmin=0.2, xmax=2.6, ymin=0, ymax=1.15,
  domain=0.2:2.6, samples=240,
  axis lines=left,
  xtick={0.5,1,1.5,2,2.5},
  ytick={0,0.5,0.707,1},
  yticklabels={$0$,$0.5$,$0.707$,$1$},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.98,0.98)}, anchor=north east, draw=none},
  clip=true,
]
  % Q = 5 tank: |Z|/R = 1 / sqrt( 1 + Q^2 (x - 1/x)^2 ), x = omega/omega0.
  \addplot[engblue, very thick] {1/sqrt( 1 + 25*(x - 1/x)^2 )};
  \addlegendentry{$Q = 5$}
  % half-power level
  \addplot[engred, dashed, thick, domain=0.2:2.6] {0.7071};
  \addlegendentry{half-power ($1/\sqrt{2}$)}
  % vertical guides at the two half-power frequencies (approx for Q=5)
  \draw[enggray, dotted] (axis cs:0.9050,0) -- (axis cs:0.9050,0.7071);
  \draw[enggray, dotted] (axis cs:1.1050,0) -- (axis cs:1.1050,0.7071);
  \node[engred, font=\footnotesize, anchor=south] at (axis cs:1.0,0.72) {$\Delta\omega$};
\end{axis}
\end{tikzpicture}
\caption[The parallel-RLC resonance curve and its half-power bandwidth]{The
normalised impedance magnitude of a parallel RLC tank against frequency, for a
quality factor $Q = 5$. The peak $|Z| = R$ occurs at the resonant frequency
$\omega_{0} = 1/\sqrt{LC}$, where the inductive and capacitive susceptances
cancel and the tank presents a pure resistance. The curve crosses the
half-power level $1/\sqrt{2}$ at the two frequencies whose spacing
$\Delta\omega = \omega_{0}/Q$ is the bandwidth. A high-$Q$ tank gives a tall,
narrow peak (a selective filter); a low-$Q$ tank gives a broad one. The
half-power points are the frequencies at which the reactive term equals the
resistive term in the impedance denominator, the same $\pm\pi/4$ phase
condition that the impedance triangle marks.}
\label{fig:vol02ch03:parallel-rlc-resonance}
\end{figure}
