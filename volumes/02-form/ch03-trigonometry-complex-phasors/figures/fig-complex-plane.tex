% Figure: complex plane (Argand diagram) showing rectangular and polar
% representations, addition by parallelogram, multiplication as rotation
% plus dilation, conjugate as reflection.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0, line cap=round, >=latex]
  % Left panel: rectangular vs polar
  \begin{scope}[xshift=-3.6cm]
    \draw[->, engblack, thin] (-0.6, 0) -- (4.0, 0) node[right, font=\small] {$\Re$};
    \draw[->, engblack, thin] (0, -0.6) -- (0, 3.7) node[above, font=\small] {$\Im$};
    \draw[engblue, thick, ->] (0, 0) -- (3, 2.5);
    \filldraw[engblue] (3, 2.5) circle (0.05);
    \node[engblue, font=\footnotesize, anchor=south west] at (3, 2.5) {$z = a + b i$};
    % rectangular projections
    \draw[engblack, dashed, thin] (3, 2.5) -- (3, 0) node[below, font=\footnotesize] {$a$};
    \draw[engblack, dashed, thin] (3, 2.5) -- (0, 2.5) node[left, font=\footnotesize] {$b$};
    % radius and angle
    \draw[engred, thick] (0.55, 0) arc (0:39.8:0.55);
    \node[engred, font=\footnotesize] at (0.85, 0.20) {$\theta$};
    \node[engred, font=\footnotesize, anchor=south east, rotate=39.8] at (1.5, 1.25) {$r = |z|$};
    \node[font=\small, anchor=north] at (1.7, -0.8) {(a) rectangular and polar};
  \end{scope}

  % Right panel: addition and conjugate
  \begin{scope}[xshift=3.0cm]
    \draw[->, engblack, thin] (-2.5, 0) -- (3.2, 0) node[right, font=\small] {$\Re$};
    \draw[->, engblack, thin] (0, -2.0) -- (0, 3.2) node[above, font=\small] {$\Im$};
    % z1 = 2 + 0.5i
    \draw[engblue, thick, ->] (0, 0) -- (2, 0.5);
    \node[engblue, font=\footnotesize, anchor=west] at (2.05, 0.45) {$z_{1}$};
    % z2 = 0.5 + 2i
    \draw[engblue, thick, ->] (0, 0) -- (0.5, 2);
    \node[engblue, font=\footnotesize, anchor=south] at (0.5, 2.05) {$z_{2}$};
    % sum z1 + z2 = 2.5 + 2.5i, parallelogram in dashed
    \draw[engblue, dashed, thin] (2, 0.5) -- (2.5, 2.5);
    \draw[engblue, dashed, thin] (0.5, 2) -- (2.5, 2.5);
    \draw[engred, thick, ->] (0, 0) -- (2.5, 2.5);
    \filldraw[engred] (2.5, 2.5) circle (0.05);
    \node[engred, font=\footnotesize, anchor=south west] at (2.5, 2.5) {$z_{1} + z_{2}$};
    % Conjugate: z = -1.5 + 1.5i and its conjugate -1.5 - 1.5i
    \filldraw[engblack] (-1.5, 1.5) circle (0.05);
    \filldraw[engblack] (-1.5, -1.5) circle (0.05);
    \draw[engblack, dotted] (-1.5, 1.5) -- (-1.5, -1.5);
    \node[font=\footnotesize, anchor=east] at (-1.55, 1.5) {$z$};
    \node[font=\footnotesize, anchor=east] at (-1.55, -1.5) {$\bar z$};
    \node[font=\small, anchor=north] at (0.5, -1.8) {(b) addition (red) and conjugate};
  \end{scope}
\end{tikzpicture}
\caption[Complex plane: rectangular, polar, addition, conjugate]{The
Argand diagram displays a complex number as a point in the plane.
Panel (a) shows the same number $z = a + b i = r e^{i\theta}$ in
rectangular and polar form; the radius $r$ is the modulus and the
angle $\theta$ is the argument, measured counterclockwise from the
positive real axis. Panel (b) shows the parallelogram rule for
addition (the sum $z_{1} + z_{2}$ in red) and the conjugate
$\bar z$ as reflection across the real axis. Multiplication is
the rotation plus dilation construction of figure
\ref{fig:vol02:ch03:rotation}.}
\label{fig:vol02:ch03:complex-plane}
\end{figure}
