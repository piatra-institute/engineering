% Figure: the symmetrical-components decomposition of an unbalanced
% three-phase voltage set into positive-, negative-, and zero-sequence
% phasor stars. Each star is drawn separately and the unbalanced set is
% drawn as the superposition.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.05, line cap=round, >=latex, font=\footnotesize]
  % --- Positive sequence: balanced, abc rotation ---
  \begin{scope}[shift={(0,0)}]
    \node[anchor=south] at (0,1.5) {positive seq.};
    \draw[engblack!30, thin] (-1.3,0) -- (1.3,0);
    \draw[engblack!30, thin] (0,-1.3) -- (0,1.3);
    \draw[engblue, thick, ->] (0,0) -- (90:1.0);
    \draw[engblue, thick, ->] (0,0) -- (-30:1.0);
    \draw[engblue, thick, ->] (0,0) -- (210:1.0);
    \node[engblue] at (90:1.18) {$a_1$};
    \node[engblue] at (-30:1.20) {$b_1$};
    \node[engblue] at (210:1.20) {$c_1$};
  \end{scope}
  % --- Negative sequence: balanced, acb rotation ---
  \begin{scope}[shift={(3.4,0)}]
    \node[anchor=south] at (0,1.5) {negative seq.};
    \draw[engblack!30, thin] (-1.3,0) -- (1.3,0);
    \draw[engblack!30, thin] (0,-1.3) -- (0,1.3);
    \draw[engred, thick, ->] (0,0) -- (90:0.55);
    \draw[engred, thick, ->] (0,0) -- (210:0.55);
    \draw[engred, thick, ->] (0,0) -- (-30:0.55);
    \node[engred] at (90:0.74) {$a_2$};
    \node[engred] at (210:0.78) {$b_2$};
    \node[engred] at (-30:0.78) {$c_2$};
  \end{scope}
  % --- Zero sequence: three equal phasors, all aligned ---
  \begin{scope}[shift={(6.8,0)}]
    \node[anchor=south] at (0,1.5) {zero seq.};
    \draw[engblack!30, thin] (-1.3,0) -- (1.3,0);
    \draw[engblack!30, thin] (0,-1.3) -- (0,1.3);
    \draw[engblack, thick, ->] (0,0) -- (60:0.45);
    \draw[engblack, thick, ->] (0.05,0.02) -- ($(0.05,0.02)+(60:0.45)$);
    \draw[engblack, thick, ->] (-0.05,-0.02) -- ($(-0.05,-0.02)+(60:0.45)$);
    \node[engblack] at (60:0.70) {$a_0\!=\!b_0\!=\!c_0$};
  \end{scope}
\end{tikzpicture}
\caption[Symmetrical components of a three-phase set]{The
symmetrical-components decomposition resolves any three-phase phasor set
into three balanced sets. The positive sequence (blue) is a balanced set
with the original $a$-$b$-$c$ rotation; the negative sequence (red) is a
balanced set with the reversed $a$-$c$-$b$ rotation; the zero sequence
(black) is three equal in-phase phasors. A balanced supply has only
positive-sequence content. A single-line-to-ground fault excites all
three sequences in equal measure at the fault point, which is the
algebraic fact that makes the sequence networks connect in series for
that fault.}
\label{fig:vol02:ch03:symmetrical-components}
\end{figure}
