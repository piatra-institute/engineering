% Figure: impulse response and step response of the second-order
% overdamped system of section 7.5, showing that the step response is
% the running integral of the impulse response.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6cm,
  xmin=0, xmax=6, ymin=0, ymax=2.2,
  xlabel={$t$}, ylabel={response},
  xtick={0,1,2,3,4,5,6},
  ytick={0,0.5,1,1.5,2},
  label style={font=\small}, tick label style={font=\scriptsize},
  grid=major, grid style={enggray!25},
  legend style={font=\scriptsize, at={(0.97,0.5)}, anchor=east, draw=enggray!60},
]
% impulse response h(t) = 0.5 (e^{-t} - e^{-3t})
\addplot[engred, very thick, domain=0:6, samples=160] {0.5*(exp(-x) - exp(-3*x))};
\addlegendentry{$h(t)=\tfrac12(e^{-t}-e^{-3t})$}
% step response (unit-step forcing) g(t) = integral of h = (1/6)(3 - ... )
% For x'' + 4x' + 3x = u(t): x = 1/3 - (1/2)e^{-t} + (1/6)e^{-3t}
\addplot[engblue, very thick, domain=0:6, samples=160] {1/3 - 0.5*exp(-x) + (1/6)*exp(-3*x)};
\addlegendentry{unit-step response}
% steady value of the unit-step response
\draw[densely dotted, enggray] (axis cs:0,0.3333) -- (axis cs:6,0.3333);
\node[font=\scriptsize, enggray, anchor=west] at (axis cs:4.2,0.40) {$1/3$};
% mark the impulse-response peak at t = (1/2)ln3 approx 0.549
\draw[densely dashed, enggray] (axis cs:0.549,0) -- (axis cs:0.549,0.385);
\node[font=\scriptsize, engred, anchor=south west] at (axis cs:0.6,0.39) {peak};
\end{axis}
\end{tikzpicture}
\caption[Impulse response and step response of a second-order system]{The
impulse response $h(t)$ and the unit-step response of the system
$\ddot{x} + 4\dot{x} + 3x = f(t)$ with characteristic roots $-1$ and
$-3$. The impulse response starts at zero, because a finite inertia
cannot move instantaneously, rises to a peak at $t = \tfrac12\ln 3
\approx 0.55$, and decays along the slow mode $e^{-t}$. The step
response is the running integral of the impulse response and climbs
monotonically to the static value $1/3 = 1/k$. Reading one plot off
the other is the time-domain statement that the step is the integral of
the impulse, the same relation the Laplace algebra delivers as division
by $s$.}
\label{fig:vol02:ch07:impulse-step}
\end{figure}
