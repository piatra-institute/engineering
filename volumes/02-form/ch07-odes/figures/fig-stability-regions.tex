% forward Euler, implicit Euler, and RK4.

\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=6.4cm,
  axis equal image,
  xmin=-4, xmax=3, ymin=-3.2, ymax=3.2,
  xlabel={$\operatorname{Re}(z)$}, ylabel={$\operatorname{Im}(z)$},
  xtick={-4,-3,-2,-1,0,1,2},
  ytick={-3,-2,-1,0,1,2,3},
  label style={font=\small}, tick label style={font=\scriptsize},
  axis lines=middle,
  grid=major, grid style={enggray!20},
  legend style={font=\scriptsize, at={(0.02,0.98)}, anchor=north west, draw=enggray!60},
]
% Implicit Euler stable region is exterior of disc |z-1|=1; shade the
% left half-plane lightly to indicate A-stability coverage.
\fill[engamber!12] (axis cs:-4,-3.2) rectangle (axis cs:0,3.2);
% Forward Euler: |1+z| <= 1, disc radius 1 centred at -1.
\addplot[engblue, very thick, domain=0:360, samples=100] ({-1 + cos(x)}, {sin(x)});
\addlegendentry{forward Euler}
% Implicit Euler boundary: |z-1| = 1, disc radius 1 centred at +1
% (stable region is the EXTERIOR of this disc).
\addplot[engred, very thick, densely dashed, domain=0:360, samples=100] ({1 + cos(x)}, {sin(x)});
\addlegendentry{implicit Euler (bd.)}
% RK4 stability boundary |R(z)|=1, R(z)=1+z+z^2/2+z^3/6+z^4/24.
% Parametric boundary curve (precomputed sample points), upper half then
% mirror is implicit by symmetry; draw full closed loop.
\addplot[engblack, very thick, smooth] coordinates {
  (0,0)(0.4,0.9)(0.55,1.6)(0.4,2.2)(0,2.55)(-0.6,2.65)(-1.3,2.5)
  (-1.9,2.0)(-2.3,1.35)(-2.55,0.75)(-2.75,0.2)(-2.78,0)
  (-2.75,-0.2)(-2.55,-0.75)(-2.3,-1.35)(-1.9,-2.0)(-1.3,-2.5)
  (-0.6,-2.65)(0,-2.55)(0.4,-2.2)(0.55,-1.6)(0.4,-0.9)(0,0)};
\addlegendentry{RK4}
\end{axis}
\end{tikzpicture}
\caption[Stability regions of three one-step methods]{Stability regions
in the complex plane $z = h\lambda$ for forward Euler, implicit Euler,
and classical RK4. A mode $\lambda$ integrated at step $h$ stays bounded
only where $z$ lies in the method's region. Forward Euler's region is
the unit disc centred at $-1$, so a fast mode ($|\lambda|$ large) forces
a small $h$. RK4's region (solid black) is the fourth-order truncation
of $e^{z}$, reaching to about $z = -2.78$ on the negative real axis and
bulging into the complex plane, larger than Euler's but still bounded.
Implicit Euler is stable everywhere \emph{outside} the dashed disc
centred at $+1$, which contains the entire left half-plane (shaded): it
is A-stable, and every decaying mode integrates stably at any step size.
The bounded explicit regions against the unbounded implicit one are the
whole reason stiff problems demand implicit methods.}
\label{fig:vol02:ch07:stability-regions}
\end{figure}
