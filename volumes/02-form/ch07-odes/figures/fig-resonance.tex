% Figure: frequency-response amplitude curves of the forced second-order
% oscillator for several damping ratios, normalised to static deflection.
% A(omega)/(F0/k) = 1/sqrt((1-r^2)^2 + (2 zeta r)^2), r = omega/omega0.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={$\omega/\omega_{0}$}, ylabel={$A\,/\,(F_{0}/k)$},
  xmin=0, xmax=2.5, ymin=0, ymax=5.5,
  axis lines=left,
  domain=0.001:2.5, samples=400,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% zeta = 0.1 -> peak ~ 1/(2 zeta) = 5.025
\addplot[engred, very thick] {1/sqrt((1-x^2)^2 + (2*0.1*x)^2)};
\addlegendentry{$\zeta=0.10$ ($Q=5$)}
% zeta = 0.25
\addplot[engorange, very thick] {1/sqrt((1-x^2)^2 + (2*0.25*x)^2)};
\addlegendentry{$\zeta=0.25$ ($Q=2$)}
% zeta = 0.5
\addplot[engblue, very thick] {1/sqrt((1-x^2)^2 + (2*0.5*x)^2)};
\addlegendentry{$\zeta=0.50$}
% zeta = 1/sqrt2 ~ 0.707, the flattest (maximally flat, no peak)
\addplot[engamber, very thick, dashed] {1/sqrt((1-x^2)^2 + (2*0.7071*x)^2)};
\addlegendentry{$\zeta=0.71$ (no peak)}
% static deflection line
\addplot[enggray, thin] {1};
\node[enggray, font=\scriptsize, anchor=south west] at (axis cs:0.02,1.02) {static deflection};
\end{axis}
\end{tikzpicture}
\caption[Resonance: frequency response of the forced
oscillator]{Steady-state amplitude of $m\ddot{x}+c\dot{x}+kx=F_{0}
\cos(\omega t)$, normalised to the static deflection $F_{0}/k$, as a
function of the frequency ratio $\omega/\omega_{0}$, for four damping
ratios. Light damping ($\zeta=0.1$, red) produces a sharp resonant
peak of height $\approx 1/(2\zeta)=Q$ just below $\omega/\omega_{0}=1$,
at $\omega_{r}=\omega_{0}\sqrt{1-2\zeta^{2}}$. As damping increases
the peak lowers and broadens. At $\zeta=1/\sqrt{2}\approx 0.71$
(amber dashed) the response is maximally flat with no peak at all,
the Butterworth condition prized in filter design. All curves
approach the static deflection at low frequency and roll off as
$\omega^{-2}$ ($-40\,\text{dB}$ per decade) at high frequency. The
peak height is the working currency of resonant-system design: a
$Q=100$ system deflects one hundred times its static value at
resonance. Computed from
$A/(F_{0}/k)=[(1-r^{2})^{2}+(2\zeta r)^{2}]^{-1/2}$ with
$r=\omega/\omega_{0}$.}
\label{fig:vol02:ch07:resonance}
\end{figure}
