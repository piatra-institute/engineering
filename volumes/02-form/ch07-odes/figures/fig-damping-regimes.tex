% Figure: the three damping regimes of the second-order homogeneous
% oscillator, on a normalised time axis. Overdamped, critically damped,
% underdamped responses to the same initial displacement, zero velocity.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7cm,
  xlabel={$\omega_{0} t$}, ylabel={$x/x_{0}$},
  xmin=0, xmax=14, ymin=-0.45, ymax=1.05,
  axis lines=left,
  domain=0:14, samples=300,
  legend pos=north east,
  legend cell align=left,
  every axis x label/.style={at={(current axis.right of origin)}, anchor=west},
  every axis y label/.style={at={(current axis.above origin)}, anchor=south},
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend style={font=\scriptsize},
]
% Overdamped, zeta = 2: roots lambda = (-zeta +- sqrt(zeta^2-1)) omega0.
% lambda1 = (-2+sqrt(3)) = -0.2679, lambda2 = (-2-sqrt(3)) = -3.732.
% x = A e^{l1 t} + B e^{l2 t}, x(0)=1, xdot(0)=0:
% A = -l2/(l1-l2), B = l1/(l1-l2). l1-l2 = 2 sqrt3 = 3.4641.
% A = 3.732/3.4641 = 1.0774, B = -0.2679/3.4641 = -0.07735.
\addplot[engamber, very thick] {1.0774*exp(-0.2679*x) - 0.07735*exp(-3.732*x)};
\addlegendentry{overdamped $\zeta=2$}
% Critically damped, zeta = 1: x = (1 + omega0 t) e^{-omega0 t}, omega0=1.
\addplot[engblue, very thick, dashed] {(1 + x)*exp(-x)};
\addlegendentry{critical $\zeta=1$}
% Underdamped, zeta = 0.15, omega_d = sqrt(1-0.0225)=0.98869.
% x = e^{-0.15 t}(cos(wd t) + (zeta/wd) sin(wd t)), zeta/wd = 0.1517.
\addplot[engred, very thick] {exp(-0.15*x)*(cos(0.98869*x) + 0.1517*sin(0.98869*x))};
\addlegendentry{underdamped $\zeta=0.15$}
% zero line
\addplot[enggray, thin] {0};
\end{axis}
\end{tikzpicture}
\caption[The three damping regimes of the second-order
oscillator]{Free response of $m\ddot{x}+c\dot{x}+kx=0$ from the
initial condition $x(0)=x_{0}$, $\dot{x}(0)=0$, on the normalised
time axis $\omega_{0}t$, for three damping ratios. The overdamped
case ($\zeta=2$, amber) returns to equilibrium slowly without
crossing zero, governed by the slower of two real exponentials. The
critically damped case ($\zeta=1$, blue dashed) is the fastest
return with no overshoot, the repeated-root response
$x_{0}(1+\omega_{0}t)e^{-\omega_{0}t}$. The underdamped case
($\zeta=0.15$, red) overshoots and rings, an exponentially decaying
sinusoid at the damped frequency $\omega_{d}=\omega_{0}\sqrt{1-\zeta^{2}}$.
Critical damping is the design target whenever overshoot is
unacceptable and settling must be fast: instrument needles,
door closers, vehicle suspensions tuned for ride rather than
handling. Computed from the closed-form solutions of the
characteristic polynomial.}
\label{fig:vol02:ch07:damping-regimes}
\end{figure}
