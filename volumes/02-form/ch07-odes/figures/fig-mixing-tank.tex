\begin{tikzpicture}

  axis lines=left,

  legend pos=south east,

% Tank concentration approaching inlet: c/c_in = 1 - e^{-t/tau}
\addplot[engblue, very thick] {1 - exp(-x)};
\addlegendentry{tank conc.\ $c/c_{\mathrm{in}}=1-e^{-t/\tau}$}
% Newton cooling toward ambient: theta = (T-T_inf)/(T0-T_inf) = e^{-t/tau}
\addplot[engred, very thick] {exp(-x)};
\addlegendentry{cooling $\frac{T-T_{\infty}}{T_{0}-T_{\infty}}=e^{-t/\tau}$}

\caption[First-order step response: mixing tank and Newton
cooling]{Two first-order linear systems sharing one mathematical
form. A perfectly stirred tank whose inlet concentration steps up at
$t=0$ approaches the inlet value as $c/c_{\mathrm{in}}=1-e^{-t/\tau}$
(blue), with $\tau=V/Q$ the residence time (tank volume over
volumetric flow). An object cooling toward ambient follows Newton's
law of cooling, $(T-T_{\infty})/(T_{0}-T_{\infty})=e^{-t/\tau}$
(red), with $\tau=mc_{p}/(hA)$ the thermal time constant. Both cross
the $63\,\%$ mark at one time constant and settle to within $1\,\%$
in five. The shared form is the engineering point: residence time,
thermal time constant, and the RC time constant are the same object
wearing three sets of units, and the reader who has internalised one
reads all three. Computed from the first-order step and decay
responses.}
\label{fig:vol02:ch07:mixing-tank}
\end{figure}
