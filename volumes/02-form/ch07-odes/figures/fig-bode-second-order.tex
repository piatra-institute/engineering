% Figure: magnitude and phase of the second-order low-pass transfer
% function versus normalised frequency, log axes, for several damping
% ratios.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{semilogxaxis}[
  name=mag,
  width=10.5cm, height=4.6cm,
  xmin=0.1, xmax=10, ymin=-42, ymax=22,
  xlabel={$\omega/\omega_{0}$}, ylabel={$|H|$ (\si{\decibel})},
  xtick={0.1,0.2,0.5,1,2,5,10},
  xticklabels={$0.1$,$0.2$,$0.5$,$1$,$2$,$5$,$10$},
  ytick={-40,-20,0,20},
  label style={font=\small}, tick label style={font=\scriptsize},
  grid=both, grid style={enggray!25},
  legend style={font=\scriptsize, at={(0.02,0.06)}, anchor=south west, draw=enggray!60},
]
% |H| = 1 / sqrt( (1-r^2)^2 + (2 zeta r)^2 ), in dB
\addplot[engred, very thick, domain=0.1:10, samples=160] {10*log10(1/((1-x^2)^2 + (2*0.1*x)^2))};
\addlegendentry{$\zeta=0.1$}
\addplot[engblue, very thick, domain=0.1:10, samples=160] {10*log10(1/((1-x^2)^2 + (2*0.3*x)^2))};
\addlegendentry{$\zeta=0.3$}
\addplot[engamber, very thick, domain=0.1:10, samples=160] {10*log10(1/((1-x^2)^2 + (2*0.707*x)^2))};
\addlegendentry{$\zeta=0.707$}
\end{semilogxaxis}
\begin{semilogxaxis}[
  at={(mag.below south west)}, anchor=north west, yshift=-0.6cm,
  width=10.5cm, height=4.0cm,
  xmin=0.1, xmax=10, ymin=0, ymax=180,
  xlabel={$\omega/\omega_{0}$}, ylabel={phase (\si{\degree})},
  xtick={0.1,0.2,0.5,1,2,5,10},
  xticklabels={$0.1$,$0.2$,$0.5$,$1$,$2$,$5$,$10$},
  ytick={0,45,90,135,180},
  label style={font=\small}, tick label style={font=\scriptsize},
  grid=both, grid style={enggray!25},
]
% phase lag = atan2(2 zeta r, 1 - r^2), degrees; pgf atan2 gives (-180,180]
\addplot[engred, very thick, domain=0.1:10, samples=200] {atan2(2*0.1*x, 1-x^2) < 0 ? atan2(2*0.1*x, 1-x^2)+180 : atan2(2*0.1*x, 1-x^2)};
\addplot[engblue, very thick, domain=0.1:10, samples=200] {atan2(2*0.3*x, 1-x^2) < 0 ? atan2(2*0.3*x, 1-x^2)+180 : atan2(2*0.3*x, 1-x^2)};
\addplot[engamber, very thick, domain=0.1:10, samples=200] {atan2(2*0.707*x, 1-x^2) < 0 ? atan2(2*0.707*x, 1-x^2)+180 : atan2(2*0.707*x, 1-x^2)};
\draw[densely dotted, enggray] (axis cs:1,0) -- (axis cs:1,90);
\end{semilogxaxis}
\end{tikzpicture}
\caption[Frequency response of the second-order low-pass transfer
function]{Magnitude (top) and phase lag (bottom) of the second-order
low-pass transfer function $H(i\omega) = \omega_{0}^{2}/[(\omega_{0}^{2}
- \omega^{2}) + 2i\zeta\omega_{0}\omega]$ against normalised frequency
$\omega/\omega_{0}$ on logarithmic axes, for three damping ratios. The
lightly damped case peaks near $\omega_{0}$ with a gain of about $Q =
1/(2\zeta)$; the $\zeta = 0.707$ case is the maximally flat Butterworth
response with no peak. Every curve rolls off at $-40\,\si{\decibel}$
per decade far above resonance, the $\omega^{-2}$ inertial limit, and
every phase lag passes through exactly $90^{\circ}$ at $\omega =
\omega_{0}$ regardless of damping. This is the same amplitude curve as
the forced mass-spring-dashpot and the series RLC circuit, read in the
frequency domain rather than the time domain.}
\label{fig:vol02:ch07:bode-second-order}
\end{figure}
