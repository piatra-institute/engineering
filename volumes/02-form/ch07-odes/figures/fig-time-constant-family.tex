% Figure: first-order step response for a family of time constants,
% showing that the single-time-constant template rescales in time
% without changing shape.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  fast/.style={draw=engred, very thick},
  mid/.style={draw=engblue, very thick},
  slow/.style={draw=engamber, very thick},
]
% Axes: t from 0 to 10 s; y/yinf from 0 to 1.1
\draw[axis] (-0.2,0) -- (10.6,0) node[anchor=west, font=\small] {$t\ (\si{\second})$};
\draw[axis] (0,-0.1) -- (0,1.25) node[anchor=south, font=\small] {$y/y_{\infty}$};
% Final-value line
\draw[densely dotted, enggray] (0,1) -- (10,1);
\node[font=\scriptsize, left] at (-0.1,1) {$1$};
\foreach \x in {2,4,6,8,10} {
  \draw (\x,-0.06) -- (\x,0.06);
  \node[font=\scriptsize, below] at (\x,-0.1) {\x};
}
% 63% marker
\draw[densely dotted, enggray] (0,0.632) -- (10,0.632);
\node[font=\scriptsize, left] at (-0.1,0.632) {$0.63$};
% tau = 0.5 (fast)
\draw[fast] plot[smooth, domain=0:10, samples=120] (\x, {1 - exp(-\x/0.5)});
\node[font=\scriptsize, engred, anchor=west] at (2.1,1.06) {$\tau=0.5\,\si{\second}$};
% tau = 2 (mid)
\draw[mid] plot[smooth, domain=0:10, samples=120] (\x, {1 - exp(-\x/2)});
\node[font=\scriptsize, engblue, anchor=west] at (4.4,0.86) {$\tau=2\,\si{\second}$};
% tau = 5 (slow)
\draw[slow] plot[smooth, domain=0:10, samples=120] (\x, {1 - exp(-\x/5)});
\node[font=\scriptsize, engamber, anchor=west] at (7.2,0.62) {$\tau=5\,\si{\second}$};
% mark the tau crossing at 63% for the mid curve
\draw[densely dashed, enggray] (2,0) -- (2,0.632);
\end{tikzpicture}
\caption[First-order step response for a family of time constants]{The
first-order linear step response $y(t) = y_{\infty}(1 - e^{-t/\tau})$
for three time constants. Every curve reaches $63\,\%$ of its final
value at one time constant (the dashed marker locates $t = \tau$ for
the middle curve), $95\,\%$ at three, and $99\,\%$ at five, regardless
of the value of $\tau$. Changing $\tau$ rescales the horizontal axis
and leaves the shape fixed. The RC charge, the mixing-tank salt mass,
the cooling body, and the RL current all trace this one curve in their
own units, which is the working content of the transport archetype:
one template, three domains, a single number $\tau$ to read off the
coefficient.}
\label{fig:vol02:ch07:time-constant-family}
\end{figure}
