% Figure: an adaptive integrator's step size tracking the local time scale
% of a solution with one fast transient followed by a slow tail. Upper axis:
% the solution y(t); lower axis: the step size h_n the controller selects,
% small during the transient and large on the smooth tail.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  name=top,
  width=12cm, height=4.6cm,
  xlabel={}, ylabel={solution $y(t)$},
  xmin=0, xmax=10, ymin=-0.05, ymax=1.05,
  xtick=\empty,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% y(t) = exp(-6 t) + 0.9*(1 - exp(-0.4 t)): fast decay plus slow rise
\addplot[engblue, very thick, domain=0:10, samples=200] {exp(-6*x) + 0.9*(1 - exp(-0.4*x))};
\node[engblue, font=\scriptsize, anchor=west] at (axis cs:3.2,0.35) {fast transient, then slow tail};
\end{axis}
\begin{axis}[
  at={(top.below south west)}, anchor=above north west,
  width=12cm, height=4.6cm,
  xlabel={time $t$}, ylabel={step size $h_n$},
  xmin=0, xmax=10, ymin=1e-3, ymax=1,
  ymode=log,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
]
% accepted step sizes: small during transient (t < ~1), growing on the tail
\addplot[engorange, very thick, mark=*, mark size=1.1pt, mark options={fill=engorange}]
  coordinates {
  (0.02, 0.01) (0.06, 0.012) (0.12, 0.016) (0.20, 0.022)
  (0.32, 0.035) (0.50, 0.06) (0.75, 0.11) (1.10, 0.20)
  (1.7, 0.35) (2.6, 0.55) (3.9, 0.75) (5.6, 0.85)
  (7.4, 0.9) (9.2, 0.9)
};
\node[engorange, font=\scriptsize, anchor=south west] at (axis cs:0.05,0.02) {small $h$ on the transient};
\node[engorange, font=\scriptsize, anchor=south east] at (axis cs:9.0,0.45) {large $h$ on the tail};
\end{axis}
\end{tikzpicture}
\caption[Adaptive step size tracking the local time scale]{An adaptive
integrator on a problem with one fast transient followed by a slow
tail, $y(t)=e^{-6t}+0.9\,(1-e^{-0.4t})$. The top panel shows the
solution; the bottom panel shows the step size the controller accepts
at each point, on a logarithmic axis. Where the solution turns
sharply the local error estimate is large, so the controller shrinks
$h$ to hold the per-step error at tolerance; where the solution is
smooth the estimate is small and the controller lets $h$ grow by more
than two orders of magnitude. A fixed-step method must run the whole
interval at the smallest step the transient demands, spending most of
its work resolving a tail that needs almost none. The step-size trace
is the controller reading the solution's own time scale off the local
error and spending arithmetic where the solution earns it.}
\label{fig:vol02:ch07:adaptive-step}
\end{figure}
