% Figure: velocity of a body falling under gravity with quadratic drag,
% approaching terminal velocity. Shown in dimensionless form: v/v_inf against
% t/tau, with the linear-drag (exponential) approach drawn for contrast.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=12cm, height=7.5cm,
  xlabel={dimensionless time $t/\tau$}, ylabel={$v/v_\infty$},
  xmin=0, xmax=4, ymin=0, ymax=1.08,
  axis lines=left,
  grid=major, grid style={enggray!20},
  tick label style={font=\scriptsize},
  label style={font=\small},
  legend pos=south east,
  legend cell align=left,
  legend style={font=\scriptsize},
]
% quadratic drag: v/v_inf = tanh(t/tau)
\addplot[engblue, very thick, domain=0:4, samples=200] {tanh(x)};
\addlegendentry{quadratic drag: $\tanh(t/\tau)$}
% linear drag for contrast: 1 - exp(-t/tau)
\addplot[engorange, very thick, dashed, domain=0:4, samples=200] {1 - exp(-x)};
\addlegendentry{linear drag: $1-e^{-t/\tau}$}
% terminal-velocity asymptote
\addplot[enggray, thin, dotted, domain=0:4] {1};
\node[enggray, font=\scriptsize, anchor=south east] at (axis cs:4,1.0) {$v_\infty$};
% marker: quadratic reaches 0.964 at t/tau = 2
\addplot[engblue, only marks, mark=*, mark size=1.3pt, mark options={fill=engblue}]
  coordinates {(2,0.964)};
\node[engblue, font=\scriptsize, anchor=north west] at (axis cs:2.05,0.9) {$96\%$ at $t=2\tau$};
\end{axis}
\end{tikzpicture}
\caption[Approach to terminal velocity under quadratic drag]{Velocity
of a body falling under gravity with drag, in dimensionless form: the
ratio $v/v_\infty$ against time in units of the drag time scale
$\tau=v_\infty/g$. Quadratic drag (blue), the high-Reynolds regime of a
skydiver or a hailstone, gives $v/v_\infty=\tanh(t/\tau)$; linear drag
(orange dashed), the low-Reynolds regime of a fine droplet or a
falling sphere in oil, gives $1-e^{-t/\tau}$. Both climb to the same
terminal velocity $v_\infty$, the value at which drag balances weight
and the acceleration vanishes, but the quadratic law approaches it
faster once the speed is an appreciable fraction of $v_\infty$, because
its retarding force grows with the square of the speed rather than
linearly. The nondimensional form collapses every mass, area, and
fluid density into the two scales $v_\infty$ and $\tau$, so one curve
serves every falling body in its drag regime.}
\label{fig:vol02:ch07:terminal-velocity}
\end{figure}
