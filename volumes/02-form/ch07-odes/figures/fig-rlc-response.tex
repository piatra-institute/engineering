% Figure: series RLC step response in the three damping regimes,
% capacitor voltage versus time, normalised to the final value.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=engblack, -{Stealth}, thick},
  over/.style={draw=engblue, very thick},
  crit/.style={draw=engblack, very thick, densely dashed},
  under/.style={draw=engred, very thick},
  label/.style={font=\small},
]
% Axes: tau = omega0 * t from 0 to 12; v/Vfinal from 0 to 1.7
\draw[axis] (-0.2,0) -- (12.4,0) node[anchor=west, label] {$\omega_{0} t$};
\draw[axis] (0,-0.1) -- (0,1.85) node[anchor=south, label] {$v_{C}/V_{0}$};
% Final-value line
\draw[densely dotted, enggray] (0,1) -- (12,1);
\node[font=\scriptsize, left] at (-0.1,1) {$1$};
\foreach \x in {2,4,6,8,10,12} {
  \draw (\x,-0.06) -- (\x,0.06);
  \node[font=\scriptsize, below] at (\x,-0.1) {\x};
}
% Overdamped, zeta = 2: vc/V0 = 1 - (r2 e^{r1 t} - r1 e^{r2 t})/(r2-r1)
% with r1,2 = -zeta +- sqrt(zeta^2-1) (in units of omega0). zeta=2 -> r=-0.268,-3.73
\draw[over] plot[smooth, domain=0:12, samples=120]
  (\x, {1 - (-3.732*exp(-0.268*\x) + 0.268*exp(-3.732*\x))/(-3.464)});
\node[font=\scriptsize, engblue, anchor=west] at (8.2,0.78) {overdamped ($\zeta=2$)};
% Critically damped, zeta = 1: vc/V0 = 1 - (1 + t) e^{-t}
\draw[crit] plot[smooth, domain=0:12, samples=120]
  (\x, {1 - (1 + \x)*exp(-\x)});
\node[font=\scriptsize, engblack, anchor=west] at (5.3,1.06) {critical ($\zeta=1$)};
% Underdamped, zeta = 0.2: vc/V0 = 1 - e^{-zeta t}(cos wd t + (zeta/wd) sin wd t)
% wd = sqrt(1-0.04) = 0.98; zeta/wd = 0.204
\draw[under] plot[smooth, domain=0:12, samples=200]
  (\x, {1 - exp(-0.2*\x)*(cos(0.98*\x r) + 0.204*sin(0.98*\x r))});
\node[font=\scriptsize, engred, anchor=south] at (3.2,1.55) {underdamped ($\zeta=0.2$)};
\end{tikzpicture}
\caption[Series RLC step response across damping regimes]{Step
response of the series RLC circuit: capacitor voltage $v_{C}$ rising
toward the source voltage $V_{0}$ after a switch closes at $t = 0$,
normalised to $V_{0}$ and plotted against the dimensionless time
$\omega_{0} t$ with $\omega_{0} = 1/\sqrt{LC}$. The overdamped case
($\zeta = 2$) rises slowly with no overshoot; the critically damped
case ($\zeta = 1$) is the fastest monotone rise; the underdamped case
($\zeta = 0.2$) overshoots its final value and rings before settling.
The first overshoot of the underdamped response is $\exp(-\pi\zeta/
\sqrt{1-\zeta^{2}})$ above the final value, here about $53\,\%$. The
same three curves describe a mass-spring-dashpot displacement, an op-
amp step response, and any second-order linear system.}
\label{fig:vol02:ch07:rlc-response}
\end{figure}
