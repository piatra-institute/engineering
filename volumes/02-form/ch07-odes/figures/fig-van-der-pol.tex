% The limit cycle of van der Pol with mu = 1, drawn as a closed curve
% approximated by a smooth ovoid passing through the known turning

  (2.0,0.0) (1.7,1.6) (0.6,2.55) (-0.9,2.0) (-2.0,0.6)
  (-2.0,-0.05) (-1.7,-1.6) (-0.6,-2.55) (0.9,-2.0) (2.0,-0.6)
};
% Trajectory starting inside (near origin) spiralling outward onto cycle

  (0.3,0.0) (0.25,0.55) (-0.1,0.7) (-0.55,0.4) (-0.6,-0.2)
  (-0.3,-0.8) (0.35,-1.0) (0.95,-0.6) (1.2,0.3) (0.9,1.35)

};

\caption[Van der Pol limit cycle]{Phase portrait of the van der Pol
oscillator $\ddot{x} - \mu(1 - x^{2})\dot{x} + x = 0$ at $\mu = 1$.
The origin is an unstable spiral: the linearisation has eigenvalues
with positive real part, so small perturbations grow. Far from the
origin the nonlinear damping term turns positive and removes energy,
so large trajectories shrink. The two effects balance on a single
isolated closed curve, the \emph{limit cycle}. Every trajectory in
the plane, whether it starts inside (blue) or outside (orange),
approaches the same cycle. The amplitude is set by the dynamics, not
by the initial condition, which distinguishes a limit cycle from the
amplitude-fixed-by-energy closed orbits of the conservative
centre.}
\label{fig:vol02:ch07:van-der-pol}
\end{figure}
