% Figure: Young's inequality read as an area picture. For a monotone
% increasing function y = f(x) through the origin, the product xy is
% covered by the area under f (from 0 to x) plus the area to the left of f
% (from 0 to y = f-inverse), with equality only when y = f(x). Taking
% f(x) = x^{p-1} recovers xy <= x^p/p + y^q/q.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
\begin{axis}[
  width=9.5cm, height=7cm,
  xlabel={$x$},
  ylabel={$y$},
  xmin=0, xmax=2.3,
  ymin=0, ymax=2.3,
  axis lines=left,
  xtick={1.5}, xticklabels={$a$},
  ytick={1.6}, yticklabels={$b$},
  tick label style={font=\footnotesize},
  label style={font=\small},
]
% the curve y = f(x) = x^{p-1}, here p = 2 so f(x) = x for a clean drawing,
% but we tilt to x^{1.5} to show the two unequal regions.
\addplot[engblack, very thick, domain=0:2.1, samples=60] {x^1.5};
\node[font=\footnotesize, engblack, anchor=south west]
  at (axis cs:1.75,2.05) {$y=f(x)$};
% region A: area under f from 0 to a (contributes x^p/p).
\addplot[engblue!22, draw=none, domain=0:1.5, samples=40] {x^1.5}
  \closedcycle;
\node[font=\small, anchor=center] at (axis cs:1.05,0.55)
  {$\displaystyle\int_{0}^{a}\! f$};
% region B: area to the left of f from 0 to b (contributes y^q/q).
% f(x)=x^1.5 => x = y^{1/1.5}=y^{0.667}. Fill between y-axis and curve for
% y in [0,b]. Draw as a band of horizontal extent x(y).
\addplot[engred!22, draw=none] coordinates {
  (0,0) (0,1.6) (1.363,1.6) (1.28,1.45) (1.17,1.28) (1.03,1.1)
  (0.86,0.9) (0.68,0.71) (0.47,0.5) (0.24,0.31) (0,0)
};
\node[font=\small, anchor=center] at (axis cs:0.42,1.15)
  {$\displaystyle\int_{0}^{b}\! f^{-1}$};
% the rectangle a-by-b outline; its area is the product ab.
\draw[engamber, thick, dashed] (axis cs:0,0) rectangle (axis cs:1.5,1.6);
\node[font=\footnotesize, engamber, anchor=north east]
  at (axis cs:1.48,1.55) {rectangle area $=ab$};
\end{axis}
\end{tikzpicture}
\caption[Young's inequality as an area balance]{Young's inequality read as
an area picture. For a monotone increasing $f$ through the origin, the
product $ab$ is the area of the dashed rectangle, and it never exceeds the
sum of the area under the curve (blue, $\int_{0}^{a} f$) and the area to
the left of the curve (red, $\int_{0}^{b} f^{-1}$). Equality holds only
when the corner $(a,b)$ sits on the curve, that is, $b=f(a)$. Taking
$f(x)=x^{p-1}$ evaluates the two areas to $a^{p}/p$ and $b^{q}/q$ with
$1/p+1/q=1$, giving $ab\le a^{p}/p + b^{q}/q$. The unweighted AM--GM bound
$ab\le\tfrac12(a^{2}+b^{2})$ is the case $p=q=2$.}
\label{fig:vol02:ch01:young-inequality}
\end{figure}
