% Figure: accumulated rounding error of a long running sum, naive
% left-to-right accumulation against compensated (Kahan) summation. The
% naive error grows roughly linearly with the number of terms n while the
% compensated error stays near the single-rounding floor, which is the
% visible form of the worst-case n u bound against the O(u) compensated
% bound.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={number of terms added $n$},
  ylabel={accumulated error (units of $u$)},
  xmode=log, ymode=log,
  xmin=100, xmax=1000000,
  ymin=0.3, ymax=200000,
  grid=both,
  grid style={engblack!12},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.03,0.97)}, anchor=north west},
]
% Naive: error grows like n u (worst case), drawn as a representative line.
\addplot[engred, thick, mark=square*, mark size=1.4pt] coordinates {
  (100,60) (1000,550) (10000,5200) (100000,52000) (1000000,150000)
};
% Compensated (Kahan): error stays near the O(1) rounding floor.
\addplot[engblue, thick, mark=*, mark size=1.6pt] coordinates {
  (100,1.1) (1000,1.3) (10000,1.5) (100000,1.7) (1000000,2.0)
};
% Reference line: the sqrt(n) typical growth of random-sign rounding.
\addplot[engorange, dashed, thick] coordinates {
  (100,10) (1000,32) (10000,100) (100000,316) (1000000,1000)
};
\legend{{naive left-to-right (grows $\sim n$)},
        {compensated (stays $\sim$ constant)},
        {$\sqrt{n}$ typical, for reference}}
\end{axis}
\end{tikzpicture}
\caption[Accumulated error of naive against compensated summation]{Accumulated
rounding error of a long running sum, in units of the unit roundoff $u$,
against the number of terms. Naive left-to-right accumulation drifts by
roughly $n u$ in the worst case, because each of the $n$ additions can add a
fresh rounding error of size $u$ times the running total. Compensated
summation carries a correction term that captures the low-order bits each
addition discards, holding the error near the single-rounding floor
regardless of $n$. The dashed line marks the $\sqrt{n}$ growth that
random-sign rounding produces on average, the typical case between the two
extremes. The three curves are the accumulation companions of the two
error-growth curves the triangle-inequality figure drew: worst case, typical
case, and a controlled case bought by a better algorithm.}
\label{fig:vol02:ch01:kahan-summation}
\end{figure}
