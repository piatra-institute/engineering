% Figure: the non-uniform spacing of floating-point numbers on the
% real line. Representable values cluster near zero and spread apart
% across each binade (power-of-two interval).
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  tick/.style={draw=engblack, thick},
  ann/.style={font=\footnotesize, align=center},
]
% Real line
\draw[->, thick] (0,0) -- (12,0) node[right] {$\R$};
\draw[thick] (0,-0.12) -- (0,0.12) node[above, ann] {$0$};

% Binade [1,2): 8 evenly spaced subticks over [3,5] cm
% Binade [2,4): 8 evenly spaced subticks over [5,9] cm (wider)
% Binade [0.5,1): 8 over [2,3] (denser)
% draw representable points as short ticks
\foreach \x in {2.0,2.125,...,3.0} {\draw[tick] (\x,-0.08)--(\x,0.08);}
\foreach \x in {3.0,3.25,...,5.0} {\draw[tick] (\x,-0.10)--(\x,0.10);}
\foreach \x in {5.0,5.5,...,9.0} {\draw[tick] (\x,-0.12)--(\x,0.12);}

% binade braces / labels
\draw[decorate, decoration={brace, amplitude=4pt}, engblue, thick]
  (3.0,0.35) -- (5.0,0.35);
\node[ann, engblue] at (4.0,0.75) {binade $[2^{1},2^{2})$};
\draw[decorate, decoration={brace, amplitude=4pt}, engred, thick]
  (5.0,0.35) -- (9.0,0.35);
\node[ann, engred] at (7.0,0.75) {binade $[2^{2},2^{3})$: gap doubles};

\node[ann] at (6.0,-0.85) {spacing $=2^{e}\cdot 2^{-p}$ across a binade with exponent $e$, $p$ mantissa bits};
\end{tikzpicture}
\caption[Floating-point numbers are not evenly spaced]{Representable
floating-point numbers are not evenly spaced on the real line. Within
each binade $[2^{e},2^{e+1})$ the spacing is constant at $2^{e}\cdot
2^{-p}$ for $p$ mantissa bits, and it doubles at every power of two.
The decimal $0.1$ falls between representable binary values and is
stored with a small error; accumulating that error is the mechanism
behind the Patriot Dhahran clock drift (Sec.~1.7). An algebraic
identity that is exact over $\R$ is only approximate over this grid.}
\label{fig:vol02:ch01:floating-point-line}
\end{figure}
