% Figure: the dependency structure of ordinary induction against strong
% induction. Ordinary induction proves P(n+1) from P(n) alone; strong
% induction proves P(n+1) from all of P(1),...,P(n). The prime-
% factorisation proof needs the strong form because it appeals to two
% smaller factors, neither of which need be the immediate predecessor.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  node distance=9mm and 12mm,
  box/.style={draw=engblack, thick, rounded corners=1pt, minimum width=8mm,
    minimum height=7mm, font=\footnotesize, fill=engblue!8},
  base/.style={draw=engblack, thick, rounded corners=1pt, minimum width=8mm,
    minimum height=7mm, font=\footnotesize, fill=englime!25},
  ar/.style={-{Stealth[length=2mm]}, engblue, thick},
  arS/.style={-{Stealth[length=2mm]}, engred, thick},
]
% ordinary induction row
\node[base] (o1) {$P(1)$};
\node[box, right=of o1] (o2) {$P(2)$};
\node[box, right=of o2] (o3) {$P(3)$};
\node[box, right=of o3] (o4) {$P(4)$};
\node[box, right=of o4] (o5) {$P(5)$};
\node[font=\footnotesize, left=6mm of o1, align=right] {ordinary\\induction};
\draw[ar] (o1) -- (o2);
\draw[ar] (o2) -- (o3);
\draw[ar] (o3) -- (o4);
\draw[ar] (o4) -- (o5);
% strong induction row, below
\node[base, below=16mm of o1] (s1) {$P(1)$};
\node[box, right=of s1] (s2) {$P(2)$};
\node[box, right=of s2] (s3) {$P(3)$};
\node[box, right=of s3] (s4) {$P(4)$};
\node[box, right=of s4] (s5) {$P(5)$};
\node[font=\footnotesize, left=6mm of s1, align=right] {strong\\induction};
% all predecessors feed P(5): straight neighbour plus curved long-range.
\draw[arS] (s4) -- (s5);
\draw[arS] (s3.south) to[bend right=28] (s5.south);
\draw[arS] (s2.south) to[bend right=34] (s5.south);
\draw[arS] (s1.south) to[bend right=40] (s5.south);
\end{tikzpicture}
\caption[Ordinary versus strong induction]{The dependency structure of the
two inductive forms. Ordinary induction (top) proves $P(n{+}1)$ from its
immediate predecessor $P(n)$ alone, so the argument advances one link at a
time. Strong induction (bottom) proves $P(n{+}1)$ from the whole
run $P(1),\ldots,P(n)$, drawn here as every earlier case feeding $P(5)$.
The green base case anchors both. The prime-factorisation theorem needs
the strong form: a composite $n{+}1=ab$ appeals to the factorisations of
$a$ and $b$, neither of which need be the immediate predecessor, so the
single arrow of ordinary induction does not reach them.}
\label{fig:vol02:ch01:strong-induction}
\end{figure}
