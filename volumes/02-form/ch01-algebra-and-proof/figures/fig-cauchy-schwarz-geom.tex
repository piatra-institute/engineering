% Figure: Cauchy-Schwarz drawn as the projection geometry of two
% vectors in the plane. The inequality is the statement that the
% projection cannot exceed the original vector's length.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  vec/.style={->, thick, draw=engblack},
  proj/.style={->, thick, draw=engblue, dashed},
  axis/.style={->, draw=engblack!50, thin},
  point/.style={circle, fill=engblack, inner sep=1.2pt},
  ann/.style={font=\small, align=left},
]
% Background axes
\draw[axis] (-0.3, 0) -- (5.2, 0);
\draw[axis] (0, -0.3) -- (0, 3.5);

% u and v vectors
\coordinate (O) at (0, 0);
\coordinate (U) at (4.5, 0.6);
\coordinate (V) at (2.3, 2.8);

\draw[vec] (O) -- (U) node[midway, below right, ann] {$u$};
\draw[vec] (O) -- (V) node[midway, above left, ann] {$v$};

% Projection of v onto u (foot of perpendicular)
% Direction u_hat = (4.5, 0.6) / |u|.
% proj = (<v,u>/|u|^2) * u  computed approximately for the diagram
\coordinate (P) at (2.41, 0.32);
\draw[proj] (O) -- (P)
  node[midway, below, ann, yshift=-3pt] {$\proj_{u}(v)$};
\draw[draw=engblack!50, thin, dashed] (V) -- (P);

\node[point] at (O) {};
\node[point] at (U) {};
\node[point] at (V) {};
\node[point] at (P) {};

% Angle theta
\draw[draw=engblack!70, thin]
  ([shift=(7.6:0.55cm)]O) arc[start angle=7.6, end angle=50.6, radius=0.55cm];
\node[ann] at (1.1, 0.55) {$\theta$};

% Inequality annotation block
\node[ann, anchor=west] at (5.6, 2.6)
  {$\inner{u, v} = \norm{u}\,\norm{v}\,\cos\theta$};
\node[ann, anchor=west] at (5.6, 1.9)
  {$\abs{\inner{u, v}} \leq \norm{u}\,\norm{v}$};
\node[ann, anchor=west] at (5.6, 1.2)
  {equality iff $u \parallel v$};
\node[ann, anchor=west] at (5.6, 0.5)
  {(Cauchy--Schwarz)};

\end{tikzpicture}
\caption[Cauchy--Schwarz as projection length]{The Cauchy--Schwarz
inequality reads geometrically as the bound on the length of the
projection of $v$ onto $u$. The dashed projection $\proj_{u}(v)$ has
length $\norm{v}\,\abs{\cos\theta}$, which never exceeds $\norm{v}$
itself; the inner product $\inner{u, v} = \norm{u}\,\norm{v}\,\cos
\theta$ therefore satisfies $\abs{\inner{u, v}} \leq \norm{u}\,
\norm{v}$, with equality precisely when $\theta = 0$ or $\theta =
\pi$ (the vectors are proportional).}
\label{fig:vol02:ch01:cauchy-schwarz}
\end{figure}
