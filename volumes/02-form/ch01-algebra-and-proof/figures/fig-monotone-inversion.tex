% Figure: a sensor response that is monotone (invertible) over its working
% span and a response that turns over (not invertible), showing why the
% inverse exists only where the response function is injective.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={temperature $T$},
  ylabel={sensor output $V$},
  xmin=0, xmax=10,
  ymin=0, ymax=6,
  grid=both,
  grid style={engblack!12},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.03,0.97)}, anchor=north west},
]
% monotone response: strictly increasing, invertible everywhere.
\addplot[engblue, very thick, domain=0:10, samples=60] {0.45*x + 0.4};
% non-monotone response: rises then turns over (a bare diode at low T).
\addplot[engred, very thick, domain=0:10, samples=80]
  {5.2 - 0.12*(x-3.2)^2};
% mark the turnover of the red curve at x = 3.2
\addplot[only marks, mark=*, engred, mark size=2pt] coordinates {(3.2,5.2)};
\node[font=\footnotesize, engred, anchor=south] at (axis cs:3.2,5.25)
  {turnover};
% a single output value that two temperatures share on the red curve.
\addplot[engamber, dashed] coordinates {(0,4.5)(10,4.5)};
\addplot[only marks, mark=square*, engamber, mark size=2pt] coordinates {
  (0.79,4.5) (5.61,4.5)
};
\node[font=\footnotesize, engamber, anchor=west] at (axis cs:6.3,4.5)
  {one $V$, two $T$};
\node[font=\footnotesize, engblue, anchor=north west] at (axis cs:6.4,3.6)
  {monotone: invertible};
\node[font=\footnotesize, engred, anchor=north] at (axis cs:2.0,2.4)
  {non-monotone};
\legend{{monotone response}, {non-monotone response}}
\end{axis}
\end{tikzpicture}
\caption[Invertibility of a sensor response]{Two sensor response
functions. The blue response is strictly increasing across its working
span, so it is injective there and the calibration inverse $T=f^{-1}(V)$
is a well-defined function: one output maps back to one temperature. The
red response turns over, so two distinct temperatures produce the same
output (the amber markers), and the response is not injective across the
turnover. On the non-monotone span the inverse is not a function; the
engineer must restrict the domain to one monotone branch, the same
domain-restriction discipline that defines $\sqrt{\cdot}$ from
$x\mapsto x^{2}$.}
\label{fig:vol02:ch01:monotone-inversion}
\end{figure}
