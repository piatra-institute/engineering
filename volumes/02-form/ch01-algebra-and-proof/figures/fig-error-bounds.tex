% Figure: the worst-case (triangle-inequality, linear sum) error bound
% against the typical-case (root-sum-of-squares, quadrature) bound, as a
% function of the number of independent error contributions. The gap
% between the two is the difference between the two questions the engineer
% can ask about accumulated error.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={number of equal error contributions $n$},
  ylabel={total error (units of one contribution $\epsilon$)},
  xmin=0, xmax=20,
  ymin=0, ymax=21,
  grid=both,
  grid style={engblack!12},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.03,0.97)}, anchor=north west},
]
% worst case: n*epsilon (triangle inequality, all errors aligned).
\addplot[engred, very thick] coordinates {
  (0,0) (2,2) (4,4) (6,6) (8,8) (10,10) (12,12) (14,14) (16,16) (18,18) (20,20)
};
% typical case: sqrt(n)*epsilon (quadrature, independent errors).
\addplot[engblue, very thick] coordinates {
  (0,0) (1,1) (2,1.414) (3,1.732) (4,2.0) (6,2.449) (8,2.828)
  (10,3.162) (12,3.464) (14,3.742) (16,4.0) (18,4.243) (20,4.472)
};
\node[font=\footnotesize, engred, anchor=south east] at (axis cs:19,19)
  {worst case $n\epsilon$};
\node[font=\footnotesize, engblue, anchor=north west] at (axis cs:12,3.4)
  {typical case $\sqrt{n}\,\epsilon$};
\legend{{worst case (triangle)}, {typical case (quadrature)}}
\end{axis}
\end{tikzpicture}
\caption[Worst-case versus typical-case error accumulation]{Two bounds on
the accumulation of $n$ equal error contributions of size $\epsilon$. The
triangle inequality gives the worst-case bound $n\epsilon$, realised when
every contribution points the same way. The root-sum-of-squares rule
gives the typical-case size $\sqrt{n}\,\epsilon$ when the contributions
are independent and can partly cancel. At $n=20$ the two differ by more
than a factor of four. The engineer chooses the worst-case bound when the
errors can conspire and the quadrature estimate when they cannot; the
quadrature refinement is built on top of the worst-case bound in Volume~I
Chapter~4.}
\label{fig:vol02:ch01:error-bounds}
\end{figure}
