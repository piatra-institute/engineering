% Figure: partial sums of a geometric series approaching the closed-form
% limit 1/(1-r). The truncation tail after n terms is r^n/(1-r), which
% sets how many terms an engineer must keep for a stated accuracy.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.5cm,
  xlabel={number of terms $n$},
  ylabel={partial sum $S_{n}=\sum_{k=0}^{n-1} r^{k}$},
  xmin=0, xmax=12,
  ymin=0, ymax=2.2,
  grid=both,
  grid style={engblack!12},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.97,0.05)}, anchor=south east},
]
% r = 0.5, limit 2.
\addplot[engblue, thick, mark=*, mark size=1.6pt] coordinates {
  (1,1.0) (2,1.5) (3,1.75) (4,1.875) (5,1.9375) (6,1.96875)
  (7,1.984375) (8,1.9921875) (9,1.99609375) (10,1.998046875)
  (11,1.9990234375) (12,1.99951171875)
};
% limit line for r = 0.5
\addplot[engblue, dashed] coordinates {(0,2)(12,2)};
\node[font=\footnotesize, engblue, anchor=east] at (axis cs:11.8,2.1)
  {$1/(1-r)=2$ at $r=0.5$};
% r = 0.8, limit 5 (off top of visible range; show the slow approach)
\addplot[engred, thick, mark=square*, mark size=1.4pt] coordinates {
  (1,1.0) (2,1.8) (3,2.44) (4,2.952) (5,3.3616) (6,3.68928)
  (7,3.951424) (8,4.1611392) (9,4.32891136) (10,4.463129088)
};
\node[font=\footnotesize, engred, anchor=west] at (axis cs:6.2,3.55)
  {$r=0.8$ (limit $5$, slower)};
\legend{{$r=0.5$}, {}, {$r=0.8$}}
\end{axis}
\end{tikzpicture}
\caption[Partial sums of a geometric series]{Partial sums of the geometric
series $\sum_{k\ge 0} r^{k}$ approaching the closed form $1/(1-r)$, valid
only for $\lvert r\rvert<1$. At $r=0.5$ the sum reaches its limit of $2$
to three figures after ten terms; the truncation tail after $n$ terms is
$r^{n}/(1-r)$, which halves with each added term. At $r=0.8$ the limit is
$5$ and the approach is much slower, because the tail shrinks only by the
factor $r$ per term. The number of terms an engineer keeps for a stated
accuracy is read off the tail, not guessed.}
\label{fig:vol02:ch01:geometric-series}
\end{figure}
