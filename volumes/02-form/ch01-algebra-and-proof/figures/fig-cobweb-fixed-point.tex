% Figure: cobweb (staircase) diagram of a contraction map iterating to its
% fixed point. The iteration x_{k+1}=h(x_k) is read off the diagram by
% bouncing between the curve y=h(x) and the diagonal y=x; the fixed point is
% where the two meet, and a slope of magnitude below one drives the staircase
% inward to it.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=9.5cm, height=8cm,
  xlabel={$x_{k}$},
  ylabel={$x_{k+1}=h(x_{k})$},
  xmin=0, xmax=3.2,
  ymin=0, ymax=3.2,
  axis equal image,
  grid=both,
  grid style={engblack!12},
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.03,0.97)}, anchor=north west},
]
% The map h(x) = 0.5 x + 1, a contraction with L = 0.5, fixed point x* = 2.
\addplot[engblue, thick, domain=0:3.2, samples=2] {0.5*x + 1};
% The diagonal y = x.
\addplot[engblack!60, dashed, domain=0:3.2, samples=2] {x};
% Fixed point marker at (2,2).
\addplot[engred, only marks, mark=*, mark size=2pt] coordinates {(2,2)};
\node[font=\footnotesize, engred, anchor=west] at (axis cs:2.08,1.86)
  {$x^{\ast}=2$};
% Cobweb staircase starting from x0 = 0.2.
% Steps: up to curve, across to diagonal, up to curve, across, ...
\addplot[engorange, thick] coordinates {
  (0.2,0.2)   % start on diagonal
  (0.2,1.1)   % up to curve h(0.2)=1.1
  (1.1,1.1)   % across to diagonal
  (1.1,1.55)  % up to curve h(1.1)=1.55
  (1.55,1.55) % across to diagonal
  (1.55,1.775)% up to curve
  (1.775,1.775)
  (1.775,1.8875)
  (1.8875,1.8875)
};
\node[font=\footnotesize, engorange, anchor=north west] at (axis cs:0.22,0.9)
  {start $x_{0}$};
\legend{{$y=h(x)$, slope $L=0.5$}, {$y=x$}, {fixed point}}
\end{axis}
\end{tikzpicture}
\caption[Cobweb diagram of a contraction iterating to its fixed point]{Cobweb
diagram of the fixed-point iteration $x_{k+1}=h(x_{k})$ for the contraction
$h(x)=\tfrac12 x + 1$, whose fixed point is where the curve $y=h(x)$ crosses
the diagonal $y=x$, at $x^{\ast}=2$. Each step is read by moving vertically
to the curve to apply the map, then horizontally to the diagonal to feed the
output back as the next input. The staircase spirals inward to the fixed
point because the slope of $h$ has magnitude $L=\tfrac12$ below one, so each
step shrinks the horizontal gap to $x^{\ast}$ by the factor $L$, the same
geometric $L^{k}$ decay the convergence proof lifts to all $k$ by induction.
A slope of magnitude above one would step the staircase outward and the
iteration would diverge, which is the visible statement of why the
contraction condition $L<1$ is not optional.}
\label{fig:vol02:ch01:cobweb-fixed-point}
\end{figure}
