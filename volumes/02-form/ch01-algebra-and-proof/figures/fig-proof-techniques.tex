% Figure: a decision tree for choosing a proof technique. The engineer
% reading a claim walks the tree to a starting technique.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  node distance=6mm,
  q/.style={draw=engblack, thick, rounded corners, fill=engblue!8,
    align=center, font=\footnotesize, text width=3.6cm, inner sep=4pt},
  t/.style={draw=engblack, thick, fill=engred!10,
    align=center, font=\footnotesize, text width=3.0cm, inner sep=4pt},
  e/.style={->, thick, draw=engblack},
  lbl/.style={font=\scriptsize\itshape, fill=white, inner sep=1pt},
]
\node[q] (start) {Claim of the form ``for all $x$, $P(x)$''?};
\node[q, below left=10mm and 1.0cm of start] (univ)
  {Quantified over the natural numbers?};
\node[q, below right=10mm and 1.0cm of start] (exist)
  {Existence or non\nobreakdash-existence claim?};

\node[t, below=of univ] (induction) {Induction (Sec.~1.4)};
\node[t, below left=10mm and -0.6cm of exist] (contra)
  {Contradiction};
\node[t, below right=10mm and -0.6cm of exist] (direct)
  {Direct or by cases};
\node[t, below=14mm of direct] (contrapos)
  {Contrapositive (when the converse is easier)};

\draw[e] (start) -- (univ) node[lbl, midway] {yes};
\draw[e] (start) -- (exist) node[lbl, midway] {no};
\draw[e] (univ) -- (induction) node[lbl, midway] {yes};
\draw[e] (univ.south east) -- (direct.north west) node[lbl, midway] {no};
\draw[e] (exist) -- (contra) node[lbl, midway] {non-existence};
\draw[e] (exist) -- (direct) node[lbl, midway] {existence};
\draw[e] (direct) -- (contrapos) node[lbl, midway] {witness hard};
\end{tikzpicture}
\caption[Choosing a proof technique]{A working decision tree for the
opening move of a proof. The tree gives a starting technique, not a
guarantee; many proofs combine techniques (the prime-factorisation
proof of Sec.~1.4 is strong induction with a case split inside the
inductive step). The value of the tree is in forcing the writer to
name the logical shape of the claim before reaching for algebra.}
\label{fig:vol02:ch01:proof-techniques}
\end{figure}
