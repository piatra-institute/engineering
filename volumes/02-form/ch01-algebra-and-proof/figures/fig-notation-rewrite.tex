% with definitions, units, and unambiguous indices.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  before/.style={draw=engred!60, thick, fill=engred!6, rounded corners,
    align=left, font=\footnotesize, text width=5.6cm, inner sep=6pt},
  after/.style={draw=engblue!60, thick, fill=engblue!6, rounded corners,
    align=left, font=\footnotesize, text width=5.6cm, inner sep=6pt},
  hd/.style={font=\footnotesize\bfseries},
  ann/.style={font=\scriptsize\itshape, align=center},
]
\node[hd] (bh) at (-3.4, 3.1) {Before};
\node[hd] (ah) at (3.4, 3.1) {After};

\node[before] (b) at (-3.4, 1.1) {
  Heat in the rod:\\[2pt]
  $q = k\,A\,dT/dx$\\[2pt]
  with $\rho = k/(c\,T)$ the diffusivity,\\
  and $i$-th node $T_i^2$ from the solver.
};
\node[after] (a) at (3.4, 1.0) {
  Conductive heat flux (Sec.~2):\\[2pt]
  $q = -k\,A\,\dfrac{dT}{dx}\quad[\si{\watt}]$\\[2pt]
  $k$ thermal conductivity $[\si{\watt\per\meter\per\kelvin}]$,\\
  $A$ area $[\si{\meter\squared}]$, $T$ temperature $[\si{\kelvin}]$.\\
  Diffusivity $\alpha = k/(\rho\,c_{p})\;[\si{\meter\squared\per\second}]$.\\
  Node temperature: $(T_i)^{2}$, index $i=1,\dots,N$.
};

\draw[->, very thick, draw=engblack] (b.east) -- (a.west);

\node[ann, anchor=north, text width=11cm] at (0,-1.0) {
  defects fixed: missing sign on Fourier's law; $\rho$ overloaded
  (density vs.\ used here for diffusivity, now $\alpha$); $c$ undefined;
  no units; $T_i^2$ ambiguous between $(T_i)^2$ and the square's $i$-th
  component; no section reference for the governing law.};
\end{tikzpicture}
\caption[A notation rewrite, before and after]{A notation rewrite. The
left fragment carries six defects of the kind Sec.~1.6 names: a dropped
sign on Fourier's law, an overloaded $\rho$ (mass density elsewhere,
diffusivity here), an undefined $c$, no units on any quantity, an
ambiguous $T_i^2$, and no cross-reference to the governing law. The
right fragment fixes each. The project of Sec.~1.8 asks the reader to
perform this operation on a real published page.}
\label{fig:vol02:ch01:notation-rewrite}
\end{figure}
