% Figure: the defective exponential. A single Jordan block at a
% negative eigenvalue produces a t e^{lambda t} term that humps before
% decaying, unlike the pure e^{lambda t} of a diagonalisable matrix.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm,
  height=8cm,
  xlabel={time $t$},
  ylabel={off-diagonal response},
  xmin=0, xmax=12,
  ymin=0, ymax=0.75,
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.97,0.97)}, anchor=north east},
  title={Defective response: the $t\,e^{\lambda t}$ hump},
  title style={font=\bfseries},
  ymajorgrids=true,
  grid style={gray!25},
]
% Defective: off-diagonal of e^{At} for J = [[l,1],[0,l]] is t e^{lt}.
% lambda = -0.5.
\addplot[engred, very thick, domain=0:12, samples=120] {x*exp(-0.5*x)};
\addlegendentry{defective: $t\,e^{-0.5t}$ (Jordan block)}
% Diagonalisable comparison: pure e^{lt} scaled to match early slope.
\addplot[engblue, thick, dashed, domain=0:12, samples=120] {exp(-0.5*x)};
\addlegendentry{diagonalisable: $e^{-0.5t}$}
% Mark the peak of t e^{-0.5 t}: at t = 2, value = 2 e^{-1} = 0.7358.
\addplot[engred, only marks, mark=*, mark size=2pt] coordinates {(2,0.7358)};
\node[font=\footnotesize, engred, anchor=south west] at (axis cs:2.1,0.71)
  {peak $2e^{-1}\approx 0.74$ at $t=2$};
\end{axis}
\end{tikzpicture}
\caption[The defective exponential and its transient hump]{The
off-diagonal entry of $e^{\mat{A}t}$ for a single $2\times2$ Jordan
block at eigenvalue $\lambda = -0.5$ (red), which equals
$t\,e^{\lambda t}$, against the pure exponential $e^{\lambda t}$ that
a diagonalisable matrix with the same eigenvalue would produce (blue
dashed). The eigenvalue is negative, so both responses decay in the
end, but the defective response first climbs to a hump, peaking at
$t = -1/\lambda = 2$ with value $2e^{-1} \approx 0.74$, before the
exponential finally wins. The extra factor of $t$ is the algebraic
signature of the defect: a repeated eigenvalue with too few
eigenvectors produces a polynomial-times-exponential response rather
than a clean exponential, and a $k\times k$ block raises the polynomial
degree to $k-1$. An engineer who models a near-collision as
diagonalisable predicts the blue curve and misses the transient the
red curve carries, which is one more reason a parameter sweep across
an eigenvalue collision earns the Schur decomposition rather than
naive diagonalisation.}
\label{fig:vol02:ch10:jordan-transient}
\end{figure}
