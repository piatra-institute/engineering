% Figure: the Marchenko-Pastur noise floor overlaid on a scree-style
% eigenvalue plot. Eigenvalues of a sample covariance matrix from
% pure noise spread between lambda_- and lambda_+; a genuine signal
% eigenvalue stands above lambda_+. The figure shows where the
% structural eigenvalues end and the noise bulk begins.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm,
  height=8cm,
  xlabel={eigenvalue $\lambda$ of the sample covariance matrix},
  ylabel={density of eigenvalues},
  xmin=0, xmax=4.2,
  ymin=0, ymax=1.3,
  tick label style={font=\footnotesize},
  label style={font=\small},
  title={Marchenko-Pastur noise bulk and the signal eigenvalues},
  title style={font=\bfseries},
  clip=false,
]
% Marchenko-Pastur density for aspect ratio gamma = n/m = 0.25:
% lambda_- = (1 - sqrt(gamma))^2 = 0.25, lambda_+ = (1 + sqrt(gamma))^2 = 2.25.
% density(x) = sqrt((lp - x)(x - lm)) / (2 pi gamma x), on [lm, lp].
\addplot[engblue, very thick, fill=engblue!18, domain=0.25:2.25,
         samples=120]
  {sqrt(max(0,(2.25 - x)*(x - 0.25))) / (2*3.14159*0.25*x)};
% Edges of the bulk.
\draw[enggray, dashed] (axis cs:0.25,0) -- (axis cs:0.25,1.3);
\draw[engamber, thick, dashed] (axis cs:2.25,0) -- (axis cs:2.25,1.3);
\node[font=\footnotesize, anchor=south, engamber, rotate=90]
  at (axis cs:2.32,0.5) {$\lambda_{+} = 2.25$};
\node[font=\footnotesize, anchor=south, enggray, rotate=90]
  at (axis cs:0.18,0.5) {$\lambda_{-} = 0.25$};
% Two signal eigenvalues standing above the bulk edge.
\draw[engred, very thick, -{Stealth}] (axis cs:3.6,0) -- (axis cs:3.6,1.0);
\draw[engred, very thick, -{Stealth}] (axis cs:2.7,0) -- (axis cs:2.7,0.7);
\node[font=\footnotesize, anchor=south, engred] at (axis cs:3.6,1.0)
  {signal};
\node[font=\footnotesize, anchor=south, engred] at (axis cs:2.7,0.7)
  {signal};
\end{axis}
\end{tikzpicture}
\caption[The Marchenko-Pastur law as a PCA significance test]{When the
data carry no structure, the eigenvalues of the sample covariance
matrix do not collapse to a single value; sampling noise spreads them
across a predictable band. For $m$ samples of $n$ standardised features
with aspect ratio $\gamma = n/m$, the noise eigenvalues fill the
interval between $\lambda_{-} = (1 - \sqrt{\gamma})^{2}$ and
$\lambda_{+} = (1 + \sqrt{\gamma})^{2}$ with the Marchenko-Pastur
density shown in blue (here $\gamma = 0.25$, so the bulk runs from
$0.25$ to $2.25$). An eigenvalue that stands clearly above
$\lambda_{+}$ (the two red spikes) is a genuine principal component;
an eigenvalue inside the bulk is indistinguishable from noise no matter
how suggestive the scree plot looks. The law gives the engineer a
quantitative significance threshold for retaining a component, in place
of the eyeballed elbow.}
\label{fig:vol02:ch10:marchenko-pastur}
\end{figure}
