% Figure: eigenvalue assignment by state feedback. The open-loop
% eigenvalues of A sit in the right half-plane (unstable). State
% feedback u = -Kx changes the matrix to A - BK, whose eigenvalues are
% placed at chosen stable locations in the left half-plane. The
% imaginary axis is the continuous-time stability frontier.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11.5cm,
  height=7.6cm,
  xlabel={real part $\operatorname{Re}\lambda$},
  ylabel={imaginary part $\operatorname{Im}\lambda$},
  xmin=-4.2, xmax=2.2,
  ymin=-3.2, ymax=3.2,
  axis lines=middle,
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.03,0.97)}, anchor=north west},
  title={Eigenvalue assignment: moving the closed-loop poles},
  title style={font=\bfseries},
]
% Shade the stable (left) half-plane lightly.
\addplot[draw=none, fill=englime, fill opacity=0.10] coordinates {
  (-4.2,-3.2) (0,-3.2) (0,3.2) (-4.2,3.2)
} \closedcycle;
% Open-loop eigenvalues (unstable): +1 and +0.5 +- 1.5 i.
\addplot[only marks, mark=o, engred, mark size=3.2pt, line width=1pt]
  coordinates { (1,0) (0.5,1.5) (0.5,-1.5) };
\addlegendentry{open-loop $\mat{A}$ (unstable)}
% Closed-loop eigenvalues (assigned): -1, -1.5 +- 1.2 i.
\addplot[only marks, mark=x, engblue, mark size=4pt, line width=1.2pt]
  coordinates { (-1,0) (-1.5,1.2) (-1.5,-1.2) };
\addlegendentry{closed-loop $\mat{A}-\mat{B}\mat{K}$ (assigned)}
% Arrows from each open-loop pole to its assigned closed-loop location.
\draw[-{Latex}, enggray, thick] (axis cs:1,0) -- (axis cs:-0.9,0);
\draw[-{Latex}, enggray, thick] (axis cs:0.5,1.5) -- (axis cs:-1.4,1.25);
\draw[-{Latex}, enggray, thick] (axis cs:0.5,-1.5) -- (axis cs:-1.4,-1.25);
% Label the stable region.
\node[font=\footnotesize\itshape, enggray, anchor=north west]
  at (axis cs:-4.1,3.0) {stable half-plane};
\end{axis}
\end{tikzpicture}
\caption[Eigenvalue assignment by state feedback]{State feedback
relocates the eigenvalues. The open-loop matrix $\mat{A}$ has three
eigenvalues in the right half-plane (red circles), so the free response
$\dot{\vect{x}}=\mat{A}\vect{x}$ grows without bound. The feedback law
$\vect{u}=-\mat{K}\vect{x}$ replaces $\mat{A}$ by the closed-loop matrix
$\mat{A}-\mat{B}\mat{K}$, and choosing $\mat{K}$ places the closed-loop
eigenvalues (blue crosses) at any desired self-conjugate set, here a
real pole at $-1$ and a damped pair at $-1.5\pm1.2i$, all inside the
shaded stable region. The design freedom is the eigenvalue locations
themselves; the constraint is that they arrive as a real spectrum,
which forces complex assignments to come in conjugate pairs. The
distance of the rightmost assigned pole from the imaginary axis is the
stability margin the designer buys.}
\label{fig:vol02:ch10:pole-placement}
\end{figure}
