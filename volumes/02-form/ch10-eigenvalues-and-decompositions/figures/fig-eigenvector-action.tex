% Figure: the action of a 2x2 matrix on a vector grid, with the two
% eigenvectors highlighted as the directions the matrix only scales.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  axis/.style={draw=enggray, thin, -{Stealth}},
  ngrid/.style={draw=engblue, thin, -{Stealth}, opacity=0.55},
  outgrid/.style={draw=engamber, thin, -{Stealth}, opacity=0.7},
  eigvec/.style={draw=engred, very thick, -{Stealth}},
  eigout/.style={draw=engred, very thick, dashed, -{Stealth}},
  label/.style={font=\small},
]

% Left panel: input vectors
\begin{scope}[shift={(0,0)}]
  \draw[axis] (-2.5,0) -- (2.5,0) node[label, anchor=west] {$x_{1}$};
  \draw[axis] (0,-2.5) -- (0,2.5) node[label, anchor=south] {$x_{2}$};
  % A small fan of input vectors, each unit length
  \foreach \ang in {0,30,60,90,120,150,180,210,240,270,300,330} {
    \draw[ngrid] (0,0) -- ({1.5*cos(\ang)},{1.5*sin(\ang)});
  }
  % Eigenvectors of A = [[4,1],[2,3]]: v1 = (1,1)/sqrt(2), v2 = (1,-2)/sqrt(5)
  \draw[eigvec] (0,0) -- ({1.5/sqrt(2)},{1.5/sqrt(2)})
    node[anchor=south west, font=\small\bfseries, engred] {$\vect{v}_{1}$};
  \draw[eigvec] (0,0) -- ({1.5/sqrt(5)},{-3.0/sqrt(5)})
    node[anchor=north west, font=\small\bfseries, engred] {$\vect{v}_{2}$};
  \node[font=\small\bfseries, anchor=south] at (0,2.7) {input};
\end{scope}

% Right panel: output vectors after applying A
\begin{scope}[shift={(7,0)}]
  \draw[axis] (-4.5,0) -- (4.5,0) node[label, anchor=west] {$y_{1}$};
  \draw[axis] (0,-4.5) -- (0,4.5) node[label, anchor=south] {$y_{2}$};
  % Image of the same fan under A = [[4,1],[2,3]]
  \foreach \ang in {0,30,60,90,120,150,180,210,240,270,300,330} {
    \pgfmathsetmacro\xnew{1.5*(4*cos(\ang) + 1*sin(\ang))}
    \pgfmathsetmacro\ynew{1.5*(2*cos(\ang) + 3*sin(\ang))}
    \draw[outgrid] (0,0) -- ({\xnew/4},{\ynew/4}); % scaled down to fit
  }
  % Image of eigenvectors: A v1 = 5 v1, A v2 = 2 v2
  % v1 image: 5 * 1.5/sqrt(2) * (1,1)/sqrt(2)/4 in scaled axes -> just show direction
  \draw[eigout] (0,0) -- ({1.5*5/(sqrt(2)*4)},{1.5*5/(sqrt(2)*4)})
    node[anchor=south west, font=\small\bfseries, engred] {$5\vect{v}_{1}$};
  \draw[eigout] (0,0) -- ({1.5*2/(sqrt(5)*4)},{-3.0*2/(sqrt(5)*4)})
    node[anchor=north west, font=\small\bfseries, engred] {$2\vect{v}_{2}$};
  \node[font=\small\bfseries, anchor=south] at (0,2.7) {$\mat{A}\,\cdot$ (input)};
\end{scope}

\end{tikzpicture}
\caption[Action of a $2 \times 2$ matrix and its eigenvectors]{The
matrix $\mat{A} = \begin{psmallmatrix} 4 & 1 \\ 2 & 3 \end{psmallmatrix}$
acts on the input fan (left, blue arrows) and produces the output fan
(right, amber arrows, drawn at one-quarter scale so the panels share a
page). Almost every input rotates as well as stretches. The two red
arrows are the eigenvectors $\vect{v}_{1} = (1,1)^{\transpose}$ and
$\vect{v}_{2} = (1,-2)^{\transpose}$; the matrix scales them by
$\lambda_{1} = 5$ and $\lambda_{2} = 2$ respectively, without rotation.
The whole chapter is the operational use of this picture: find the
directions the matrix leaves alone, and the matrix becomes a list of
numbers in their basis.}
\label{fig:vol02:ch10:eigenvector-action}
\end{figure}
