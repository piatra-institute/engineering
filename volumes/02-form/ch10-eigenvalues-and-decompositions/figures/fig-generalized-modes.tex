% Figure: mode shapes of a two-degree-of-freedom system with unequal
% masses, solved as a generalized eigenproblem K x = lambda M x. The
% two modes are plotted as displacement patterns of the two masses,
% showing in-phase (low frequency) and out-of-phase (high frequency)
% motion with the heavier mass moving less in the high mode.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=1.0]
% Baseline reference lines for the two masses (positions 0 and 1).
\foreach \x in {0,1} {
  \draw[enggray, thin] (\x-0.05,0) -- (\x+0.05,0) node[below=2pt, font=\footnotesize, black] {};
}
% Mode 1 (in phase), drawn on the upper band, baseline y = 1.6.
\begin{scope}[yshift=1.6cm]
\draw[enggray, thin, dashed] (-0.4,0) -- (1.4,0);
\node[anchor=east, font=\footnotesize] at (-0.5,0)
  {mode 1 (in phase, $\omega_{1}$)};
% mass 1 displacement +0.45, mass 2 displacement +0.30 (heavier, less)
\filldraw[engblue] (0,0.45) circle (3pt);
\filldraw[engblue] (1,0.30) circle (3pt);
\draw[engblue, thick, -{Stealth}] (0,0) -- (0,0.42);
\draw[engblue, thick, -{Stealth}] (1,0) -- (1,0.27);
\node[font=\footnotesize, anchor=west] at (0.06,0.45) {$m_1$};
\node[font=\footnotesize, anchor=west] at (1.06,0.30) {$m_2$};
\end{scope}
% Mode 2 (out of phase), lower band, baseline y = 0.
\draw[enggray, thin, dashed] (-0.4,0) -- (1.4,0);
\node[anchor=east, font=\footnotesize] at (-0.5,0)
  {mode 2 (out of phase, $\omega_{2}$)};
% mass 1 displacement +0.50, mass 2 displacement -0.22 (heavier moves less)
\filldraw[engred] (0,0.50) circle (3pt);
\filldraw[engred] (1,-0.22) circle (3pt);
\draw[engred, thick, -{Stealth}] (0,0) -- (0,0.47);
\draw[engred, thick, -{Stealth}] (1,0) -- (1,-0.19);
\node[font=\footnotesize, anchor=west] at (0.06,0.50) {$m_1$};
\node[font=\footnotesize, anchor=west] at (1.06,-0.22) {$m_2$};
\end{tikzpicture}
\caption[Generalized-eigenproblem mode shapes for unequal masses]{The
two natural modes of a two-mass, three-spring chain with unequal masses
$m_{1} < m_{2}$, obtained from the generalized eigenproblem
$\mat{K}\vect{x} = \lambda \mat{M}\vect{x}$ rather than from
$\mat{K}$ alone. In the low-frequency mode the masses move in phase; in
the high-frequency mode they move in opposition. The mass matrix
weights the eigenvectors: they are orthogonal in the $\mat{M}$-inner
product, $\vect{x}_{i}\transpose \mat{M}\vect{x}_{j} = 0$ for
$i \neq j$, not in the ordinary one, and the heavier mass carries
smaller amplitude in the out-of-phase mode because the same modal
energy buys it less displacement. Ignoring $\mat{M}$ and diagonalising
$\mat{K}$ alone would give the wrong mode shapes and the wrong
frequencies whenever the masses are not equal.}
\label{fig:vol02:ch10:generalized-modes}
\end{figure}
