% Figure: the Ky Fan maximum principle. The sum of the k largest
% eigenvalues equals the maximum trace of the form over any
% k-dimensional orthonormal subspace. The bars are the sorted
% eigenvalues; the step curve is the running partial sum (partial
% trace), which is the best achievable captured variance at each k and
% saturates at the full trace.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=13cm,
  height=7.2cm,
  xlabel={number of retained directions $k$},
  ylabel={eigenvalue and partial trace},
  xmin=0.3, xmax=6.7,
  ymin=0, ymax=13.5,
  xtick={1,2,3,4,5,6},
  ybar,
  bar width=14pt,
  tick label style={font=\footnotesize},
  label style={font=\small},
  legend style={font=\footnotesize, at={(0.97,0.55)}, anchor=east},
  title={Ky Fan: partial trace as best captured variance},
  title style={font=\bfseries},
]
% Sorted eigenvalues.
\addplot[draw=engblue, fill=engblue!25] coordinates {
  (1,6.0) (2,3.2) (3,1.6) (4,0.9) (5,0.5) (6,0.3)
};
\addlegendentry{eigenvalue $\lambda_{k}$}
\end{axis}
% Overlay the partial-trace step curve on the same coordinates using a
% second axis with matching limits (line plot, not bar).
\begin{axis}[
  width=13cm,
  height=7.2cm,
  xmin=0.3, xmax=6.7,
  ymin=0, ymax=13.5,
  axis lines=none,
  legend style={font=\footnotesize, at={(0.97,0.72)}, anchor=east},
]
% partial sums: 6, 9.2, 10.8, 11.7, 12.2, 12.5
\addplot[engred, very thick, mark=*, mark size=2pt] coordinates {
  (1,6.0) (2,9.2) (3,10.8) (4,11.7) (5,12.2) (6,12.5)
};
\addlegendentry{partial trace $\sum_{i\le k}\lambda_{i}$}
% Full-trace asymptote.
\addplot[enggray, dashed, domain=0.3:6.7, samples=2] {12.5};
\addlegendentry{full trace $\tr\mat{A}=12.5$}
\end{axis}
\end{tikzpicture}
\caption[The Ky Fan maximum principle]{The Ky Fan maximum principle
reads the captured variance off the spectrum. The bars are the sorted
eigenvalues $\lambda_{1}\ge\cdots\ge\lambda_{6}$ of a symmetric positive
matrix; the red step curve is the partial trace $\sum_{i\le k}
\lambda_{i}$. Ky Fan's theorem states that this partial sum is exactly
the largest trace $\vect{q}$ any $k$-dimensional orthonormal frame can
extract from the form, achieved by the frame of the top $k$
eigenvectors. The curve is therefore the ceiling on how much variance
$k$ principal directions can capture, and the gap to the full-trace
asymptote (grey dashed) is the tail energy the truncation discards. The
plot is the eigenvalue statement of the same optimality Eckart-Young
gives for the singular values: PCA keeps the top $k$ eigenvectors
because no other $k$-frame reaches higher on this curve.}
\label{fig:vol02:ch10:ky-fan-partial-trace}
\end{figure}
