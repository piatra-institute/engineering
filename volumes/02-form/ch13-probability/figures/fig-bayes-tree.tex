% Bayes tree for the turbine-blade inspection example.
% Prior 0.004 cracked; sensitivity 0.98; specificity 0.95.
% Leaf counts are per 10,000 blades inspected.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small,
  edge/.style={thick, ->, >=stealth},
  node distance=0.4cm]

% Root
\node (root) at (0, 0) {\textbf{10\,000 blades}};

% First split: cracked / sound
\node (cr) at (5, 2.4) {cracked: $40$};
\node (so) at (5, -2.4) {sound: $9960$};

\draw[edge] (root) -- (cr)
  node[midway, above, sloped, font=\scriptsize] {$\Pr=0.004$};
\draw[edge] (root) -- (so)
  node[midway, below, sloped, font=\scriptsize] {$\Pr=0.996$};

% Second split from cracked
\node (crp) at (10.6, 3.5) {$+$ true positive: $39.2$};
\node (crn) at (10.6, 1.4) {$-$ false negative: $0.8$};
\draw[edge] (cr) -- (crp)
  node[midway, above, sloped, font=\scriptsize] {sens $0.98$};
\draw[edge] (cr) -- (crn)
  node[midway, below, sloped, font=\scriptsize] {$0.02$};

% Second split from sound
\node (sop) at (10.6, -1.4) {$+$ false positive: $498.0$};
\node (son) at (10.6, -3.5) {$-$ true negative: $9462.0$};
\draw[edge] (so) -- (sop)
  node[midway, above, sloped, font=\scriptsize] {$1-\text{spec}=0.05$};
\draw[edge] (so) -- (son)
  node[midway, below, sloped, font=\scriptsize] {spec $0.95$};

% Posterior annotation
\node[anchor=west, font=\scriptsize, text width=5.0cm] at (0, -4.6)
  {Of the $39.2 + 498.0 = 537.2$ positives, only $39.2$ are real:
   $\Pr(\text{cracked}\mid +) = 39.2/537.2 \approx 0.073$. The
   false-positive leaf dominates because the sound population is
   $249$ times larger than the cracked one.};

\end{tikzpicture}
\caption[Bayes tree for a rare-defect inspection]{The base-rate
calculation as a probability tree, with leaf counts per $10\,000$
blades inspected at prevalence $0.004$, sensitivity $0.98$, and
specificity $0.95$. The posterior probability that a positive blade
is genuinely cracked is the ratio of the true-positive leaf to the
sum of the two positive leaves, about $7.3\,\%$. The tree makes
visible why a high-accuracy test still yields mostly false positives
on a rare defect: the large sound population generates more false
alarms than the small cracked population generates true alarms.
Source: computed by the authors from the stated test characteristics.}
\label{fig:vol02:ch13:bayes-tree}
\end{figure}
