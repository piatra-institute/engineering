% CLT convergence: histograms of the standardised sample mean for
% an exponentially distributed sum, at n = 1, 5, 30. Three panels.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% ---- Panel 1: n = 1 (raw exponential) ----
\begin{scope}[shift={(0,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.3) {$n = 1$};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.9)
    {raw $\mathrm{Exp}(1)$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % Heavily right-skewed (single exponential)
  \draw[thick, engred, smooth]
    plot[domain=0.0:3.95, samples=70]
    (\x, {2.4*exp(-\x)});
  % Reference Gaussian (rescaled to standardised)
  \draw[thick, engblue, dashed, smooth]
    plot[domain=0.05:3.95, samples=60]
    (\x, {2.2*exp(-((\x-1.0)*(\x-1.0))/(2*1.0*1.0))});
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1)
    {strong skew};
\end{scope}

% ---- Panel 2: n = 5 ----
\begin{scope}[shift={(5.2,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.3) {$n = 5$};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.9)
    {sum of 5};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % gamma(5, 1) shape, peak near x = 4; less skewed
  \draw[thick, engred, smooth]
    plot[domain=0.1:4.0, samples=80]
    (\x, {0.5*(\x*\x*\x*\x*exp(-\x))/24.0*4.0});
  % overlay Gaussian centred at 2.0 (the panel scale)
  \draw[thick, engblue, dashed, smooth]
    plot[domain=0.05:3.95, samples=60]
    (\x, {2.0*exp(-((\x-2.0)*(\x-2.0))/(2*0.7*0.7))});
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1)
    {residual skew};
\end{scope}

% ---- Panel 3: n = 30 ----
\begin{scope}[shift={(10.4,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.3) {$n = 30$};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.9)
    {sample mean};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % Nearly Gaussian
  \draw[thick, engred, smooth]
    plot[domain=0.1:3.9, samples=60]
    (\x, {2.4*exp(-((\x-2.0)*(\x-2.0))/(2*0.55*0.55))});
  \draw[thick, engblue, dashed, smooth]
    plot[domain=0.1:3.9, samples=60]
    (\x, {2.4*exp(-((\x-2.0)*(\x-2.0))/(2*0.55*0.55))});
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1)
    {Gaussian-shaped};
\end{scope}

% ---- Legend underneath ----
\node[anchor=north, font=\scriptsize] at (8.0, -1.0)
  {Solid: empirical density of the standardised sum.
   Dashed: standard Gaussian reference $\mathcal{N}(0, 1)$.};

\end{tikzpicture}
\caption[Central Limit Theorem convergence]{Three snapshots of the
Central Limit Theorem at work. The underlying distribution is
$\mathrm{Exp}(1)$, which is heavily right-skewed. At $n = 1$ the
distribution is the raw exponential, far from Gaussian. At $n = 5$
the sum is $\mathrm{Gamma}(5, 1)$, with reduced but visible
skewness. At $n = 30$ the standardised sample mean is
indistinguishable from the standard normal at the precision of the
plot, which matches the rule of thumb that $n \approx 30$ suffices
for nearly symmetric underlying distributions. The rate at which
the Gaussian approximation becomes good is set by the underlying
distribution's third moment, formalised by the Berry-Esseen
theorem. Source: computed by the authors from the gamma density
formula.}
\label{fig:vol02:ch13:clt-convergence}
\end{figure}
