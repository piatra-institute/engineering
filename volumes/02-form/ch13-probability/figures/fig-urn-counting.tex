\centering
\begin{tikzpicture}[
  ball/.style={circle, draw=engblack, minimum size=0.45cm,
    inner sep=0pt, font=\scriptsize},
  node distance=1.2cm,
  edge/.style={-{Stealth}, thick, draw=engblack!70},
]

% ---- Left: the urn ----
\begin{scope}[shift={(0, 0)}]
  \draw[thick, engblack] (0.6, -0.2) -- (0.4, 2.6) -- (3.6, 2.6)
    -- (3.4, -0.2) -- cycle;
  \node[anchor=south, font=\small\bfseries] at (2.0, 2.8)
    {Urn};
  % Balls
  \foreach \x/\y in {1.0/0.2, 1.6/0.2, 2.2/0.2, 2.8/0.2, 1.3/0.7}
    \node[ball, fill=engred!50] at (\x, \y) {};
  \foreach \x/\y in {1.9/0.7, 2.5/0.7, 1.0/1.2}
    \node[ball, fill=engblue!50] at (\x, \y) {};

  \node[anchor=west, font=\scriptsize] at (3.7, 1.4)
    {$5$ red};
  \node[anchor=west, font=\scriptsize] at (3.7, 1.0)

    {$2$ green};
  \node[anchor=west, font=\scriptsize] at (3.7, 0.2)
    {total $10$};
\end{scope}

% ---- Right: first-draw tree ----
\begin{scope}[shift={(7.5, 0)}]
  \node[font=\small\bfseries] at (2.0, 2.7) {First draw};
  \node[ball, fill=engred!50] (rR) at (0, 1.6) {};
  \node[ball, fill=engblue!50] (rB) at (2, 1.6) {};
  \node[ball, fill=englime!60] (rG) at (4, 1.6) {};
  \node[anchor=south, font=\scriptsize] at (0, 2.0)
    {$\Pr(R) = 5/10$};
  \node[anchor=south, font=\scriptsize] at (2, 2.0)
    {$\Pr(B) = 3/10$};
  \node[anchor=south, font=\scriptsize] at (4, 2.0)

  \node[font=\small\bfseries] at (2.0, 0.3) {After R drawn};

  \node[ball, fill=engblue!50] (sB) at (2, -1.0) {};
  \node[ball, fill=englime!60] (sG) at (4, -1.0) {};
  \node[anchor=north, font=\scriptsize] at (0, -1.3)

\caption[The urn and the sampling tree]{The urn of section
13.1's running example: $5$ red, $3$ blue, $2$ green balls. The
first draw has the indicated marginal probabilities. The right
panel shows the conditional probabilities for the second draw
given that the first was red: the population shrinks to nine balls
with one fewer red, so the conditional distribution differs from
the marginal. Sampling without replacement makes successive draws
dependent; sampling with replacement makes them independent. The
counting answer for any event involving multiple draws is the
product of conditionals down the corresponding branch of the
tree. Source: the standard urn model of
\cite[ch.~1]{text:ross-probability}.}
\label{fig:vol02:ch13:urn-counting}
\end{figure}
