% Gambler's ruin: the overflow (absorption-at-N) probability r_k as a
% function of the starting position k, for the fair walk and two biased
% walks, showing how a small drift concentrates the outcome.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[font=\small]
\begin{axis}[
  width=10.5cm, height=6.4cm,
  xlabel={starting tokens $k$ (buffer size $N=100$)},
  ylabel={$\Pr(\text{overflow before empty})$},
  xmin=0, xmax=100, ymin=0, ymax=1.03,
  xtick={0,20,40,60,80,100},
  ytick={0,0.25,0.5,0.75,1.0},
  grid=both, grid style={enggray!25},
  tick label style={font=\scriptsize},
  label style={font=\scriptsize},
  legend style={font=\scriptsize, at={(0.02,0.98)}, anchor=north west,
    draw=enggray!50},
]
  % fair walk: r_k = k/N, a straight line
  \addplot[thick, engblue, domain=0:100, samples=2] {x/100};
  \addlegendentry{$p=0.50$ (fair): $r_k=k/N$}

  % biased p = 0.52, q = 0.48 : r_k = (1-(q/p)^k)/(1-(q/p)^N),
  % q/p = 0.923077, (q/p)^100 approx 0.000432
  % r_k = (1 - 0.923077^k)/(1 - 0.000432)
  \addplot[thick, engred, mark=none, smooth] coordinates {
    (0,0) (5,0.3315) (10,0.5527) (15,0.7003) (20,0.7988)
    (25,0.8645) (30,0.9083) (40,0.9591) (50,0.9789)
    (60,0.9906) (70,0.9958) (80,0.9981) (90,0.9993) (100,1.0)
  };
  \addlegendentry{$p=0.52$ (slight up-drift)}

  % biased p = 0.48, q = 0.52 : q/p = 1.08333, mirror of above
  \addplot[thick, engorange, dashed, mark=none, smooth] coordinates {
    (0,0) (10,0.0007) (20,0.0019) (30,0.0042) (40,0.0094)
    (50,0.0211) (60,0.0355) (70,0.0917) (75,0.1355) (80,0.2012)
    (85,0.2997) (90,0.4473) (95,0.6685) (100,1.0)
  };
  \addlegendentry{$p=0.48$ (slight down-drift)}

  % mark the midpoint outcomes
  \addplot[only marks, mark=*, engred, mark size=2pt] coordinates {(50,0.9789)};
  \addplot[only marks, mark=*, engblue, mark size=2pt] coordinates {(50,0.5)};
  \node[anchor=west, font=\scriptsize, engred] at (axis cs:52,0.90)
    {$r_{50}\approx0.98$};
  \node[anchor=north west, font=\scriptsize, engblue] at (axis cs:52,0.46)
    {$r_{50}=0.50$};
\end{axis}
\end{tikzpicture}
\caption[Gambler's ruin and the effect of drift]{The overflow
probability of a bounded buffer of size $N=100$ that gains a token with
probability $p$ and loses one with probability $1-p$ each tick, absorbing
at empty ($0$) or full ($N$), as a function of the starting level $k$. The
fair walk ($p=0.5$, blue) gives the linear law $r_k = k/N$: the overflow
probability is just the fractional starting position. A four-percentage-
point upward drift ($p=0.52$, red) bends the curve sharply so that a
buffer starting half full overflows with probability $0.98$ rather than
$0.50$; a symmetric downward drift ($p=0.48$, orange dashed) makes
overflow from the midpoint nearly impossible. A tiny bias dominates a long
walk, which is why first-passage outcomes concentrate and why the
expected time to absorption, $k(N-k)$ in the fair case, is the timescale
over which a balanced buffer wanders before it hits a boundary. Source:
computed by the authors from the gambler's-ruin recurrence,
\cite[ch.~6]{text:wasserman2004}.}
\label{fig:vol02:ch13:gambler-ruin}
\end{figure}
