% Bathtub hazard curve: three regimes stitched together. A falling
% infant-mortality phase (Weibull shape k<1), a flat useful-life phase
% where the exponential applies (constant hazard), and a rising
% wear-out phase (Weibull shape k>1). Plot units are arbitrary; the
% shape is what carries the lesson.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1.05cm, y=1.05cm, font=\small]

% Axes
\draw[->, thick] (0, 0) -- (10.4, 0) node[right] {age $t$};
\draw[->, thick] (0, 0) -- (0, 4.4) node[above] {hazard $h(t)$};

% Region shading boundaries at t = 3.0 and t = 7.0
\fill[engblue!8]  (0,0) rectangle (3.0, 4.2);
\fill[englime!12] (3.0,0) rectangle (7.0, 4.2);
\fill[engred!8]   (7.0,0) rectangle (10.2, 4.2);

% Infant-mortality: falling branch, clamped positive. A decaying
% curve from a high value down to the flat level 1.0.
\draw[very thick, engblue, smooth]
  plot[domain=0.35:3.0, samples=60]
  (\x, {1.0 + 3.0*exp(-1.3*\x)});

% Useful life: flat constant hazard at level 1.0
\draw[very thick, englime!60!black]
  (3.0, 1.0) -- (7.0, 1.0);

% Wear-out: rising branch from 1.0 upward, a quadratic climb clamped
% inside the frame.
\draw[very thick, engred, smooth]
  plot[domain=7.0:10.0, samples=50]
  (\x, {1.0 + 0.32*(\x-7.0)*(\x-7.0)});

% Region dividers
\draw[dashed, enggray] (3.0, 0) -- (3.0, 4.2);
\draw[dashed, enggray] (7.0, 0) -- (7.0, 4.2);

% Region labels
\node[anchor=south, engblue, font=\scriptsize, align=center]
  at (1.5, 3.5) {infant\\mortality\\$k<1$};
\node[anchor=south, englime!45!black, font=\scriptsize, align=center]
  at (5.0, 3.5) {useful life\\$h=\lambda$ (constant)\\exponential};
\node[anchor=south, engred, font=\scriptsize, align=center]
  at (8.6, 3.5) {wear-out\\$k>1$};

% Constant-level reference tick
\node[anchor=east, font=\scriptsize] at (-0.05, 1.0) {$\lambda$};
\draw (0, 1.0) -- (-0.08, 1.0);

\end{tikzpicture}
\caption[The bathtub hazard curve]{The bathtub curve of reliability
engineering, three hazard regimes stitched together. Early in life the
hazard falls as manufacturing defects are culled, a Weibull shape
$k<1$. Through the useful-life plateau the hazard is constant at
$\lambda$, the one regime where the exponential lifetime model of
section 13.4 holds and the memoryless property applies. Late in life
the hazard rises as wear, fatigue, and corrosion accumulate, a Weibull
shape $k>1$. The exponential is the right model only in the flat middle;
fitting it across a full lifetime averages away the two regimes that
drive the earliest and the latest failures. Source: schematic by the
authors; the hazard-survival identity $S(t)=\exp(-\int_0^t h)$ is
derived in section 13.4.}
\label{fig:vol02ch13:hazard-bathtub}
\end{figure}
