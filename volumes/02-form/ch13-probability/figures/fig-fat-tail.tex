% Gaussian vs heavy-tailed density: same centre, very different tail
% mass. Log-scaled vertical conceptual sketch to expose the tail gap.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

\draw[->, thick] (-0.2, 0) -- (9.2, 0) node[right] {standardised loss $z$};
\draw[->, thick] (0, -0.2) -- (0, 5.2) node[above] {density (log scale)};

% Gaussian (on log scale: parabola opening down): log f ~ -z^2/2
% map z in [0,4.5] to x in [0,9]; y = 4.6 - 0.45*z^2
\draw[thick, engblue, smooth]
  plot[domain=0:4.4, samples=80] ({2*\x}, {max(-0.1, 4.6 - 0.45*\x*\x)});
\node[engblue, anchor=west] at (2.0, 3.6) {Gaussian};

% Heavy tail (power law: log f ~ -alpha*log(1+z)); decays much slower
\draw[thick, engred, smooth]
  plot[domain=0:4.4, samples=80] ({2*\x}, {max(-0.1, 4.6 - 1.7*ln(1+\x))});
\node[engred, anchor=west] at (5.6, 2.3) {heavy-tailed};

% Highlight the tail gap at z = 3.5 (x = 7)
\pgfmathsetmacro{\zg}{3.5}
\pgfmathsetmacro{\yg}{4.6 - 0.45*\zg*\zg}
\pgfmathsetmacro{\yr}{4.6 - 1.7*ln(1+\zg)}
\draw[<->, thick, gray] (2*\zg, \yg) -- (2*\zg, \yr);
\node[anchor=west, font=\scriptsize, text width=3.0cm] at (2*\zg+0.2, {(\yg+\yr)/2})
  {tail gap: orders of magnitude more mass than the Gaussian
   assigns};

\node[anchor=north, font=\scriptsize] at (2*\zg, -0.15) {$z \approx 3.5$};

\end{tikzpicture}
\caption[Gaussian versus heavy-tailed tail mass]{Two densities with
the same centre and the same nominal spread, plotted on a
logarithmic vertical scale to expose the tails. The Gaussian
(parabolic on this scale) assigns vanishingly small probability
beyond three standard deviations; the heavy-tailed law (nearly
linear decay on this scale) assigns far more. A risk model that
fits the centre with a Gaussian and extrapolates to the tail
understates the probability of a large loss by the height of the
gap, which grows without bound as the threshold moves out. This is
the mechanism behind the underpriced joint-default risk of the 2008
crisis discussed in this section. Source: schematic by the authors.}
\label{fig:vol02:ch13:fat-tail}
\end{figure}
