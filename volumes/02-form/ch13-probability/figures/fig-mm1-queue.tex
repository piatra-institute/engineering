% M/M/1 congestion wall: the mean number in system E[N] = rho/(1-rho)
% as a function of utilisation rho. The curve is flat until rho ~ 0.7,
% then climbs steeply and diverges as rho -> 1. Plotted from the closed
% form; the vertical asymptote at rho = 1 is the saturation wall.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=10.5cm, height=7.0cm,
  xlabel={utilisation $\rho=\lambda/\mu$},
  ylabel={mean number in system $\E[N]$},
  xmin=0, xmax=1, ymin=0, ymax=12,
  domain=0:0.92, samples=140,
  legend pos=north west,
  legend cell align=left,
  grid=both,
  grid style={enggray!18},
  tick label style={font=\small},
  label style={font=\small},
  legend style={font=\scriptsize},
]

% E[N] = rho / (1 - rho), clamped by the plot ymax
\addplot[very thick, engblue, domain=0:0.92, samples=140]
  {x/(1-x)};
\addlegendentry{$\E[N]=\rho/(1-\rho)$}

% Markers at rho = 0.8 (E[N] = 4) and rho = 0.9 (E[N] = 9)
\addplot[only marks, mark=*, engred, mark size=2pt]
  coordinates {(0.8, 4) (0.9, 9)};
\node[anchor=west, engred, font=\scriptsize] at (axis cs:0.8, 4)
  {$\rho=0.8:\ \E[N]=4$};
\node[anchor=east, engred, font=\scriptsize] at (axis cs:0.9, 9)
  {$\rho=0.9:\ \E[N]=9$};

% Guide lines for the two markers
\draw[dashed, enggray] (axis cs:0.8,0) -- (axis cs:0.8,4);
\draw[dashed, enggray] (axis cs:0.9,0) -- (axis cs:0.9,9);

\end{axis}
\end{tikzpicture}
\caption[The M/M/1 congestion wall]{The mean number in an M/M/1 queue,
$\E[N]=\rho/(1-\rho)$, as a function of utilisation $\rho=\lambda/\mu$.
The queue is nearly empty at low load and stays modest up to about
$\rho=0.7$, then the $\rho/(1-\rho)$ term climbs steeply and diverges as
$\rho\to 1$. Raising utilisation from $\rho=0.8$ (four in system) to
$\rho=0.9$ (nine in system) more than doubles the mean occupancy for a
$12.5\,\%$ rise in load, the saturation non-linearity a capacity plan
must respect. The birth-death derivation of section 13.7 gives the
geometric stationary law $\pi_n=(1-\rho)\rho^n$ behind this mean, and
the design exercises push it to the heavy-tailed-service M/G/1 case
where the wait is worse still. Source: computed by the authors from the
M/M/1 stationary distribution.}
\label{fig:vol02ch13:mm1-queue}
\end{figure}
