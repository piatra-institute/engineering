% Poisson process: a rate-lambda arrival timeline with exponential
% inter-arrival gaps, and the thinning operation that splits it into a
% rare critical-alarm sub-stream.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1.15cm, y=1cm, font=\small]

% ---- Full arrival stream (rate lambda) ----
\draw[->, thick, black] (0,3) -- (10.6,3)
  node[right, font=\scriptsize] {time};
% arrival marks (irregular gaps, exponential-looking spacing)
\foreach \t/\lab in {0.7/1, 1.5/2, 2.0/3, 3.4/4, 3.9/5, 4.3/6,
                     5.8/7, 7.0/8, 7.4/9, 8.9/10} {
  \draw[thick, engblue] (\t,2.82) -- (\t,3.18);
}
% brace-marked inter-arrival gap
\draw[decorate, decoration={brace, amplitude=4pt, mirror}, enggray]
  (2.0,2.6) -- (3.4,2.6);
\node[anchor=north, font=\scriptsize, enggray] at (2.7,2.55)
  {gap $\sim\mathrm{Exp}(\lambda)$};
\node[anchor=south west, font=\scriptsize\bfseries, engblue] at (0,3.3)
  {full stream: rate $\lambda$};

% ---- Thinning: each arrival critical w.p. q ----
\draw[->, thick, black] (0,1) -- (10.6,1);
% keep a sparse subset as "critical" (marks 4 and 8 above kept)
\foreach \t in {3.4, 7.0} {
  \draw[thick, engred] (\t,0.82) -- (\t,1.18);
}
% ghosted non-critical marks
\foreach \t in {0.7,1.5,2.0,3.9,4.3,5.8,7.4,8.9} {
  \draw[thick, enggray!45] (\t,0.86) -- (\t,1.14);
}
\node[anchor=south west, font=\scriptsize\bfseries, engred] at (0,1.3)
  {critical sub-stream: rate $q\lambda$};

% thinning arrows connecting kept arrivals
\foreach \t in {3.4, 7.0} {
  \draw[->, thin, engred!70, dashed] (\t,2.78) -- (\t,1.22);
}
\node[anchor=west, font=\scriptsize, engred] at (7.15,2.0)
  {kept w.p. $q$};

% superposition note
\node[anchor=north west, font=\scriptsize, align=left, enggray]
  at (0,0.4)
  {Superposition: merging rate-$\lambda_1$ and rate-$\lambda_2$ streams
   gives rate $\lambda_1+\lambda_2$.};

\end{tikzpicture}
\caption[The Poisson process, thinning, and superposition]{A Poisson
process of rate $\lambda$ as a stream of arrival instants (top, blue) with
independent inter-arrival gaps that are $\mathrm{Exp}(\lambda)$
distributed, the bridge between the discrete count in a window and the
continuous wait between events. Thinning keeps each arrival independently
with probability $q$ (here the critical alarms, red), and the retained
sub-stream is itself Poisson with rate $q\lambda$; the discarded arrivals
(grey) form an independent Poisson stream of rate $(1-q)\lambda$. The dual
operation, superposition, merges independent Poisson streams into one of
the summed rate. These closure properties are why the Poisson process is
the default model for arrivals in queueing, photon counts in detectors,
and request streams in networked systems. Source: schematic drawn by the
authors; see \cite[ch.~6]{text:wasserman2004}.}
\label{fig:vol02:ch13:poisson-process}
\end{figure}
