% Poisson-process closure: superposition and thinning on a timeline.
% Two independent input streams (rates lambda_1, lambda_2) merge into a
% pooled stream of rate lambda_1 + lambda_2 (superposition); the pooled
% stream is then thinned by keeping each arrival with probability q,
% yielding a Poisson stream of rate q(lambda_1 + lambda_2). Marks are
% placed at fixed illustrative instants, not simulated.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small,
  mark/.style={engblack, thick}]

% Stream A (rate lambda_1)
\draw[->, thick] (0, 4) -- (11.2, 4);
\node[anchor=east, engblue, font=\scriptsize] at (-0.1, 4)
  {stream A, $\lambda_1$};
\foreach \t in {0.7, 2.1, 3.0, 5.4, 7.2, 9.6} {
  \draw[engblue, thick] (\t, 3.78) -- (\t, 4.22);
}

% Stream B (rate lambda_2)
\draw[->, thick] (0, 3) -- (11.2, 3);
\node[anchor=east, engred, font=\scriptsize] at (-0.1, 3)
  {stream B, $\lambda_2$};
\foreach \t in {1.3, 4.0, 4.8, 6.5, 8.4, 10.3} {
  \draw[engred, thick] (\t, 2.78) -- (\t, 3.22);
}

% Superposition arrow
\node[anchor=west, enggray, font=\scriptsize] at (11.0, 3.5)
  {merge};
\draw[->, enggray] (11.1, 3.9) .. controls (11.7, 3.5) .. (11.1, 3.1);

% Pooled stream (rate lambda_1 + lambda_2): union of the two mark sets
\draw[->, thick] (0, 1.8) -- (11.2, 1.8);
\node[anchor=east, font=\scriptsize, align=right] at (-0.1, 1.8)
  {pooled\\$\lambda_1+\lambda_2$};
\foreach \t in {0.7, 1.3, 2.1, 3.0, 4.0, 4.8, 5.4, 6.5, 7.2, 8.4, 9.6, 10.3} {
  \draw[engblack, thick] (\t, 1.58) -- (\t, 2.02);
}

% Thinning: keep a subset with probability q (kept marks highlighted)
\draw[->, thick] (0, 0.4) -- (11.2, 0.4);
\node[anchor=east, font=\scriptsize, align=right] at (-0.1, 0.4)
  {thinned\\$q(\lambda_1+\lambda_2)$};
\foreach \t in {1.3, 4.0, 7.2, 10.3} {
  \draw[englime!55!black, very thick] (\t, 0.18) -- (\t, 0.62);
}
% Ghost marks for the discarded arrivals
\foreach \t in {0.7, 2.1, 3.0, 4.8, 5.4, 6.5, 8.4, 9.6} {
  \draw[enggray!50, thin] (\t, 0.28) -- (\t, 0.52);
}

% Downward connector between pooled and thinned
\node[anchor=west, enggray, font=\scriptsize] at (5.2, 1.1)
  {keep each with prob.\ $q$};

\end{tikzpicture}
\caption[Poisson superposition and thinning]{The two closure
operations that make the Poisson process the default arrival model.
\emph{Superposition}: merging two independent Poisson streams of rates
$\lambda_1$ and $\lambda_2$ gives a Poisson stream of rate
$\lambda_1+\lambda_2$, so the aggregate arrivals at a server fed by many
sources are themselves Poisson. \emph{Thinning}: keeping each pooled
arrival independently with probability $q$ (heavy marks kept, faint
marks discarded) gives a Poisson stream of rate $q(\lambda_1+\lambda_2)$,
so a severity-filtered or sampled substream stays Poisson. The marks are
placed at illustrative instants rather than simulated. These two
properties are why the depot-alarm worked example of section 13.7
answers four questions from one model. Source: schematic by the
authors.}
\label{fig:vol02ch13:poisson-superposition}
\end{figure}
