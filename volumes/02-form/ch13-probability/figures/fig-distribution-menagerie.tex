% Full-page figure: six-panel distribution menagerie for Chapter 13.
% Two rows of three panels. Each panel: thin axes, named distribution
% with its parameter, density or PMF. Stylised for recognition.
\begin{figure}[p]
\centering
\begin{tikzpicture}[x=1cm, y=1cm, font=\small]

% ---- Panel 1: Bernoulli / binomial Bin(10, 0.3) ----
\begin{scope}[shift={(0,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Binomial};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\mathrm{Bin}(10,\,0.3)$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % Tabulated PMF for n=10, p=0.3
  \foreach \k/\p in {0/0.028, 1/0.121, 2/0.233, 3/0.267, 4/0.200,
                     5/0.103, 6/0.037, 7/0.009, 8/0.001, 9/0.000,
                     10/0.000} {
    \pgfmathsetmacro\xpos{0.15 + \k*0.36}
    \pgfmathsetmacro\height{\p*9.0}
    \draw[fill=engblue!60, draw=engblue]
      (\xpos, 0) rectangle (\xpos+0.30, \height);
  }
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {$k$ successes};
\end{scope}

% ---- Panel 2: Poisson, lambda = 4 ----
\begin{scope}[shift={(5.2,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Poisson};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\lambda=4$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  \foreach \k/\p in {0/0.018, 1/0.073, 2/0.147, 3/0.195, 4/0.195,
                     5/0.156, 6/0.104, 7/0.060, 8/0.030, 9/0.013,
                     10/0.005} {
    \pgfmathsetmacro\xpos{0.15 + \k*0.36}
    \pgfmathsetmacro\height{\p*12.0}
    \draw[fill=engblue!60, draw=engblue]
      (\xpos, 0) rectangle (\xpos+0.30, \height);
  }
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {$k$ events};
\end{scope}

% ---- Panel 3: Geometric, p = 0.4 ----
\begin{scope}[shift={(10.4,0)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Geometric};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$p=0.4$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  \foreach \k/\p in {1/0.400, 2/0.240, 3/0.144, 4/0.086, 5/0.052,
                     6/0.031, 7/0.019, 8/0.011, 9/0.007, 10/0.004} {
    \pgfmathsetmacro\xpos{0.15 + (\k - 1)*0.36}
    \pgfmathsetmacro\height{\p*6.0}
    \draw[fill=engblue!60, draw=engblue]
      (\xpos, 0) rectangle (\xpos+0.30, \height);
  }
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1)
    {$k$ trials to first success};
\end{scope}

% ---- Panel 4: Normal N(0, 1) ----
\begin{scope}[shift={(0,-4.8)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Normal};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\mathcal{N}(0, 1)$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  \draw[thick, engblue, smooth]
    plot[domain=0.05:3.95, samples=70]
    (\x, {2.4*exp(-((\x-2.0)*(\x-2.0))/(2*0.55*0.55))});
  \draw[dashed, enggray] (2.0, 0) -- (2.0, 2.4);
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {$\mu$};
\end{scope}

% ---- Panel 5: Exponential, lambda = 1 ----
\begin{scope}[shift={(5.2,-4.8)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Exponential};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\lambda=1$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  \draw[thick, engred, smooth]
    plot[domain=0.0:3.95, samples=70]
    (\x, {2.4*exp(-\x*1.0)});
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {waiting time};
\end{scope}

% ---- Panel 6: Gamma, alpha=3, lambda=1 ----
\begin{scope}[shift={(10.4,-4.8)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Gamma};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\alpha=3,\ \lambda=1$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % gamma(3,1) density: x^2 e^{-x} / 2; max at x=2 with value 2/e^2 = 0.27
  \draw[thick, engamber, smooth]
    plot[domain=0.05:3.95, samples=70]
    (\x, {8.5*(\x*\x*exp(-\x))/2.0});
  \draw[dashed, enggray] (2.0, 0) -- (2.0, 2.3);
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {mode};
\end{scope}

% ---- Panel 7: Uniform [0, 1] ----
\begin{scope}[shift={(0,-9.6)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Uniform};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\mathrm{Unif}(0, 1)$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  \draw[thick, engblue] (0.4, 0) -- (0.4, 1.8) -- (3.6, 1.8) -- (3.6, 0);
  \node[anchor=north, font=\scriptsize] at (0.4, -0.1) {$a$};
  \node[anchor=north, font=\scriptsize] at (3.6, -0.1) {$b$};
\end{scope}

% ---- Panel 8: Beta(2, 5) ----
\begin{scope}[shift={(5.2,-9.6)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Beta};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {$\alpha=2,\ \beta=5$};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % beta(2, 5) density on [0, 1]; scale x to [0, 4]
  \draw[thick, engred, smooth]
    plot[domain=0.02:3.98, samples=80]
    (\x, {0.6*30*(\x/4.0)*pow(1-\x/4.0, 4)});
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {$x \in [0,1]$};
\end{scope}

% ---- Panel 9: Cauchy ----
\begin{scope}[shift={(10.4,-9.6)}]
  \node[anchor=south, font=\small\bfseries] at (2.0, 3.2) {Cauchy};
  \node[anchor=south, font=\scriptsize] at (2.0, 2.8)
    {no mean};
  \draw[->, thick] (-0.1, 0) -- (4.3, 0);
  \draw[->, thick] (0, -0.1) -- (0, 2.7);
  % standard Cauchy: 1/(pi(1+(x-mu)^2))
  \draw[thick, engamber, smooth]
    plot[domain=0.05:3.95, samples=80]
    (\x, {2.0/(1.0 + ((\x-2.0)*(\x-2.0)))});
  \draw[dashed, enggray] (2.0, 0) -- (2.0, 2.0);
  \node[anchor=north, font=\scriptsize] at (2.0, -0.1) {$x_{0}$};
\end{scope}

\end{tikzpicture}
\caption[The distribution menagerie of Chapter 13]{The nine
distributions an engineer should recognise on sight: three discrete
(binomial, Poisson, geometric) in the top row; three continuous on
the half line (normal, exponential, gamma) in the middle; three
shape variants (uniform on a bounded interval, beta as a flexible
shape on $[0, 1]$, Cauchy with no finite mean) in the bottom. Each
panel labels the parameter that controls the shape. The vertical
scales are stylised for recognition; the relative proportions
within each panel reflect each distribution's actual mass or
density.}
\label{fig:vol02:ch13:distribution-menagerie}
\end{figure}
