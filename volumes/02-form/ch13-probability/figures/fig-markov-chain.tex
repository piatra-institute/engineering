% Markov-chain state-transition graph for the two-state weather
% chain of section 13.7, plus a running-occupancy convergence sketch.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
  state/.style={draw=engblack, circle, minimum size=1.2cm,
    align=center, font=\small\bfseries, fill=engblue!12},
  edge/.style={-{Stealth}, thick, draw=engblack!80, font=\scriptsize},
]

% ---- Left: state-transition graph ----
\begin{scope}[shift={(0, 0)}]
  \node[state] (S) at (0, 0) {Sunny};
  \node[state] (R) at (4, 0) {Rainy};

  % Self-loops (drawn explicitly to avoid relying on extra libraries)
  \draw[edge] (S) .. controls (-2.0, 1.0) and (-2.0, -1.0) .. (S)
    node[midway, left, font=\scriptsize] {$0.9$};
  \draw[edge] (R) .. controls (6.0, 1.0) and (6.0, -1.0) .. (R)
    node[midway, right, font=\scriptsize] {$0.5$};

  % Transitions between states
  \draw[edge, bend left=18] (S) to
    node[above, font=\scriptsize] {$0.1$} (R);
  \draw[edge, bend left=18] (R) to
    node[below, font=\scriptsize] {$0.5$} (S);

  \node[anchor=north, font=\small\bfseries] at (2.0, -2.0)
    {Transition graph};
\end{scope}

% ---- Right: running occupancy converging to stationary 5/6 ----
\begin{scope}[shift={(8.0, -1.2)}]
  % Axes
  \draw[->, thick] (-0.2, 0) -- (5.5, 0);
  \draw[->, thick] (0, -0.2) -- (0, 2.8);
  % Stationary line at y = 2.5 (representing pi_1 = 5/6 = 0.833)
  \draw[dashed, engred, thick] (0, 2.5) -- (5.3, 2.5);
  \node[anchor=west, font=\scriptsize, engred] at (5.3, 2.5)
    {$\pi_{1} = 5/6$};

  % Stylised running estimate, oscillating then settling near 2.5
  \draw[thick, engblue]
    (0.0, 0.0) -- (0.2, 3.0) -- (0.5, 1.6) -- (0.8, 2.9)
    -- (1.2, 2.0) -- (1.7, 2.7) -- (2.2, 2.3) -- (2.8, 2.6)
    -- (3.4, 2.45) -- (4.0, 2.55) -- (4.6, 2.48) -- (5.2, 2.52);

  % Axis labels
  \node[anchor=north, font=\scriptsize] at (2.6, -0.15)
    {steps $t$ (log scale, $10^{1}$ to $10^{5}$)};
  \node[anchor=east, font=\scriptsize, rotate=90]
    at (-0.35, 1.4) {running fraction};
  \node[anchor=north, font=\small\bfseries] at (2.6, -0.8)
    {Convergence of running fraction};
\end{scope}

\end{tikzpicture}
\caption[The two-state weather Markov chain and its convergence]{The
two-state weather chain of section~\ref{sec:vol02:ch13:processes}.
Left: the transition graph, with the self-loops giving the
probability of staying in each state and the inter-state arrows
giving the probability of switching. The stationary distribution
solves $\vect{\pi}\mat{P} = \vect{\pi}$ and gives
$\pi_{1} = 5/6 \approx 0.833$ and $\pi_{2} = 1/6 \approx 0.167$.
Right: a stylised trace of the running fraction
$\hat\pi_{1}^{(t)} = (1/t) |\{s \leq t : X_{s} = 1\}|$ as a
function of $t$, converging to the stationary value at rate
roughly $1/\sqrt{t}$. The project of section~\ref{sec:vol02:ch13:project}
asks the reader to produce this plot for a chain of their choice.
Source: parameters from \cite[ch.~6]{text:wasserman2004}; running
trace stylised for illustration.}
\label{fig:vol02:ch13:markov-chain}
\end{figure}
