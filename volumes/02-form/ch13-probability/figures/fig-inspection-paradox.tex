% The inspection (length-biased) paradox. For a renewal process with
% interval density f(x), an interval hit by a random inspection instant
% is sampled with probability proportional to its own length, so the
% observed density is the length-biased x f(x) / E[X], shifted to the
% right of the underlying f(x). Panel plots a gamma-like underlying
% density and its length-biased counterpart, both stylised, with the two
% means marked. Approved colours only.
\begin{figure}[htbp]
\centering
\begin{tikzpicture}
\begin{axis}[
  width=11cm, height=6.6cm,
  xlabel={interval length $x$},
  ylabel={density},
  xmin=0, xmax=8, ymin=0, ymax=0.55,
  domain=0.001:8, samples=160,
  legend pos=north east,
  legend cell align=left,
  grid=both, grid style={enggray!18},
  tick label style={font=\small},
  label style={font=\small},
  legend style={font=\scriptsize},
]

% Underlying interval density: Gamma(shape 2, rate 1), f(x) = x e^{-x},
% mean 2. Peak at x = 1, value 1/e ~ 0.368.
\addplot[very thick, engblue, domain=0.001:8, samples=160]
  {x*exp(-x)};
\addlegendentry{underlying $f(x)$, mean $2$}

% Length-biased density: x f(x) / E[X] = x * x e^{-x} / 2 = x^2 e^{-x}/2,
% which is Gamma(shape 3, rate 1), mean 3. Peak at x = 2, value
% 4 e^{-2}/2 = 2 e^{-2} ~ 0.271.
\addplot[very thick, engred, domain=0.001:8, samples=160]
  {0.5*x*x*exp(-x)};
\addlegendentry{observed (length-biased), mean $3$}

% mean markers
\draw[dashed, engblue] (axis cs:2,0) -- (axis cs:2,0.37);
\node[anchor=south, engblue, font=\scriptsize] at (axis cs:2,0.37)
  {$\E[X]=2$};
\draw[dashed, engred] (axis cs:3,0) -- (axis cs:3,0.28);
\node[anchor=south west, engred, font=\scriptsize] at (axis cs:3,0.28)
  {observed mean $=3$};

\end{axis}
\end{tikzpicture}
\caption[The inspection paradox]{An interval sampled by a random
inspection instant is not a typical interval: a long interval covers
more of the timeline and is therefore more likely to be the one the
inspection lands in. The observed length density (red) is the underlying
density (blue) reweighted by length, $x\,f(x)/\E[X]$, which for the
gamma interval shown shifts the mean from $2$ to $3$. An engineer who
estimates a mean service interval, a mean time between failures, or a
mean bus headway by timing the interval a random visit falls into,
rather than by sampling intervals directly, overstates the mean by the
ratio $\E[X^2]/\E[X]^2 = 1 + \mathrm{CV}^2$. The bias grows with the
spread of the intervals and vanishes only for a constant interval.
Source: schematic by the authors; the length-bias factor is derived in
the worked example of section 13.6.}
\label{fig:vol02ch13:inspection-paradox}
\end{figure}
