% Projection of v onto a course direction: along-track and cross-track.
% Illustrates v = v_parallel + v_perp via the dot product.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % Course direction (unit vector c-hat along x).
  \draw[enggray, thick, ->] (0,0) -- (5.2,0)
    node[right, font=\footnotesize]{$\unitvec{c}$ (desired course)};

  % Velocity vector v = (4, 2).
  \draw[engblue, very thick, ->] (0,0) -- (4,2)
    node[above right, font=\footnotesize]{$\vect{v}$};

  % Along-track component (projection onto c-hat).
  \draw[engred, very thick, ->] (0,0) -- (4,0)
    node[below, font=\footnotesize, midway]{$\vect{v}_{\parallel} = (\vect{v}\cdot\unitvec{c})\,\unitvec{c}$};

  % Cross-track component.
  \draw[engorange, very thick, ->] (4,0) -- (4,2)
    node[right, font=\footnotesize, midway]{$\vect{v}_{\perp}$};

  % Right-angle mark at the foot of the perpendicular.
  \draw[enggray] (3.8,0) -- (3.8,0.2) -- (4.0,0.2);

  % Angle mark at origin.
  \draw[enggray] (0.7,0) arc (0:26.57:0.7);
  \node[font=\footnotesize] at (0.95,0.22) {$\theta$};

\end{tikzpicture}
