% Peak field E_max in a coaxial cable versus the radius ratio b/a, for fixed
% outer radius b and fixed voltage. The normalised curve a*ln(b/a) that
% E_max is inversely proportional to peaks at b/a = e, so E_max is minimised
% there.
\begin{tikzpicture}[>=Stealth, scale=1.0]
  \begin{axis}[
    width=10cm, height=6.5cm,
    xlabel={ratio $b/a$}, ylabel={normalised peak field $E_{\max}\,b / V_{ab}$},
    xmin=1, xmax=8, ymin=0, ymax=4,
    axis lines=left,
    label style={font=\footnotesize},
    tick label style={font=\footnotesize},
    samples=120,
    domain=1.05:8,
  ]
    % E_max * b / V = (b/a) / ln(b/a). Let t = b/a; plot t/ln(t).
    \addplot[engblue, very thick] {x/ln(x)};

    % Mark the minimum at b/a = e.
    \draw[engred, dashed] (axis cs:2.718,0) -- (axis cs:2.718,2.718);
    \fill[engred] (axis cs:2.718,2.718) circle (2pt);
    \node[font=\footnotesize, engred, anchor=south west]
      at (axis cs:2.718,2.75) {minimum at $b/a = e$};

  \end{axis}
\end{tikzpicture}
