% Wake velocity profile behind a bluff body: u(y) = U (1 - 0.5 exp(-(2y/d)^2)).
% The shaded momentum deficit u(u - U) integrated across the wake gives the
% drag per unit span.
\begin{tikzpicture}[>=Stealth, scale=1.0]
  \begin{axis}[
    width=9cm, height=6.5cm,
    xlabel={streamwise velocity $u/U$},
    ylabel={transverse position $y/d$},
    xmin=0, xmax=1.15, ymin=-2, ymax=2,
    axis lines=left,
    label style={font=\footnotesize},
    tick label style={font=\footnotesize},
    samples=120,
  ]
    % Free-stream reference line u = U.
    \draw[enggray, dashed] (axis cs:1,-2) -- (axis cs:1,2);
    \node[font=\footnotesize, enggray, anchor=south] at (axis cs:1,1.6) {$U$};

    % Wake profile: x-axis is u/U, y-axis is y/d, so plot parametrically in y.
    \addplot[engblue, very thick, domain=-2:2, variable=\t] (
      {1 - 0.5*exp(-(2*\t)^2)}, {\t});
    \node[font=\footnotesize, engblue, anchor=west] at (axis cs:0.55,-1.2) {$u(y)$};

    % Centreline marker.
    \fill[engred] (axis cs:0.5,0) circle (2pt);
    \node[font=\footnotesize, engred, anchor=west] at (axis cs:0.52,0.2)
      {centreline deficit $U/2$};

  \end{axis}
\end{tikzpicture}
