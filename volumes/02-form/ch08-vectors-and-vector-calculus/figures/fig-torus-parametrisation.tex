% Torus parametrisation: the two angles u (tube) and v (centreline).
% A cross-section of the tube shows the tube radius a and the centreline
% radius R, with the area factor (R + a cos u) larger on the outer rim.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % Centreline axis of symmetry.
  \draw[enggray, dashed] (0,-2.4) -- (0,2.4);
  \node[font=\footnotesize, enggray] at (0.25,2.3) {axis};

  % Two tube cross-sections (circles of radius a) at the outer and inner rim.
  \def\R{2.4}
  \def\a{0.7}

  % Outer tube cross-section.
  \draw[engblue, thick] (\R,0) circle (\a);
  \fill (\R,0) circle (0.04);
  \node[font=\footnotesize, engblue, above] at (\R,\a) {outer rim ($u=0$)};

  % Inner tube cross-section (mirrored).
  \draw[engblue, thick] (-\R,0) circle (\a);
  \fill (-\R,0) circle (0.04);
  \node[font=\footnotesize, engblue, above] at (-\R,\a) {inner rim};

  % Centreline radius R (from axis to a tube centre).
  \draw[engred, thick, <->] (0,-0.35) -- (\R,-0.35);
  \node[font=\footnotesize, engred, below] at ({\R/2},-0.35) {$R$};

  % Tube radius a on the outer cross-section.
  \draw[engorange, thick, <->] (\R,0) -- ({\R+\a},0);
  \node[font=\footnotesize, engorange, above right] at (\R,0.02) {$a$};

  % Area factor annotation.
  \node[font=\footnotesize] at (0,-2.0)
    {area element $\propto a\,(R + a\cos u)$};

\end{tikzpicture}
