% Material derivative: a fixed sensor versus a parcel carried by a steady flow.
% A steady temperature field rising along a duct; the parcel heats while the
% wall sensor reads a constant value.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % Duct walls.
  \draw[enggray, thick] (0,0) -- (8,0);
  \draw[enggray, thick] (0,2) -- (8,2);

  % Temperature shading suggested by tick labels along the axis.
  \foreach \x/\T in {0.5/20, 2.5/70, 4.5/120, 6.5/170} {
    \draw[enggray, dashed] (\x,0) -- (\x,2);
    \node[font=\footnotesize, below=2pt, enggray] at (\x,0) {$\T\,^{\circ}\mathrm{C}$};
  }

  % Flow arrow.
  \draw[engblue, very thick, ->] (0.6,1.4) -- (7.4,1.4)
    node[above left, font=\footnotesize]{$\vect{v} = (U,0)$};

  % Fixed wall sensor.
  \fill[engred] (4.5,2) circle (0.07);
  \node[font=\footnotesize, above=3pt, engred] at (4.5,2)
    {fixed sensor: $\partial T/\partial t = 0$};

  % Moving parcel at three instants.
  \foreach \x in {1.2,3.2,5.2} {
    \draw[engorange, thick, fill=engorange!20] (\x,0.7) circle (0.22);
  }
  \node[font=\footnotesize, below=2pt, engorange] at (3.2,0.45)
    {parcel: $\mathrm{D}T/\mathrm{D}t = U\,\partial T/\partial x$};

  \node[font=\footnotesize] at (4,-1.1)
    {steady field, moving parcel still heats};

\end{tikzpicture}
