% Helmholtz decomposition of F = (x - y, x + y) into a radial (source)
% part (x, y) and a rotational part (-y, x).
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % --- Panel A: the full field F = (x - y, x + y) ---
  \begin{scope}[xshift=0cm]
    \draw[enggray] (0,0) circle (1.5);
    \foreach \ang in {0,45,90,135,180,225,270,315} {
      \pgfmathsetmacro{\xs}{1.0*cos(\ang)}
      \pgfmathsetmacro{\ys}{1.0*sin(\ang)}
      \pgfmathsetmacro{\fx}{\xs - \ys}
      \pgfmathsetmacro{\fy}{\xs + \ys}
      \draw[->, engblue, thick] (\xs,\ys) -- ({\xs + 0.32*\fx},{\ys + 0.32*\fy});
    }
    \node[font=\footnotesize, below=1.7cm] at (0,0) {$\vect{F} = (x - y,\ x + y)$};
  \end{scope}

  % --- Panel B: radial (curl-free) part (x, y) ---
  \begin{scope}[xshift=4.6cm]
    \draw[enggray] (0,0) circle (1.5);
    \foreach \ang in {0,45,90,135,180,225,270,315} {
      \pgfmathsetmacro{\xs}{1.0*cos(\ang)}
      \pgfmathsetmacro{\ys}{1.0*sin(\ang)}
      \draw[->, engred, thick] (\xs,\ys) -- ({1.45*\xs},{1.45*\ys});
    }
    \node[font=\footnotesize, below=1.7cm] at (0,0) {$\grad\Phi = (x, y)$ (all divergence)};
  \end{scope}

  % --- Panel C: rotational (divergence-free) part (-y, x) ---
  \begin{scope}[xshift=9.2cm]
    \draw[enggray] (0,0) circle (1.5);
    \foreach \ang in {0,45,90,135,180,225,270,315} {
      \pgfmathsetmacro{\xs}{1.0*cos(\ang)}
      \pgfmathsetmacro{\ys}{1.0*sin(\ang)}
      \draw[->, engorange, thick] (\xs,\ys) -- ({\xs - 0.45*\ys},{\ys + 0.45*\xs});
    }
    \node[font=\footnotesize, below=1.7cm] at (0,0) {$(-y, x)$ (all curl)};
  \end{scope}

\end{tikzpicture}
