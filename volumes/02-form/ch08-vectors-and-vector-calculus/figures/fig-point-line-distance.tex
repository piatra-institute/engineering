% Distance from a point P to a line through A with direction d-hat, read from
% the cross product: dist = || (P - A) x d-hat ||, the height of the
% parallelogram spanned by (P - A) and d-hat.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % The line through A with direction d-hat.
  \draw[enggray, thick] (-1.0,0) -- (5.4,0);
  \fill[engblack] (0,0) circle (2pt);
  \node[font=\footnotesize, anchor=north east] at (0,-0.05){$A$};
  \draw[enggray, thick, ->] (0,0) -- (1.6,0)
    node[below, font=\footnotesize, midway]{$\unitvec{d}$};

  % The external point P.
  \fill[engblue] (3.5,2.2) circle (2pt);
  \node[font=\footnotesize, engblue, anchor=south] at (3.5,2.3){$P$};

  % The vector from A to P.
  \draw[engblue, very thick, ->] (0,0) -- (3.5,2.2)
    node[midway, above left, font=\footnotesize]{$\vect{P}-\vect{A}$};

  % The foot of the perpendicular and the distance.
  \draw[engred, very thick, ->] (3.5,0) -- (3.5,2.2)
    node[midway, right, font=\footnotesize]{$D$};
  \draw[enggray] (3.3,0) -- (3.3,0.2) -- (3.5,0.2);

  % The parallelogram (P-A) and d-hat span; its height is the distance.
  \draw[engorange, dashed] (3.5,2.2) -- (5.1,2.2);
  \draw[engorange, dashed] (1.6,0) -- (5.1,2.2);
  \node[font=\scriptsize, engorange, anchor=west] at (4.4,1.1)
    {area $=\norm{(\vect{P}-\vect{A})\times\unitvec{d}}$};

\end{tikzpicture}
