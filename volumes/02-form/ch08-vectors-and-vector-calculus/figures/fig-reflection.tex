% Reflection of a vector off a plane with unit normal n-hat.
% v' = v - 2 (v . n-hat) n-hat: the normal component reverses, the
% tangential component is preserved.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % The reflecting surface (drawn as a horizontal line in this section view).
  \draw[enggray, thick] (-3.2,0) -- (3.2,0)
    node[right, font=\footnotesize]{surface};
  % Hatching below the surface to mark the solid side.
  \foreach \x in {-3.0,-2.6,...,3.0}{
    \draw[enggray] (\x,0) -- (\x+0.3,-0.3);
  }

  % Unit normal at the impact point (origin).
  \draw[enggray, thick, ->] (0,0) -- (0,1.4)
    node[right, font=\footnotesize]{$\unitvec{n}$};

  % Incoming vector v arriving at the origin (from upper left, pointing down-right).
  \draw[engblue, very thick, ->] (-2.4,1.8) -- (0,0)
    node[midway, above left, font=\footnotesize]{$\vect{v}$};

  % Reflected vector v' leaving the origin (up-right), tangential part kept,
  % normal part reversed.
  \draw[engred, very thick, ->] (0,0) -- (2.4,1.8)
    node[midway, above right, font=\footnotesize]{$\vect{v}'$};

  % Angle-of-incidence and angle-of-reflection marks against the normal.
  \draw[engorange] (0,0.9) arc (90:143:0.9);
  \node[font=\scriptsize, engorange] at (-0.55,1.05){$\alpha$};
  \draw[engorange] (0,0.9) arc (90:37:0.9);
  \node[font=\scriptsize, engorange] at (0.55,1.05){$\alpha$};

\end{tikzpicture}
