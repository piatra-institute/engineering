% The centroid of a region read from a boundary integral. The first-moment
% integral over the interior is converted to a surface integral of a
% quadratic field, so the centre of mass is fixed by boundary data alone.
\begin{tikzpicture}[>=Stealth, scale=1.0]

  % An irregular closed region (a smooth blob) as a section of the object.
  \draw[engblue, very thick]
    plot[smooth cycle, tension=0.8] coordinates{
      (-2.2,0.2) (-1.4,1.6) (0.2,2.0) (1.8,1.3)
      (2.3,-0.3) (1.2,-1.5) (-0.6,-1.7) (-2.0,-0.9)};
  \node[font=\footnotesize, engblue, anchor=west] at (1.9,1.5){$\partial V$};

  % Outward normals sampled at a few boundary points.
  \draw[enggray, ->] (0.2,2.0) -- (0.35,2.7);
  \draw[enggray, ->] (2.3,-0.3) -- (3.0,-0.4);
  \draw[enggray, ->] (-2.0,-0.9) -- (-2.7,-1.3);
  \draw[enggray, ->] (-0.6,-1.7) -- (-0.7,-2.4);
  \node[font=\scriptsize, enggray, anchor=south west] at (0.35,2.55){$\unitvec{n}$};

  % The centroid, marked inside.
  \fill[engred] (-0.1,0.05) circle (2.5pt);
  \node[font=\footnotesize, engred, anchor=south west] at (-0.05,0.1)
    {$\bar{\vect{r}}$};

  % The origin and a position vector to a boundary sample.
  \fill[engblack] (-3.4,-2.6) circle (1.6pt);
  \node[font=\scriptsize, anchor=north east] at (-3.4,-2.6){$O$};
  \draw[engorange, thick, ->] (-3.4,-2.6) -- (0.2,2.0)
    node[midway, above left, font=\scriptsize]{$\vect{r}$};

\end{tikzpicture}
