\begin{itemize}

  \item Walter Vincenti, \textit{What Engineers Know and How They Know It}.
  \item George Polya, \textit{How to Solve It}.
  \item Richard Feynman, \textit{The Feynman Lectures on Physics}.
  \item Charles Perrow, \textit{Normal Accidents}.
  \item James C.\ Scott, \textit{Seeing Like a State}.
  \item Edward Tufte, \textit{The Visual Display of Quantitative Information}.
  \item David MacKay, \textit{Sustainable Energy: Without the Hot Air}.
  \item Donella Meadows, \textit{Thinking in Systems}.
\end{itemize}
